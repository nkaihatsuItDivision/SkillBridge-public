    並行プログラミングの基本概念

    同期処理 vs 並行処理

    import java.util.concurrent.*;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;

public class ConcurrencyBasics {
    // 同期処理の例（ブロッキング）
    public static void synchronousExample() {
        System.out.println(&quot;同期処理開始&quot;);

        // 時間のかかる処理をシミュレート
        try {
            Thread.sleep(3000); // 3秒間ブロック
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }

        System.out.println(&quot;同期処理完了&quot;);
        // この間、プログラムは他の処理を実行できない
    }

    // 並行処理の例（ノンブロッキング）
    public static void asynchronousExample() throws InterruptedException, ExecutionException {
        System.out.println(&quot;並行処理開始&quot;);

        ExecutorService executor = Executors.newFixedThreadPool(3);

        // 複数のタスクを並行実行
        Future&lt;String&gt; task1 = executor.submit(() -&gt; performTask(&quot;タスク1&quot;, 2000));
        Future&lt;String&gt; task2 = executor.submit(() -&gt; performTask(&quot;タスク2&quot;, 1500));
        Future&lt;String&gt; task3 = executor.submit(() -&gt; performTask(&quot;タスク3&quot;, 3000));

        System.out.println(&quot;すべてのタスクを開始しました&quot;);

        // 結果を取得（必要に応じてブロック）
        System.out.println(task1.get());
        System.out.println(task2.get());
        System.out.println(task3.get());

        System.out.println(&quot;すべてのタスクが完了しました&quot;);
        executor.shutdown();
    }

    private static String performTask(String taskName, int delayMs) {
        try {
            System.out.println(taskName + &quot;開始 - &quot; + getCurrentTime());
            Thread.sleep(delayMs);
            System.out.println(taskName + &quot;完了 - &quot; + getCurrentTime());
            return taskName + &quot;の結果&quot;;
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            return taskName + &quot;が中断されました&quot;;
        }
    }

    private static String getCurrentTime() {
        return LocalDateTime.now().format(DateTimeFormatter.ofPattern(&quot;HH:mm:ss.SSS&quot;));
    }
}

    Threadクラスの基本的な使い方

    public class ThreadBasics {
    // Threadクラスを継承する方法
    static class WorkerThread extends Thread {
        private String threadName;

        public WorkerThread(String name) {
            this.threadName = name;
        }

        @Override
        public void run() {
            for (int i = 1; i &lt;= 5; i++) {
                System.out.printf(&quot;%s: カウント %d%n&quot;, threadName, i);
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    System.out.println(threadName + &quot;が中断されました&quot;);
                    Thread.currentThread().interrupt();
                    return;
                }
            }
            System.out.println(threadName + &quot;完了&quot;);
        }
    }

    // Runnableインターフェースを実装する方法（推奨）
    static class WorkerRunnable implements Runnable {
        private String taskName;

        public WorkerRunnable(String name) {
            this.taskName = name;
        }

        @Override
        public void run() {
            for (int i = 1; i &lt;= 5; i++) {
                System.out.printf(&quot;%s: 処理 %d%n&quot;, taskName, i);
                try {
                    Thread.sleep(800);
                } catch (InterruptedException e) {
                    System.out.println(taskName + &quot;が中断されました&quot;);
                    Thread.currentThread().interrupt();
                    return;
                }
            }
            System.out.println(taskName + &quot;完了&quot;);
        }
    }

    public static void demonstrateThreads() throws InterruptedException {
        System.out.println(&quot;=== Thread継承方式 ===&quot;);
        WorkerThread thread1 = new WorkerThread(&quot;スレッド1&quot;);
        WorkerThread thread2 = new WorkerThread(&quot;スレッド2&quot;);

        thread1.start();
        thread2.start();

        // メインスレッドが他のスレッドの完了を待つ
        thread1.join();
        thread2.join();

        System.out.println(&quot;\n=== Runnable実装方式 ===&quot;);
        Thread thread3 = new Thread(new WorkerRunnable(&quot;タスクA&quot;));
        Thread thread4 = new Thread(new WorkerRunnable(&quot;タスクB&quot;));

        thread3.start();
        thread4.start();

        thread3.join();
        thread4.join();

        System.out.println(&quot;すべてのスレッドが完了しました&quot;);
    }

    // ラムダ式を使用したより簡潔な書き方
    public static void demonstrateWithLambda() throws InterruptedException {
        Thread lambdaThread = new Thread(() -&gt; {
            for (int i = 1; i &lt;= 3; i++) {
                System.out.println(&quot;ラムダスレッド: &quot; + i);
                try {
                    Thread.sleep(500);
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                    return;
                }
            }
        });

        lambdaThread.start();
        lambdaThread.join();
        System.out.println(&quot;ラムダスレッド完了&quot;);
    }
}

    ExecutorServiceによるスレッドプール管理

    基本的なExecutorServiceの使用

    import java.util.concurrent.atomic.AtomicInteger;

public class ExecutorServiceBasics {
    // タスクの実行結果を返さない処理
    public static void executeTasksWithoutReturn() throws InterruptedException {
        ExecutorService executor = Executors.newFixedThreadPool(3);

        for (int i = 1; i &lt;= 10; i++) {
            final int taskId = i;
            executor.execute(() -&gt; {
                System.out.printf(&quot;タスク%d実行中 (スレッド: %s)%n&quot;, 
                                taskId, Thread.currentThread().getName());
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
                System.out.printf(&quot;タスク%d完了%n&quot;, taskId);
            });
        }

        executor.shutdown(); // 新しいタスクの受付を停止

        // すべてのタスクの完了を待つ
        if (!executor.awaitTermination(10, TimeUnit.SECONDS)) {
            System.out.println(&quot;タイムアウト - 強制終了します&quot;);
            executor.shutdownNow();
        }

        System.out.println(&quot;すべてのタスクが完了しました&quot;);
    }

    // タスクの実行結果を返す処理
    public static void submitTasksWithReturn() throws InterruptedException, ExecutionException {
        ExecutorService executor = Executors.newFixedThreadPool(3);
        List&lt;Future&lt;Integer&gt;&gt; futures = new ArrayList&lt;&gt;();

        // 複数のタスクを投入
        for (int i = 1; i &lt;= 5; i++) {
            final int input = i;
            Future&lt;Integer&gt; future = executor.submit(() -&gt; {
                System.out.printf(&quot;計算開始: %d の二乗%n&quot;, input);
                Thread.sleep(1000); // 処理時間をシミュレート
                int result = input * input;
                System.out.printf(&quot;計算完了: %d^2 = %d%n&quot;, input, result);
                return result;
            });
            futures.add(future);
        }

        // 結果を取得
        System.out.println(&quot;=== 計算結果 ===&quot;);
        for (int i = 0; i &lt; futures.size(); i++) {
            Future&lt;Integer&gt; future = futures.get(i);
            Integer result = future.get(); // ブロッキング呼び出し
            System.out.printf(&quot;タスク%d結果: %d%n&quot;, i + 1, result);
        }

        executor.shutdown();
    }

    // 様々なタイプのExecutorService
    public static void demonstrateExecutorTypes() throws InterruptedException {
        System.out.println(&quot;=== 固定サイズスレッドプール ===&quot;);
        ExecutorService fixedPool = Executors.newFixedThreadPool(2);
        executeTaskBatch(fixedPool, &quot;固定&quot;, 3);

        System.out.println(&quot;\n=== キャッシュスレッドプール ===&quot;);
        ExecutorService cachedPool = Executors.newCachedThreadPool();
        executeTaskBatch(cachedPool, &quot;キャッシュ&quot;, 3);

        System.out.println(&quot;\n=== 単一スレッドプール ===&quot;);
        ExecutorService singlePool = Executors.newSingleThreadExecutor();
        executeTaskBatch(singlePool, &quot;単一&quot;, 3);

        System.out.println(&quot;\n=== スケジュールされたスレッドプール ===&quot;);
        ScheduledExecutorService scheduledPool = Executors.newScheduledThreadPool(2);

        // 遅延実行
        scheduledPool.schedule(() -&gt; 
            System.out.println(&quot;3秒後に実行されるタスク&quot;), 3, TimeUnit.SECONDS);

        // 定期実行
        AtomicInteger counter = new AtomicInteger(0);
        ScheduledFuture&lt;?&gt; periodicTask = scheduledPool.scheduleAtFixedRate(() -&gt; {
            int count = counter.incrementAndGet();
            System.out.println(&quot;定期タスク実行回数: &quot; + count);
            if (count &gt;= 3) {
                // 3回実行したら停止（実際のアプリケーションでは適切なキャンセル処理を実装）
            }
        }, 1, 2, TimeUnit.SECONDS);

        // 6秒後にキャンセル
        scheduledPool.schedule(() -&gt; periodicTask.cancel(false), 6, TimeUnit.SECONDS);

        Thread.sleep(8000); // デモ用の待機
        scheduledPool.shutdown();
    }

    private static void executeTaskBatch(ExecutorService executor, String poolType, int taskCount) 
            throws InterruptedException {
        for (int i = 1; i &lt;= taskCount; i++) {
            final int taskId = i;
            executor.execute(() -&gt; {
                System.out.printf(&quot;[%s] タスク%d (スレッド: %s)%n&quot;, 
                                poolType, taskId, Thread.currentThread().getName());
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            });
        }

        executor.shutdown();
        executor.awaitTermination(5, TimeUnit.SECONDS);
    }
}

                    CompletableFuture.supplyAsync(() -&gt; {
                        try { Thread.sleep(1000); } catch (InterruptedException e) { 
                            Thread.currentThread().interrupt(); throw new RuntimeException(e); }

    public class SynchronizationBasics {
    // 同期化されていないカウンター（スレッドセーフでない）
    static class UnsafeCounter {
        private int count = 0;

        public void increment() {
            count++; // この操作はアトミックではない
        }

        public int getCount() {
            return count;
        }
    }

    // synchronizedを使用したスレッドセーフなカウンター
    static class SafeCounter {
        private int count = 0;

        public synchronized void increment() {
            count++;
        }

        public synchronized int getCount() {
            return count;
        }
    }

    // AtomicIntegerを使用したカウンター
    static class AtomicCounter {
        private AtomicInteger count = new AtomicInteger(0);

        public void increment() {
            count.incrementAndGet();
        }

        public int getCount() {
            return count.get();
        }
    }

    public static void demonstrateSynchronization() throws InterruptedException {
        final int THREAD_COUNT = 10;
        final int INCREMENTS_PER_THREAD = 1000;

        // 同期化されていないカウンター
        UnsafeCounter unsafeCounter = new UnsafeCounter();
        Thread[] unsafeThreads = new Thread[THREAD_COUNT];

        for (int i = 0; i &lt; THREAD_COUNT; i++) {
            unsafeThreads[i] = new Thread(() -&gt; {
                for (int j = 0; j &lt; INCREMENTS_PER_THREAD; j++) {
                    unsafeCounter.increment();
                }
            });
            unsafeThreads[i].start();
        }

        for (Thread thread : unsafeThreads) {
            thread.join();
        }

        System.out.printf(&quot;同期化なし - 期待値: %d, 実際の値: %d%n&quot;,
                THREAD_COUNT * INCREMENTS_PER_THREAD, unsafeCounter.getCount());

        // synchronizedを使用したカウンター
        SafeCounter safeCounter = new SafeCounter();
        Thread[] safeThreads = new Thread[THREAD_COUNT];

        for (int i = 0; i &lt; THREAD_COUNT; i++) {
            safeThreads[i] = new Thread(() -&gt; {
                for (int j = 0; j &lt; INCREMENTS_PER_THREAD; j++) {
                    safeCounter.increment();
                }
            });
            safeThreads[i].start();
        }

        for (Thread thread : safeThreads) {
            thread.join();
        }

        System.out.printf(&quot;synchronized使用 - 期待値: %d, 実際の値: %d%n&quot;,
                THREAD_COUNT * INCREMENTS_PER_THREAD, safeCounter.getCount());

        // AtomicIntegerを使用したカウンター
        AtomicCounter atomicCounter = new AtomicCounter();
        Thread[] atomicThreads = new Thread[THREAD_COUNT];

        for (int i = 0; i &lt; THREAD_COUNT; i++) {
            atomicThreads[i] = new Thread(() -&gt; {
                for (int j = 0; j &lt; INCREMENTS_PER_THREAD; j++) {
                    atomicCounter.increment();
                }
            });
            atomicThreads[i].start();
        }

        for (Thread thread : atomicThreads) {
            thread.join();
        }

        System.out.printf(&quot;AtomicInteger使用 - 期待値: %d, 実際の値: %d%n&quot;,
                THREAD_COUNT * INCREMENTS_PER_THREAD, atomicCounter.getCount());
    }
}

    高度な同期化テクニック

    public class AdvancedSynchronization {
    // ReadWriteLockを使用した読み書き分離
    static class ReadWriteCache&lt;K, V&gt; {
        private final Map&lt;K, V&gt; cache = new HashMap&lt;&gt;();
        private final ReadWriteLock lock = new ReentrantReadWriteLock();
        private final Lock readLock = lock.readLock();
        private final Lock writeLock = lock.writeLock();

        public V get(K key) {
            readLock.lock();
            try {
                return cache.get(key);
            } finally {
                readLock.unlock();
            }
        }

        public void put(K key, V value) {
            writeLock.lock();
            try {
                cache.put(key, value);
            } finally {
                writeLock.unlock();
            }
        }

        public int size() {
            readLock.lock();
            try {
                return cache.size();
            } finally {
                readLock.unlock();
            }
        }
    }

    // CountDownLatchを使用した同期
    public static void demonstrateCountDownLatch() throws InterruptedException {
        final int WORKER_COUNT = 5;
        CountDownLatch startSignal = new CountDownLatch(1);
        CountDownLatch doneSignal = new CountDownLatch(WORKER_COUNT);

        for (int i = 0; i &lt; WORKER_COUNT; i++) {
            final int workerId = i + 1;
            new Thread(() -&gt; {
                try {
                    startSignal.await(); // 開始シグナルを待つ
                    System.out.printf(&quot;ワーカー%d開始%n&quot;, workerId);

                    // 作業をシミュレート
                    Thread.sleep(1000 + (int)(Math.random() * 1000));

                    System.out.printf(&quot;ワーカー%d完了%n&quot;, workerId);
                    doneSignal.countDown(); // 完了をシグナル
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            }).start();
        }

        System.out.println(&quot;すべてのワーカーを開始します&quot;);
        startSignal.countDown(); // 全ワーカーに開始シグナル

        doneSignal.await(); // 全ワーカーの完了を待つ
        System.out.println(&quot;すべてのワーカーが完了しました&quot;);
    }

    // CyclicBarrierを使用した同期
    public static void demonstrateCyclicBarrier() throws InterruptedException {
        final int THREAD_COUNT = 3;
        CyclicBarrier barrier = new CyclicBarrier(THREAD_COUNT, () -&gt; {
            System.out.println(&quot;=== すべてのスレッドが同期ポイントに到達 ===&quot;);
        });

        for (int i = 0; i &lt; THREAD_COUNT; i++) {
            final int threadId = i + 1;
            new Thread(() -&gt; {
                try {
                    // フェーズ1
                    System.out.printf(&quot;スレッド%d: フェーズ1開始%n&quot;, threadId);
                    Thread.sleep(1000 + (int)(Math.random() * 2000));
                    System.out.printf(&quot;スレッド%d: フェーズ1完了、同期待ち%n&quot;, threadId);

                    barrier.await(); // 他のスレッドを待つ

                    // フェーズ2
                    System.out.printf(&quot;スレッド%d: フェーズ2開始%n&quot;, threadId);
                    Thread.sleep(1000 + (int)(Math.random() * 2000));
                    System.out.printf(&quot;スレッド%d: フェーズ2完了%n&quot;, threadId);

                } catch (InterruptedException | BrokenBarrierException e) {
                    Thread.currentThread().interrupt();
                }
            }).start();
        }

        Thread.sleep(10000); // デモ用の待機
    }

    要求仕様： - 複数ファイルの並行処理 - 進捗状況の表示 - エラーハンドリング - スレッドプールによるリソース管理

    import java.io.*;
import java.nio.file.*;
import java.util.*;
import java.util.concurrent.*;
import java.util.concurrent.atomic.AtomicInteger;
import java.util.concurrent.atomic.AtomicLong;

public class ConcurrentFileProcessor {

    static class ProcessingResult {
        private final String fileName;
        private final boolean success;
        private final String errorMessage;
        private final long processingTime;
        private final long fileSize;

        public ProcessingResult(String fileName, boolean success, String errorMessage, 
                              long processingTime, long fileSize) {
            this.fileName = fileName;
            this.success = success;
            this.errorMessage = errorMessage;
            this.processingTime = processingTime;
            this.fileSize = fileSize;
        }

        // getterメソッド
        public String getFileName() { return fileName; }
        public boolean isSuccess() { return success; }
        public String getErrorMessage() { return errorMessage; }
        public long getProcessingTime() { return processingTime; }
        public long getFileSize() { return fileSize; }
    }

    interface ProgressListener {
        void onProgress(String fileName, int percentage);
        void onComplete(String fileName, boolean success);
        void onError(String fileName, String error);
    }

    private final ExecutorService executor;
    private final int maxConcurrentFiles;
    private final AtomicInteger completedFiles = new AtomicInteger(0);
    private final AtomicLong totalBytesProcessed = new AtomicLong(0);

    public ConcurrentFileProcessor(int maxConcurrentFiles) {
        this.maxConcurrentFiles = maxConcurrentFiles;
        this.executor = Executors.newFixedThreadPool(maxConcurrentFiles);
    }

    // 単一ファイルの処理
    public CompletableFuture&lt;ProcessingResult&gt; processFileAsync(String filePath, 
                                                               ProgressListener progressListener) {
        return CompletableFuture.supplyAsync(() -&gt; {
            long startTime = System.currentTimeMillis();
            String fileName = Paths.get(filePath).getFileName().toString();

            try {
                Path path = Paths.get(filePath);
                if (!Files.exists(path)) {
                    throw new FileNotFoundException(&quot;ファイルが見つかりません: &quot; + filePath);
                }

                long fileSize = Files.size(path);
                progressListener.onProgress(fileName, 0);

                // ファイル処理をシミュレート
                try (BufferedReader reader = Files.newBufferedReader(path)) {
                    String line;
                    long bytesRead = 0;
                    int lineCount = 0;

                    while ((line = reader.readLine()) != null) {
                        // 行の処理をシミュレート
                        processLine(line);

                        bytesRead += line.getBytes().length + System.lineSeparator().getBytes().length;
                        lineCount++;

                        // 進捗更新（1000行ごと）
                        if (lineCount % 1000 == 0) {
                            int progress = (int) ((bytesRead * 100) / fileSize);
                            progressListener.onProgress(fileName, Math.min(progress, 100));
                        }

                        // 処理時間をシミュレート
                        if (lineCount % 10000 == 0) {
                            Thread.sleep(10);
                        }
                    }
                }

                long processingTime = System.currentTimeMillis() - startTime;
                totalBytesProcessed.addAndGet(fileSize);
                completedFiles.incrementAndGet();

                progressListener.onProgress(fileName, 100);
                progressListener.onComplete(fileName, true);

                return new ProcessingResult(fileName, true, null, processingTime, fileSize);

            } catch (Exception e) {
                long processingTime = System.currentTimeMillis() - startTime;
                progressListener.onError(fileName, e.getMessage());
                progressListener.onComplete(fileName, false);

                return new ProcessingResult(fileName, false, e.getMessage(), processingTime, 0);
            }
        }, executor);
    }

    private void processLine(String line) {
        // 実際の行処理ロジック（ここでは単純にシミュレート）
        if (line.trim().isEmpty()) return;

        // 重い処理をシミュレート
        for (int i = 0; i &lt; line.length(); i++) {
            Math.sqrt(i + line.charAt(i));
        }
    }

    // 複数ファイルの並行処理
    public CompletableFuture&lt;List&lt;ProcessingResult&gt;&gt; processFilesAsync(List&lt;String&gt; filePaths,
                                                                      ProgressListener progressListener) {
        System.out.printf(&quot;=== %d個のファイルの処理を開始 ===%n&quot;, filePaths.size());

        List&lt;CompletableFuture&lt;ProcessingResult&gt;&gt; futures = filePaths.stream()
                .map(filePath -&gt; processFileAsync(filePath, progressListener))
                .collect(Collectors.toList());

        // すべてのファイル処理の完了を待つ
        return CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]))
                .thenApply(v -&gt; futures.stream()
                        .map(CompletableFuture::join)
                        .collect(Collectors.toList()));
    }

    // バッチ処理の管理
    public void processBatch(List&lt;String&gt; filePaths, Duration timeout) throws InterruptedException, ExecutionException, TimeoutException {
        AtomicInteger processedCount = new AtomicInteger(0);
        int totalFiles = filePaths.size();

        ProgressListener progressListener = new ProgressListener() {
            @Override
            public void onProgress(String fileName, int percentage) {
                if (percentage % 25 == 0 &amp;&amp; percentage &gt; 0) {
                    System.out.printf(&quot;[%s] 進捗: %d%%%n&quot;, fileName, percentage);
                }
            }

            @Override
            public void onComplete(String fileName, boolean success) {
                int completed = processedCount.incrementAndGet();
                System.out.printf(&quot;[%s] %s (%d/%d)%n&quot;, 
                                fileName, 
                                success ? &quot;✓ 完了&quot; : &quot;✗ 失敗&quot;, 
                                completed, totalFiles);
            }

            @Override
            public void onError(String fileName, String error) {
                System.err.printf(&quot;[%s] エラー: %s%n&quot;, fileName, error);
            }
        };

        long startTime = System.currentTimeMillis();

        try {
            List&lt;ProcessingResult&gt; results = processFilesAsync(filePaths, progressListener)
                    .get(timeout.toMillis(), TimeUnit.MILLISECONDS);

            long totalTime = System.currentTimeMillis() - startTime;

            // 結果サマリーの表示
            displayResults(results, totalTime);

        } catch (TimeoutException e) {
            System.err.println(&quot;処理がタイムアウトしました&quot;);
            throw e;
        }
    }

    private void displayResults(List&lt;ProcessingResult&gt; results, long totalTime) {
        long successCount = results.stream().mapToLong(r -&gt; r.isSuccess() ? 1 : 0).sum();
        long totalSize = results.stream().mapToLong(ProcessingResult::getFileSize).sum();
        double averageTime = results.stream().mapToLong(ProcessingResult::getProcessingTime).average().orElse(0);

        System.out.println(&quot;\n=== 処理結果サマリー ===&quot;);
        System.out.printf(&quot;成功: %d/%d%n&quot;, successCount, results.size());
        System.out.printf(&quot;総サイズ: %s%n&quot;, formatBytes(totalSize));
        System.out.printf(&quot;総時間: %d ms%n&quot;, totalTime);
        System.out.printf(&quot;平均処理時間: %.1f ms%n&quot;, averageTime);

        if (totalTime &gt; 0) {
            double throughput = (totalSize / 1024.0 / 1024.0) / (totalTime / 1000.0);
            System.out.printf(&quot;スループット: %.2f MB/s%n&quot;, throughput);
        }

        // エラーがあった場合の詳細表示
        List&lt;ProcessingResult&gt; failures = results.stream()
                .filter(r -&gt; !r.isSuccess())
                .collect(Collectors.toList());

        if (!failures.isEmpty()) {
            System.out.println(&quot;\n=== 失敗したファイル ===&quot;);
            failures.forEach(failure -&gt; 
                System.out.printf(&quot;- %s: %s%n&quot;, failure.getFileName(), failure.getErrorMessage()));
        }
    }

    private String formatBytes(long bytes) {
        String[] units = {&quot;B&quot;, &quot;KB&quot;, &quot;MB&quot;, &quot;GB&quot;, &quot;TB&quot;};
        double size = bytes;
        int unitIndex = 0;

        while (size &gt;= 1024 &amp;&amp; unitIndex &lt; units.length - 1) {
            size /= 1024;
            unitIndex++;
        }

        return String.format(&quot;%.1f %s&quot;, size, units[unitIndex]);
    }

    public void shutdown() {
        executor.shutdown();
        try {
            if (!executor.awaitTermination(5, TimeUnit.SECONDS)) {
                executor.shutdownNow();
            }
        } catch (InterruptedException e) {
            executor.shutdownNow();
            Thread.currentThread().interrupt();
        }
    }

    // 使用例
    public static void main(String[] args) {
        ConcurrentFileProcessor processor = new ConcurrentFileProcessor(4);

        try {
            // テスト用ファイルの作成
            List&lt;String&gt; testFiles = createTestFiles();

            // バッチ処理の実行
            processor.processBatch(testFiles, Duration.ofMinutes(5));

        } catch (Exception e) {
            System.err.println(&quot;処理中にエラーが発生しました: &quot; + e.getMessage());
            e.printStackTrace();
        } finally {
            processor.shutdown();
        }
    }

    private static List&lt;String&gt; createTestFiles() throws IOException {
        List&lt;String&gt; filePaths = new ArrayList&lt;&gt;();
        Path tempDir = Files.createTempDirectory(&quot;concurrent_processing_test&quot;);

        for (int i = 1; i &lt;= 5; i++) {
            Path testFile = tempDir.resolve(&quot;test_file_&quot; + i + &quot;.txt&quot;);
            try (PrintWriter writer = new PrintWriter(Files.newBufferedWriter(testFile))) {
                for (int line = 1; line &lt;= 10000; line++) {
                    writer.printf(&quot;ファイル%d - 行%d: これはテストデータです。%n&quot;, i, line);
                }
            }
            filePaths.add(testFile.toString());
        }

        System.out.println(&quot;テストファイルを作成しました: &quot; + tempDir);
        return filePaths;
    }

    まとめ

    並行プログラミングの本質

    • 並行性と並列性：複数のタスクを効率的に実行するための仕組み
    • スレッドセーフティ：共有リソースへの安全なアクセスの保証
    • 非同期処理：ブロッキングを避けて応答性の高いアプリケーションを実現
    • リソース管理：スレッドプールによる効率的なリソース活用

    実践的な並行プログラミングパターン

    • Producer-Consumer：生産者と消費者の協調動作
    • Fork-Join：タスクの分割統治による並列処理
    • Pipeline：データの段階的処理による効率化
    • Actor Model：メッセージパッシングによる並行制御

    パフォーマンス最適化の考慮点

    • 適切なスレッド数：CPUコア数とI/O待機時間に応じた調整
    • ロックの最小化：細粒度ロックやlock-freeアルゴリズムの活用
    • メモリ可視性：volatileやAtomicクラスによる適切なメモリ管理
    • デッドロックの回避：ロック順序の統一や時間制限の設定
