    1. 電卓システムの要件定義と設計

    🎯 ステップ1: 要件定義

    レッスン16で学んだ設計手法を活用して、まずは要件を明確にしましょう。

    基本要件

    1. 四則演算（+、-、*、/）の実行
2. 連続計算機能（前の結果を次の計算に使用）
3. 計算履歴の保存と表示（最大20件）
4. エラーハンドリング（ゼロ除算、不正入力対応）
5. ユーザーフレンドリーなメニューシステム
6. 美しい出力フォーマット

    非機能要件

    - 起動から終了まで連続使用可能
- 初心者でも迷わず使える操作性
- エラー時の適切なメッセージ表示
- 計算結果の見やすい表示

    新しいクラスの作成

    1. Eclipse で右クリック → 新規 → クラス

    基本構造の実装

    import java.util.Scanner;

public class Calculator {
    // データ（フィールド）
    private static double currentResult = 0.0;
    private static String[] history = new String[20];
    private static int historyCount = 0;
    private static Scanner scanner = new Scanner(System.in);

    public static void main(String[] args) {
        System.out.println(&quot;=== 電卓プログラム ===&quot;);
        System.out.println(&quot;Week 2 実践課題1&quot;);
        System.out.println();

        // メインループ
        boolean running = true;
        while (running) {
            showMenu();

            System.out.print(&quot;選択してください (1-5): &quot;);
            int choice = scanner.nextInt();
            scanner.nextLine(); // 改行文字を消費

            switch (choice) {
                case 1:
                    performCalculation();
                    break;
                case 2:
                    showHistory();
                    break;
                case 3:
                    clearHistory();
                    break;
                case 4:
                    showCurrentResult();
                    break;
                case 5:
                    System.out.println(&quot;電卓を終了します。ありがとうございました！&quot;);
                    running = false;
                    break;
                default:
                    System.out.println(&quot;❌ 1-5の数字を入力してください。&quot;);
            }

            if (running) {
                System.out.println(&quot;\nEnterキーを押して続行...&quot;);
                scanner.nextLine();
            }
        }

        scanner.close();
    }

    // メニュー表示メソッド（仮実装）
    private static void showMenu() {
        System.out.println(&quot;┌─────────────────────────────┐&quot;);
        System.out.println(&quot;│          電卓メニュー         │&quot;);
        System.out.println(&quot;├─────────────────────────────┤&quot;);
        System.out.println(&quot;│ 1. 計算実行                   │&quot;);
        System.out.println(&quot;│ 2. 計算履歴表示               │&quot;);
        System.out.println(&quot;│ 3. 履歴クリア                 │&quot;);
        System.out.println(&quot;│ 4. 現在の結果表示             │&quot;);
        System.out.println(&quot;│ 5. 終了                      │&quot;);
        System.out.println(&quot;└─────────────────────────────┘&quot;);
        System.out.printf(&quot;現在の結果: %.2f%n&quot;, currentResult);
        System.out.println();
    }

    // その他のメソッドは後で実装
    private static void performCalculation() {
        System.out.println(&quot;計算機能（実装予定）&quot;);
    }

    private static void showHistory() {
        System.out.println(&quot;履歴表示機能（実装予定）&quot;);
    }

    private static void clearHistory() {
        System.out.println(&quot;履歴クリア機能（実装予定）&quot;);
    }

    private static void showCurrentResult() {
        System.out.printf(&quot;現在の計算結果: %.2f%n&quot;, currentResult);
    }
}

    🛠️ ステップ5: 計算機能の詳細実装

    計算実行メソッドの実装

    performCalculation メソッドを以下のように実装してください：

    private static void performCalculation() {
    System.out.println(&quot;=== 計算実行 ===&quot;);

    try {
        // 最初の数値取得
        double num1;
        System.out.print(&quot;連続計算しますか？ (y/n): &quot;);
        String useCurrent = scanner.nextLine().toLowerCase();

        if (useCurrent.equals(&quot;y&quot;) || useCurrent.equals(&quot;yes&quot;)) {
            num1 = currentResult;
            System.out.printf(&quot;最初の数値: %.2f (現在の結果)%n&quot;, num1);
        } else {
            num1 = getValidNumber(&quot;最初の数値を入力してください: &quot;);
        }

        // 演算子取得
        char operator = getValidOperator();

        // 2番目の数値取得
        double num2 = getValidNumber(&quot;2番目の数値を入力してください: &quot;);

        // 計算実行
        double result = calculate(num1, operator, num2);

        // 結果表示と履歴追加
        String calculation = String.format(&quot;%.2f %c %.2f = %.2f&quot;, 
                                         num1, operator, num2, result);
        displayResult(calculation, result);
        addToHistory(calculation);

        // 現在の結果を更新
        currentResult = result;

    } catch (Exception e) {
        System.out.println(&quot;❌ エラーが発生しました: &quot; + e.getMessage());
    }
}

    数値入力検証メソッドの実装

    private static double getValidNumber(String prompt) {
    while (true) {
        try {
            System.out.print(prompt);
            String input = scanner.nextLine();
            return Double.parseDouble(input);
        } catch (NumberFormatException e) {
            System.out.println(&quot;❌ 有効な数値を入力してください。&quot;);
        }
    }
}

    演算子入力検証メソッドの実装

    private static char getValidOperator() {
    while (true) {
        System.out.print(&quot;演算子を入力してください (+, -, *, /): &quot;);
        String input = scanner.nextLine();

        if (input.length() == 1) {
            char operator = input.charAt(0);
            if (operator == &apos;+&apos; || operator == &apos;-&apos; || 
                operator == &apos;*&apos; || operator == &apos;/&apos;) {
                return operator;
            }
        }

        System.out.println(&quot;❌ 有効な演算子 (+, -, *, /) を入力してください。&quot;);
    }
}

    計算実行メソッドの実装

    private static double calculate(double num1, char operator, double num2) {
    switch (operator) {
        case &apos;+&apos;:
            return num1 + num2;
        case &apos;-&apos;:
            return num1 - num2;
        case &apos;*&apos;:
            return num1 * num2;
        case &apos;/&apos;:
            if (num2 == 0) {
                throw new ArithmeticException(&quot;ゼロで割ることはできません&quot;);
            }
            return num1 / num2;
        default:
            throw new IllegalArgumentException(&quot;無効な演算子: &quot; + operator);
    }
}

    4. 段階的実装：履歴機能の実装

    🛠️ ステップ7: 履歴管理機能の実装

    履歴追加メソッドの実装

    private static void addToHistory(String calculation) {
    if (historyCount &lt; history.length) {
        history[historyCount] = calculation;
        historyCount++;
    } else {
        // 履歴が満杯の場合、古いものから削除
        for (int i = 0; i &lt; history.length - 1; i++) {
            history[i] = history[i + 1];
        }
        history[history.length - 1] = calculation;
    }
}

    履歴表示メソッドの実装

    private static void showHistory() {
    System.out.println(&quot;=== 計算履歴 ===&quot;);

    if (historyCount == 0) {
        System.out.println(&quot;履歴はありません。&quot;);
        return;
    }

    System.out.println(&quot;┌────┬─────────────────────────────┐&quot;);
    System.out.println(&quot;│No. │          計算式             │&quot;);
    System.out.println(&quot;├────┼─────────────────────────────┤&quot;);

    int maxDisplay = Math.min(historyCount, history.length);
    for (int i = maxDisplay - 1; i &gt;= 0; i--) {
        if (history[i] != null) {
            System.out.printf(&quot;│%3d │ %-27s │%n&quot;, 
                            maxDisplay - i, history[i]);
        }
    }

    System.out.println(&quot;└────┴─────────────────────────────┘&quot;);
    System.out.printf(&quot;履歴件数: %d / %d%n&quot;, 
                     Math.min(historyCount, history.length), history.length);
}

    履歴クリア機能の実装

    private static void clearHistory() {
    System.out.print(&quot;本当に履歴をクリアしますか？ (y/n): &quot;);
    String confirm = scanner.nextLine().toLowerCase();

    if (confirm.equals(&quot;y&quot;) || confirm.equals(&quot;yes&quot;)) {
        for (int i = 0; i &lt; history.length; i++) {
            history[i] = null;
        }
        historyCount = 0;
        currentResult = 0.0;
        System.out.println(&quot;✅ 履歴をクリアしました。&quot;);
    } else {
        System.out.println(&quot;❌ キャンセルしました。&quot;);
    }
}

    import java.util.Scanner;

public class Calculator {
    // データ（フィールド）
    private static double currentResult = 0.0;
    private static String[] history = new String[20];
    private static int historyCount = 0;
    private static Scanner scanner = new Scanner(System.in);

    public static void main(String[] args) {
        System.out.println(&quot;=== 電卓プログラム ===&quot;);
        System.out.println(&quot;Week 2 実践課題1&quot;);
        System.out.println(&quot;作成者: あなたの名前&quot;);
        System.out.println();

        // メインループ
        boolean running = true;
        while (running) {
            showMenu();

            try {
                System.out.print(&quot;選択してください (1-5): &quot;);
                int choice = scanner.nextInt();
                scanner.nextLine(); // 改行文字を消費

                switch (choice) {
                    case 1:
                        performCalculation();
                        break;
                    case 2:
                        showHistory();
                        break;
                    case 3:
                        clearHistory();
                        break;
                    case 4:
                        showCurrentResult();
                        break;
                    case 5:
                        System.out.println(&quot;電卓を終了します。ありがとうございました！&quot;);
                        running = false;
                        break;
                    default:
                        System.out.println(&quot;❌ 1-5の数字を入力してください。&quot;);
                }

                if (running) {
                    System.out.println(&quot;\nEnterキーを押して続行...&quot;);
                    scanner.nextLine();
                    System.out.println(&quot;\n&quot; + &quot;=&quot;.repeat(50) + &quot;\n&quot;);
                }
            } catch (Exception e) {
                System.out.println(&quot;❌ 有効な数字を入力してください。&quot;);
                scanner.nextLine(); // 不正な入力をクリア
            }
        }

        scanner.close();
    }

    private static void showMenu() {
        System.out.println(&quot;┌─────────────────────────────┐&quot;);
        System.out.println(&quot;│          電卓メニュー            │&quot;);
        System.out.println(&quot;├─────────────────────────────┤&quot;);
        System.out.println(&quot;│ 1. 計算実行                    │&quot;);
        System.out.println(&quot;│ 2. 計算履歴表示                │&quot;);
        System.out.println(&quot;│ 3. 履歴クリア                  │&quot;);
        System.out.println(&quot;│ 4. 現在の結果表示              │&quot;);
        System.out.println(&quot;│ 5. 終了                       │&quot;);
        System.out.println(&quot;└─────────────────────────────┘&quot;);
        System.out.printf(&quot;現在の結果: %.2f%n&quot;, currentResult);
        System.out.println();
    }

    private static void performCalculation() {
        System.out.println(&quot;=== 計算実行 ===&quot;);

        try {
            // 最初の数値取得
            double num1;
            System.out.print(&quot;連続計算しますか？ (y/n): &quot;);
            String useCurrent = scanner.nextLine().toLowerCase();

            if (useCurrent.equals(&quot;y&quot;) || useCurrent.equals(&quot;yes&quot;)) {
                num1 = currentResult;
                System.out.printf(&quot;最初の数値: %.2f (現在の結果)%n&quot;, num1);
            } else {
                num1 = getValidNumber(&quot;最初の数値を入力してください: &quot;);
            }

            // 演算子取得
            char operator = getValidOperator();

            // 2番目の数値取得
            double num2 = getValidNumber(&quot;2番目の数値を入力してください: &quot;);

            // 計算実行
            double result = calculate(num1, operator, num2);

            // 結果表示と履歴追加
            String calculation = String.format(&quot;%.2f %c %.2f = %.2f&quot;, 
                                             num1, operator, num2, result);
            displayResult(calculation, result);
            addToHistory(calculation);

            // 現在の結果を更新
            currentResult = result;

        } catch (Exception e) {
            System.out.println(&quot;❌ エラーが発生しました: &quot; + e.getMessage());
        }
    }

    private static double getValidNumber(String prompt) {
        while (true) {
            try {
                System.out.print(prompt);
                String input = scanner.nextLine();
                return Double.parseDouble(input);
            } catch (NumberFormatException e) {
                System.out.println(&quot;❌ 有効な数値を入力してください。&quot;);
            }
        }
    }

    private static char getValidOperator() {
        while (true) {
            System.out.print(&quot;演算子を入力してください (+, -, *, /): &quot;);
            String input = scanner.nextLine();

            if (input.length() == 1) {
                char operator = input.charAt(0);
                if (operator == &apos;+&apos; || operator == &apos;-&apos; || 
                    operator == &apos;*&apos; || operator == &apos;/&apos;) {
                    return operator;
                }
            }

            System.out.println(&quot;❌ 有効な演算子 (+, -, *, /) を入力してください。&quot;);
        }
    }

    private static double calculate(double num1, char operator, double num2) {
        switch (operator) {
            case &apos;+&apos;:
                return num1 + num2;
            case &apos;-&apos;:
                return num1 - num2;
            case &apos;*&apos;:
                return num1 * num2;
            case &apos;/&apos;:
                if (num2 == 0) {
                    throw new ArithmeticException(&quot;ゼロで割ることはできません&quot;);
                }
                return num1 / num2;
            default:
                throw new IllegalArgumentException(&quot;無効な演算子: &quot; + operator);
        }
    }

    private static void displayResult(String calculation, double result) {
        System.out.println();
        System.out.println(&quot;┌─────────────────────────────┐&quot;);
        System.out.println(&quot;│          計算結果            │&quot;);
        System.out.println(&quot;├─────────────────────────────┤&quot;);
        System.out.printf(&quot;│ %-27s │%n&quot;, calculation);
        System.out.println(&quot;└─────────────────────────────┘&quot;);

        // 結果の特徴を表示
        if (result &gt; 1000000) {
            System.out.println(&quot;💡 大きな数値ですね！&quot;);
        } else if (result &lt; 0) {
            System.out.println(&quot;💡 負の数値になりました。&quot;);
        } else if (result == 0) {
            System.out.println(&quot;💡 結果はゼロです。&quot;);
        }

        System.out.println();
    }

    private static void addToHistory(String calculation) {
        if (historyCount &lt; history.length) {
            history[historyCount] = calculation;
            historyCount++;
        } else {
            // 履歴が満杯の場合、古いものから削除
            for (int i = 0; i &lt; history.length - 1; i++) {
                history[i] = history[i + 1];
            }
            history[history.length - 1] = calculation;
        }
    }

    private static void showHistory() {
        System.out.println(&quot;=== 計算履歴 ===&quot;);

        if (historyCount == 0) {
            System.out.println(&quot;📝 履歴はありません。計算を実行してください。&quot;);
            return;
        }

        System.out.println(&quot;┌────┬─────────────────────────────┐&quot;);
        System.out.println(&quot;│No. │          計算式             │&quot;);
        System.out.println(&quot;├────┼─────────────────────────────┤&quot;);

        int maxDisplay = Math.min(historyCount, history.length);
        for (int i = maxDisplay - 1; i &gt;= 0; i--) {
            if (history[i] != null) {
                System.out.printf(&quot;│%3d │ %-27s │%n&quot;, 
                                maxDisplay - i, history[i]);
            }
        }

        System.out.println(&quot;└────┴─────────────────────────────┘&quot;);
        System.out.printf(&quot;履歴件数: %d / %d%n&quot;, 
                         Math.min(historyCount, history.length), history.length);
    }

    private static void clearHistory() {
        System.out.print(&quot;本当に履歴をクリアしますか？ (y/n): &quot;);
        String confirm = scanner.nextLine().toLowerCase();

        if (confirm.equals(&quot;y&quot;) || confirm.equals(&quot;yes&quot;)) {
            for (int i = 0; i &lt; history.length; i++) {
                history[i] = null;
            }
            historyCount = 0;
            currentResult = 0.0;
            System.out.println(&quot;✅ 履歴と現在の結果をクリアしました。&quot;);
        } else {
            System.out.println(&quot;❌ キャンセルしました。&quot;);
        }
    }

    private static void showCurrentResult() {
        System.out.println(&quot;=== 現在の結果 ===&quot;);
        System.out.printf(&quot;結果: %.2f%n&quot;, currentResult);

        if (currentResult == 0) {
            System.out.println(&quot;💡 ヒント: 計算を実行すると、結果がここに保存されます。&quot;);
        }
    }
}
