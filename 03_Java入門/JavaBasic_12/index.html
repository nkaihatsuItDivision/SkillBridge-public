    2. System.out.print vs println の使い分け

    基本的な違い

    これまで主に使っていた println と、新しく学ぶ print の違いを確認しましょう。

    public class PrintComparison {
    public static void main(String[] args) {
        // println：改行あり（これまで使っていた方法）
        System.out.println(&quot;Hello&quot;);
        System.out.println(&quot;World&quot;);

        System.out.println(&quot;---&quot;); // 区切り線

        // print：改行なし（新しい方法）
        System.out.print(&quot;Hello&quot;);
        System.out.print(&quot;World&quot;);
        System.out.println(); // 手動で改行

        System.out.println(&quot;---&quot;); // 区切り線

        // print を活用した連続出力
        System.out.print(&quot;商品: &quot;);
        System.out.print(&quot;ノートPC &quot;);
        System.out.print(&quot;価格: &quot;);
        System.out.println(&quot;89,800円&quot;);
    }
}

    3. エスケープシーケンス：特殊文字の活用

    エスケープシーケンスとは？

    エスケープシーケンス は、通常の文字では表現できない 特殊文字 を表現する方法です。

    📘 エスケープシーケンスの視覚化

    基本的なエスケープシーケンス

    public class EscapeSequenceExample {
    public static void main(String[] args) {
        // \n：改行（newline）
        System.out.println(&quot;1行目\n2行目\n3行目&quot;);

        System.out.println(&quot;---&quot;);

        // \t：タブ（tab）
        System.out.println(&quot;名前\t年齢\t職業&quot;);
        System.out.println(&quot;田中\t25\t会社員&quot;);
        System.out.println(&quot;佐藤\t30\tエンジニア&quot;);

        System.out.println(&quot;---&quot;);

        // \&quot;：ダブルクォート
        System.out.println(&quot;彼は\&quot;こんにちは\&quot;と言いました&quot;);

        // \\：バックスラッシュ
        System.out.println(&quot;ファイルパス: C:\\Users\\Documents\\file.txt&quot;);
    }
}

    実行結果: 

    1行目
2行目
3行目
---

    4. System.out.printf：プロ級のフォーマット出力

    printfとは？

    printf は 「print formatted」 の略で、フォーマット指定 を使って美しい出力を作る専用メソッドです。

    基本的な使い方

    public class PrintfBasic {
    public static void main(String[] args) {
        String name = &quot;田中太郎&quot;;
        int age = 25;
        double height = 170.5;

        // 基本的なprintf
        System.out.printf(&quot;名前: %s\n&quot;, name);
        System.out.printf(&quot;年齢: %d歳\n&quot;, age);
        System.out.printf(&quot;身長: %.1fcm\n&quot;, height);

        // 複数の値を同時に出力
        System.out.printf(&quot;こんにちは、%sさん！あなたは%d歳ですね。\n&quot;, name, age);
    }
}

    実行結果: 

    名前: 田中太郎
年齢: 25歳
身長: 170.5cm
こんにちは、田中太郎さん！あなたは25歳ですね。

    📘 printfの書式指定子

    主要なフォーマット指定子

    指定子 意味 使用例 結果
    %s 文字列（String） printf("名前: %s", "田中") 名前: 田中
    %d 整数（decimal） printf("年齢: %d", 25) 年齢: 25
    %f 小数（float） printf("価格: %.2f", 1234.5) 価格: 1234.50
    %c 文字（character） printf("等級: %c", 'A') 等級: A
    %x 16進数（hexadecimal） printf("色: #%06x", 255) 色: #0000ff

    5. printfの高度な活用：桁数指定と桁揃え

    桁数指定で美しい表を作成

    public class AdvancedPrintf {
    public static void main(String[] args) {
        System.out.println(&quot;=== 売上レポート ===&quot;);

        // ヘッダー（右揃えで統一）
        System.out.printf(&quot;%10s %8s %6s %10s\n&quot;, &quot;商品名&quot;, &quot;単価&quot;, &quot;数量&quot;, &quot;売上&quot;);
        System.out.println(&quot;------------------------------------------&quot;);

        // データ行（数値は右揃え、文字列は左揃え）
        System.out.printf(&quot;%-10s %8d %6d %10d\n&quot;, &quot;ノートPC&quot;, 89800, 12, 1077600);
        System.out.printf(&quot;%-10s %8d %6d %10d\n&quot;, &quot;マウス&quot;, 2980, 25, 74500);
        System.out.printf(&quot;%-10s %8d %6d %10d\n&quot;, &quot;キーボード&quot;, 5400, 18, 97200);

        System.out.println(&quot;------------------------------------------&quot;);
        System.out.printf(&quot;%-10s %8s %6s %10d\n&quot;, &quot;合計&quot;, &quot;&quot;, &quot;&quot;, 1249300);
    }
}

    実行結果:

------------------------------------------

マウス          2980     25      74500

------------------------------------------
合計                              1249300

    小数点桁数の制御

    public class DecimalFormatting {
    public static void main(String[] args) {
        double[] prices = {1234.567, 89.9, 12345.0, 0.99};

        System.out.println(&quot;価格一覧（小数点以下2桁）:&quot;);
        System.out.println(&quot;------------------------&quot;);

        for (int i = 0; i &lt; prices.length; i++) {
            System.out.printf(&quot;商品%d: %8.2f円\n&quot;, i + 1, prices[i]);
        }

        System.out.println(&quot;------------------------&quot;);

        // 平均価格の計算と表示
        double total = 0;
        for (double price : prices) {
            total += price;
        }
        double average = total / prices.length;

        System.out.printf(&quot;平均価格: %.2f円\n&quot;, average);
    }
}

商品3: 12345.00円
商品4:     0.99円
------------------------

    6. 実践課題：レッスン11のショッピングシステムを改善しよう

    まずは自分で考えてみよう！

    レッスン11で作成したショッピング計算システムの出力を、今回学んだ技術で改善してみましょう。

    💡 ヒント1: printfで桁揃え 商品名は左揃え（%-15s）、価格は右揃え（%8d）で統一感を出しましょう。

    💡 ヒント2: エスケープシーケンス活用 \tや\nを効果的に使って、見やすいレイアウトを作成しましょう。

    💡 ヒント3: 区切り線で視覚的分離 =や-を使って、情報をグループ化しましょう。

    🎯 期待される実行結果:

    完成例

    public class BeautifulShoppingSystem {
    public static void main(String[] args) {
        // 商品データ
        String product1 = &quot;ノートPC&quot;;
        int price1 = 89800;
        int quantity1 = 2;

        String product2 = &quot;マウス&quot;;
        int price2 = 2980;
        int quantity2 = 1;

        // 小計計算
        int subtotal1 = price1 * quantity1;
        int subtotal2 = price2 * quantity2;
        int totalSubtotal = subtotal1 + subtotal2;

        // 税計算
        double taxRate = 0.10;
        int tax = (int) (totalSubtotal * taxRate);
        int finalTotal = totalSubtotal + tax;

        // 美しいレシート出力
        System.out.println(&quot;========================================&quot;);
        System.out.println(&quot;           ショッピングレシート&quot;);
        System.out.println(&quot;========================================&quot;);

        // ヘッダー
        System.out.printf(&quot;%-15s %8s %6s %10s\n&quot;, &quot;商品名&quot;, &quot;単価&quot;, &quot;数量&quot;, &quot;小計&quot;);
        System.out.println(&quot;----------------------------------------&quot;);

        // 商品明細
        System.out.printf(&quot;%-15s %,8d %6d %,10d円\n&quot;, product1, price1, quantity1, subtotal1);
        System.out.printf(&quot;%-15s %,8d %6d %,10d円\n&quot;, product2, price2, quantity2, subtotal2);

        System.out.println(&quot;----------------------------------------&quot;);

        // 集計部分（右揃え）
        System.out.printf(&quot;%32s %,10d円\n&quot;, &quot;小計:&quot;, totalSubtotal);
        System.out.printf(&quot;%32s %9.0f%%\n&quot;, &quot;税率:&quot;, taxRate * 100);
        System.out.printf(&quot;%32s %,10d円\n&quot;, &quot;税額:&quot;, tax);

        System.out.println(&quot;========================================&quot;);
        System.out.printf(&quot;%32s %,10d円\n&quot;, &quot;合計:&quot;, finalTotal);
        System.out.println(&quot;========================================&quot;);

        System.out.println(&quot;ありがとうございました！&quot;);
    }
}

    新しく使った技術のポイント:

    1. %,d: 数値に3桁区切りのカンマを自動挿入
    2. %-15s: 左揃えで幅15文字
    3. %8d: 右揃えで幅8文字
    4. %6d: 右揃えで幅6文字
    5. %10s: 右揃えで幅10文字

    プロフェッショナルな出力の特徴

    1. 一貫性のあるフォーマット

    // ✅ 良い例：統一されたフォーマット
System.out.printf(&quot;商品A: %,8d円\n&quot;, 12000);
System.out.printf(&quot;商品B: %,8d円\n&quot;, 3500);
System.out.printf(&quot;商品C: %,8d円\n&quot;, 890);

// ❌ 悪い例：バラバラなフォーマット
System.out.println(&quot;商品A: 12000円&quot;);
System.out.println(&quot;商品B: 3,500円&quot;);
System.out.println(&quot;商品C: ¥890&quot;);

    2. 視覚的なグループ化

    // ✅ 良い例：セクションごとに区切り
System.out.println(&quot;=== お客様情報 ===&quot;);
System.out.printf(&quot;名前: %s\n&quot;, name);
System.out.printf(&quot;年齢: %d歳\n&quot;, age);

System.out.println(&quot;\n=== 購入商品 ===&quot;);
System.out.printf(&quot;商品: %s\n&quot;, product);
System.out.printf(&quot;価格: %,d円\n&quot;, price);

    3. 適切な精度とフォーマット

    // ✅ 金額は整数、割合は小数点1桁
System.out.printf(&quot;金額: %,d円\n&quot;, amount);
System.out.printf(&quot;割引率: %.1f%%\n&quot;, discountRate);

// ✅ 測定値は適切な桁数
System.out.printf(&quot;身長: %.1fcm\n&quot;, height);
System.out.printf(&quot;体重: %.1fkg\n&quot;, weight);

    💡 よくある質問

    Q1: printfとprintlnはどう使い分けるべきですか？

    A: 以下のように使い分けましょう：

    • printf: フォーマットが必要な場合（桁揃え、小数点制御など）
    • println: シンプルな出力で十分な場合
    // printfが適している場面
System.out.printf(&quot;価格: %,8d円\n&quot;, price);

// printlnで十分な場面
System.out.println(&quot;処理が完了しました&quot;);

    Q2: printfで改行はどうやって入れるのですか？

    A: printfでは \n を使って改行します：

    System.out.printf(&quot;1行目\n2行目\n&quot;);
// または
System.out.printf(&quot;1行目%n2行目%n&quot;); // %nも使える

    Q3: 文字列の中にダブルクォートを表示したいのですが？

    A: \" エスケープシーケンスを使います：

    System.out.println(&quot;彼は\&quot;こんにちは\&quot;と言いました&quot;);
// 結果: 彼は&quot;こんにちは&quot;と言いました

    Q4: 数値の桁数が固定されていない場合、どうやって桁揃えをしますか？

    A: printfの幅指定機能を使います：

    // 右揃えで8桁確保
System.out.printf(&quot;%8d\n&quot;, 123);     // &quot;     123&quot;
System.out.printf(&quot;%8d\n&quot;, 45678);   // &quot;   45678&quot;

// 左揃えで8桁確保
System.out.printf(&quot;%-8d\n&quot;, 123);    // &quot;123     &quot;

    Q5: カンマ区切りの数値表示ができません

    A: %,d を使います：

    System.out.printf(&quot;%,d円\n&quot;, 1234567);  // 1,234,567円
