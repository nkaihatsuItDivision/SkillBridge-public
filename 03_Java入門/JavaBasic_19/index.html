    📚 Week 2技術の総整理

    🛠️ ステップ1: 学んだ技術の確認プログラム

    まず、Week 2で学んだ技術を 1つのプログラムに集約 してみましょう。

    新しいクラス Week2Review を作成してください：

    import java.util.Scanner;

public class Week2Review {
    public static void main(String[] args) {
        System.out.println("=== Week 2 学習技術の総復習 ===");

        // データ型の確認
        System.out.println("\n1. データ型の活用");
        demonstrateDataTypes();

        // 演算子の確認
        System.out.println("\n2. 演算子の活用");
        demonstrateOperators();

        // 条件分岐と繰り返しの確認
        System.out.println("\n3. 制御構文の活用");
        demonstrateControlStructures();

        // 入出力の確認
        System.out.println("\n4. 入出力の活用");
        demonstrateInputOutput();

        System.out.println("\n=== Week 2 復習完了！ ===");
    }

    // 1. データ型デモンストレーション
    public static void demonstrateDataTypes() {
        // 基本データ型
        byte byteValue = 127;
        short shortValue = 32767;
        int intValue = 2147483647;
        long longValue = 9223372036854775807L;
        float floatValue = 3.14f;
        double doubleValue = 3.14159265359;
        char charValue = &apos;A&apos;;
        boolean booleanValue = true;

        // 参照型
        String stringValue = "Week 2で学習した技術";

        System.out.println("基本データ型8種類 + String型を活用");
        System.out.println("int: " + intValue);
        System.out.println("double: " + doubleValue);
        System.out.println("boolean: " + booleanValue);
        System.out.println("String: " + stringValue);
    }

    // 2. 演算子デモンストレーション
    public static void demonstrateOperators() {
        int a = 10, b = 3;

        // 算術演算子
        System.out.println("算術演算子:");
        System.out.println(a + " + " + b + " = " + (a + b));
        System.out.println(a + " - " + b + " = " + (a - b));
        System.out.println(a + " * " + b + " = " + (a * b));
        System.out.println(a + " / " + b + " = " + (a / b));
        System.out.println(a + " % " + b + " = " + (a % b));

        // 比較演算子
        System.out.println("比較演算子:");
        System.out.println(a + " > " + b + " = " + (a > b));
        System.out.println(a + " == " + b + " = " + (a == b));

        // 論理演算子
        boolean x = true, y = false;
        System.out.println("論理演算子:");
        System.out.println(x + " &amp;&amp; " + y + " = " + (x &amp;&amp; y));
        System.out.println(x + " || " + y + " = " + (x || y));
    }

    // 3. 制御構文デモンストレーション
    public static void demonstrateControlStructures() {
        // 条件分岐
        int score = 85;
        System.out.println("条件分岐（if-else）:");
        if (score >= 80) {
            System.out.println("優秀な成績です！");
        } else if (score >= 60) {
            System.out.println("合格です！");
        } else {
            System.out.println("頑張りましょう！");
        }

        // 繰り返し処理
        System.out.println("繰り返し処理（for文）:");
        for (int i = 1; i <= 5; i++) {
            System.out.print(i + " ");
        }
        System.out.println();

        // while文

    public class ArrayReview {
    public static void main(String[] args) {
        System.out.println("=== 配列の便利さ ===");

        // 1. 複数のデータを効率的に管理
        String[] students = {"田中", "佐藤", "山田", "鈴木", "高橋"};

        System.out.println("学生一覧:");
        for (int i = 0; i < students.length; i++) {
            System.out.println((i + 1) + ". " + students[i]);
        }

        // 2. 計算結果を保存
        int[] scores = {85, 90, 78, 92, 88};

        // 合計点の計算
        int total = 0;
        for (int score : scores) {  // 拡張for文（for-each）
            total += score;
        }

        double average = (double) total / scores.length;
        System.out.println("平均点: " + average);

        // 3. データの検索
        String searchName = "山田";
        int index = -1;

        for (int i = 0; i < students.length; i++) {
            if (students[i].equals(searchName)) {
                index = i;
                break;
            }
        }

        if (index != -1) {
            System.out.println(searchName + "さんの成績: " + scores[index] + "点");
        }

        System.out.println("\n配列は確かに便利！でも...");
    }
}

    import java.util.Scanner;

public class ArrayLimitations {
    public static void main(String[] args) {
        System.out.println("=== 配列の限界を体験 ===");

        // 問題1: サイズが固定されている
        demonstrateFixedSize();

        // 問題2: 要素の追加・削除が困難
        demonstrateInsertionDifficulty();

        // 問題3: メモリの無駄使い
        demonstrateMemoryWaste();

        System.out.println("\n配列だけでは限界がある...");
        System.out.println("もっと柔軟なデータ構造が必要だ！");
    }

    // 問題1: サイズ固定の問題
    public static void demonstrateFixedSize() {
        System.out.println("\n【問題1: サイズが固定】");

        // 学生数を5人と決めて配列を作成
        String[] students = new String[5];
        students[0] = "田中";
        students[1] = "佐藤";
        students[2] = "山田";
        students[3] = "鈴木";
        students[4] = "高橋";

        System.out.println("現在の学生: " + java.util.Arrays.toString(students));

        // 6人目を追加したい！でも...
        System.out.println("6人目の学生「渡辺」を追加したい...");
        System.out.println("でも配列のサイズは5で固定！");

        // 解決法: 新しい配列を作って全てコピー
        String[] newStudents = new String[6];
        for (int i = 0; i < students.length; i++) {
            newStudents[i] = students[i];
        }
        newStudents[5] = "渡辺";

        System.out.println("新しい配列: " + java.util.Arrays.toString(newStudents));
        System.out.println("→ 毎回新しい配列を作るのは面倒...");
    }

    // 問題2: 要素の挿入・削除の困難さ
    public static void demonstrateInsertionDifficulty() {
        System.out.println("\n【問題2: 要素の挿入・削除が困難】");

        int[] numbers = {10, 20, 30, 40, 50};
        System.out.println("元の配列: " + java.util.Arrays.toString(numbers));

        // 3番目の位置（インデックス2）に25を挿入したい
        System.out.println("3番目に25を挿入したい...");

        // 手順が複雑：
        // 1. 新しい配列を作成
        int[] newNumbers = new int[6];

        // 2. 挿入位置より前をコピー
        for (int i = 0; i < 2; i++) {
            newNumbers[i] = numbers[i];
        }

        // 3. 新しい値を挿入
        newNumbers[2] = 25;

        // 4. 挿入位置より後をコピー（1つずつ後ろにずらす）
        for (int i = 2; i < numbers.length; i++) {
            newNumbers[i + 1] = numbers[i];
        }

        System.out.println("結果: " + java.util.Arrays.toString(newNumbers));
        System.out.println("→ 挿入だけでこんなに複雑...");
    }

    // 問題3: メモリの無駄使い
    public static void demonstrateMemoryWaste() {
        System.out.println("\n【問題3: メモリの無駄使い】");

        // 100人分の配列を用意したが、実際は3人だけ
        String[] classRoom = new String[100];
        classRoom[0] = "田中";
        classRoom[1] = "佐藤";
        classRoom[2] = "山田";

        int usedSlots = 3;
        int totalSlots = classRoom.length;
        int wastedSlots = totalSlots - usedSlots;

        System.out.println("配列サイズ: " + totalSlots);
        System.out.println("使用中: " + usedSlots);
        System.out.println("無駄: " + wastedSlots + " (97%が無駄！)");

        System.out.println("→ サイズを事前に決めるのは難しい...");
    }
}
