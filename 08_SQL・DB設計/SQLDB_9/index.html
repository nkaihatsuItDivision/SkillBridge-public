    🚀 インデックス戦略の考え方

    実用システムでは、検索パフォーマンスが非常に重要です。適切なインデックス設計により、検索速度を大幅に向上させることができます。

    インデックスの基本概念

    -- インデックスなしの検索（フルテーブルスキャン）
SELECT * FROM students WHERE last_name = '田中';  -- 全行を順次検索

-- インデックス有りの検索（高速検索）
CREATE INDEX idx_last_name ON students(last_name);
SELECT * FROM students WHERE last_name = '田中';  -- インデックスを使用した高速検索

    実用的なインデックス設計例

    1. 単一カラムインデックス 

    -- 学生検索で頻繁に使用される項目

CREATE INDEX idx_email ON students(email);                    -- メール検索
CREATE INDEX idx_status ON students(status);                  -- 在籍状況検索

    2. 複合インデックス 

    -- 複数条件での検索を最適化
CREATE INDEX idx_name_full ON students(last_name, first_name);  -- 氏名での検索
CREATE INDEX idx_dept_status ON students(department_id, status);  -- 学科・状況での検索
CREATE INDEX idx_grade_lookup ON grades(student_id, academic_year, semester);  -- 成績検索

    3. 用途別インデックス戦略

    -- 日常業務での検索パターンに対応
-- ケース1: 学生一覧（学科・学年別）

-- ケース3: 教員の担当科目検索
CREATE INDEX idx_teacher_subjects ON subjects(department_id, semester);

-- ケース4: 統計・レポート用
CREATE INDEX idx_statistics ON grades(department_id, grade_value, academic_year);

    インデックス設計の実践的な考え方

    WHERE句で頻繁に使用される列 

    -- よく実行される検索パターンを分析

-- → department_id, status のインデックスが必要

SELECT s.*, d.department_name 
FROM students s 
JOIN departments d ON s.department_id = d.department_id;
