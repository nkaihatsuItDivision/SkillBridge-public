    </select>

    <!-- Week 9.1: 新規登録で使用 -->
    <insert id="save" useGeneratedKeys="true" keyProperty="studentId">
        INSERT INTO students (name, email, enrollment_date, gpa, department_id)
        VALUES (#{name}, #{email}, #{enrollmentDate}, #{gpa}, #{departmentId})
    </insert>

    <!-- Week 9.2: 更新機能で使用 -->
    <update id="update">

            name = #{name},
            email = #{email},
            gpa = #{gpa},
            department_id = #{departmentId}
        WHERE student_id = #{studentId}
    </update>

    <!-- Week 10.1: 動的検索で使用（MyBatisの高度機能） -->
    <select id="search" resultMap="studentResultMap">
        SELECT student_id, name, email, enrollment_date, gpa, department_id
        FROM students
        <where>
            <if test="name != null and name != ''">
                AND name LIKE CONCAT('%', #{name}, '%')
            </if>
            <if test="email != null and email != ''">
                AND email LIKE CONCAT('%', #{email}, '%')
            </if>
            <if test="departmentId != null and departmentId > 0">
                AND department_id = #{departmentId}
            </if>
            <if test="gpa != null and gpa > 0">
                AND gpa >= #{gpa}
            </if>
        </where>
        ORDER BY student_id
    </select>

    <!-- Week 10.2: 削除機能で使用 -->
    <delete id="delete">
        DELETE FROM students WHERE student_id = #{id}
    </delete>

    <!-- Week 10.3: 論理削除で使用 -->
    <update id="logicalDelete">
        UPDATE students SET deleted_at = NOW() WHERE student_id = #{id}
    </update>

    package com.example.demo.form;

import java.time.LocalDate;
import javax.validation.constraints.*;

    private Integer studentId;

    // Week 9.1: バリデーション学習で使用

    @Size(max = 100, message = "学生名は100文字以内で入力してください")
    private String name;

    // Week 9.2: Email形式チェックの学習で使用
    @NotBlank(message = "メールアドレスは必須です")
    @Email(message = "正しいメールアドレス形式で入力してください")
    private String email;

    // Week 9.3: 日付入力の学習で使用
    @NotNull(message = "入学日は必須です")
    private LocalDate enrollmentDate;

    // Week 9.4: 数値範囲チェックの学習で使用
    @NotNull(message = "GPAは必須です")
    @DecimalMin(value = "0.0", message = "GPAは0.0以上で入力してください")
    @DecimalMax(value = "4.0", message = "GPAは4.0以下で入力してください")
    private Double gpa;

    @NotNull(message = "学科を選択してください")
    @Min(value = 1, message = "正しい学科を選択してください")
    private Integer departmentId;
}
