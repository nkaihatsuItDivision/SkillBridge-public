    🔄 Week 5システムとの連携

    Week 5システムの概要振り返り

    Week 5では、継承・ポリモーフィズム・抽象化・例外処理・Lombokを統合した600-700行規模のOOP応用版学生管理システムを作成しました。

    Week 5システムの核心データ構造

    // 基本学生クラス
public class Student {
    private String studentId;      // 学生ID (例: "S001")
    private String name;           // 名前 (例: "田中太郎")
    private int age;               // 年齢 (例: 20)
    private String department;     // 学部 (例: "情報工学部")
    private List<Grade> grades;    // 成績リスト
}

// 成績データクラス
public class Grade {
    private String subject;        // 科目名 (例: "プログラミング")
    private int score;            // 点数 (例: 85)
    private String semester;      // 学期 (例: "前期")
    private LocalDate examDate;   // 試験日
}

CREATE TABLE student (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(10) NOT NULL UNIQUE,
    name VARCHAR(100) NOT NULL,
    age INT NOT NULL CHECK (age >= 15 AND age <= 100),
    department VARCHAR(100) NOT NULL,
    email VARCHAR(255),
    phone VARCHAR(20),
    address TEXT,
    enrollment_date DATE,
    status ENUM('active', 'inactive', 'graduated') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 科目マスタテーブル
CREATE TABLE subject (
    id INT AUTO_INCREMENT PRIMARY KEY,
    subject_code VARCHAR(10) NOT NULL UNIQUE,
    subject_name VARCHAR(100) NOT NULL,
    credits INT NOT NULL CHECK (credits > 0),
    department VARCHAR(100),
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 成績管理テーブル
CREATE TABLE score (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(10) NOT NULL,
    subject_code VARCHAR(10) NOT NULL,
    score INT NOT NULL CHECK (score >= 0 AND score <= 100),
    semester VARCHAR(20) NOT NULL,
    exam_date DATE,
    grade_letter ENUM('A', 'B', 'C', 'D', 'F') GENERATED ALWAYS AS (
        CASE 
            WHEN score >= 90 THEN 'A'
            WHEN score >= 80 THEN 'B'
            WHEN score >= 70 THEN 'C'
            WHEN score >= 60 THEN 'D'
            ELSE 'F'
        END
    ) STORED,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES student(student_id) ON DELETE CASCADE,
    FOREIGN KEY (subject_code) REFERENCES subject(subject_code),
    UNIQUE KEY unique_student_subject_semester (student_id, subject_code, semester)
);

    -- 学生別GPA計算（4.0スケール）
SELECT 
    s.student_id,
    s.name,
    s.department,
    COUNT(sc.id) as total_subjects,
    ROUND(
        SUM(
            CASE 
                WHEN sc.score >= 90 THEN 4.0 * sub.credits
                WHEN sc.score >= 80 THEN 3.0 * sub.credits
                WHEN sc.score >= 70 THEN 2.0 * sub.credits
                WHEN sc.score >= 60 THEN 1.0 * sub.credits

WHERE s.status = 'active'
GROUP BY s.student_id, s.name, s.department

    -- 将来の成績予測に使える分析データ
CREATE VIEW student_performance_summary AS
SELECT 
    s.student_id,
    s.name,
    s.age,
    s.department,
    COUNT(sc.id) as subjects_taken,
    ROUND(AVG(sc.score), 2) as current_gpa_raw,
    ROUND(
        SUM(
            CASE 
                WHEN sc.score >= 90 THEN 4.0 * sub.credits
                WHEN sc.score >= 80 THEN 3.0 * sub.credits
                WHEN sc.score >= 70 THEN 2.0 * sub.credits
                WHEN sc.score >= 60 THEN 1.0 * sub.credits
                ELSE 0.0
            END
        ) / SUM(sub.credits), 2
    ) as current_gpa,
    SUM(sub.credits) as credits_earned,
    COUNT(CASE WHEN sc.score >= 90 THEN 1 END) as excellent_grades,

    -- 学生成績サマリービュー
CREATE VIEW student_grade_summary AS
SELECT 
    s.student_id,
    s.name,
    s.department,
    COUNT(sc.id) as subject_count,
    ROUND(AVG(sc.score), 2) as average_score,
    MAX(sc.score) as highest_score,
    MIN(sc.score) as lowest_score,
    ROUND(
        SUM(
            CASE 
                WHEN sc.score >= 90 THEN 4.0 * sub.credits
                WHEN sc.score >= 80 THEN 3.0 * sub.credits
                WHEN sc.score >= 70 THEN 2.0 * sub.credits
                WHEN sc.score >= 60 THEN 1.0 * sub.credits
                ELSE 0.0
            END
        ) / NULLIF(SUM(sub.credits), 0), 2
    ) as gpa
FROM student s
LEFT JOIN score sc ON s.student_id = sc.student_id
LEFT JOIN subject sub ON sc.subject_code = sub.subject_code
WHERE s.status = 'active'
GROUP BY s.student_id, s.name, s.department;
