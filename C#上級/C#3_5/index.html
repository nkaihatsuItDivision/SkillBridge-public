    using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

// ガベージコレクションの動作を理解するためのデモクラス
public class GCBehaviorDemo
{
    public static void DemonstrateGenerationalGC()
    {
        Console.WriteLine(&quot;=== 世代別ガベージコレクション動作デモ ===&quot;);

        // 開始時の状況を記録
        ShowMemoryAndGCInfo(&quot;開始時&quot;);

        // 短時間で消える小さなオブジェクトを大量作成（Gen 0対象）
        CreateShortLivedObjects();
        ShowMemoryAndGCInfo(&quot;短時間オブジェクト作成後&quot;);

        // 長時間残る大きなオブジェクトを作成（LOH対象）
        var longLivedData = CreateLongLivedObjects();
        ShowMemoryAndGCInfo(&quot;長時間オブジェクト作成後&quot;);

        // 強制的にフルGCを実行
        ForceFullGC();
        ShowMemoryAndGCInfo(&quot;フルGC実行後&quot;);

        // 長時間オブジェクトを保持して世代昇格を確認
        GC.KeepAlive(longLivedData);
        ShowGCGenerations();
    }

    private static void CreateShortLivedObjects()
    {
        // 小さなオブジェクトを大量作成してすぐ破棄
        for (int i = 0; i &lt; 100000; i++)
        {
            var tempList = new List&lt;int&gt; { i, i + 1, i + 2 };
            var tempString = $&quot;Temporary string {i}&quot;;
            // ローカル変数はすぐにスコープを外れ、Gen 0 GCの対象となる
        }
    }

    private static byte[][] CreateLongLivedObjects()
    {
        // 長時間保持される大きなオブジェクト（LOH対象）
        var longLivedData = new byte[10][];
        for (int i = 0; i &lt; 10; i++)
        {
            // 85KB以上のオブジェクトはLarge Object Heap (LOH)に配置
            longLivedData[i] = new byte[100 * 1024]; // 100KB
        }
        return longLivedData;
    }

    private static void ForceFullGC()
    {
        // 本来は推奨されないが、デモのため実行
        GC.Collect();
        GC.WaitForPendingFinalizers();
        GC.Collect();
    }

    private static void ShowMemoryAndGCInfo(string timing)
    {
        var totalMemory = GC.GetTotalMemory(false);
        Console.WriteLine($&quot;\n【{timing}】&quot;);
        Console.WriteLine($&quot;総メモリ使用量: {totalMemory:N0} bytes ({totalMemory / 1024.0 / 1024.0:F2} MB)&quot;);
        Console.WriteLine($&quot;Gen 0 collections: {GC.CollectionCount(0)}&quot;);
        Console.WriteLine($&quot;Gen 1 collections: {GC.CollectionCount(1)}&quot;);
        Console.WriteLine($&quot;Gen 2 collections: {GC.CollectionCount(2)}&quot;);
    }

    private static void ShowGCGenerations()
    {
        Console.WriteLine(&quot;\n=== オブジェクト世代情報 ===&quot;);
        var testObject = new object();
        Console.WriteLine($&quot;新規オブジェクトの世代: {GC.GetGeneration(testObject)}&quot;);
        Console.WriteLine($&quot;最大世代: {GC.MaxGeneration}&quot;);
    }
}

    IDisposableパターンの正しい実装

    using System;
using System.IO;
using System.Net.Http;
using System.Threading;

// 正しいIDisposableパターンの実装例
public class ResourceManager : IDisposable
{
    private readonly FileStream _fileStream;
    private readonly HttpClient _httpClient;
    private readonly Timer _timer;
    private bool _disposed = false;

    public ResourceManager(string filePath)
    {
        try
        {
            _fileStream = new FileStream(filePath, FileMode.OpenOrCreate, FileAccess.ReadWrite);
            _httpClient = new HttpClient { Timeout = TimeSpan.FromSeconds(30) };
            _timer = new Timer(OnTimerElapsed, null, TimeSpan.Zero, TimeSpan.FromSeconds(5));

            Console.WriteLine($&quot;ResourceManager initialized for: {filePath}&quot;);
        }
        catch
        {
            // 初期化中にエラーが発生した場合、既に作成されたリソースを確実に解放
            _fileStream?.Dispose();
            _httpClient?.Dispose();
            _timer?.Dispose();
            throw;
        }
    }

    private void OnTimerElapsed(object state)
    {
        if (!_disposed)
        {
            Console.WriteLine($&quot;Timer tick: {DateTime.Now:HH:mm:ss} - Memory: {GC.GetTotalMemory(false):N0} bytes&quot;);
        }
    }

    public void WriteData(string data)
    {
        ThrowIfDisposed();

        var bytes = System.Text.Encoding.UTF8.GetBytes(data + Environment.NewLine);
        _fileStream.Write(bytes, 0, bytes.Length);
        _fileStream.Flush();

        Console.WriteLine($&quot;Data written: {data}&quot;);
    }

    public async Task&lt;string&gt; FetchDataAsync(string url)
    {
        ThrowIfDisposed();

        try
        {
            var response = await _httpClient.GetStringAsync(url);
            Console.WriteLine($&quot;Data fetched from: {url}&quot;);
            return response;
        }
        catch (HttpRequestException ex)
        {
            Console.WriteLine($&quot;HTTP error: {ex.Message}&quot;);
            throw;
        }
    }

    // パブリックなDisposeメソッド
    public void Dispose()
    {
        Dispose(disposing: true);
        GC.SuppressFinalize(this); // ファイナライザーの実行を抑制
    }

    // 実際のリソース解放処理（継承を考慮した設計）
    protected virtual void Dispose(bool disposing)
    {
        if (!_disposed)
        {
            if (disposing)
            {
                // マネージドリソースの解放
                Console.WriteLine(&quot;Disposing managed resources...&quot;);
                _fileStream?.Dispose();
                _httpClient?.Dispose();
                _timer?.Dispose();
            }

            // アンマネージドリソースの解放（この例では該当なし）
            Console.WriteLine(&quot;Disposing unmanaged resources...&quot;);

            _disposed = true;
        }
    }

    // ファイナライザー（通常は実装不要だが、アンマネージドリソースがある場合のパターン）
    ~ResourceManager()
    {
        Console.WriteLine(&quot;Finalizer called - this should be rare!&quot;);
        Dispose(disposing: false);
    }

    private void ThrowIfDisposed()
    {
        if (_disposed)
        {
            throw new ObjectDisposedException(nameof(ResourceManager));
        }
    }
}

// using文とusing宣言の効果的な使用例
public class ResourceUsageExamples
{
    public async Task DemonstrateResourceManagement()
    {
        Console.WriteLine(&quot;=== リソース管理デモ ===&quot;);

        // using文による確実なリソース解放
        using (var resourceManager = new ResourceManager(&quot;demo.txt&quot;))
        {
            resourceManager.WriteData(&quot;Sample data 1&quot;);
            resourceManager.WriteData(&quot;Sample data 2&quot;);

            // 非同期処理での適切なリソース管理
            await ProcessMultipleFilesAsync();

            // このブロックを抜ける時に自動的にDispose()が呼ばれる
        }

        Console.WriteLine(&quot;Resource management completed&quot;);
    }

    private async Task ProcessMultipleFilesAsync()
    {
        var fileNames = new[] { &quot;file1.txt&quot;, &quot;file2.txt&quot;, &quot;file3.txt&quot; };

        foreach (var fileName in fileNames)
        {
            // using宣言（C# 8.0以降）でより簡潔に
            using var reader = new StreamReader(fileName);

            try
            {
                var content = await reader.ReadToEndAsync();
                Console.WriteLine($&quot;Processed {fileName}: {content?.Length ?? 0} characters&quot;);
            }
            catch (FileNotFoundException)
            {
                Console.WriteLine($&quot;File not found: {fileName}&quot;);
            }
            // readerはメソッド終了時に自動的にDispose()される
        }
    }
}

    イベントハンドラーによるメモリリーク対策

    using System;
using System.Collections.Generic;
using System.Linq;
using System.Runtime.CompilerServices;

// 問題のあるコード例：メモリリークが発生する
public class EventPublisher
{
    public event Action&lt;string&gt; DataChanged;

    public void PublishData(string data)
    {
        Console.WriteLine($&quot;Publishing: {data}&quot;);
        DataChanged?.Invoke(data);
    }

    public int GetSubscriberCount()
    {
        return DataChanged?.GetInvocationList().Length ?? 0;
    }
}

// 問題のあるサブスクライバー（イベント購読解除を忘れる）
public class ProblematicSubscriber
{
    private readonly int _id;

    public ProblematicSubscriber(int id)
    {
        _id = id;
    }

    public void SubscribeTo(EventPublisher publisher)
    {
        publisher.DataChanged += OnDataChanged;
        Console.WriteLine($&quot;Subscriber {_id} subscribed&quot;);
    }

    private void OnDataChanged(string data)
    {
        Console.WriteLine($&quot;Subscriber {_id} received: {data}&quot;);
    }

    // Disposeメソッドがない！
    // イベント購読解除を忘れると、GCされずメモリリークの原因となる
}

// 正しいイベントハンドリング実装
public class SafeEventSubscriber : IDisposable
{
    private readonly int _id;
    private EventPublisher _publisher;
    private bool _disposed = false;

    public SafeEventSubscriber(int id)
    {
        _id = id;
    }

    public void SubscribeTo(EventPublisher publisher)
    {
        if (_disposed)
            throw new ObjectDisposedException(nameof(SafeEventSubscriber));

        _publisher = publisher;
        _publisher.DataChanged += OnDataChanged;
        Console.WriteLine($&quot;Safe subscriber {_id} subscribed&quot;);
    }

    private void OnDataChanged(string data)
    {
        Console.WriteLine($&quot;Safe subscriber {_id} received: {data}&quot;);
    }

    public void Dispose()
    {
        if (!_disposed)
        {
            // イベント購読を確実に解除
            if (_publisher != null)
            {
                _publisher.DataChanged -= OnDataChanged;
                Console.WriteLine($&quot;Safe subscriber {_id} unsubscribed&quot;);
                _publisher = null;
            }

            _disposed = true;
        }
    }
}

// WeakEventパターンによる高度な実装
public class WeakEventManager
{
    private readonly List&lt;WeakReference&gt; _subscribers = new();
    private readonly object _lock = new object();

    public void Subscribe(Action&lt;string&gt; handler)
    {
        lock (_lock)
        {
            _subscribers.Add(new WeakReference(handler));
        }
    }

    public void Publish(string data)
    {
        List&lt;WeakReference&gt; aliveSubscribers;

        lock (_lock)
        {
            aliveSubscribers = new List&lt;WeakReference&gt;();

            foreach (var weakRef in _subscribers)
            {
                if (weakRef.Target is Action&lt;string&gt; handler)
                {
                    try
                    {
                        handler(data);
                        aliveSubscribers.Add(weakRef);
                    }
                    catch (Exception ex)
                    {
                        Console.WriteLine($&quot;Handler execution failed: {ex.Message}&quot;);
                    }
                }
                // 既にGCされたハンドラーは自動的に除外される
            }

            _subscribers.Clear();
            _subscribers.AddRange(aliveSubscribers);
        }

        Console.WriteLine($&quot;Event published to {aliveSubscribers.Count} active subscribers&quot;);
    }

    public int GetActiveSubscriberCount()
    {
        lock (_lock)
        {
            return _subscribers.Count(wr =&gt; wr.IsAlive);
        }
    }
}

    大きなオブジェクトヒープ（LOH）の効果的な管理

    using System;
using System.Buffers;
using System.Collections.Concurrent;

public class LargeObjectOptimizer
{
    private const int LOH_THRESHOLD = 85000; // 85KB以上はLOHに配置
    private readonly ConcurrentQueue&lt;byte[]&gt; _arrayPool = new();
    private int _poolHits = 0;
    private int _poolMisses = 0;

    // 問題のあるコード：大きな配列を頻繁に作成・破棄
    public void DemonstrateLOHProblem()
    {
        Console.WriteLine(&quot;=== LOH問題のデモンストレーション ===&quot;);

        var beforeMemory = GC.GetTotalMemory(true);
        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        for (int i = 0; i &lt; 1000; i++)
        {
            // 100KBの配列を作成（LOHに配置される）
            var largeArray = new byte[100 * 1024];

            // 簡単な処理をシミュレート
            ProcessLargeArray(largeArray);

            if (i % 100 == 0)
            {
                var currentMemory = GC.GetTotalMemory(false);
                Console.WriteLine($&quot;Iteration {i}: Memory usage = {currentMemory - beforeMemory:N0} bytes&quot;);
            }
        }

        stopwatch.Stop();
        var afterMemory = GC.GetTotalMemory(false);

        Console.WriteLine($&quot;Problem approach completed in {stopwatch.ElapsedMilliseconds}ms&quot;);
        Console.WriteLine($&quot;Memory delta: {afterMemory - beforeMemory:N0} bytes&quot;);
        Console.WriteLine(&quot;Large arrays created. Memory may not be immediately freed due to LOH behavior.&quot;);
    }

    // 改善版：カスタムArrayPoolの使用
    public void DemonstrateArrayPooling()
    {
        Console.WriteLine(&quot;\n=== Array Pooling最適化デモ ===&quot;);

        var beforeMemory = GC.GetTotalMemory(true);
        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        for (int i = 0; i &lt; 1000; i++)
        {
            var array = RentArray(100 * 1024);

            ProcessLargeArray(array);

            ReturnArray(array);

            if (i % 100 == 0)
            {
                var currentMemory = GC.GetTotalMemory(false);
                Console.WriteLine($&quot;Iteration {i}: Memory = {currentMemory - beforeMemory:N0} bytes, &quot; +
                                 $&quot;Pool hits: {_poolHits}, misses: {_poolMisses}&quot;);
            }
        }

        stopwatch.Stop();
        var afterMemory = GC.GetTotalMemory(false);

        Console.WriteLine($&quot;Pooling approach completed in {stopwatch.ElapsedMilliseconds}ms&quot;);
        Console.WriteLine($&quot;Memory delta: {afterMemory - beforeMemory:N0} bytes&quot;);
        Console.WriteLine($&quot;Final pool size: {_arrayPool.Count} arrays&quot;);
        Console.WriteLine($&quot;Pool efficiency: {_poolHits}/{_poolHits + _poolMisses} hits&quot;);
    }

    public byte[] RentArray(int minimumSize)
    {
        if (minimumSize &gt;= LOH_THRESHOLD)
        {
            // プールから再利用可能な配列を探す
            if (_arrayPool.TryDequeue(out var pooledArray) &amp;&amp; pooledArray.Length &gt;= minimumSize)
            {
                _poolHits++;
                return pooledArray;
            }
        }

        _poolMisses++;
        return new byte[minimumSize];
    }

    public void ReturnArray(byte[] array)
    {
        if (array.Length &gt;= LOH_THRESHOLD)
        {
            // 配列をクリアして（セキュリティのため）プールに戻す
            Array.Clear(array, 0, array.Length);
            _arrayPool.Enqueue(array);
        }
    }

    // .NET標準のArrayPool&lt;T&gt;を使用した最適化例
    public void UsingSystemArrayPool()
    {
        Console.WriteLine(&quot;\n=== System.ArrayPool使用例 ===&quot;);

        var pool = ArrayPool&lt;byte&gt;.Shared;
        var beforeMemory = GC.GetTotalMemory(true);
        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        for (int i = 0; i &lt; 1000; i++)
        {
            // 配列を借りる（実際のサイズは要求より大きい可能性がある）
            var array = pool.Rent(100 * 1024);

            try
            {
                ProcessLargeArray(array);

                if (i % 100 == 0)
                {
                    Console.WriteLine($&quot;ArrayPool iteration {i}: Array size = {array.Length:N0}&quot;);
                }
            }
            finally
            {
                // 必ず返却する（clearArrayでデータをクリア）
                pool.Return(array, clearArray: true);
            }
        }

        stopwatch.Stop();
        var afterMemory = GC.GetTotalMemory(false);

        Console.WriteLine($&quot;ArrayPool approach completed in {stopwatch.ElapsedMilliseconds}ms&quot;);
        Console.WriteLine($&quot;Memory delta: {afterMemory - beforeMemory:N0} bytes&quot;);
    }

    private void ProcessLargeArray(byte[] array)
    {
        // 配列処理をシミュレート
        int checksum = 0;
        for (int i = 0; i &lt; Math.Min(1000, array.Length); i++)
        {
            array[i] = (byte)(i % 256);
            checksum += array[i];
        }
    }
}

    3. パフォーマンス測定 - 科学的なボトルネック特定

    高精度測定ユーティリティ

    using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Text;

public static class PerformanceMeasurement
{
    public static TimeSpan MeasureExecutionTime(Action action, int iterations = 1)
    {
        // ウォームアップ実行（JITコンパイルの影響を除去）
        action();

        // GCの影響を最小化
        GC.Collect();
        GC.WaitForPendingFinalizers();
        GC.Collect();

        var stopwatch = Stopwatch.StartNew();

        for (int i = 0; i &lt; iterations; i++)
        {
            action();
        }

        stopwatch.Stop();

    public static (TimeSpan time, long memoryUsed, int gen0, int gen1, int gen2) MeasureTimeAndMemory(
        Action action, int iterations = 1)
    {
        var gcBefore = (GC.CollectionCount(0), GC.CollectionCount(1), GC.CollectionCount(2));
        var memoryBefore = GC.GetTotalMemory(true);

        var time = MeasureExecutionTime(action, iterations);

        var memoryAfter = GC.GetTotalMemory(false);
        var gcAfter = (GC.CollectionCount(0), GC.CollectionCount(1), GC.CollectionCount(2));

        return (
            time, 
            memoryAfter - memoryBefore,
            gcAfter.Item1 - gcBefore.Item1,
            gcAfter.Item2 - gcBefore.Item2,
            gcAfter.Item3 - gcBefore.Item3
        );
    }

        // 改善度の計算と表示

        Console.WriteLine($&quot;  速度: {method2Name}は{method1Name}の{speedImprovement:F2}倍の速度&quot;);

    public static PerformanceStatistics MeasureWithStatistics(Action action, int runs = 10)
    {
        var times = new List&lt;double&gt;();
        var memories = new List&lt;long&gt;();

        for (int i = 0; i &lt; runs; i++)
        {
            var (time, memory, _, _, _) = MeasureTimeAndMemory(action);
            times.Add(time.TotalMilliseconds);
            memories.Add(memory);
        }

        return new PerformanceStatistics
        {
            TimeAverage = times.Average(),
            TimeStdDev = CalculateStandardDeviation(times),
            TimeMin = times.Min(),
            TimeMax = times.Max(),
            MemoryAverage = memories.Average(),
            MemoryStdDev = CalculateStandardDeviation(memories.Select(m =&gt; (double)m)),
            Runs = runs
        };
    }

    private static double CalculateStandardDeviation(IEnumerable&lt;double&gt; values)
    {
        var enumerable = values.ToList();
        var average = enumerable.Average();
        var sumOfSquaresOfDifferences = enumerable.Select(val =&gt; (val - average) * (val - average)).Sum();
        return Math.Sqrt(sumOfSquaresOfDifferences / enumerable.Count);
    }

    using System;

public class SpanOptimizations
{
    // 従来の方法（メモリ割り当てが多い）
    public int ProcessStringTraditional(string input)
    {
        // 文字列分割で新しい配列が作成される
        var parts = input.Split(&apos;,&apos;);
        int sum = 0;

        foreach (var part in parts)
        {
            // Trimで新しい文字列が作成される
            var trimmed = part.Trim();
            if (int.TryParse(trimmed, out var number))
            {
                sum += number;
            }
        }

        return sum;
    }

    // Span&lt;T&gt;を使用した最適化版（メモリ割り当てを大幅に削減）
    public int ProcessStringWithSpan(ReadOnlySpan&lt;char&gt; input)
    {
        int sum = 0;
        int start = 0;

        for (int i = 0; i &lt;= input.Length; i++)
        {
            if (i == input.Length || input[i] == &apos;,&apos;)
            {
                // 部分文字列を新しい文字列として作成せずに参照
                var part = input.Slice(start, i - start).Trim();

                if (int.TryParse(part, out var number))
                {
                    sum += number;
                }

                start = i + 1;
            }
        }

        return sum;
    }

    public int ProcessWithStackAlloc(ReadOnlySpan&lt;int&gt; input)
    {
        // スタックに小さなバッファを確保（ヒープ割り当てなし）
        Span&lt;int&gt; buffer = stackalloc int[256];

        int sum = 0;
        int bufferIndex = 0;

        for (int i = 0; i &lt; input.Length; i++)
        {
            if (input[i] &gt; 0)
            {
                buffer[bufferIndex++] = input[i];

                if (bufferIndex == buffer.Length)
                {
                    // バッファが満杯になったら処理
                    sum += ProcessBuffer(buffer);
                    bufferIndex = 0;
                }
            }
        }

        // 残りのデータを処理
        if (bufferIndex &gt; 0)
        {
            sum += ProcessBuffer(buffer.Slice(0, bufferIndex));
        }

        return sum;
    }

    private int ProcessBuffer(ReadOnlySpan&lt;int&gt; buffer)
    {
        int sum = 0;
        foreach (var value in buffer)
        {
            sum += value;
        }
        return sum;
    }

    {
        var buffer = _bytePool.Rent(8192); // 8KB buffer

        try
        {

            var totalBytesRead = 0;
            int bytesRead;

            // 非同期読み込みでMemory&lt;T&gt;を使用
            while ((bytesRead = await dataStream.ReadAsync(memory.Slice(totalBytesRead))) &gt; 0)
            {
                totalBytesRead += bytesRead;

                if (totalBytesRead &gt;= memory.Length)
                {
                    // バッファが満杯の場合の処理
                    var processedData = ProcessMemorySegment(memory.Slice(0, totalBytesRead));
                    Console.WriteLine($&quot;Processed {totalBytesRead} bytes: {processedData}&quot;);
                    totalBytesRead = 0;
                }
            }

            // 残りのデータを処理
            if (totalBytesRead &gt; 0)
            {
                var finalData = ProcessMemorySegment(memory.Slice(0, totalBytesRead));
                return finalData;
            }

            return &quot;Processing completed&quot;;
        }
        finally
        {
            _bytePool.Return(buffer);
        }
    }

        }

            {

        }

        return true;
    }

        return processed;

    重要なポイントとまとめ

    技術別効果とトレードオフ

    技術 主な効果 適用場面 パフォーマンス向上 実装コスト 注意点
    IDisposable リソース管理 ファイル、DB、HTTP ⭐⭐ ⭐⭐ 必ず実装する
    ArrayPool<T> GC圧力軽減 大きな配列の頻繁な作成 ⭐⭐⭐⭐ ⭐⭐ LOH対象のオブジェクト
    Span<T> ゼロアロケーション 文字列・配列処理 ⭐⭐⭐⭐⭐ ⭐⭐⭐ 非同期で使用不可
    ValueTask<T> 割り当て削減 頻繁な非同期呼び出し ⭐⭐⭐ ⭐⭐ キャッシュヒット時に有効
    並行度制御 リソース効率化 APIクライアント、I/O処理 ⭐⭐⭐⭐ ⭐⭐⭐ 適切な制限値設定

    パフォーマンス最適化の戦略的アプローチ

    1. 測定駆動最適化

    1. ベースライン測定 → 2. ボトルネック特定 → 3. 最適化実装 → 4. 効果検証 → 5. 継続改善

    2. メモリ管理の段階的改善

    Level 1: IDisposable実装とusing文の徹底
Level 2: イベントハンドラーリークの防止
Level 3: ArrayPool&lt;T&gt;による大きなオブジェクトの再利用
Level 4: Span&lt;T&gt;/Memory&lt;T&gt;によるゼロアロケーション
Level 5: カスタムメモリ管理とプロファイリング

    3. 非同期処理最適化の原則

    • 適切な並行度: CPU集約的処理はCPUコア数、I/O処理はより多く
    • ValueTask<T>活用: キャッシュヒット率が高い場面で威力を発揮
    • ConfigureAwait(false): ライブラリコードでは必須
    • SemaphoreSlim: リソース制限とバックプレッシャー制御

    実践的なベストプラクティス

    メモリ効率コーディング指針

    1. using文の徹底: IDisposable実装クラスは必ずusing文で使用
    2. イベント購読解除: 長寿命オブジェクトのイベント購読は必ず解除
    3. 大きなオブジェクト: 85KB以上のオブジェクトはArrayPool<T>使用を検討
    4. 文字列処理: 頻繁な操作はSpan<T>、ReadOnlySpan<T>を活用
    5. 非同期キャッシュ: ValueTask<T>でヒープ割り当てを削減

    パフォーマンス測定のコツ

    1. ウォームアップ: JITコンパイルの影響を除去
    2. GC制御: 測定前にフルGCを実行
    3. 統計的有意性: 複数回実行して平均と標準偏差を算出
    4. 環境の一貫性: 同一環境・同一条件での測定
    5. 実用的な負荷: 本番に近いデータサイズと処理パターン

    最適化の判断基準

    いつ最適化すべきか

    • パフォーマンス要件に達していない場合
    • メモリ使用量が予算を超過している場合
    • スケーラビリティに問題がある場合
    • ユーザー体験に悪影響がある場合

    いつ最適化すべきでないか

    • 測定していない推測に基づく最適化
    • 読みやすさを大幅に損なう場合
    • 保守性とのトレードオフが悪い場合
    • ビジネス価値が不明確な場合
