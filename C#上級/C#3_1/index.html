    非同期プログラミングとは何か

    非同期処理の必要性とメリット

    非同期プログラミングは、I/Oバウンドな処理（ファイルアクセス、ネットワーク通信、データベースアクセス）において、スレッドをブロックせずに他の処理を継続できるようにする技術です。

    一言で表すと：「待ち時間中に他の仕事ができる効率的な処理方式」

    従来の同期処理の問題点

    1. UIフリーズ問題

    // 問題のあるコード：UIスレッドでの同期処理
public partial class MainWindow : Window
{
    private void DownloadButton_Click(object sender, RoutedEventArgs e)
    {
        // UIスレッドで重い処理を実行（3秒間UIがフリーズ）
        var data = DownloadLargeFile(&quot;https://example.com/largefile.zip&quot;);
        ResultTextBox.Text = $&quot;ダウンロード完了: {data.Length} bytes&quot;;
    }

    private byte[] DownloadLargeFile(string url)
    {
        using var client = new HttpClient();
        // 同期的な呼び出し - UIスレッドをブロック
        return client.GetByteArrayAsync(url).Result; // 危険！
    }
}

    非同期処理による解決

    1. UI応答性の確保

    // 改善されたコード：非同期処理でUI応答性を保つ
public partial class MainWindow : Window
{
    private async void DownloadButton_Click(object sender, RoutedEventArgs e)
    {
        try
        {
            // UIはブロックされず、ユーザー操作可能
            ProgressBar.Visibility = Visibility.Visible;
            DownloadButton.IsEnabled = false;

            var data = await DownloadLargeFileAsync(&quot;https://example.com/largefile.zip&quot;);
            ResultTextBox.Text = $&quot;ダウンロード完了: {data.Length} bytes&quot;;
        }
        catch (Exception ex)
        {
            MessageBox.Show($&quot;エラー: {ex.Message}&quot;);
        }
        finally
        {
            ProgressBar.Visibility = Visibility.Hidden;
            DownloadButton.IsEnabled = true;
        }
    }

    private async Task&lt;byte[]&gt; DownloadLargeFileAsync(string url)
    {
        using var client = new HttpClient();
        return await client.GetByteArrayAsync(url); // 安全な非同期呼び出し
    }
}

    [ApiController]
public class GoodController : ControllerBase
{
    private readonly IDataService _dataService;

    public async Task&lt;IActionResult&gt; GetDataAsync(CancellationToken cancellationToken = default)

    3. 並行処理による高速化

    // 並行実行による処理時間短縮
public async Task&lt;string&gt; GetCombinedDataAsync()
{
    // 3つのAPI呼び出しを並行実行
    var task1 = CallExternalApi1Async(); // 2秒
    var task2 = CallExternalApi2Async(); // 2秒
    var task3 = CallExternalApi3Async(); // 2秒

    // 全て完了するまで待機（約2秒で完了）
    var results = await Task.WhenAll(task1, task2, task3);

    return CombineData(results[0], results[1], results[2]);
}

    async/awaitの内部動作とSynchronizationContext

    ステートマシンによる実装

    // 開発者が書くコード
public async Task&lt;string&gt; GetDataAsync()
{
    Console.WriteLine(&quot;処理開始&quot;);
    var result = await SomeAsyncOperation();
    Console.WriteLine(&quot;処理完了&quot;);
    return result;
}

// コンパイラが生成するステートマシン（概念的な表現）
public struct GetDataAsyncStateMachine : IAsyncStateMachine
{
    public int state;
    public AsyncTaskMethodBuilder&lt;string&gt; builder;
    public string result;
    public TaskAwaiter&lt;string&gt; awaiter;

    public void MoveNext()
    {
        switch (state)
        {
            case 0:
                Console.WriteLine(&quot;処理開始&quot;);
                awaiter = SomeAsyncOperation().GetAwaiter();
                if (!awaiter.IsCompleted)
                {
                    state = 1;
                    builder.AwaitUnsafeOnCompleted(ref awaiter, ref this);
                    return;
                }
                goto case 1;

            case 1:
                result = awaiter.GetResult();
                Console.WriteLine(&quot;処理完了&quot;);
                builder.SetResult(result);
                break;
        }
    }
}

    SynchronizationContextの重要性

    public class SynchronizationContextDemo
{
    // UIアプリケーションでの例
    public async Task UpdateUIAsync()
    {
        // UIスレッド（SynchronizationContext有り）
        Console.WriteLine($&quot;開始スレッド: {Thread.CurrentThread.ManagedThreadId}&quot;);

        // 非同期処理（バックグラウンドスレッド）
        await Task.Delay(1000);

        // UIスレッドに自動的に戻る（SynchronizationContextの働き）
        Console.WriteLine($&quot;継続スレッド: {Thread.CurrentThread.ManagedThreadId}&quot;);

        // UIコントロールを安全に更新可能
        // SomeLabel.Text = &quot;更新完了&quot;;
    }

    // ライブラリコードでの例
    public async Task ProcessDataAsync()
    {
        Console.WriteLine($&quot;開始スレッド: {Thread.CurrentThread.ManagedThreadId}&quot;);

        // ConfigureAwait(false)でSynchronizationContextを無視
        await Task.Delay(1000).ConfigureAwait(false);

        // 任意のスレッドプールスレッドで継続
        Console.WriteLine($&quot;継続スレッド: {Thread.CurrentThread.ManagedThreadId}&quot;);

        // UIコントロールアクセスは不可（別スレッドのため）
    }
}

    デッドロックの理解と回避

    デッドロックが発生する典型的なパターン

    1. .Result/.Wait()によるデッドロック

    // 危険：デッドロックを引き起こすコード
public class DeadlockExample
{
    public string GetDataSynchronously()
    {
        // デッドロック発生！
        return GetDataAsync().Result;
    }

    private async Task&lt;string&gt; GetDataAsync()
    {
        await Task.Delay(1000);
        return &quot;データ&quot;;
    }
}

// デッドロックの発生メカニズム：
// 1. UIスレッドがGetDataSynchronously()を呼び出し
// 2. .Resultでブロックしながら非同期処理の完了を待機
// 3. GetDataAsync()完了後、UIスレッドに戻ろうとする
// 4. しかしUIスレッドは.Resultでブロック中
// 5. デッドロック発生

    2. ConfigureAwait(false)による解決

    // 解決策1：ConfigureAwait(false)を使用
public class DeadlockFixed1
{
    public string GetDataSynchronously()
    {
        // ConfigureAwait(false)でデッドロック回避
        return GetDataAsync().ConfigureAwait(false).GetAwaiter().GetResult();
    }

    private async Task&lt;string&gt; GetDataAsync()
    {
        // ライブラリコードでは常にConfigureAwait(false)を使用
        await Task.Delay(1000).ConfigureAwait(false);
        return &quot;データ&quot;;
    }
}

// 解決策2：完全に非同期にする（推奨）
public class DeadlockFixed2
{
    public async Task&lt;string&gt; GetDataAsync()
    {
        return await GetDataInternalAsync();
    }

    private async Task&lt;string&gt; GetDataInternalAsync()
    {
        await Task.Delay(1000);
        return &quot;データ&quot;;
    }
}

    デッドロック検知と診断

    public class DeadlockDiagnostics
{
    public void DetectPotentialDeadlock()
    {
        var timer = new System.Diagnostics.Stopwatch();
        timer.Start();

        try
        {
            // タイムアウト付きで同期的待機
            var task = GetDataAsync();
            if (!task.Wait(TimeSpan.FromSeconds(5)))
            {
                Console.WriteLine(&quot;潜在的なデッドロックを検出！&quot;);
                LogCurrentThreadState();
            }
        }
        catch (AggregateException ex)
        {
            Console.WriteLine($&quot;例外発生: {ex.InnerException?.Message}&quot;);
        }
        finally
        {
            timer.Stop();
            Console.WriteLine($&quot;実行時間: {timer.ElapsedMilliseconds}ms&quot;);
        }
    }

    private void LogCurrentThreadState()
    {
        Console.WriteLine($&quot;現在のスレッド: {Thread.CurrentThread.ManagedThreadId}&quot;);
        Console.WriteLine($&quot;SynchronizationContext: {SynchronizationContext.Current?.GetType().Name ?? &quot;null&quot;}&quot;);
        Console.WriteLine($&quot;スレッドプールスレッド: {Thread.CurrentThread.IsThreadPoolThread}&quot;);
    }

    private async Task&lt;string&gt; GetDataAsync()
    {
        await Task.Delay(1000);
        return &quot;データ&quot;;
    }
}

    CancellationTokenの実践的活用

    基本的なキャンセル処理

    public class CancellationBasics
{
    public async Task&lt;string&gt; ProcessDataAsync(CancellationToken cancellationToken = default)
    {
        try
        {
            // 1. 処理開始時点でキャンセルチェック
            cancellationToken.ThrowIfCancellationRequested();

            Console.WriteLine(&quot;処理開始&quot;);

            // 2. 長時間処理中の定期的なキャンセルチェック
            for (int i = 0; i &lt; 100; i++)
            {
                cancellationToken.ThrowIfCancellationRequested();

                // 何らかの処理
                await Task.Delay(50, cancellationToken);

                Console.WriteLine($&quot;進捗: {i + 1}/100&quot;);
            }

            // 3. 外部API呼び出し時のキャンセルトークン転送
            using var httpClient = new HttpClient();
            var response = await httpClient.GetStringAsync(&quot;https://api.example.com/data&quot;, cancellationToken);

            return response;
        }
        catch (OperationCanceledException)
        {
            Console.WriteLine(&quot;処理がキャンセルされました&quot;);
            throw; // 再スローして呼び出し元に通知
        }
    }

    // 使用例
    public async Task ExampleUsage()
    {
        using var cts = new CancellationTokenSource();

        // 5秒後に自動キャンセル
        cts.CancelAfter(TimeSpan.FromSeconds(5));

        try
        {
            var result = await ProcessDataAsync(cts.Token);
            Console.WriteLine($&quot;結果: {result}&quot;);
        }
        catch (OperationCanceledException)
        {
            Console.WriteLine(&quot;処理はタイムアウトしました&quot;);
        }
    }
}

    高度なキャンセルパターン

    public class ConcurrentProcessing
{
    // 全てのタスクの完了を待機
    public async Task&lt;string[]&gt; ProcessAllAsync(string[] urls)
    {

        {
            using var client = new HttpClient();
            return await client.GetStringAsync(url);
        });

        try
        {
            // 全てのタスクが完了するまで待機
            var results = await Task.WhenAll(tasks);
            return results;
        }
        catch (Exception ex)
        {
            // いずれかのタスクが失敗した場合
            Console.WriteLine($&quot;一部のタスクが失敗: {ex.Message}&quot;);

            // 個別の結果を確認
            foreach (var task in tasks)
            {
                if (task.IsFaulted)
                {
                    Console.WriteLine($&quot;失敗したタスク: {task.Exception?.InnerException?.Message}&quot;);
                }
                else if (task.IsCompletedSuccessfully)
                {
                    Console.WriteLine($&quot;成功したタスク: {task.Result.Substring(0, Math.Min(50, task.Result.Length))}...&quot;);
                }
            }

            throw;
        }
    }

    // 最初に完了したタスクの結果を取得
    public async Task&lt;string&gt; GetFirstResponseAsync(string[] urls)
    {
        var tasks = urls.Select(async url =&gt;
        {
            using var client = new HttpClient();
            client.Timeout = TimeSpan.FromSeconds(5);
            return await client.GetStringAsync(url);
        }).ToArray();

        try
        {
            // 最初に完了したタスクを待機
            var completedTask = await Task.WhenAny(tasks);

            // 完了したタスクの結果を取得（例外がある場合は再スロー）
            return await completedTask;
        }
        finally
        {
            // 残りのタスクのキャンセルや後処理
            foreach (var task in tasks.Where(t =&gt; !t.IsCompleted))
            {
                // 実際の実装では適切なキャンセル処理を実装
                _ = task.ContinueWith(t =&gt; t.Exception?.Handle(ex =&gt; true), 
                    TaskContinuationOptions.OnlyOnFaulted);
            }
        }
    }

    // 制限付き並行処理
    public async Task&lt;List&lt;TResult&gt;&gt; ProcessConcurrentlyAsync&lt;TInput, TResult&gt;(
        IEnumerable&lt;TInput&gt; items,
        Func&lt;TInput, Task&lt;TResult&gt;&gt; processor,
        int maxConcurrency = 5)
    {
        using var semaphore = new SemaphoreSlim(maxConcurrency, maxConcurrency);
        var results = new ConcurrentBag&lt;TResult&gt;();

        var tasks = items.Select(async item =&gt;
        {
            await semaphore.WaitAsync();
            try
            {
                var result = await processor(item);
                results.Add(result);
            }
            finally
            {
                semaphore.Release();
            }
        });

        await Task.WhenAll(tasks);
        return results.ToList();
    }
}

    public class AdvancedWebCrawler
{
    private readonly HttpClient _httpClient;
    private readonly SemaphoreSlim _rateLimiter;
    private readonly int _maxRetries;
    private readonly TimeSpan _retryDelay;

    public AdvancedWebCrawler(int maxConcurrency = 5, int maxRetries = 3, TimeSpan? retryDelay = null)
    {
        _httpClient = new HttpClient { Timeout = TimeSpan.FromSeconds(30) };
        _rateLimiter = new SemaphoreSlim(maxConcurrency, maxConcurrency);
        _maxRetries = maxRetries;
        _retryDelay = retryDelay ?? TimeSpan.FromSeconds(1);
    }

    public async Task&lt;CrawlResult&gt; CrawlUrlsAsync(
        IEnumerable&lt;string&gt; urls,
        IProgress&lt;CrawlProgress&gt; progress = null,
        CancellationToken cancellationToken = default)
    {
        var urlList = urls.ToList();
        var results = new ConcurrentBag&lt;PageResult&gt;();
        var completed = 0;

        try
        {
            var tasks = urlList.Select(async url =&gt;
            {
                var result = await CrawlSingleUrlAsync(url, cancellationToken);
                results.Add(result);

                var currentCompleted = Interlocked.Increment(ref completed);
                progress?.Report(new CrawlProgress 
                { 
                    CompletedUrls = currentCompleted, 
                    TotalUrls = urlList.Count,
                    CurrentUrl = url,
                    IsSuccess = result.IsSuccess
                });

                return result;
            });

            await Task.WhenAll(tasks);

            return new CrawlResult
            {
                Results = results.ToList(),
                SuccessCount = results.Count(r =&gt; r.IsSuccess),
                FailureCount = results.Count(r =&gt; !r.IsSuccess),
                TotalProcessed = results.Count
            };
        }
        catch (OperationCanceledException)
        {
            return new CrawlResult
            {
                Results = results.ToList(),
                SuccessCount = results.Count(r =&gt; r.IsSuccess),
                FailureCount = results.Count(r =&gt; !r.IsSuccess),
                TotalProcessed = results.Count,
                WasCancelled = true
            };
        }
    }

    private async Task&lt;PageResult&gt; CrawlSingleUrlAsync(string url, CancellationToken cancellationToken)
    {
        await _rateLimiter.WaitAsync(cancellationToken);

        try
        {
            return await RetryWithExponentialBackoffAsync(async () =&gt;
            {
                var response = await _httpClient.GetAsync(url, cancellationToken);
                response.EnsureSuccessStatusCode();

                var content = await response.Content.ReadAsStringAsync(cancellationToken);

                return new PageResult
                {
                    Url = url,
                    IsSuccess = true,
                    Content = content,
                    StatusCode = response.StatusCode,
                    ResponseTime = DateTime.UtcNow,
                    ContentLength = content.Length
                };
            }, cancellationToken);
        }
        catch (Exception ex)
        {
            return new PageResult
            {
                Url = url,
                IsSuccess = false,
                ErrorMessage = ex.Message,
                ResponseTime = DateTime.UtcNow
            };
        }
        finally
        {
            _rateLimiter.Release();
        }
    }

    private async Task&lt;T&gt; RetryWithExponentialBackoffAsync&lt;T&gt;(
        Func&lt;Task&lt;T&gt;&gt; operation,
        CancellationToken cancellationToken)
    {
        Exception lastException = null;

        for (int attempt = 0; attempt &lt; _maxRetries; attempt++)
        {
            try
            {
                return await operation();
            }
            catch (Exception ex) when (!(ex is OperationCanceledException))
            {
                lastException = ex;

                if (attempt &lt; _maxRetries - 1)
                {
                    var delay = TimeSpan.FromMilliseconds(_retryDelay.TotalMilliseconds * Math.Pow(2, attempt));
                    await Task.Delay(delay, cancellationToken);
                }
            }
        }

        throw lastException ?? new Exception(&quot;Unknown error during retry operation&quot;);
    }

    public void Dispose()
    {
        _httpClient?.Dispose();
        _rateLimiter?.Dispose();
    }
}

    public class RealTimeDataProcessor&lt;T&gt; where T : class
{
    private readonly Channel&lt;T&gt; _inputChannel;
    private readonly ChannelWriter&lt;T&gt; _writer;
    private readonly ChannelReader&lt;T&gt; _reader;
    private readonly Func&lt;T, Task&lt;ProcessedData&lt;T&gt;&gt;&gt; _processor;
    private readonly SemaphoreSlim _processingSlots;
    private readonly CancellationTokenSource _shutdownCts;
    private readonly List&lt;Task&gt; _processingTasks;
    private volatile bool _isRunning;

    public event Func&lt;ProcessedData&lt;T&gt;, Task&gt; OnDataProcessed;
    public event Func&lt;Exception, Task&gt; OnProcessingError;

    public RealTimeDataProcessor(
        Func&lt;T, Task&lt;ProcessedData&lt;T&gt;&gt;&gt; processor,
        int bufferSize = 10000,
        int maxConcurrency = Environment.ProcessorCount)
    {
        _processor = processor;
        _processingSlots = new SemaphoreSlim(maxConcurrency, maxConcurrency);
        _shutdownCts = new CancellationTokenSource();
        _processingTasks = new List&lt;Task&gt;();

        var options = new BoundedChannelOptions(bufferSize)
        {
            FullMode = BoundedChannelFullMode.Wait,
            SingleReader = false,
            SingleWriter = false
        };

        _inputChannel = Channel.CreateBounded&lt;T&gt;(options);
        _writer = _inputChannel.Writer;
        _reader = _inputChannel.Reader;
    }

    public async Task&lt;bool&gt; EnqueueAsync(T data, CancellationToken cancellationToken = default)
    {
        try
        {
            if (!_isRunning) return false;

            return await _writer.WriteAsync(data, cancellationToken);
        }
        catch (OperationCanceledException)
        {
            return false;
        }
    }

    public void StartProcessing(int workerCount = 1)
    {
        if (_isRunning) return;

        _isRunning = true;

        for (int i = 0; i &lt; workerCount; i++)
        {
            var workerTask = ProcessDataWorkerAsync(i, _shutdownCts.Token);
            _processingTasks.Add(workerTask);
        }
    }

    private async Task ProcessDataWorkerAsync(int workerId, CancellationToken cancellationToken)
    {
        try
        {
            await foreach (var data in _reader.ReadAllAsync(cancellationToken))
            {
                await _processingSlots.WaitAsync(cancellationToken);

                try
                {
                    var processedData = await ProcessWithRetryAsync(data, cancellationToken);

                    if (OnDataProcessed != null)
                    {
                        await OnDataProcessed(processedData);
                    }
                }
                catch (Exception ex)
                {
                    if (OnProcessingError != null)
                    {
                        await OnProcessingError(ex);
                    }
                }
                finally
                {
                    _processingSlots.Release();
                }
            }
        }
        catch (OperationCanceledException)
        {
            // 期待される例外：シャットダウン時
        }
        catch (Exception ex)
        {
            // 予期しない例外
            if (OnProcessingError != null)
            {
                await OnProcessingError(ex);
            }
        }
    }

    private async Task&lt;ProcessedData&lt;T&gt;&gt; ProcessWithRetryAsync(T data, CancellationToken cancellationToken)
    {
        const int maxRetries = 3;
        Exception lastException = null;

        for (int attempt = 0; attempt &lt; maxRetries; attempt++)
        {
            try
            {
                return await _processor(data);
            }
            catch (Exception ex) when (!(ex is OperationCanceledException))
            {
                lastException = ex;

                if (attempt &lt; maxRetries - 1)
                {
                    var delay = TimeSpan.FromMilliseconds(100 * Math.Pow(2, attempt));
                    await Task.Delay(delay, cancellationToken);
                }
            }
        }

        // 全ての試行が失敗した場合は失敗としてマーク
        return new ProcessedData&lt;T&gt;
        {
            OriginalData = data,
            IsSuccess = false,
            ErrorMessage = lastException?.Message,
            ProcessedAt = DateTime.UtcNow
        };
    }

    public async Task StopAsync(TimeSpan timeout = default)
    {
        if (!_isRunning) return;

        if (timeout == default) timeout = TimeSpan.FromSeconds(30);

        // 新しいデータの受け入れを停止
        _writer.Complete();
        _isRunning = false;

        // シャットダウンをシグナル
        _shutdownCts.Cancel();

        try
        {
            // すべてのワーカータスクの完了を待機（タイムアウト付き）
            await Task.WhenAll(_processingTasks).WaitAsync(timeout);
        }
        catch (TimeoutException)
        {
            // タイムアウト時は強制終了
            Console.WriteLine(&quot;処理タスクのタイムアウト - 強制終了します&quot;);
        }
    }

    public void Dispose()
    {
        StopAsync().Wait(TimeSpan.FromSeconds(10));
        _processingSlots?.Dispose();
        _shutdownCts?.Dispose();
    }
}

    public class HighAvailabilityFileSyncSystem
{
    private readonly Dictionary&lt;string, FileSystemWatcher&gt; _watchers;
    private readonly ConcurrentQueue&lt;SyncOperation&gt; _syncQueue;
    private readonly SemaphoreSlim _syncSemaphore;
    private readonly CancellationTokenSource _shutdownCts;
    private readonly ILogger&lt;HighAvailabilityFileSyncSystem&gt; _logger;
    private volatile bool _isRunning;

    public event Func&lt;SyncProgressEventArgs, Task&gt; OnSyncProgress;
    public event Func&lt;SyncConflictEventArgs, Task&lt;ConflictResolution&gt;&gt; OnConflictDetected;

    public HighAvailabilityFileSyncSystem(ILogger&lt;HighAvailabilityFileSyncSystem&gt; logger = null)
    {
        _watchers = new Dictionary&lt;string, FileSystemWatcher&gt;();
        _syncQueue = new ConcurrentQueue&lt;SyncOperation&gt;();
        _syncSemaphore = new SemaphoreSlim(3, 3); // 最大3つの同時同期
        _shutdownCts = new CancellationTokenSource();
        _logger = logger ?? new ConsoleLogger&lt;HighAvailabilityFileSyncSystem&gt;();
    }

    public async Task&lt;Result&lt;bool&gt;&gt; StartSyncAsync(SyncConfiguration config)
    {
        try
        {
            if (_isRunning) return Result&lt;bool&gt;.Failure(&quot;同期は既に実行中です&quot;);

            _isRunning = true;

            // 監視対象ディレクトリの設定
            foreach (var syncPair in config.SyncPairs)
            {
                SetupDirectoryWatcher(syncPair.SourcePath, syncPair.TargetPath);
            }

            // 同期処理ワーカーを開始
            var workerTasks = Enumerable.Range(0, config.WorkerCount)
                .Select(i =&gt; SyncWorkerAsync(i, _shutdownCts.Token))
                .ToArray();

            // 初期同期を実行
            await PerformInitialSyncAsync(config);

            _logger.LogInformation(&quot;ファイル同期システムが開始されました&quot;);
            return Result&lt;bool&gt;.Success(true);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, &quot;同期システムの開始に失敗しました&quot;);
            return Result&lt;bool&gt;.Failure($&quot;開始に失敗: {ex.Message}&quot;);
        }
    }

    private void SetupDirectoryWatcher(string sourcePath, string targetPath)
    {
        if (!Directory.Exists(sourcePath))
        {
            Directory.CreateDirectory(sourcePath);
        }

        var watcher = new FileSystemWatcher(sourcePath)
        {
            IncludeSubdirectories = true,
            NotifyFilter = NotifyFilters.FileName | NotifyFilters.DirectoryName | 
                          NotifyFilters.Size | NotifyFilters.LastWrite
        };

        watcher.Created += (s, e) =&gt; EnqueueSyncOperation(e.FullPath, targetPath, SyncOperationType.Create);
        watcher.Changed += (s, e) =&gt; EnqueueSyncOperation(e.FullPath, targetPath, SyncOperationType.Update);
        watcher.Deleted += (s, e) =&gt; EnqueueSyncOperation(e.FullPath, targetPath, SyncOperationType.Delete);
        watcher.Renamed += (s, e) =&gt; EnqueueSyncOperation(e.OldFullPath, targetPath, SyncOperationType.Rename, e.FullPath);

        watcher.EnableRaisingEvents = true;
        _watchers[sourcePath] = watcher;

        _logger.LogInformation($&quot;ディレクトリ監視開始: {sourcePath} -&gt; {targetPath}&quot;);
    }

    private void EnqueueSyncOperation(string sourcePath, string targetBasePath, SyncOperationType operationType, string newPath = null)
    {
        var operation = new SyncOperation
        {
            Id = Guid.NewGuid(),
            SourcePath = sourcePath,
            TargetBasePath = targetBasePath,
            OperationType = operationType,
            NewPath = newPath,
            QueuedAt = DateTime.UtcNow,
            RetryCount = 0
        };

        _syncQueue.Enqueue(operation);
        _logger.LogDebug($&quot;同期操作をキューに追加: {operationType} - {sourcePath}&quot;);
    }

    private async Task SyncWorkerAsync(int workerId, CancellationToken cancellationToken)
    {
        _logger.LogInformation($&quot;同期ワーカー {workerId} を開始&quot;);

        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                if (_syncQueue.TryDequeue(out var operation))
                {
                    await _syncSemaphore.WaitAsync(cancellationToken);

                    try
                    {
                        await ProcessSyncOperationAsync(operation, workerId, cancellationToken);
                    }
                    finally
                    {
                        _syncSemaphore.Release();
                    }
                }
                else
                {
                    // キューが空の場合は少し待機
                    await Task.Delay(100, cancellationToken);
                }
            }
            catch (OperationCanceledException)
            {
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $&quot;ワーカー {workerId} でエラーが発生&quot;);
            }
        }

        _logger.LogInformation($&quot;同期ワーカー {workerId} を停止&quot;);
    }

    private async Task ProcessSyncOperationAsync(SyncOperation operation, int workerId, CancellationToken cancellationToken)
    {
        const int maxRetries = 3;

        try
        {
            _logger.LogDebug($&quot;ワーカー {workerId}: 処理開始 - {operation.OperationType} {operation.SourcePath}&quot;);

            var result = await ExecuteSyncOperationWithRetryAsync(operation, maxRetries, cancellationToken);

            if (result.IsSuccess)
            {
                await ReportProgressAsync(new SyncProgressEventArgs
                {
                    OperationId = operation.Id,
                    OperationType = operation.OperationType,
                    SourcePath = operation.SourcePath,
                    Status = SyncStatus.Completed,
                    WorkerId = workerId
                });

                _logger.LogInformation($&quot;同期完了: {operation.OperationType} - {operation.SourcePath}&quot;);
            }
            else
            {
                await ReportProgressAsync(new SyncProgressEventArgs
                {
                    OperationId = operation.Id,
                    OperationType = operation.OperationType,
                    SourcePath = operation.SourcePath,
                    Status = SyncStatus.Failed,
                    ErrorMessage = result.ErrorMessage,
                    WorkerId = workerId
                });

                _logger.LogError($&quot;同期失敗: {operation.OperationType} - {operation.SourcePath}: {result.ErrorMessage}&quot;);
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $&quot;同期処理中に予期しないエラー: {operation.SourcePath}&quot;);
        }
    }

    private async Task&lt;Result&lt;bool&gt;&gt; ExecuteSyncOperationWithRetryAsync(
        SyncOperation operation, int maxRetries, CancellationToken cancellationToken)
    {
        for (int attempt = 0; attempt &lt; maxRetries; attempt++)
        {
            try
            {
                var result = await ExecuteSyncOperationAsync(operation, cancellationToken);
                if (result.IsSuccess) return result;

                if (attempt &lt; maxRetries - 1)
                {
                    var delay = TimeSpan.FromSeconds(Math.Pow(2, attempt));
                    _logger.LogWarning($&quot;同期リトライ {attempt + 1}/{maxRetries}: {operation.SourcePath} (遅延: {delay.TotalSeconds}秒)&quot;);
                    await Task.Delay(delay, cancellationToken);
                }
                else
                {
                    return result;
                }
            }
            catch (Exception ex)
            {
                if (attempt == maxRetries - 1)
                {
                    return Result&lt;bool&gt;.Failure($&quot;最大リトライ回数到達: {ex.Message}&quot;);
                }
            }
        }

        return Result&lt;bool&gt;.Failure(&quot;未知のエラー&quot;);
    }

    private async Task&lt;Result&lt;bool&gt;&gt; ExecuteSyncOperationAsync(SyncOperation operation, CancellationToken cancellationToken)
    {
        try
        {
            var targetPath = CalculateTargetPath(operation.SourcePath, operation.TargetBasePath);

            switch (operation.OperationType)
            {
                case SyncOperationType.Create:
                case SyncOperationType.Update:
                    return await SyncFileAsync(operation.SourcePath, targetPath, cancellationToken);

                case SyncOperationType.Delete:
                    return await DeleteFileAsync(targetPath, cancellationToken);

                case SyncOperationType.Rename:
                    var newTargetPath = CalculateTargetPath(operation.NewPath, operation.TargetBasePath);
                    return await RenameFileAsync(targetPath, newTargetPath, cancellationToken);

                default:
                    return Result&lt;bool&gt;.Failure($&quot;未対応の操作タイプ: {operation.OperationType}&quot;);
            }
        }
        catch (Exception ex)
        {
            return Result&lt;bool&gt;.Failure(ex.Message);
        }
    }

    private async Task&lt;Result&lt;bool&gt;&gt; SyncFileAsync(string sourcePath, string targetPath, CancellationToken cancellationToken)
    {
        try
        {
            if (!File.Exists(sourcePath))
            {
                return Result&lt;bool&gt;.Success(true); // ファイルが削除された場合はスキップ
            }

            // ディレクトリの作成
            var targetDirectory = Path.GetDirectoryName(targetPath);
            if (!Directory.Exists(targetDirectory))
            {
                Directory.CreateDirectory(targetDirectory);
            }

            // 競合チェック
            if (File.Exists(targetPath))
            {
                var conflict = await DetectConflictAsync(sourcePath, targetPath);
                if (conflict != ConflictType.None)
                {
                    var resolution = await ResolveConflictAsync(sourcePath, targetPath, conflict);
                    if (resolution == ConflictResolution.Skip)
                    {
                        return Result&lt;bool&gt;.Success(true);
                    }
                }
            }

            // ファイルコピー
            using var sourceStream = new FileStream(sourcePath, FileMode.Open, FileAccess.Read, FileShare.Read);
            using var targetStream = new FileStream(targetPath, FileMode.Create, FileAccess.Write);

            await sourceStream.CopyToAsync(targetStream, cancellationToken);

            // メタデータの同期
            var sourceInfo = new FileInfo(sourcePath);
            var targetInfo = new FileInfo(targetPath);
            targetInfo.CreationTime = sourceInfo.CreationTime;
            targetInfo.LastWriteTime = sourceInfo.LastWriteTime;

            return Result&lt;bool&gt;.Success(true);
        }
        catch (Exception ex)
        {
            return Result&lt;bool&gt;.Failure($&quot;ファイル同期エラー: {ex.Message}&quot;);
        }
    }

    private async Task&lt;ConflictType&gt; DetectConflictAsync(string sourcePath, string targetPath)
    {
        try
        {
            var sourceInfo = new FileInfo(sourcePath);
            var targetInfo = new FileInfo(targetPath);

            // サイズとタイムスタンプで競合を判定
            if (sourceInfo.Length != targetInfo.Length)
            {
                return ConflictType.SizeMismatch;
            }

            if (Math.Abs((sourceInfo.LastWriteTime - targetInfo.LastWriteTime).TotalSeconds) &gt; 1)
            {
                return ConflictType.TimestampMismatch;
            }

            return ConflictType.None;
        }
        catch
        {
            return ConflictType.Unknown;
        }
    }

    private async Task&lt;ConflictResolution&gt; ResolveConflictAsync(string sourcePath, string targetPath, ConflictType conflictType)
    {
        if (OnConflictDetected != null)
        {
            var conflictArgs = new SyncConflictEventArgs
            {
                SourcePath = sourcePath,
                TargetPath = targetPath,
                ConflictType = conflictType,
                SourceLastModified = File.GetLastWriteTime(sourcePath),
                TargetLastModified = File.GetLastWriteTime(targetPath)
            };

            return await OnConflictDetected(conflictArgs);
        }

        // デフォルト：より新しいファイルを優先
        var sourceTime = File.GetLastWriteTime(sourcePath);
        var targetTime = File.GetLastWriteTime(targetPath);

        return sourceTime &gt; targetTime ? ConflictResolution.OverwriteTarget : ConflictResolution.Skip;
    }

    private async Task&lt;Result&lt;bool&gt;&gt; DeleteFileAsync(string targetPath, CancellationToken cancellationToken)
    {
        try
        {
            if (File.Exists(targetPath))
            {
                File.Delete(targetPath);
            }
            else if (Directory.Exists(targetPath))
            {
                Directory.Delete(targetPath, true);
            }

            return Result&lt;bool&gt;.Success(true);
        }
        catch (Exception ex)
        {
            return Result&lt;bool&gt;.Failure($&quot;削除エラー: {ex.Message}&quot;);
        }
    }

    private async Task&lt;Result&lt;bool&gt;&gt; RenameFileAsync(string oldPath, string newPath, CancellationToken cancellationToken)
    {
        try
        {
            if (File.Exists(oldPath))
            {
                var newDirectory = Path.GetDirectoryName(newPath);
                if (!Directory.Exists(newDirectory))
                {
                    Directory.CreateDirectory(newDirectory);
                }

                File.Move(oldPath, newPath);
            }

            return Result&lt;bool&gt;.Success(true);
        }
        catch (Exception ex)
        {
            return Result&lt;bool&gt;.Failure($&quot;リネームエラー: {ex.Message}&quot;);
        }
    }

    private string CalculateTargetPath(string sourcePath, string targetBasePath)
    {
        // ソースパスから対応するターゲットパスを計算
        var sourceDirectory = Path.GetDirectoryName(sourcePath);
        var fileName = Path.GetFileName(sourcePath);

        // 相対パスを計算してターゲットベースパスと結合
        // 簡略化された実装
        return Path.Combine(targetBasePath, fileName);
    }

    private async Task PerformInitialSyncAsync(SyncConfiguration config)
    {
        foreach (var syncPair in config.SyncPairs)
        {
            if (Directory.Exists(syncPair.SourcePath))
            {
                var files = Directory.GetFiles(syncPair.SourcePath, &quot;*&quot;, SearchOption.AllDirectories);
                foreach (var file in files)
                {
                    EnqueueSyncOperation(file, syncPair.TargetPath, SyncOperationType.Create);
                }
            }
        }
    }

    private async Task ReportProgressAsync(SyncProgressEventArgs args)
    {
        if (OnSyncProgress != null)
        {
            await OnSyncProgress(args);
        }
    }

    public async Task StopAsync(TimeSpan timeout = default)
    {
        if (!_isRunning) return;

        _logger.LogInformation(&quot;ファイル同期システムを停止中...&quot;);

        _isRunning = false;
        _shutdownCts.Cancel();

        // ファイルシステム監視を停止
        foreach (var watcher in _watchers.Values)
        {
            watcher.EnableRaisingEvents = false;
            watcher.Dispose();
        }
        _watchers.Clear();

        _logger.LogInformation(&quot;ファイル同期システムが停止されました&quot;);
    }

    public void Dispose()
    {
        StopAsync().Wait(TimeSpan.FromSeconds(10));
        _syncSemaphore?.Dispose();
        _shutdownCts?.Dispose();
    }
}

    public class FileSyncExample
{
    public async Task RunFileSyncSystemAsync()
    {
        var syncSystem = new HighAvailabilityFileSyncSystem();

        // 進捗監視
        syncSystem.OnSyncProgress += async args =&gt;
        {
            Console.WriteLine($&quot;同期進捗: {args.Status} - {args.OperationType} - {Path.GetFileName(args.SourcePath)}&quot;);
        };

        // 競合解決
        syncSystem.OnConflictDetected += async args =&gt;
        {
            Console.WriteLine($&quot;競合検出: {args.ConflictType} - {args.SourcePath}&quot;);
            Console.WriteLine($&quot;  ソース更新: {args.SourceLastModified}&quot;);
            Console.WriteLine($&quot;  ターゲット更新: {args.TargetLastModified}&quot;);

            // より新しいファイルを優先
            return args.SourceLastModified &gt; args.TargetLastModified 
                ? ConflictResolution.OverwriteTarget 
                : ConflictResolution.Skip;
        };

        var config = new SyncConfiguration
        {
            SyncPairs = new List&lt;SyncPair&gt;
            {
                new SyncPair { SourcePath = @&quot;C:\Source&quot;, TargetPath = @&quot;C:\Target&quot; }
            },
            WorkerCount = 2
        };

        var result = await syncSystem.StartSyncAsync(config);

        if (result.IsSuccess)
        {
            Console.WriteLine(&quot;ファイル同期システムが開始されました&quot;);
            Console.WriteLine(&quot;Enterキーで停止します...&quot;);
            Console.ReadLine();

            await syncSystem.StopAsync();
        }
        else
        {
            Console.WriteLine($&quot;同期システム開始エラー: {result.ErrorMessage}&quot;);
        }

        syncSystem.Dispose();
    }
}
