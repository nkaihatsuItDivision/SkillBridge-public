    はじめに - エンタープライズアプリケーションとは

    エンタープライズアプリケーションは、大規模な企業や組織で使用される業務用システムです。以下の特徴を持ちます：

    エンタープライズアプリケーションの特徴

    • 高い可用性: 24時間365日の稼働が求められる
    • スケーラビリティ: 大量のユーザーと取引に対応
    • セキュリティ: 重要なビジネスデータの保護
    • 保守性: 長期間にわたる機能追加と改修
    • 統合性: 他のシステムとの連携

    今回学習する技術要素の統合

    本プロジェクトでは、これまでの8レッスンで学んだ全ての技法を統合します：

    1. 非同期プログラミング - 全てのI/O操作での効率的な非同期処理
    2. LINQ高度活用 - 複雑なデータクエリと分析処理
    3. デザインパターン - Repository、Factory、Observer、Strategy、Commandパターン
    4. 例外処理 - 包括的なエラーハンドリングとカスタム例外
    5. メモリ管理 - キャッシュ、オブジェクトプーリング、パフォーマンス最適化
    6. リフレクション - 動的バリデーション、設定システム、プラグインアーキテクチャ
    7. 並行プログラミング - 高性能な非同期通知・バックグラウンド処理システム
    8. 統合アーキテクチャ - レイヤードアーキテクチャとマイクロサービス指向設計

    • 多層アーキテクチャ: ドメイン層、アプリケーション層、インフラストラクチャ層、プレゼンテーション層

public abstract class BaseEntity : IEquatable&lt;BaseEntity&gt;
{
    public Guid Id { get; protected set; } = Guid.NewGuid();
    public DateTime CreatedAt { get; protected set; } = DateTime.UtcNow;
    public DateTime UpdatedAt { get; protected set; } = DateTime.UtcNow;
    public string CreatedBy { get; protected set; }
    public string UpdatedBy { get; protected set; }

    // ドメインイベント（Observerパターン統合）
    private readonly List&lt;IDomainEvent&gt; _domainEvents = new();
    public IReadOnlyList&lt;IDomainEvent&gt; DomainEvents =&gt; _domainEvents.AsReadOnly();

    protected void AddDomainEvent(IDomainEvent domainEvent)
    {
        _domainEvents.Add(domainEvent);
    }

    public void ClearDomainEvents()
    {
        _domainEvents.Clear();
    }

    protected void UpdateTimestamp(string updatedBy = null)
    {
        UpdatedAt = DateTime.UtcNow;
        UpdatedBy = updatedBy;
    }

    protected void SetCreatedBy(string createdBy)
    {
        CreatedBy = createdBy;
    }

    // エンティティ等価性の実装
    public bool Equals(BaseEntity other)
    {
        if (ReferenceEquals(null, other)) return false;
        if (ReferenceEquals(this, other)) return true;
        return Id.Equals(other.Id);
    }

    public override bool Equals(object obj) =&gt; Equals(obj as BaseEntity);
    public override int GetHashCode() =&gt; Id.GetHashCode();
}

    public PasswordHash PasswordHash { get; private set; }
    public UserRole Role { get; private set; }
    public UserStatus Status { get; private set; } = UserStatus.Active;

    private readonly List&lt;TaskItem&gt; _assignedTasks = new();

    protected User() { }

    // ファクトリーメソッドパターン
    public static User Create(string email, string name, string password, UserRole role = UserRole.Member)
    {
        var user = new User();
        user.SetEmail(email);
        user.SetName(name);
        user.SetPassword(password);
        user.Role = role;
        user.Preferences = UserPreferences.CreateDefault();

        // ドメインイベント発行
        user.AddDomainEvent(new UserCreatedEvent(user.Id, user.Email.Value, user.Name.FullName));

    }

    public void SetEmail(string email)
    {
        var newEmail = Email.Create(email);
        if (Email?.Value != newEmail.Value)
        {
            Email = newEmail;
            UpdateTimestamp();
            AddDomainEvent(new UserEmailChangedEvent(Id, newEmail.Value));
        }
    }

    public void SetName(string firstName, string lastName)
    {
        Name = Name.Create(firstName, lastName);
        UpdateTimestamp();
    }

    public void SetPassword(string password)
    {
        PasswordHash = PasswordHash.Create(password);
        UpdateTimestamp();
        AddDomainEvent(new UserPasswordChangedEvent(Id));
    }

    public void ChangeRole(UserRole newRole, string changedBy)
    {
        if (Role != newRole)
        {
            var oldRole = Role;
            Role = newRole;
            UpdateTimestamp(changedBy);
            AddDomainEvent(new UserRoleChangedEvent(Id, oldRole, newRole));
        }
    }

    public void Deactivate(string reason = null)
    {
        Status = UserStatus.Inactive;
        UpdateTimestamp();
        AddDomainEvent(new UserDeactivatedEvent(Id, reason));
    }

    {
        Status = UserStatus.Active;
        UpdateTimestamp();
        AddDomainEvent(new UserActivatedEvent(Id));

    public bool CanManageProject(Project project)
    {

    }

    // Domain/ValueObjects/Email.cs
public class Email : ValueObject
{
    public string Value { get; private set; }

    private Email(string value)
    {
        Value = value;
    }

    public static Email Create(string email)
    {
        if (string.IsNullOrWhiteSpace(email))
            throw new DomainException(&quot;メールアドレスは必須です&quot;, &quot;EMAIL_REQUIRED&quot;);

        email = email.Trim().ToLowerInvariant();

        if (!IsValidEmail(email))
            throw new DomainException(&quot;無効なメールアドレス形式です&quot;, &quot;EMAIL_INVALID_FORMAT&quot;);

        return new Email(email);
    }

    private static bool IsValidEmail(string email)
    {
        try
        {
            var addr = new System.Net.Mail.MailAddress(email);
            return addr.Address == email;
        }
        catch
        {
            return false;
        }
    }

    protected override IEnumerable&lt;object&gt; GetEqualityComponents()
    {
        yield return Value;
    }
}

    // Domain/Entities/TaskItem.cs
public class TaskItem : BaseEntity, IAggregateRoot
{
    public Title Title { get; private set; }
    public Description Description { get; private set; }
    public TaskStatus Status { get; private set; } = TaskStatus.Todo;
    public TaskPriority Priority { get; private set; } = TaskPriority.Medium;
    public DateTime? DueDate { get; private set; }
    public Guid AssignedUserId { get; private set; }
    public Guid ProjectId { get; private set; }
    public EstimatedHours EstimatedHours { get; private set; }
    public ActualHours ActualHours { get; private set; }

    // ナビゲーションプロパティ
    public User AssignedUser { get; private set; }
    public Project Project { get; private set; }

    private readonly List&lt;TaskComment&gt; _comments = new();
    public IReadOnlyList&lt;TaskComment&gt; Comments =&gt; _comments.AsReadOnly();

    private readonly List&lt;TaskAttachment&gt; _attachments = new();
    public IReadOnlyList&lt;TaskAttachment&gt; Attachments =&gt; _attachments.AsReadOnly();

    protected TaskItem() { }

    public static TaskItem Create(string title, string description, Guid assignedUserId, 
        Guid projectId, DateTime? dueDate = null, TaskPriority priority = TaskPriority.Medium)
    {
        var task = new TaskItem();
        task.SetTitle(title);
        task.SetDescription(description);
        task.AssignedUserId = assignedUserId;
        task.ProjectId = projectId;
        task.DueDate = dueDate;
        task.Priority = priority;
        task.EstimatedHours = EstimatedHours.Create(0);
        task.ActualHours = ActualHours.Create(0);

        task.AddDomainEvent(new TaskCreatedEvent(task.Id, task.Title.Value, assignedUserId, projectId));

        return task;
    }

    public void SetTitle(string title)
    {
        var newTitle = Title.Create(title);
        if (Title?.Value != newTitle.Value)
        {
            Title = newTitle;
            UpdateTimestamp();
            AddDomainEvent(new TaskTitleChangedEvent(Id, newTitle.Value));
        }
    }

    public void SetDescription(string description)
    {
        Description = Description.Create(description);
        UpdateTimestamp();
    }

    public void ChangeStatus(TaskStatus newStatus, string changedBy = null)
    {
        if (Status == newStatus) return;

        ValidateStatusTransition(Status, newStatus);

        var oldStatus = Status;
        Status = newStatus;
        UpdateTimestamp(changedBy);

        // ステータス変更時の特別な処理
        if (newStatus == TaskStatus.InProgress)
        {
            AddDomainEvent(new TaskStartedEvent(Id, AssignedUserId));
        }
        else if (newStatus == TaskStatus.Completed)
        {
            AddDomainEvent(new TaskCompletedEvent(Id, AssignedUserId, ActualHours.Value));
        }

        AddDomainEvent(new TaskStatusChangedEvent(Id, oldStatus, newStatus));
    }

    public void SetPriority(TaskPriority priority)
    {
        if (Priority != priority)
        {
            var oldPriority = Priority;
            Priority = priority;
            UpdateTimestamp();
            AddDomainEvent(new TaskPriorityChangedEvent(Id, oldPriority, priority));
        }
    }

    public void SetDueDate(DateTime? dueDate)
    {
        DueDate = dueDate;
        UpdateTimestamp();
        AddDomainEvent(new TaskDueDateChangedEvent(Id, dueDate));
    }

    public void AddComment(string content, Guid authorId)
    {
        var comment = TaskComment.Create(Id, content, authorId);
        _comments.Add(comment);
        UpdateTimestamp();
        AddDomainEvent(new TaskCommentAddedEvent(Id, comment.Id, authorId));
    }

    public void LogTimeSpent(decimal hours, string description = null)
    {
        ActualHours = ActualHours.Add(hours);
        UpdateTimestamp();
        AddDomainEvent(new TaskTimeLoggedEvent(Id, hours, description));
    }

    // ビジネスルールの実装
    public bool IsOverdue =&gt; DueDate.HasValue &amp;&amp; DueDate.Value &lt; DateTime.UtcNow &amp;&amp; Status != TaskStatus.Completed;

    public bool CanChangeStatus(TaskStatus newStatus)
    {
        return IsValidStatusTransition(Status, newStatus);
    }

    private static void ValidateStatusTransition(TaskStatus currentStatus, TaskStatus newStatus)
    {
        if (!IsValidStatusTransition(currentStatus, newStatus))
        {
            throw new DomainException(
                $&quot;無効なステータス遷移: {currentStatus} → {newStatus}&quot;, 
                &quot;TASK_INVALID_STATUS_TRANSITION&quot;);
        }
    }

    private static bool IsValidStatusTransition(TaskStatus current, TaskStatus target)
    {
        return current switch
        {
            TaskStatus.Todo =&gt; target is TaskStatus.InProgress or TaskStatus.Cancelled,
            TaskStatus.InProgress =&gt; target is TaskStatus.Completed or TaskStatus.Todo or TaskStatus.Cancelled,
            TaskStatus.Completed =&gt; target is TaskStatus.InProgress,
            TaskStatus.Cancelled =&gt; target is TaskStatus.Todo,
            _ =&gt; false
        };
    }

                await CacheInMemoryAsync(memoryCacheKey, distributedCachedTask);

            .Include(t =&gt; t.AssignedUser)
            .Include(t =&gt; t.Project)
            .Include(t =&gt; t.Comments.OrderByDescending(c =&gt; c.CreatedAt))

            .FirstOrDefaultAsync(t =&gt; t.Id == id, cancellationToken);

    public async Task&lt;IEnumerable&lt;TaskItem&gt;&gt; GetTasksBatchAsync(IEnumerable&lt;Guid&gt; taskIds, CancellationToken cancellationToken = default)
    {
        var taskIdList = taskIds.ToList();
        const int batchSize = 100;

        // 大量のIDを分割して並列処理
        var batches = taskIdList
            .Select((id, index) =&gt; new { id, index })
            .GroupBy(x =&gt; x.index / batchSize)
            .Select(g =&gt; g.Select(x =&gt; x.id).ToList());

        var tasks = new List&lt;TaskItem&gt;();

        await Parallel.ForEachAsync(batches, 
            new ParallelOptions { CancellationToken = cancellationToken, MaxDegreeOfParallelism = 4 },
            async (batch, ct) =&gt;
            {
                var batchTasks = await _context.Tasks
                    .Where(t =&gt; batch.Contains(t.Id))
                    .Include(t =&gt; t.AssignedUser)
                    .Include(t =&gt; t.Project)
                    .ToListAsync(ct);

                lock (tasks)
                {
                    tasks.AddRange(batchTasks);
                }
            });

        return tasks;
    }

    private async Task CacheTaskAsync(string memoryCacheKey, string distributedCacheKey, TaskItem task)
    {
        await _cacheSemaphore.WaitAsync();
        try
        {
            // メモリキャッシュ
            _cache.Set(memoryCacheKey, task, CacheExpiry);

            // 分散キャッシュ
            var json = JsonSerializer.Serialize(task);
            await _distributedCache.SetStringAsync(distributedCacheKey, json, new DistributedCacheEntryOptions
            {
                AbsoluteExpirationRelativeToNow = DistributedCacheExpiry
            });
        }
        finally
        {
            _cacheSemaphore.Release();
        }
    }

    private async Task CacheInMemoryAsync(string cacheKey, TaskItem task)
    {
        await _cacheSemaphore.WaitAsync();
        try
        {
            _cache.Set(cacheKey, task, CacheExpiry);
        }
        finally
        {
            _cacheSemaphore.Release();
        }
    }

// Infrastructure/UnitOfWork/UnitOfWork.cs
public class UnitOfWork : IUnitOfWork
{
    private readonly ApplicationDbContext _context;
    private readonly IDomainEventDispatcher _domainEventDispatcher;
    private IDbContextTransaction _currentTransaction;

    public ITaskRepository Tasks { get; }
    public IUserRepository Users { get; }
    public IProjectRepository Projects { get; }

    public UnitOfWork(
        ApplicationDbContext context,
        ITaskRepository taskRepository,
        IUserRepository userRepository,
        IProjectRepository projectRepository,
        IDomainEventDispatcher domainEventDispatcher)
    {
        _context = context;
        _domainEventDispatcher = domainEventDispatcher;
        Tasks = taskRepository;
        Users = userRepository;
        Projects = projectRepository;
    }

    public async Task&lt;int&gt; SaveChangesAsync(CancellationToken cancellationToken = default)
    {
        // ドメインイベントの収集（Observerパターン）
        var domainEvents = _context.ChangeTracker
            .Entries&lt;BaseEntity&gt;()
            .Where(x =&gt; x.Entity.DomainEvents.Any())
            .SelectMany(x =&gt; x.Entity.DomainEvents)
            .ToList();

        try
        {
            var result = await _context.SaveChangesAsync(cancellationToken);

            // ドメインイベントの発行
            foreach (var domainEvent in domainEvents)
            {
                await _domainEventDispatcher.DispatchAsync(domainEvent, cancellationToken);
            }

            // イベント履歴のクリア
            _context.ChangeTracker
                .Entries&lt;BaseEntity&gt;()
                .Where(x =&gt; x.Entity.DomainEvents.Any())
                .ToList()
                .ForEach(entity =&gt; entity.Entity.ClearDomainEvents());

            return result;
        }
        catch (Exception)
        {
            // エラー時はロールバック
            if (_currentTransaction != null)
            {
                await RollbackTransactionAsync(cancellationToken);
            }
            throw;
        }
    }

    public async Task&lt;IDbContextTransaction&gt; BeginTransactionAsync(CancellationToken cancellationToken = default)
    {
        _currentTransaction = await _context.Database.BeginTransactionAsync(cancellationToken);
        return _currentTransaction;
    }

    public async Task CommitTransactionAsync(CancellationToken cancellationToken = default)
    {
        if (_currentTransaction == null)
            throw new InvalidOperationException(&quot;アクティブなトランザクションがありません&quot;);

        try
        {
            await _currentTransaction.CommitAsync(cancellationToken);
        }
        finally
        {
            await _currentTransaction.DisposeAsync();
            _currentTransaction = null;
        }
    }

    public async Task RollbackTransactionAsync(CancellationToken cancellationToken = default)
    {
        if (_currentTransaction == null) return;

        try
        {
            await _currentTransaction.RollbackAsync(cancellationToken);
        }
        finally
        {
            await _currentTransaction.DisposeAsync();
            _currentTransaction = null;
        }
    }

    public void Dispose()
    {
        _currentTransaction?.Dispose();
        _context.Dispose();
    }
}

    // Infrastructure/Middleware/GlobalExceptionMiddleware.cs
public class GlobalExceptionMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger&lt;GlobalExceptionMiddleware&gt; _logger;
    private readonly IWebHostEnvironment _environment;

    public GlobalExceptionMiddleware(RequestDelegate next, ILogger&lt;GlobalExceptionMiddleware&gt; logger, IWebHostEnvironment environment)
    {
        _next = next;
        _logger = logger;
        _environment = environment;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        try
        {
            await _next(context);
        }
        catch (Exception ex)
        {
            await HandleExceptionAsync(context, ex);
        }
    }

    private async Task HandleExceptionAsync(HttpContext context, Exception exception)
    {
        var correlationId = Guid.NewGuid().ToString();

        _logger.LogError(exception, &quot;予期しないエラーが発生しました。CorrelationId: {CorrelationId}&quot;, correlationId);

        var response = exception switch
        {
            DomainException domainEx =&gt; new ApiErrorResponse
            {
                StatusCode = 400,
                Message = domainEx.Message,
                ErrorCode = domainEx.ErrorCode,
                CorrelationId = correlationId,
                Details = _environment.IsDevelopment() ? exception.StackTrace : null
            },

            ValidationException validationEx =&gt; new ApiErrorResponse
            {
                StatusCode = 422,
                Message = &quot;バリデーションエラーが発生しました&quot;,
                ErrorCode = &quot;VALIDATION_ERROR&quot;,
                CorrelationId = correlationId,
                ValidationErrors = validationEx.Errors?.ToDictionary(x =&gt; x.PropertyName, x =&gt; x.ErrorMessage)
            },

            UnauthorizedAccessException =&gt; new ApiErrorResponse
            {
                StatusCode = 401,
                Message = &quot;認証が必要です&quot;,
                ErrorCode = &quot;UNAUTHORIZED&quot;,
                CorrelationId = correlationId
            },

            _ =&gt; new ApiErrorResponse
            {
                StatusCode = 500,
                Message = &quot;内部サーバーエラーが発生しました&quot;,
                ErrorCode = &quot;INTERNAL_ERROR&quot;,
                CorrelationId = correlationId,
                Details = _environment.IsDevelopment() ? exception.StackTrace : null
            }
        };

        context.Response.StatusCode = response.StatusCode;
        context.Response.ContentType = &quot;application/json&quot;;

        var jsonResponse = JsonSerializer.Serialize(response, new JsonSerializerOptions
        {
            PropertyNamingPolicy = JsonNamingPolicy.CamelCase,
            WriteIndented = true
        });

        await context.Response.WriteAsync(jsonResponse);
    }

    // Infrastructure/Monitoring/PerformanceMiddleware.cs
public class PerformanceMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger&lt;PerformanceMiddleware&gt; _logger;
    private readonly DiagnosticSource _diagnosticSource;

    public PerformanceMiddleware(RequestDelegate next, ILogger&lt;PerformanceMiddleware&gt; logger, DiagnosticSource diagnosticSource)
    {
        _next = next;
        _logger = logger;
        _diagnosticSource = diagnosticSource;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        var stopwatch = Stopwatch.StartNew();
        var initialMemory = GC.GetTotalMemory(false);

        try
        {
            await _next(context);
        }
        finally
        {
            stopwatch.Stop();
            var finalMemory = GC.GetTotalMemory(false);
            var memoryDelta = finalMemory - initialMemory;

            var performanceData = new
            {
                Method = context.Request.Method,
                Path = context.Request.Path.Value,
                StatusCode = context.Response.StatusCode,
                ElapsedMilliseconds = stopwatch.ElapsedMilliseconds,
                MemoryUsed = memoryDelta,
                ThreadId = Thread.CurrentThread.ManagedThreadId
            };

            // 診断情報の記録
            if (_diagnosticSource.IsEnabled(&quot;TaskManagement.RequestPerformance&quot;))
            {
                _diagnosticSource.Write(&quot;TaskManagement.RequestPerformance&quot;, performanceData);
            }

            // パフォーマンス警告
            if (stopwatch.ElapsedMilliseconds &gt; 1000)
            {
                _logger.LogWarning(&quot;遅いリクエストを検出: {Method} {Path} - {ElapsedMs}ms, Memory: {MemoryUsed} bytes&quot;,
                    context.Request.Method,
                    context.Request.Path,
                    stopwatch.ElapsedMilliseconds,
                    memoryDelta);
            }
            else
            {
                _logger.LogInformation(&quot;リクエスト完了: {Method} {Path} - {ElapsedMs}ms&quot;,
                    context.Request.Method,
                    context.Request.Path,
                    stopwatch.ElapsedMilliseconds);
            }
        }
    }
}

    6.3 セキュリティとバリデーション

    // Application/Behaviors/ValidationBehavior.cs
public class ValidationBehavior&lt;TRequest, TResponse&gt; : IPipelineBehavior&lt;TRequest, TResponse&gt;
    where TRequest : IRequest&lt;TResponse&gt;
{
    private readonly IEnumerable&lt;IValidator&lt;TRequest&gt;&gt; _validators;
    private readonly ILogger&lt;ValidationBehavior&lt;TRequest, TResponse&gt;&gt; _logger;

    public ValidationBehavior(IEnumerable&lt;IValidator&lt;TRequest&gt;&gt; validators, ILogger&lt;ValidationBehavior&lt;TRequest, TResponse&gt;&gt; logger)
    {
        _validators = validators;
        _logger = logger;
    }

    public async Task&lt;TResponse&gt; Handle(TRequest request, RequestHandlerDelegate&lt;TResponse&gt; next, CancellationToken cancellationToken)
    {
        if (!_validators.Any()) return await next();

        var context = new ValidationContext&lt;TRequest&gt;(request);

        var validationResults = await Task.WhenAll(
            _validators.Select(v =&gt; v.ValidateAsync(context, cancellationToken))
        );

        var failures = validationResults
            .SelectMany(r =&gt; r.Errors)
            .Where(f =&gt; f != null)
            .ToList();

        if (failures.Any())
        {
            _logger.LogWarning(&quot;バリデーションエラー: {RequestType} - {Errors}&quot;,
                typeof(TRequest).Name,
                string.Join(&quot;, &quot;, failures.Select(f =&gt; f.ErrorMessage)));

            throw new ValidationException(failures);
        }

        return await next();
    }
}
