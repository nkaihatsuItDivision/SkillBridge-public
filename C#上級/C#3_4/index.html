    はじめに - 高度な例外処理の重要性

    例外処理は、堅牢なアプリケーションを構築するための最も重要な技術の一つです。基本的なtry-catchだけでは対応できない複雑なエラーシナリオに対処するため、高度な技法が必要となります。

    なぜ高度な例外処理が必要なのか

    • システムの信頼性向上: 予期しないエラーからの適切な回復
    • 運用効率の改善: 詳細なログとモニタリングによる迅速な問題解決
    • ユーザー体験の向上: 分かりやすいエラーメッセージと適切な誘導
    • 開発生産性: 統一されたエラーハンドリングパターンによる効率的な開発

    今回学習する5つの高度な技法： 1. カスタム例外設計 - ビジネスロジックに特化した例外体系 2. Resultパターン - 例外を使わない軽量なエラー処理 3. Retryパターン - 一時的障害からの自動回復 4. グローバル例外ハンドリング - アプリケーション全体の統一的なエラー処理 5. 統合エラーサービス - 包括的なエラー管理システム

    1. カスタム例外設計 - ビジネスロジックに特化した例外体系

    技法の目的

    ビジネスドメイン固有の例外を設計し、エラーの種類と対処法を明確にします。

    実際の使用例

    • バリデーションエラーの詳細管理
    • 認証・認可エラーの分類
    • リソース不足やタイムアウトエラー
    • 外部サービス連携エラー

    基本的なカスタム例外階層

    // ビジネス例外の基底クラス
public abstract class BusinessException : Exception
{
    public string ErrorCode { get; }
    public DateTime Timestamp { get; }
    public Dictionary&lt;string, object&gt; AdditionalData { get; }

    protected BusinessException(string errorCode, string message) 
        : base(message)
    {
        ErrorCode = errorCode;
        Timestamp = DateTime.UtcNow;
        AdditionalData = new Dictionary&lt;string, object&gt;();
    }

    protected BusinessException(string errorCode, string message, Exception innerException) 
        : base(message, innerException)
    {
        ErrorCode = errorCode;
        Timestamp = DateTime.UtcNow;
        AdditionalData = new Dictionary&lt;string, object&gt;();
    }

    // 追加情報の設定
    public void AddData(string key, object value)
    {
        AdditionalData[key] = value;
    }
}

    実用的なカスタム例外の実装

    // バリデーション例外
public class ValidationException : BusinessException
{
    public Dictionary&lt;string, List&lt;string&gt;&gt; ValidationErrors { get; }

    public ValidationException(string message) 
        : base(&quot;VALIDATION_ERROR&quot;, message)
    {
        ValidationErrors = new Dictionary&lt;string, List&lt;string&gt;&gt;();
    }

    public ValidationException(Dictionary&lt;string, List&lt;string&gt;&gt; validationErrors) 
        : base(&quot;VALIDATION_ERROR&quot;, &quot;バリデーションエラーが発生しました&quot;)
    {
        ValidationErrors = validationErrors;
    }

    public void AddError(string field, string error)
    {
        if (!ValidationErrors.ContainsKey(field))
        {
            ValidationErrors[field] = new List&lt;string&gt;();
        }
        ValidationErrors[field].Add(error);
    }

    public override string Message
    {
        get
        {
            if (!ValidationErrors.Any()) return base.Message;

            var errors = ValidationErrors.SelectMany(kvp =&gt; 
                kvp.Value.Select(error =&gt; $&quot;{kvp.Key}: {error}&quot;));
            return string.Join(Environment.NewLine, errors);
        }
    }
}

    使用例：バリデーション例外の活用

    public class UserRegistrationService
{
    public void RegisterUser(string email, string password, int age)
    {
        var validation = new ValidationException(&quot;ユーザー登録でエラーが発生しました&quot;);

        // メールアドレスの検証
        if (string.IsNullOrWhiteSpace(email))
        {
            validation.AddError(&quot;Email&quot;, &quot;メールアドレスは必須です&quot;);
        }
        else if (!IsValidEmail(email))
        {
            validation.AddError(&quot;Email&quot;, &quot;有効なメールアドレスを入力してください&quot;);
        }

        // パスワードの検証
        if (string.IsNullOrWhiteSpace(password))
        {
            validation.AddError(&quot;Password&quot;, &quot;パスワードは必須です&quot;);
        }
        else if (password.Length &lt; 8)
        {
            validation.AddError(&quot;Password&quot;, &quot;パスワードは8文字以上で入力してください&quot;);
        }

        // 年齢の検証
        if (age &lt; 0 || age &gt; 120)
        {
            validation.AddError(&quot;Age&quot;, &quot;年齢は0歳から120歳の間で入力してください&quot;);
        }

        // エラーがある場合は例外をスロー
        if (validation.ValidationErrors.Any())
        {
            throw validation;
        }

        // 正常な処理を続行
        Console.WriteLine($&quot;ユーザー登録成功: {email}&quot;);
    }

    private bool IsValidEmail(string email)
    {
        try
        {
            var addr = new System.Net.Mail.MailAddress(email);
            return addr.Address == email;
        }
        catch
        {
            return false;
        }
    }
}

    2. Resultパターン - 例外を使わない軽量なエラー処理

    技法の目的

    例外のパフォーマンスオーバーヘッドを避けつつ、型安全なエラー処理を実現します。

    実際の使用例

    • API呼び出し結果の管理
    • ファイル操作の結果処理
    • バリデーション結果の返却
    • チェーン処理での中間結果管理

    Result<T>クラスの実装

    public class Result&lt;T&gt;
{
    public bool IsSuccess { get; private set; }
    public T Value { get; private set; }
    public string ErrorMessage { get; private set; }
    public Exception Exception { get; private set; }
    public string ErrorCode { get; private set; }

    private Result(T value)
    {
        IsSuccess = true;
        Value = value;
    }

    private Result(string errorMessage, string errorCode = null, Exception exception = null)
    {
        IsSuccess = false;
        ErrorMessage = errorMessage;
        ErrorCode = errorCode;
        Exception = exception;
    }

    // 成功時のファクトリメソッド
    public static Result&lt;T&gt; Success(T value) =&gt; new Result&lt;T&gt;(value);

    // 失敗時のファクトリメソッド
    public static Result&lt;T&gt; Failure(string errorMessage, string errorCode = null) =&gt; 
        new Result&lt;T&gt;(errorMessage, errorCode);

    public static Result&lt;T&gt; Failure(Exception exception, string errorCode = null) =&gt; 
        new Result&lt;T&gt;(exception.Message, errorCode, exception);

    // 関数型プログラミング風のメソッド
    public TResult Match&lt;TResult&gt;(Func&lt;T, TResult&gt; onSuccess, Func&lt;string, TResult&gt; onFailure)
    {
        return IsSuccess ? onSuccess(Value) : onFailure(ErrorMessage);
    }

    // 結果の変換
    public Result&lt;TNew&gt; Map&lt;TNew&gt;(Func&lt;T, TNew&gt; mapper)
    {
        return IsSuccess 
            ? Result&lt;TNew&gt;.Success(mapper(Value))
            : Result&lt;TNew&gt;.Failure(ErrorMessage, ErrorCode, Exception);
    }

    // チェーン処理
    public Result&lt;TNew&gt; Bind&lt;TNew&gt;(Func&lt;T, Result&lt;TNew&gt;&gt; binder)
    {
        return IsSuccess 
            ? binder(Value)
            : Result&lt;TNew&gt;.Failure(ErrorMessage, ErrorCode, Exception);
    }
}

// 戻り値がない場合のResult
public class Result
{
    public bool IsSuccess { get; private set; }
    public string ErrorMessage { get; private set; }
    public Exception Exception { get; private set; }
    public string ErrorCode { get; private set; }

    private Result(bool isSuccess, string errorMessage = null, string errorCode = null, Exception exception = null)
    {
        IsSuccess = isSuccess;
        ErrorMessage = errorMessage;
        ErrorCode = errorCode;
        Exception = exception;
    }

    public static Result Success() =&gt; new Result(true);
    public static Result Failure(string errorMessage, string errorCode = null) =&gt; 
        new Result(false, errorMessage, errorCode);
    public static Result Failure(Exception exception, string errorCode = null) =&gt; 
        new Result(false, exception.Message, errorCode, exception);

    // Result&lt;T&gt;への変換
    public static implicit operator Result&lt;bool&gt;(Result result)
    {
        return result.IsSuccess 
            ? Result&lt;bool&gt;.Success(true)
            : Result&lt;bool&gt;.Failure(result.ErrorMessage, result.ErrorCode, result.Exception);
    }
}

    Resultパターンの実用例

    public class FileService
{
    public async Task&lt;Result&lt;string&gt;&gt; ReadFileAsync(string filePath)
    {
        try
        {
            if (string.IsNullOrWhiteSpace(filePath))
            {
                return Result&lt;string&gt;.Failure(&quot;ファイルパスが指定されていません&quot;, &quot;INVALID_PATH&quot;);
            }

            if (!File.Exists(filePath))
            {
                return Result&lt;string&gt;.Failure($&quot;ファイルが存在しません: {filePath}&quot;, &quot;FILE_NOT_FOUND&quot;);
            }

            var content = await File.ReadAllTextAsync(filePath);
            return Result&lt;string&gt;.Success(content);
        }
        catch (UnauthorizedAccessException ex)
        {
            return Result&lt;string&gt;.Failure(ex, &quot;ACCESS_DENIED&quot;);
        }
        catch (IOException ex)
        {
            return Result&lt;string&gt;.Failure(ex, &quot;IO_ERROR&quot;);
        }
        catch (Exception ex)
        {
            return Result&lt;string&gt;.Failure(ex, &quot;UNKNOWN_ERROR&quot;);
        }
    }

    public async Task&lt;Result&gt; WriteFileAsync(string filePath, string content)
    {
        try
        {
            if (string.IsNullOrWhiteSpace(filePath))
            {
                return Result.Failure(&quot;ファイルパスが指定されていません&quot;, &quot;INVALID_PATH&quot;);
            }

            // ディレクトリが存在しない場合は作成
            var directory = Path.GetDirectoryName(filePath);
            if (!string.IsNullOrEmpty(directory) &amp;&amp; !Directory.Exists(directory))
            {
                Directory.CreateDirectory(directory);
            }

            await File.WriteAllTextAsync(filePath, content);
            return Result.Success();
        }
        catch (Exception ex)
        {
            return Result.Failure(ex, &quot;WRITE_ERROR&quot;);
        }
    }
}

// 使用例：Resultパターンのチェーン処理
public class DocumentProcessor
{
    private readonly FileService _fileService = new FileService();

    public async Task&lt;Result&lt;string&gt;&gt; ProcessDocumentAsync(string inputPath, string outputPath)
    {
        // チェーン処理でファイル読み込み → 変換 → 保存
        var result = await _fileService.ReadFileAsync(inputPath);

        return await result
            .Map(content =&gt; content.ToUpper()) // 内容を大文字変換
            .Bind(async processedContent =&gt;     // 結果を次の処理にチェーン
            {
                var writeResult = await _fileService.WriteFileAsync(outputPath, processedContent);
                return writeResult.IsSuccess 
                    ? Result&lt;string&gt;.Success(processedContent)
                    : Result&lt;string&gt;.Failure(writeResult.ErrorMessage, writeResult.ErrorCode);
            });
    }

    public async Task DemonstrateResultPattern()
    {
        var result = await ProcessDocumentAsync(&quot;input.txt&quot;, &quot;output.txt&quot;);

        // Match を使った分岐処理
        var message = result.Match(
            success =&gt; $&quot;処理成功: {success.Length}文字を処理しました&quot;,
            error =&gt; $&quot;処理失敗: {error}&quot;
        );

        Console.WriteLine(message);
    }
}

    3. Retryパターン - 一時的障害からの自動回復

    技法の目的

    ネットワークエラーやリソース不足など、一時的な障害からの自動回復を実現します。

    実際の使用例

    • 外部API呼び出しの再試行
    • データベース接続の復旧処理
    • ファイルアクセスの競合回避
    • クラウドサービスの一時的障害対応

    汎用的なRetryPolicyの実装

    public class RetryPolicy
{
    private readonly int _maxRetries;
    private readonly TimeSpan _baseDelay;
    private readonly Func&lt;Exception, bool&gt; _shouldRetry;
    private readonly Func&lt;int, TimeSpan&gt; _delayStrategy;

    public RetryPolicy(
        int maxRetries = 3, 
        TimeSpan? baseDelay = null,
        Func&lt;Exception, bool&gt; shouldRetry = null,
        RetryStrategy strategy = RetryStrategy.FixedDelay)
    {
        _maxRetries = maxRetries;
        _baseDelay = baseDelay ?? TimeSpan.FromSeconds(1);
        _shouldRetry = shouldRetry ?? DefaultShouldRetry;
        _delayStrategy = GetDelayStrategy(strategy);
    }

    // デフォルトの再試行判定
    private static bool DefaultShouldRetry(Exception ex)
    {
        return ex is HttpRequestException ||
               ex is TaskCanceledException ||
               ex is SocketException ||
               ex is IOException;
    }

    // 遅延戦略の取得
    private Func&lt;int, TimeSpan&gt; GetDelayStrategy(RetryStrategy strategy)
    {
        return strategy switch
        {
            RetryStrategy.FixedDelay =&gt; attempt =&gt; _baseDelay,
            RetryStrategy.LinearBackoff =&gt; attempt =&gt; TimeSpan.FromMilliseconds(_baseDelay.TotalMilliseconds * (attempt + 1)),
            RetryStrategy.ExponentialBackoff =&gt; attempt =&gt; TimeSpan.FromMilliseconds(_baseDelay.TotalMilliseconds * Math.Pow(2, attempt)),
            RetryStrategy.RandomJitter =&gt; attempt =&gt; TimeSpan.FromMilliseconds(_baseDelay.TotalMilliseconds + Random.Shared.Next(0, 1000)),
            _ =&gt; attempt =&gt; _baseDelay
        };
    }

    // 同期版の実行
    public T Execute&lt;T&gt;(Func&lt;T&gt; operation, string operationName = null)
    {
        Exception lastException = null;

        for (int attempt = 0; attempt &lt;= _maxRetries; attempt++)
        {
            try
            {
                return operation();
            }
            catch (Exception ex) when (_shouldRetry(ex) &amp;&amp; attempt &lt; _maxRetries)
            {
                lastException = ex;

                var delay = _delayStrategy(attempt);
                Console.WriteLine($&quot;{operationName ?? &quot;操作&quot;} 失敗 (試行 {attempt + 1}/{_maxRetries + 1}): {ex.Message}&quot;);
                Console.WriteLine($&quot;{delay.TotalSeconds}秒後に再試行します...&quot;);

                Thread.Sleep(delay);
            }
        }

        throw new AggregateException($&quot;操作が {_maxRetries + 1} 回失敗しました: {operationName}&quot;, lastException);
    }

    // 非同期版の実行
    public async Task&lt;T&gt; ExecuteAsync&lt;T&gt;(Func&lt;Task&lt;T&gt;&gt; operation, string operationName = null)
    {
        Exception lastException = null;

        for (int attempt = 0; attempt &lt;= _maxRetries; attempt++)
        {
            try
            {
                return await operation();
            }
            catch (Exception ex) when (_shouldRetry(ex) &amp;&amp; attempt &lt; _maxRetries)
            {
                lastException = ex;

                var delay = _delayStrategy(attempt);
                Console.WriteLine($&quot;{operationName ?? &quot;操作&quot;} 失敗 (試行 {attempt + 1}/{_maxRetries + 1}): {ex.Message}&quot;);
                Console.WriteLine($&quot;{delay.TotalSeconds}秒後に再試行します...&quot;);

                await Task.Delay(delay);
            }
        }

        throw new AggregateException($&quot;操作が {_maxRetries + 1} 回失敗しました: {operationName}&quot;, lastException);
    }
}

public enum RetryStrategy
{
    FixedDelay,         // 固定間隔
    LinearBackoff,      // 線形増加
    ExponentialBackoff, // 指数関数的増加
    RandomJitter        // ランダムジッター
}

    実用的なRetryパターンの活用例

    public class ApiClient
{
    private readonly HttpClient _httpClient;
    private readonly RetryPolicy _retryPolicy;

    public ApiClient(HttpClient httpClient = null)
    {
        _httpClient = httpClient ?? new HttpClient();

        // API呼び出し用のRetryポリシー
        _retryPolicy = new RetryPolicy(
            maxRetries: 3,
            baseDelay: TimeSpan.FromSeconds(2),
            shouldRetry: ShouldRetryApiCall,
            strategy: RetryStrategy.ExponentialBackoff
        );
    }

    private bool ShouldRetryApiCall(Exception ex)
    {
        // HTTP例外の場合、ステータスコードによって判定
        if (ex is HttpRequestException httpEx)
        {
            // 一時的なエラーの場合のみ再試行
            return true; // 実際にはレスポンスコードを確認
        }

        // タイムアウトは再試行
        if (ex is TaskCanceledException timeoutEx &amp;&amp; timeoutEx.InnerException is TimeoutException)
        {
            return true;
        }

        // ネットワークエラーは再試行
        return ex is SocketException || ex is IOException;
    }

    public async Task&lt;Result&lt;string&gt;&gt; GetDataAsync(string endpoint)
    {
        try
        {
            var result = await _retryPolicy.ExecuteAsync(async () =&gt;
            {
                var response = await _httpClient.GetAsync(endpoint);
                response.EnsureSuccessStatusCode();
                return await response.Content.ReadAsStringAsync();
            }, $&quot;API呼び出し: {endpoint}&quot;);

            return Result&lt;string&gt;.Success(result);
        }
        catch (AggregateException ex)
        {
            return Result&lt;string&gt;.Failure(ex.InnerException ?? ex, &quot;API_CALL_FAILED&quot;);
        }
        catch (Exception ex)
        {
            return Result&lt;string&gt;.Failure(ex, &quot;UNEXPECTED_ERROR&quot;);
        }
    }

    public async Task&lt;Result&lt;T&gt;&gt; PostDataAsync&lt;T&gt;(string endpoint, object data)
    {
        try
        {
            var result = await _retryPolicy.ExecuteAsync(async () =&gt;
            {
                var json = JsonSerializer.Serialize(data);
                var content = new StringContent(json, Encoding.UTF8, &quot;application/json&quot;);

                var response = await _httpClient.PostAsync(endpoint, content);
                response.EnsureSuccessStatusCode();

                var responseContent = await response.Content.ReadAsStringAsync();
                return JsonSerializer.Deserialize&lt;T&gt;(responseContent);
            }, $&quot;API POST: {endpoint}&quot;);

            return Result&lt;T&gt;.Success(result);
        }
        catch (Exception ex)
        {
            return Result&lt;T&gt;.Failure(ex, &quot;API_POST_FAILED&quot;);
        }
    }
}

    ASP.NET Core用のグローバル例外ミドルウェア

    public class GlobalExceptionMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger&lt;GlobalExceptionMiddleware&gt; _logger;
    private readonly IWebHostEnvironment _environment;

    public GlobalExceptionMiddleware(
        RequestDelegate next, 
        ILogger&lt;GlobalExceptionMiddleware&gt; logger,
        IWebHostEnvironment environment)
    {
        _next = next;
        _logger = logger;
        _environment = environment;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        try
        {
            await _next(context);
        }
        catch (Exception ex)
        {
            await HandleExceptionAsync(context, ex);
        }
    }

    private async Task HandleExceptionAsync(HttpContext context, Exception exception)
    {
        // ログ記録
        var errorId = Guid.NewGuid().ToString(&quot;N&quot;)[..8];
        var logMessage = $&quot;[{errorId}] {exception.GetType().Name}: {exception.Message}&quot;;

        _logger.LogError(exception, logMessage);

        // レスポンス設定
        context.Response.ContentType = &quot;application/json&quot;;
        context.Response.StatusCode = GetStatusCode(exception);

        var response = CreateErrorResponse(exception, errorId);

        var jsonResponse = JsonSerializer.Serialize(response, new JsonSerializerOptions
        {
            PropertyNamingPolicy = JsonNamingPolicy.CamelCase,
            WriteIndented = _environment.IsDevelopment()
        });

        await context.Response.WriteAsync(jsonResponse);
    }

    private static int GetStatusCode(Exception exception) =&gt; exception switch
    {
        ValidationException =&gt; 400,
        AuthorizationException =&gt; 403,
        ResourceNotFoundException =&gt; 404,
        ExternalServiceException =&gt; 502,
        TimeoutException =&gt; 504,
        _ =&gt; 500
    };

    private object CreateErrorResponse(Exception exception, string errorId)
    {
        var baseResponse = new
        {
            error = new
            {
                id = errorId,
                message = GetUserFriendlyMessage(exception),
                timestamp = DateTime.UtcNow,
                type = exception.GetType().Name
            }
        };

        // 開発環境では詳細情報を追加
        if (_environment.IsDevelopment())
        {
            return new
            {
                error = new
                {
                    id = errorId,
                    message = GetUserFriendlyMessage(exception),
                    timestamp = DateTime.UtcNow,
                    type = exception.GetType().Name,
                    details = exception.Message,
                    stackTrace = exception.StackTrace,
                    data = GetExceptionData(exception)
                }
            };
        }

        return baseResponse;
    }

    private static string GetUserFriendlyMessage(Exception exception) =&gt; exception switch
    {
        ValidationException validationEx =&gt; &quot;入力内容に問題があります。確認して再試行してください。&quot;,
        AuthorizationException =&gt; &quot;この操作を実行する権限がありません。&quot;,
        ResourceNotFoundException =&gt; &quot;要求されたリソースが見つかりません。&quot;,
        ExternalServiceException =&gt; &quot;外部サービスとの通信でエラーが発生しました。しばらく時間をおいて再試行してください。&quot;,
        TimeoutException =&gt; &quot;処理に時間がかかりすぎています。再試行してください。&quot;,
        _ =&gt; &quot;システムエラーが発生しました。管理者にお問い合わせください。&quot;
    };

    private static object GetExceptionData(Exception exception)
    {
        return exception switch
        {
            ValidationException validationEx =&gt; validationEx.ValidationErrors,
            BusinessException businessEx =&gt; new 
            { 
                businessEx.ErrorCode, 
                businessEx.Timestamp,
                businessEx.AdditionalData
            },
            _ =&gt; null
        };
    }
}

    WPF/WinForms用のグローバル例外ハンドラー

    public class GlobalExceptionHandler
{
    private readonly ILogger _logger;
    private readonly bool _isProductionEnvironment;

    public GlobalExceptionHandler(ILogger logger, bool isProductionEnvironment = false)
    {
        _logger = logger;
        _isProductionEnvironment = isProductionEnvironment;

        SetupExceptionHandlers();
    }

    private void SetupExceptionHandlers()
    {
        // アプリケーションドメインの未処理例外
        AppDomain.CurrentDomain.UnhandledException += OnUnhandledException;

        // UIスレッドの未処理例外
        Application.Current.DispatcherUnhandledException += OnDispatcherUnhandledException;

        // タスクの未処理例外
        TaskScheduler.UnobservedTaskException += OnUnobservedTaskException;
    }

    private void OnUnhandledException(object sender, UnhandledExceptionEventArgs e)
    {
        var exception = (Exception)e.ExceptionObject;
        var errorId = LogException(exception, &quot;Critical unhandled exception&quot;);

        ShowCriticalErrorDialog(exception, errorId, e.IsTerminating);

        if (e.IsTerminating)
        {
            Environment.Exit(1);
        }
    }

    private void OnDispatcherUnhandledException(object sender, DispatcherUnhandledExceptionEventArgs e)
    {
        var errorId = LogException(e.Exception, &quot;UI thread unhandled exception&quot;);

        if (ShouldHandleException(e.Exception))
        {
            ShowErrorDialog(e.Exception, errorId);
            e.Handled = true; // 例外を処理済みとしてアプリを継続
        }
        else
        {
            ShowCriticalErrorDialog(e.Exception, errorId, true);
        }
    }

    private void OnUnobservedTaskException(object sender, UnobservedTaskExceptionEventArgs e)
    {
        var errorId = LogException(e.Exception, &quot;Unobserved task exception&quot;);
        e.SetObserved(); // 例外を観測済みとしてマーク

        // バックグラウンド処理のエラーはユーザーに通知（軽微な場合）
        if (ShouldNotifyUser(e.Exception))
        {
            ShowBackgroundErrorNotification(e.Exception, errorId);
        }
    }

    private string LogException(Exception exception, string context)
    {
        var errorId = Guid.NewGuid().ToString(&quot;N&quot;)[..8];
        var logMessage = $&quot;[{errorId}] {context}: {exception.Message}&quot;;

        _logger.LogError(exception, logMessage);

        return errorId;
    }

    private bool ShouldHandleException(Exception exception)
    {
        // ビジネス例外は通常処理を継続可能
        return exception is BusinessException ||
               exception is ValidationException ||
               exception is HttpRequestException;
    }

    private bool ShouldNotifyUser(Exception exception)
    {
        // 重要度が高い例外のみユーザーに通知
        return exception is ExternalServiceException ||
               exception is IOException;
    }

    private void ShowErrorDialog(Exception exception, string errorId)
    {
        var message = GetUserFriendlyMessage(exception);
        var title = &quot;エラーが発生しました&quot;;

        if (!_isProductionEnvironment)
        {
            message += $&quot;\n\nエラーID: {errorId}\n詳細: {exception.Message}&quot;;
        }

        MessageBox.Show(message, title, MessageBoxButton.OK, MessageBoxImage.Warning);
    }

    private void ShowCriticalErrorDialog(Exception exception, string errorId, bool willTerminate)
    {
        var message = willTerminate 
            ? &quot;重大なエラーが発生したため、アプリケーションを終了します。&quot;
            : &quot;重大なエラーが発生しました。&quot;;

        if (!_isProductionEnvironment)
        {
            message += $&quot;\n\nエラーID: {errorId}\n詳細: {exception.Message}&quot;;
        }

        MessageBox.Show(message, &quot;重大なエラー&quot;, MessageBoxButton.OK, MessageBoxImage.Error);
    }

    private void ShowBackgroundErrorNotification(Exception exception, string errorId)
    {
        // トースト通知やステータスバーでの軽微な通知
        Console.WriteLine($&quot;バックグラウンド処理でエラーが発生しました (ID: {errorId})&quot;);
    }

    private string GetUserFriendlyMessage(Exception exception) =&gt; exception switch
    {
        ValidationException =&gt; &quot;入力内容に問題があります。確認して再試行してください。&quot;,
        AuthorizationException =&gt; &quot;この操作を実行する権限がありません。&quot;,
        ResourceNotFoundException =&gt; &quot;要求されたデータが見つかりません。&quot;,
        ExternalServiceException =&gt; &quot;外部サービスとの通信でエラーが発生しました。&quot;,
        TimeoutException =&gt; &quot;処理に時間がかかりすぎています。再試行してください。&quot;,
        IOException =&gt; &quot;ファイルの読み書きでエラーが発生しました。&quot;,
        UnauthorizedAccessException =&gt; &quot;ファイルやリソースへのアクセス権限がありません。&quot;,
        _ =&gt; &quot;予期しないエラーが発生しました。&quot;
    };
}

    5. 統合エラーサービス - 包括的なエラー管理システム

    技法の目的

    ログ記録、監視、通知、分析を統合した包括的なエラー管理システムを構築します。

    実際の使用例

    • エンタープライズアプリケーションの運用監視
    • マイクロサービス間のエラー追跡
    • ビジネスインテリジェンスでのエラー分析
    • 自動復旧とエスカレーション

    統合エラーサービスの実装

    public interface IErrorHandlingService
{
    Task&lt;Result&lt;T&gt;&gt; ExecuteAsync&lt;T&gt;(Func&lt;Task&lt;T&gt;&gt; operation, string operationName = null, string context = null);
    Task&lt;Result&gt; ExecuteAsync(Func&lt;Task&gt; operation, string operationName = null, string context = null);
    Task LogErrorAsync(Exception exception, string context = null, Dictionary&lt;string, object&gt; additionalData = null);
    void HandleError(Exception exception, string userMessage = null, bool shouldNotifyUser = true);
    Task&lt;ErrorStatistics&gt; GetErrorStatisticsAsync(DateTime from, DateTime to);
}

public class ErrorHandlingService : IErrorHandlingService
{
    private readonly ILogger&lt;ErrorHandlingService&gt; _logger;
    private readonly RetryPolicy _defaultRetryPolicy;
    private readonly IErrorNotificationService _notificationService;
    private readonly IErrorMetricsCollector _metricsCollector;
    private readonly IErrorPersistenceService _persistenceService;

    public ErrorHandlingService(
        ILogger&lt;ErrorHandlingService&gt; logger,
        IErrorNotificationService notificationService = null,
        IErrorMetricsCollector metricsCollector = null,
        IErrorPersistenceService persistenceService = null)
    {
        _logger = logger;
        _notificationService = notificationService;
        _metricsCollector = metricsCollector;
        _persistenceService = persistenceService;

        _defaultRetryPolicy = new RetryPolicy(
            maxRetries: 2,
            baseDelay: TimeSpan.FromMilliseconds(500),
            strategy: RetryStrategy.ExponentialBackoff
        );
    }

    public async Task&lt;Result&lt;T&gt;&gt; ExecuteAsync&lt;T&gt;(Func&lt;Task&lt;T&gt;&gt; operation, string operationName = null, string context = null)
    {
        var stopwatch = Stopwatch.StartNew();
        var operationId = Guid.NewGuid().ToString(&quot;N&quot;)[..8];

        try
        {
            // メトリクス収集開始
            _metricsCollector?.RecordOperationStart(operationName, operationId);

            var result = await _defaultRetryPolicy.ExecuteAsync(operation, operationName);

            stopwatch.Stop();
            _metricsCollector?.RecordOperationSuccess(operationName, operationId, stopwatch.Elapsed);

            return Result&lt;T&gt;.Success(result);
        }
        catch (BusinessException businessEx)
        {
            stopwatch.Stop();
            await HandleBusinessExceptionAsync(businessEx, operationName, context, operationId, stopwatch.Elapsed);
            return Result&lt;T&gt;.Failure(businessEx.Message, businessEx.ErrorCode);
        }
        catch (Exception ex)
        {
            stopwatch.Stop();
            await HandleSystemExceptionAsync(ex, operationName, context, operationId, stopwatch.Elapsed);
            return Result&lt;T&gt;.Failure(&quot;システムエラーが発生しました。しばらく時間をおいて再試行してください。&quot;, &quot;SYSTEM_ERROR&quot;);
        }
    }

    public async Task&lt;Result&gt; ExecuteAsync(Func&lt;Task&gt; operation, string operationName = null, string context = null)
    {
        var result = await ExecuteAsync(async () =&gt;
        {
            await operation();
            return true;
        }, operationName, context);

        return result.IsSuccess ? Result.Success() : Result.Failure(result.ErrorMessage, result.ErrorCode);
    }

    public async Task LogErrorAsync(Exception exception, string context = null, Dictionary&lt;string, object&gt; additionalData = null)
    {
        var errorEntry = new ErrorEntry
        {
            Id = Guid.NewGuid(),
            Timestamp = DateTime.UtcNow,
            ExceptionType = exception.GetType().Name,
            Message = exception.Message,
            StackTrace = exception.StackTrace,
            Context = context,
            Severity = GetErrorSeverity(exception),
            AdditionalData = additionalData ?? new Dictionary&lt;string, object&gt;()
        };

        // 構造化ログ出力
        _logger.LogError(exception, 
            &quot;Error logged - ID: {ErrorId}, Type: {ExceptionType}, Context: {Context}&quot;,
            errorEntry.Id, errorEntry.ExceptionType, context);

        // 永続化
        if (_persistenceService != null)
        {
            await _persistenceService.SaveErrorAsync(errorEntry);
        }

        // メトリクス収集
        _metricsCollector?.RecordError(errorEntry);

        // 重要度に応じた通知
        if (ShouldNotifyOperation(exception))
        {
            await _notificationService?.NotifyOperationTeamAsync(errorEntry);
        }
    }

    private async Task HandleBusinessExceptionAsync(BusinessException exception, string operationName, string context, string operationId, TimeSpan duration)
    {
        var additionalData = new Dictionary&lt;string, object&gt;
        {
            [&quot;OperationName&quot;] = operationName,
            [&quot;OperationId&quot;] = operationId,
            [&quot;Duration&quot;] = duration.TotalMilliseconds,
            [&quot;ErrorCode&quot;] = exception.ErrorCode
        };

        additionalData.AddRange(exception.AdditionalData);

        await LogErrorAsync(exception, context, additionalData);
        _metricsCollector?.RecordOperationFailure(operationName, operationId, duration, exception.ErrorCode);
    }

    private async Task HandleSystemExceptionAsync(Exception exception, string operationName, string context, string operationId, TimeSpan duration)
    {
        var additionalData = new Dictionary&lt;string, object&gt;
        {
            [&quot;OperationName&quot;] = operationName,
            [&quot;OperationId&quot;] = operationId,
            [&quot;Duration&quot;] = duration.TotalMilliseconds,
            [&quot;IsCritical&quot;] = IsCriticalError(exception)
        };

        await LogErrorAsync(exception, context, additionalData);
        _metricsCollector?.RecordOperationFailure(operationName, operationId, duration, &quot;SYSTEM_ERROR&quot;);

        // 重大エラーの場合は即座にエスカレーション
        if (IsCriticalError(exception))
        {
            await _notificationService?.NotifyDevelopmentTeamAsync(new ErrorEntry
            {
                Id = Guid.NewGuid(),
                Timestamp = DateTime.UtcNow,
                ExceptionType = exception.GetType().Name,
                Message = exception.Message,
                Context = context,
                Severity = ErrorSeverity.Critical,
                AdditionalData = additionalData
            });
        }
    }

    public void HandleError(Exception exception, string userMessage = null, bool shouldNotifyUser = true)
    {
        var message = userMessage ?? GetUserFriendlyMessage(exception);
        var severity = GetErrorSeverity(exception);

        if (shouldNotifyUser)
        {
            NotifyUser(message, severity);
        }
    }

    public async Task&lt;ErrorStatistics&gt; GetErrorStatisticsAsync(DateTime from, DateTime to)
    {
        if (_persistenceService == null)
        {
            throw new InvalidOperationException(&quot;Error persistence service is not configured&quot;);
        }

        return await _persistenceService.GetErrorStatisticsAsync(from, to);
    }

    private ErrorSeverity GetErrorSeverity(Exception exception) =&gt; exception switch
    {
        ValidationException =&gt; ErrorSeverity.Warning,
        AuthorizationException =&gt; ErrorSeverity.Warning,
        ResourceNotFoundException =&gt; ErrorSeverity.Info,
        ExternalServiceException =&gt; ErrorSeverity.Error,
        BusinessException =&gt; ErrorSeverity.Warning,
        OutOfMemoryException =&gt; ErrorSeverity.Critical,
        StackOverflowException =&gt; ErrorSeverity.Critical,
        _ =&gt; ErrorSeverity.Error
    };

    private bool IsCriticalError(Exception exception) =&gt; exception switch
    {
        OutOfMemoryException =&gt; true,
        StackOverflowException =&gt; true,
        AccessViolationException =&gt; true,
        _ =&gt; false
    };

    private bool ShouldNotifyOperation(Exception exception)
    {
        return GetErrorSeverity(exception) &gt;= ErrorSeverity.Error;
    }

    private string GetUserFriendlyMessage(Exception exception) =&gt; exception switch
    {
        ValidationException =&gt; &quot;入力内容に問題があります。確認して再試行してください。&quot;,
        AuthorizationException =&gt; &quot;この操作を実行する権限がありません。&quot;,
        ResourceNotFoundException =&gt; &quot;要求されたデータが見つかりません。&quot;,
        ExternalServiceException =&gt; &quot;外部サービスとの通信でエラーが発生しました。&quot;,
        TimeoutException =&gt; &quot;処理に時間がかかりすぎています。再試行してください。&quot;,
        _ =&gt; &quot;システムエラーが発生しました。管理者にお問い合わせください。&quot;
    };

    private void NotifyUser(string message, ErrorSeverity severity)
    {
        // 実装例（実際のアプリケーションでは適切なUI表示メカニズムを使用）
        var severityText = severity switch
        {
            ErrorSeverity.Info =&gt; &quot;情報&quot;,
            ErrorSeverity.Warning =&gt; &quot;警告&quot;,
            ErrorSeverity.Error =&gt; &quot;エラー&quot;,
            ErrorSeverity.Critical =&gt; &quot;重大エラー&quot;,
            _ =&gt; &quot;通知&quot;
        };

        Console.WriteLine($&quot;[{severityText}] {message}&quot;);
    }
}

    採点基準: 1. 正しい実装 (30%): 例外クラスとエラーハンドリングが適切 2. Resultパターンの活用 (25%): 型安全なエラー処理 3. ログとモニタリング (20%): 適切なログ出力と追跡可能性 4. エラー分類 (15%): ビジネス例外とシステム例外の適切な分類 5. ユーザビリティ (10%): 分かりやすいエラーメッセージ

    技法別特徴比較表

    技法 主な目的 適用場面 メリット デメリット 実装の複雑さ

    統合エラーサービス 包括的管理 エンタープライズ 監視、分析、自動化 アーキテクチャの複雑化 ⭐⭐⭐⭐⭐

    エラー処理戦略の選択指針

    エラーの性質を分析
├─ 予測可能なビジネスエラー
│  ├─ バリデーションエラー → カスタム例外 + Resultパターン
│  └─ 権限・リソースエラー → カスタム例外
├─ 一時的な障害
│  └─ ネットワーク・I/Oエラー → Retryパターン
├─ システム全体の例外
│  └─ 未処理例外 → グローバルハンドリング
└─ エンタープライズレベル
   └─ 監視・分析・自動化 → 統合エラーサービス

    成功の鍵

    1. 適切な技法選択: エラーの性質に応じた最適な処理方法の選択
    2. 段階的実装: 小さく始めて徐々に高度化

    避けるべき罠

    • 例外の乱用: パフォーマンスに影響する過度な例外使用
    • 情報不足: デバッグに必要な情報が不足したエラー処理
    • セキュリティリスク: 機密情報を含むエラーメッセージの露出
    • 複雑化: 過度に複雑なエラーハンドリングアーキテクチャ
