    public void ComplexFilteringExample()
    {
        Console.WriteLine("=== 複雑なフィルタリング例 ===");

        // マネージャーかつ開発部門で、かつ特定スキルを持つ従業員
        var complexFilter = employees
            .Where(e => e.Position.Contains("マネージャー") || e.Position.Contains("リード"))
            .Where(e => e.Department == "開発" || e.Department == "QA")
            .Where(e => e.Skills.Any(skill => skill.Contains("C#") || skill.Contains("テスト")))
            .Where(e => e.Salary >= 500000)
            .Where(e => DateTime.Now.Year - e.HireDate.Year >= 3);

        foreach (var emp in complexFilter)
        {
            Console.WriteLine($"{emp.Name} - {emp.Position} ({emp.Department}) - スキル: {string.Join(", ", emp.Skills)}");
        }
    }

    // 高度なクエリ例2: ネストしたLINQとSelectMany
    public void NestedLinqExample()
    {
        Console.WriteLine("\n=== ネストしたLINQとSelectMany ===");

        // 全従業員のスキルを平坦化し、頻度順で表示
        var skillFrequency = employees
            .SelectMany(e => e.Skills, (employee, skill) => new { Employee = employee, Skill = skill })
            .GroupBy(x => x.Skill)
            .Select(g => new
            {
                Skill = g.Key,
                Count = g.Count(),
                Employees = g.Select(x => x.Employee.Name).ToList(),
                AverageSalary = g.Average(x => x.Employee.Salary)
            })
            .OrderByDescending(x => x.Count);

        foreach (var skill in skillFrequency)
        {
            Console.WriteLine($"{skill.Skill}: {skill.Count}人保有, 平均給与: ¥{skill.AverageSalary:N0}");
            Console.WriteLine($"  保有者: {string.Join(", ", skill.Employees)}");
        }
    }

    // 高度なクエリ例3: 複雑なグループ化と集約
    public void ComplexGroupingExample()
    {
        Console.WriteLine("\n=== 複雑なグループ化と集約 ===");

        // 部署別、年代別の詳細統計
        var departmentAgeAnalysis = employees
            .GroupBy(e => e.Department)
            .Select(deptGroup => new
            {
                Department = deptGroup.Key,
                TotalEmployees = deptGroup.Count(),
                AverageSalary = deptGroup.Average(e => e.Salary),
                SalaryRange = new
                {
                    Min = deptGroup.Min(e => e.Salary),
                    Max = deptGroup.Max(e => e.Salary)
                },
                AgeGroups = deptGroup
                    .GroupBy(e => (e.Age / 10) * 10) // 年代でグループ化
                    .Select(ageGroup => new
                    {
                        AgeRange = $"{ageGroup.Key}代",
                        Count = ageGroup.Count(),
                        Names = ageGroup.Select(e => e.Name).ToList()
                    })
                    .OrderBy(ag => ag.AgeRange),
                MostCommonSkills = deptGroup
                    .SelectMany(e => e.Skills)
                    .GroupBy(skill => skill)
                    .OrderByDescending(g => g.Count())
                    .Take(3)
                    .Select(g => $"{g.Key}({g.Count()}人)")
                    .ToList()
            })
            .OrderByDescending(x => x.TotalEmployees);

        foreach (var dept in departmentAgeAnalysis)
        {
            Console.WriteLine($"\n【{dept.Department}部】");
            Console.WriteLine($"  総人数: {dept.TotalEmployees}人");
            Console.WriteLine($"  平均給与: ¥{dept.AverageSalary:N0}");
            Console.WriteLine($"  給与範囲: ¥{dept.SalaryRange.Min:N0} - ¥{dept.SalaryRange.Max:N0}");
            Console.WriteLine($"  主要スキル: {string.Join(", ", dept.MostCommonSkills)}");

            Console.WriteLine("  年代別構成:");
            foreach (var ageGroup in dept.AgeGroups)
            {

        {

        }

        if (!string.IsNullOrWhiteSpace(criteria.DepartmentFilter))
        {
            query = query.Where(e => e.Department == criteria.DepartmentFilter);
        }

        // 給与範囲フィルター
        if (criteria.MinSalary.HasValue)
        {
            query = query.Where(e => e.Salary >= criteria.MinSalary.Value);
        }

        if (criteria.MaxSalary.HasValue)
        {
            query = query.Where(e => e.Salary <= criteria.MaxSalary.Value);
        }

        // 年齢範囲フィルター
        if (criteria.MinAge.HasValue)
        {
            query = query.Where(e => e.Age >= criteria.MinAge.Value);
        }

        if (criteria.MaxAge.HasValue)
        {
            query = query.Where(e => e.Age <= criteria.MaxAge.Value);
        }

        // 必須スキルフィルター（すべてのスキルを持っている必要がある）
        if (criteria.RequiredSkills.Any())
        {
            foreach (var skill in criteria.RequiredSkills)
            {
                query = query.Where(e => e.Skills.Contains(skill, StringComparer.OrdinalIgnoreCase));
            }
        }

        // 入社日フィルター
        if (criteria.HiredAfter.HasValue)
        {
            query = query.Where(e => e.HireDate >= criteria.HiredAfter.Value);
        }

        if (criteria.HiredBefore.HasValue)
        {
            query = query.Where(e => e.HireDate <= criteria.HiredBefore.Value);
        }

        // 動的ソート
        query = ApplySorting(query, criteria.SortBy, criteria.SortDescending);

        return query.ToList();
    }

    private IQueryable<Employee> ApplySorting(IQueryable<Employee> query, string sortBy, bool descending)
    {
        Expression<Func<Employee, object>> keySelector = sortBy?.ToLower() switch
        {
            "name" => e => e.Name,
            "department" => e => e.Department,
            "salary" => e => e.Salary,
            "age" => e => e.Age,
            "hiredate" => e => e.HireDate,
            "position" => e => e.Position,
            _ => e => e.Name
        };

        return descending 
            ? query.OrderByDescending(keySelector)
            : query.OrderBy(keySelector);
    }

    public static class AdvancedLinqExtensions
{
    // バッチ処理（指定サイズでグループ化）
    public static IEnumerable<IEnumerable<T>> Batch<T>(this IEnumerable<T> source, int batchSize)
    {
        if (batchSize <= 0) throw new ArgumentException("Batch size must be positive", nameof(batchSize));

        var batch = new List<T>(batchSize);
        foreach (var item in source)
        {
            batch.Add(item);
            if (batch.Count == batchSize)
            {
                yield return batch;
                batch = new List<T>(batchSize);
            }
        }

        if (batch.Count > 0)
        {
            yield return batch;
        }
    }

    // 移動平均の計算
    public static IEnumerable<decimal> MovingAverage(this IEnumerable<decimal> source, int windowSize)
    {
        if (windowSize <= 0) throw new ArgumentException("Window size must be positive", nameof(windowSize));

        var window = new Queue<decimal>();
        decimal sum = 0;

        foreach (var value in source)
        {
            window.Enqueue(value);
            sum += value;

            if (window.Count > windowSize)
            {
                sum -= window.Dequeue();
            }

            if (window.Count == windowSize)
            {
                yield return sum / windowSize;
            }
        }
    }

    // 条件に基づく分割
    public static (IEnumerable<T> matching, IEnumerable<T> nonMatching) PartitionBy<T>(
        this IEnumerable<T> source, Func<T, bool> predicate)
    {
        var sourceList = source.ToList(); // 複数回列挙を避ける
        return (sourceList.Where(predicate), sourceList.Where(x => !predicate(x)));
    }

    // 統計情報の一括取得
    public static StatisticsResult<T> GetStatistics<T>(this IEnumerable<T> source, Func<T, decimal> selector)
    {
        var values = source.Select(selector).ToList();

        if (!values.Any())
        {
            return new StatisticsResult<T>
            {
                Count = 0,
                Sum = 0,
                Average = 0,
                Min = 0,
                Max = 0,
                Median = 0,
                StandardDeviation = 0
            };
        }

        var sorted = values.OrderBy(x => x).ToList();
        var count = values.Count;
        var sum = values.Sum();
        var average = values.Average();
        var min = values.Min();
        var max = values.Max();

        var median = count % 2 == 0
            ? (sorted[count / 2 - 1] + sorted[count / 2]) / 2
            : sorted[count / 2];

        var variance = values.Sum(v => Math.Pow((double)(v - average), 2)) / count;
        var standardDeviation = (decimal)Math.Sqrt(variance);

        return new StatisticsResult<T>
        {
            Count = count,
            Sum = sum,
            Average = average,
            Min = min,
            Max = max,
            Median = median,
            StandardDeviation = standardDeviation
        };
    }

    // 階層構造のフラット化
    public static IEnumerable<T> Flatten<T>(this IEnumerable<T> source, Func<T, IEnumerable<T>> childSelector)
    {
        var stack = new Stack<T>(source.Reverse());

        while (stack.Count > 0)
        {
            var current = stack.Pop();
            yield return current;

            var children = childSelector(current);
            if (children != null)
            {
                foreach (var child in children.Reverse())
                {
                    stack.Push(child);
                }
            }
        }
    }

    // 文字列結合の高度版
    public static string JoinWith<T>(this IEnumerable<T> source, 
        string separator, 
        Func<T, string> selector = null,
        string prefix = "",
        string suffix = "")
    {
        selector ??= item => item?.ToString() ?? "";
        var result = string.Join(separator, source.Select(selector));
        return $"{prefix}{result}{suffix}";
    }

    private double CalculateAttritionRisk(Employee employee)
    {
        var experience = (DateTime.Now - employee.HireDate).Days / 365.25;
        var salaryPercentile = employees.Count(e => e.Salary <= employee.Salary) / (double)employees.Count;

        // 簡略化されたリスクスコア計算
        var riskScore = 0.0;

        if (experience < 1) riskScore += 0.3; // 新人は退職リスクが高い
        if (experience > 5) riskScore += 0.2; // ベテランも転職を考えやすい
        if (salaryPercentile < 0.3) riskScore += 0.4; // 低給与はリスク
        if (employee.Age < 30) riskScore += 0.2; // 若手は転職しやすい

        return Math.Min(riskScore, 1.0);
    }

    public CustomerSegmentAnalysis AnalyzeCustomerSegments()

            .Join(sales, c => c.Id, s => s.CustomerId, (c, s) => new { Customer = c, Sale = s })
            .Join(products, cs => cs.Sale.ProductId, p => p.Id, (cs, p) => new { cs.Customer, cs.Sale, Product = p })

            .Select(g => new SegmentMetrics

    public ProductProfitabilityAnalysis AnalyzeProductProfitability()
    {

            .GroupJoin(sales, p => p.Id, s => s.ProductId, (p, salesGroup) => new { Product = p, Sales = salesGroup })
            .Select(pg => new ProductMetrics
            {
                Product = pg.Product,
                TotalRevenue = pg.Sales.Sum(s => s.Quantity * s.UnitPrice * (1 - s.Discount)),
                TotalCost = pg.Sales.Sum(s => s.Quantity * pg.Product.Cost),
                GrossProfit = pg.Sales.Sum(s => s.Quantity * s.UnitPrice * (1 - s.Discount)) - 
                             pg.Sales.Sum(s => s.Quantity * pg.Product.Cost),
                UnitsSold = pg.Sales.Sum(s => s.Quantity),
                ProfitMargin = pg.Sales.Any() ? 
                    (pg.Sales.Sum(s => s.Quantity * s.UnitPrice * (1 - s.Discount)) - 
                     pg.Sales.Sum(s => s.Quantity * pg.Product.Cost)) / 
                    pg.Sales.Sum(s => s.Quantity * s.UnitPrice * (1 - s.Discount)) : 0,
                InventoryTurnover = pg.Product.StockQuantity > 0 ? 
                    (double)pg.Sales.Sum(s => s.Quantity) / pg.Product.StockQuantity : 0
            })
            .Where(pm => pm.UnitsSold > 0)
            .OrderByDescending(pm => pm.GrossProfit)
            .ToList();

                TotalSpent = cg.Sales.Sum(s => s.Quantity * s.UnitPrice * (1 - s.Discount)),
                PreferredCategories = cg.Sales
                    .Join(products, s => s.ProductId, p => p.Id, (s, p) => p.Category)
                    .GroupBy(cat => cat)
                    .OrderByDescending(g => g.Count())
                    .Take(3)
                    .Select(g => g.Key)
                    .ToList(),
                Pricesensitivity = CalculatePriceSensitivity(cg.Sales),
                LoyaltyScore = CalculateLoyaltyScore(cg.Customer, cg.Sales),
                ChurnRisk = CalculateChurnRisk(cg.Customer, cg.Sales)
            })
            .Where(bp => bp.PurchaseFrequency > 0)
            .ToList();

        var segmentBehaviors = behaviorPatterns
            .GroupBy(bp => bp.Customer.Segment)
            .Select(g => new SegmentBehavior
            {
                Segment = g.Key,
                AveragePurchaseFrequency = g.Average(bp => bp.PurchaseFrequency),
                AverageProductDiversity = g.Average(bp => bp.ProductDiversity),
                AverageLoyaltyScore = g.Average(bp => bp.LoyaltyScore),
                HighRiskCustomers = g.Count(bp => bp.ChurnRisk > 0.7)
            })
            .ToList();

        return new CustomerBehaviorAnalysis
        {
            BehaviorPatterns = behaviorPatterns,
            SegmentBehaviors = segmentBehaviors,
            HighValueCustomers = behaviorPatterns
                .OrderByDescending(bp => bp.TotalSpent)
                .Take(50)
                .ToList(),
            AtRiskCustomers = behaviorPatterns
                .Where(bp => bp.ChurnRisk > 0.7)
                .OrderByDescending(bp => bp.TotalSpent)
                .ToList()
        };

    }

    private double CalculateQuarterlyGrowthRate(int year, int quarter)

    private double CalculateSeasonalityIndex(string category)
    {
        // 季節性指数の計算（簡略化）
        return category switch
        {
            "衣料品" => 1.3,
            "電子機器" => 0.9,
            _ => 1.0
        };

    private double CalculateAverageDaysBetweenPurchases(List<Sale> sales)
    {
        if (sales.Count <= 1) return 0;

        var sortedDates = sales.OrderBy(s => s.SaleDate).Select(s => s.SaleDate).ToList();
        var intervals = sortedDates.Zip(sortedDates.Skip(1), (first, second) => (second - first).Days);
        return intervals.Any() ? intervals.Average() : 0;
    }

    private double CalculatePriceSensitivity(List<Sale> sales)
    {
        // 価格感応度の計算（割引の平均利用率）
        return sales.Any() ? sales.Average(s => (double)s.Discount) : 0;
    }

    private double CalculateLoyaltyScore(Customer customer, List<Sale> sales)
    {
        var daysSinceRegistration = (DateTime.Now - customer.RegistrationDate).Days;
        var purchaseDays = sales.Select(s => s.SaleDate.Date).Distinct().Count();
        var loyaltyBase = daysSinceRegistration > 0 ? (double)purchaseDays / daysSinceRegistration : 0;

        // 購入頻度と継続期間を考慮
        var frequencyBonus = Math.Min(sales.Count / 10.0, 1.0);
        return Math.Min(loyaltyBase + frequencyBonus, 1.0);
    }

    private double CalculateChurnRisk(Customer customer, List<Sale> sales)
    {
        if (!sales.Any()) return 1.0;

        var daysSinceLastPurchase = (DateTime.Now - sales.Max(s => s.SaleDate)).Days;
        var averageDaysBetweenPurchases = CalculateAverageDaysBetweenPurchases(sales);

        if (averageDaysBetweenPurchases == 0) return 0.5;

        var riskRatio = daysSinceLastPurchase / averageDaysBetweenPurchases;
        return Math.Min(riskRatio / 3.0, 1.0); // 平均間隔の3倍以上で高リスク
    }

    public class LinqPerformanceBenchmark
{
    public static void RunBenchmarkSuite()
    {
        Console.WriteLine("=== LINQ パフォーマンステストスイート ===\n");

        var testData = GenerateTestData(100_000);

        BenchmarkAggregationOperations(testData);

        // 3. 結合操作のベンチマーク
        BenchmarkJoinOperations(testData);

        // 4. カスタム拡張メソッドのベンチマーク
        BenchmarkCustomExtensions(testData);
    }

    private static List<TestEntity> GenerateTestData(int count)
    {
        var random = new Random();
        return Enumerable.Range(1, count)
            .Select(i => new TestEntity
            {
                Id = i,
                Value = random.Next(1, 1000),
                Category = $"Category{random.Next(1, 11)}",
                IsActive = random.NextDouble() > 0.3,
                CreatedDate = DateTime.Now.AddDays(-random.Next(0, 365))
            })
            .ToList();
    }

    private static void BenchmarkFilteringOperations(List<TestEntity> data)
    {
        Console.WriteLine("--- フィルタリング操作ベンチマーク ---");

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // Where with multiple conditions
        stopwatch.Restart();
        var result1 = data.Where(x => x.IsActive && x.Value > 500 && x.Category.StartsWith("Category1")).ToList();
        stopwatch.Stop();
        Console.WriteLine($"複数条件Where: {stopwatch.ElapsedMilliseconds}ms ({result1.Count} 件)");

        // Where with WhereIf extension
        stopwatch.Restart();
        var result2 = data
            .WhereIf(true, x => x.IsActive)
            .WhereIf(true, x => x.Value > 500)
            .WhereIf(true, x => x.Category.StartsWith("Category1"))
            .ToList();
        stopwatch.Stop();
        Console.WriteLine($"WhereIf拡張メソッド: {stopwatch.ElapsedMilliseconds}ms ({result2.Count} 件)");

        // Take vs First
        stopwatch.Restart();
        var result3 = data.Where(x => x.Value > 900).Take(1).ToList();
        stopwatch.Stop();
        var takeTime = stopwatch.ElapsedMilliseconds;

        stopwatch.Restart();
        var result4 = data.Where(x => x.Value > 900).FirstOrDefault();
        stopwatch.Stop();
        var firstTime = stopwatch.ElapsedMilliseconds;

        Console.WriteLine($"Take(1): {takeTime}ms vs FirstOrDefault: {firstTime}ms");
        Console.WriteLine();
    }

    private static void BenchmarkAggregationOperations(List<TestEntity> data)
    {
        Console.WriteLine("--- 集約操作ベンチマーク ---");

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // Multiple aggregations - inefficient
        stopwatch.Restart();
        var count1 = data.Count();
        var sum1 = data.Sum(x => x.Value);
        var avg1 = data.Average(x => x.Value);
        var max1 = data.Max(x => x.Value);
        stopwatch.Stop();
        Console.WriteLine($"個別集約操作: {stopwatch.ElapsedMilliseconds}ms");

        // Single pass aggregation - efficient
        stopwatch.Restart();
        var stats = data.GetStatistics(x => x.Value);
        stopwatch.Stop();
        Console.WriteLine($"一括統計取得: {stopwatch.ElapsedMilliseconds}ms");

        Console.WriteLine($"結果: Count={stats.Count}, Sum={stats.Sum}, Avg={stats.Average:F2}, Max={stats.Max}");
        Console.WriteLine();
    }

    private static void BenchmarkJoinOperations(List<TestEntity> data)
    {
        Console.WriteLine("--- 結合操作ベンチマーク ---");

        var categories = data.Select(x => x.Category).Distinct()
            .Select(cat => new { Category = cat, Description = $"Description for {cat}" })
            .ToList();

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // LINQ Join
        stopwatch.Restart();
        var joined1 = data.Join(categories, 
            d => d.Category, 
            c => c.Category, 
            (d, c) => new { d.Id, d.Value, c.Description })
            .ToList();
        stopwatch.Stop();
        Console.WriteLine($"LINQ Join: {stopwatch.ElapsedMilliseconds}ms ({joined1.Count} 件)");

        // Dictionary lookup (faster for large datasets)
        stopwatch.Restart();
        var categoryDict = categories.ToDictionary(c => c.Category, c => c.Description);
        var joined2 = data.Select(d => new 
        { 
            d.Id, 
            d.Value, 
            Description = categoryDict.TryGetValue(d.Category, out var desc) ? desc : "Unknown" 
        }).ToList();
        stopwatch.Stop();
        Console.WriteLine($"Dictionary Lookup: {stopwatch.ElapsedMilliseconds}ms ({joined2.Count} 件)");
        Console.WriteLine();
    }

    private static void BenchmarkCustomExtensions(List<TestEntity> data)
    {
        Console.WriteLine("--- カスタム拡張メソッドベンチマーク ---");

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // Batch processing
        stopwatch.Restart();
        var batches = data.Batch(1000).ToList();
        stopwatch.Stop();
        Console.WriteLine($"Batch処理 (1000件ずつ): {stopwatch.ElapsedMilliseconds}ms ({batches.Count} バッチ)");

        // DistinctBy
        stopwatch.Restart();
        var distinct1 = data.DistinctBy(x => x.Category).ToList();
        stopwatch.Stop();
        Console.WriteLine($"DistinctBy: {stopwatch.ElapsedMilliseconds}ms ({distinct1.Count} 件)");

        // Partition
        stopwatch.Restart();
        var (active, inactive) = data.PartitionBy(x => x.IsActive);
        var activeCount = active.Count();
        var inactiveCount = inactive.Count();
        stopwatch.Stop();
        Console.WriteLine($"PartitionBy: {stopwatch.ElapsedMilliseconds}ms (Active: {activeCount}, Inactive: {inactiveCount})");
        Console.WriteLine();
    }
