    1.2 条件付きクエリの動的構築

    実際のアプリケーションでは、ユーザーの入力や設定に基づいて動的にクエリを構築する必要があります。

    public class DynamicQueryBuilder
{
    public class SearchCriteria
    {
        public string NameFilter { get; set; }
        public string DepartmentFilter { get; set; }
        public decimal? MinSalary { get; set; }
        public decimal? MaxSalary { get; set; }
        public int? MinAge { get; set; }
        public int? MaxAge { get; set; }
        public List&lt;string&gt; RequiredSkills { get; set; } = new List&lt;string&gt;();
        public DateTime? HiredAfter { get; set; }
        public DateTime? HiredBefore { get; set; }
        public string SortBy { get; set; } = &quot;Name&quot;;
        public bool SortDescending { get; set; } = false;
    }

    private List&lt;Employee&gt; employees;

    public IEnumerable&lt;Employee&gt; SearchEmployees(SearchCriteria criteria)
    {
        IQueryable&lt;Employee&gt; query = employees.AsQueryable();

        // 名前フィルター（部分一致）
        if (!string.IsNullOrWhiteSpace(criteria.NameFilter))
        {
            query = query.Where(e =&gt; e.Name.Contains(criteria.NameFilter, StringComparison.OrdinalIgnoreCase));
        }

        // 部署フィルター
        if (!string.IsNullOrWhiteSpace(criteria.DepartmentFilter))
        {
            query = query.Where(e =&gt; e.Department == criteria.DepartmentFilter);
        }

        // 給与範囲フィルター
        if (criteria.MinSalary.HasValue)
        {
            query = query.Where(e =&gt; e.Salary &gt;= criteria.MinSalary.Value);
        }

        if (criteria.MaxSalary.HasValue)
        {
            query = query.Where(e =&gt; e.Salary &lt;= criteria.MaxSalary.Value);
        }

        // 年齢範囲フィルター
        if (criteria.MinAge.HasValue)
        {
            query = query.Where(e =&gt; e.Age &gt;= criteria.MinAge.Value);
        }

        if (criteria.MaxAge.HasValue)
        {
            query = query.Where(e =&gt; e.Age &lt;= criteria.MaxAge.Value);
        }

        // 必須スキルフィルター（すべてのスキルを持っている必要がある）
        if (criteria.RequiredSkills.Any())
        {
            foreach (var skill in criteria.RequiredSkills)
            {
                query = query.Where(e =&gt; e.Skills.Contains(skill, StringComparer.OrdinalIgnoreCase));
            }
        }

        // 入社日フィルター
        if (criteria.HiredAfter.HasValue)
        {
            query = query.Where(e =&gt; e.HireDate &gt;= criteria.HiredAfter.Value);
        }

        if (criteria.HiredBefore.HasValue)
        {
            query = query.Where(e =&gt; e.HireDate &lt;= criteria.HiredBefore.Value);
        }

        // 動的ソート
        query = ApplySorting(query, criteria.SortBy, criteria.SortDescending);

        return query.ToList();
    }

    private IQueryable&lt;Employee&gt; ApplySorting(IQueryable&lt;Employee&gt; query, string sortBy, bool descending)
    {
        Expression&lt;Func&lt;Employee, object&gt;&gt; keySelector = sortBy?.ToLower() switch
        {
            &quot;name&quot; =&gt; e =&gt; e.Name,
            &quot;department&quot; =&gt; e =&gt; e.Department,
            &quot;salary&quot; =&gt; e =&gt; e.Salary,
            &quot;age&quot; =&gt; e =&gt; e.Age,
            &quot;hiredate&quot; =&gt; e =&gt; e.HireDate,
            &quot;position&quot; =&gt; e =&gt; e.Position,
            _ =&gt; e =&gt; e.Name
        };

        return descending 
            ? query.OrderByDescending(keySelector)
            : query.OrderBy(keySelector);
    }

    // 使用例
    public void DemonstrateDynamicQueries()
    {
        Console.WriteLine(&quot;=== 動的クエリの例 ===&quot;);

        // 例1: 開発部門で C# スキルを持つ高給取り
        var criteria1 = new SearchCriteria
        {
            DepartmentFilter = &quot;開発&quot;,
            RequiredSkills = new List&lt;string&gt; { &quot;C#&quot; },
            MinSalary = 550000,
            SortBy = &quot;Salary&quot;,
            SortDescending = true
        };

        var result1 = SearchEmployees(criteria1);
        Console.WriteLine(&quot;開発部門のC#エンジニア（高給取り順）:&quot;);
        foreach (var emp in result1)
        {
            Console.WriteLine($&quot;  {emp.Name} - ¥{emp.Salary:N0}&quot;);
        }

        // 例2: 若手エンジニア（年齢・入社日条件）
        var criteria2 = new SearchCriteria
        {
            MaxAge = 30,
            HiredAfter = new DateTime(2020, 1, 1),
            SortBy = &quot;Age&quot;
        };

        var result2 = SearchEmployees(criteria2);
        Console.WriteLine(&quot;\n若手社員（2020年以降入社、30歳以下）:&quot;);
        foreach (var emp in result2)
        {
            Console.WriteLine($&quot;  {emp.Name} - {emp.Age}歳 (入社: {emp.HireDate:yyyy/MM/dd})&quot;);
        }
    }

    2.1 効率的なLINQ操作の選択

    LINQは便利ですが、使い方によってはパフォーマンスに大きく影響します。適切な操作を選択することで劇的に性能を改善できます。

    public class LinqPerformanceOptimization
{
    private List&lt;int&gt; largeDataSet;
    private List&lt;Employee&gt; employees;

    public LinqPerformanceOptimization()
    {
        // 大きなデータセットを生成（パフォーマンステスト用）
        largeDataSet = Enumerable.Range(1, 1_000_000).ToList();
        employees = GenerateEmployees(100_000);
    }

    private List&lt;Employee&gt; GenerateEmployees(int count)

    public static class AdvancedLinqExtensions
{
    // バッチ処理（指定サイズでグループ化）
    public static IEnumerable&lt;IEnumerable&lt;T&gt;&gt; Batch&lt;T&gt;(this IEnumerable&lt;T&gt; source, int batchSize)
    {
        if (batchSize &lt;= 0) throw new ArgumentException(&quot;Batch size must be positive&quot;, nameof(batchSize));

        var batch = new List&lt;T&gt;(batchSize);
        foreach (var item in source)
        {
            batch.Add(item);
            if (batch.Count == batchSize)
            {
                yield return batch;
                batch = new List&lt;T&gt;(batchSize);
            }
        }

        if (batch.Count &gt; 0)
        {
            yield return batch;
        }
    }

    // 移動平均の計算
    public static IEnumerable&lt;decimal&gt; MovingAverage(this IEnumerable&lt;decimal&gt; source, int windowSize)
    {
        if (windowSize &lt;= 0) throw new ArgumentException(&quot;Window size must be positive&quot;, nameof(windowSize));

        var window = new Queue&lt;decimal&gt;();
        decimal sum = 0;

        foreach (var value in source)
        {
            window.Enqueue(value);
            sum += value;

            if (window.Count &gt; windowSize)
            {
                sum -= window.Dequeue();
            }

            if (window.Count == windowSize)
            {
                yield return sum / windowSize;
            }
        }
    }

    // 条件に基づく分割
    public static (IEnumerable&lt;T&gt; matching, IEnumerable&lt;T&gt; nonMatching) PartitionBy&lt;T&gt;(
        this IEnumerable&lt;T&gt; source, Func&lt;T, bool&gt; predicate)
    {
        var sourceList = source.ToList(); // 複数回列挙を避ける
        return (sourceList.Where(predicate), sourceList.Where(x =&gt; !predicate(x)));
    }

    // 統計情報の一括取得
    public static StatisticsResult&lt;T&gt; GetStatistics&lt;T&gt;(this IEnumerable&lt;T&gt; source, Func&lt;T, decimal&gt; selector)
    {
        var values = source.Select(selector).ToList();

        if (!values.Any())
        {
            return new StatisticsResult&lt;T&gt;
            {
                Count = 0,
                Sum = 0,
                Average = 0,
                Min = 0,
                Max = 0,
                Median = 0,
                StandardDeviation = 0
            };
        }

        var sorted = values.OrderBy(x =&gt; x).ToList();
        var count = values.Count;
        var sum = values.Sum();
        var average = values.Average();
        var min = values.Min();
        var max = values.Max();

        var median = count % 2 == 0
            ? (sorted[count / 2 - 1] + sorted[count / 2]) / 2
            : sorted[count / 2];

        var variance = values.Sum(v =&gt; Math.Pow((double)(v - average), 2)) / count;
        var standardDeviation = (decimal)Math.Sqrt(variance);

        return new StatisticsResult&lt;T&gt;
        {
            Count = count,
            Sum = sum,
            Average = average,
            Min = min,
            Max = max,
            Median = median,
            StandardDeviation = standardDeviation
        };
    }

    // 階層構造のフラット化
    public static IEnumerable&lt;T&gt; Flatten&lt;T&gt;(this IEnumerable&lt;T&gt; source, Func&lt;T, IEnumerable&lt;T&gt;&gt; childSelector)
    {
        var stack = new Stack&lt;T&gt;(source.Reverse());

        while (stack.Count &gt; 0)
        {
            var current = stack.Pop();
            yield return current;

            var children = childSelector(current);
            if (children != null)
            {
                foreach (var child in children.Reverse())
                {
                    stack.Push(child);
                }
            }
        }
    }

    // 文字列結合の高度版
    public static string JoinWith&lt;T&gt;(this IEnumerable&lt;T&gt; source, 
        string separator, 
        Func&lt;T, string&gt; selector = null,
        string prefix = &quot;&quot;,
        string suffix = &quot;&quot;)
    {
        selector ??= item =&gt; item?.ToString() ?? &quot;&quot;;
        var result = string.Join(separator, source.Select(selector));
        return $&quot;{prefix}{result}{suffix}&quot;;
    }

    private double CalculateAttritionRisk(Employee employee)
    {
        var experience = (DateTime.Now - employee.HireDate).Days / 365.25;
        var salaryPercentile = employees.Count(e =&gt; e.Salary &lt;= employee.Salary) / (double)employees.Count;

        // 簡略化されたリスクスコア計算
        var riskScore = 0.0;

        if (experience &lt; 1) riskScore += 0.3; // 新人は退職リスクが高い
        if (experience &gt; 5) riskScore += 0.2; // ベテランも転職を考えやすい
        if (salaryPercentile &lt; 0.3) riskScore += 0.4; // 低給与はリスク
        if (employee.Age &lt; 30) riskScore += 0.2; // 若手は転職しやすい

        return Math.Min(riskScore, 1.0);
    }

        return 0.15; // ダミー値
    }

    private double CalculateSeasonalityIndex(string category)
    {
        // 季節性指数の計算（簡略化）
        return category switch
        {
            &quot;衣料品&quot; =&gt; 1.3,
            &quot;電子機器&quot; =&gt; 0.9,
            _ =&gt; 1.0
        };
    }

    private double CalculateAverageDaysBetweenPurchases(List&lt;Sale&gt; sales)
    {
        if (sales.Count &lt;= 1) return 0;

        var sortedDates = sales.OrderBy(s =&gt; s.SaleDate).Select(s =&gt; s.SaleDate).ToList();
        var intervals = sortedDates.Zip(sortedDates.Skip(1), (first, second) =&gt; (second - first).Days);
        return intervals.Any() ? intervals.Average() : 0;
    }

    private double CalculatePriceSensitivity(List&lt;Sale&gt; sales)
    {
        // 価格感応度の計算（割引の平均利用率）
        return sales.Any() ? sales.Average(s =&gt; (double)s.Discount) : 0;
    }

    private double CalculateLoyaltyScore(Customer customer, List&lt;Sale&gt; sales)
    {
        var daysSinceRegistration = (DateTime.Now - customer.RegistrationDate).Days;
        var purchaseDays = sales.Select(s =&gt; s.SaleDate.Date).Distinct().Count();
        var loyaltyBase = daysSinceRegistration &gt; 0 ? (double)purchaseDays / daysSinceRegistration : 0;

        // 購入頻度と継続期間を考慮
        var frequencyBonus = Math.Min(sales.Count / 10.0, 1.0);
        return Math.Min(loyaltyBase + frequencyBonus, 1.0);
    }

    private double CalculateChurnRisk(Customer customer, List&lt;Sale&gt; sales)
    {
        if (!sales.Any()) return 1.0;

        var daysSinceLastPurchase = (DateTime.Now - sales.Max(s =&gt; s.SaleDate)).Days;
        var averageDaysBetweenPurchases = CalculateAverageDaysBetweenPurchases(sales);

        if (averageDaysBetweenPurchases == 0) return 0.5;

        var riskRatio = daysSinceLastPurchase / averageDaysBetweenPurchases;
        return Math.Min(riskRatio / 3.0, 1.0); // 平均間隔の3倍以上で高リスク
    }

    public class LinqPerformanceBenchmark
{
    public static void RunBenchmarkSuite()
    {
        Console.WriteLine(&quot;=== LINQ パフォーマンステストスイート ===\n&quot;);

        var testData = GenerateTestData(100_000);

        // 1. フィルタリング操作のベンチマーク
        BenchmarkFilteringOperations(testData);

        // 2. 集約操作のベンチマーク
        BenchmarkAggregationOperations(testData);

        // 3. 結合操作のベンチマーク
        BenchmarkJoinOperations(testData);

        // 4. カスタム拡張メソッドのベンチマーク
        BenchmarkCustomExtensions(testData);
    }

    private static List&lt;TestEntity&gt; GenerateTestData(int count)
    {
        var random = new Random();
        return Enumerable.Range(1, count)
            .Select(i =&gt; new TestEntity
            {
                Id = i,
                Value = random.Next(1, 1000),
                Category = $&quot;Category{random.Next(1, 11)}&quot;,
                IsActive = random.NextDouble() &gt; 0.3,
                CreatedDate = DateTime.Now.AddDays(-random.Next(0, 365))
            })
            .ToList();
    }

    private static void BenchmarkFilteringOperations(List&lt;TestEntity&gt; data)
    {
        Console.WriteLine(&quot;--- フィルタリング操作ベンチマーク ---&quot;);

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // Where with multiple conditions
        stopwatch.Restart();
        var result1 = data.Where(x =&gt; x.IsActive &amp;&amp; x.Value &gt; 500 &amp;&amp; x.Category.StartsWith(&quot;Category1&quot;)).ToList();
        stopwatch.Stop();
        Console.WriteLine($&quot;複数条件Where: {stopwatch.ElapsedMilliseconds}ms ({result1.Count} 件)&quot;);

        // Where with WhereIf extension
        stopwatch.Restart();
        var result2 = data
            .WhereIf(true, x =&gt; x.IsActive)
            .WhereIf(true, x =&gt; x.Value &gt; 500)
            .WhereIf(true, x =&gt; x.Category.StartsWith(&quot;Category1&quot;))
            .ToList();
        stopwatch.Stop();
        Console.WriteLine($&quot;WhereIf拡張メソッド: {stopwatch.ElapsedMilliseconds}ms ({result2.Count} 件)&quot;);

        // Take vs First
        stopwatch.Restart();
        var result3 = data.Where(x =&gt; x.Value &gt; 900).Take(1).ToList();
        stopwatch.Stop();
        var takeTime = stopwatch.ElapsedMilliseconds;

        stopwatch.Restart();
        var result4 = data.Where(x =&gt; x.Value &gt; 900).FirstOrDefault();
        stopwatch.Stop();
        var firstTime = stopwatch.ElapsedMilliseconds;

        Console.WriteLine($&quot;Take(1): {takeTime}ms vs FirstOrDefault: {firstTime}ms&quot;);
        Console.WriteLine();
    }

    private static void BenchmarkAggregationOperations(List&lt;TestEntity&gt; data)
    {
        Console.WriteLine(&quot;--- 集約操作ベンチマーク ---&quot;);

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // Multiple aggregations - inefficient
        stopwatch.Restart();
        var count1 = data.Count();
        var sum1 = data.Sum(x =&gt; x.Value);
        var avg1 = data.Average(x =&gt; x.Value);
        var max1 = data.Max(x =&gt; x.Value);
        stopwatch.Stop();
        Console.WriteLine($&quot;個別集約操作: {stopwatch.ElapsedMilliseconds}ms&quot;);

        // Single pass aggregation - efficient
        stopwatch.Restart();
        var stats = data.GetStatistics(x =&gt; x.Value);
        stopwatch.Stop();
        Console.WriteLine($&quot;一括統計取得: {stopwatch.ElapsedMilliseconds}ms&quot;);

        Console.WriteLine($&quot;結果: Count={stats.Count}, Sum={stats.Sum}, Avg={stats.Average:F2}, Max={stats.Max}&quot;);
        Console.WriteLine();
    }

    private static void BenchmarkJoinOperations(List&lt;TestEntity&gt; data)
    {
        Console.WriteLine(&quot;--- 結合操作ベンチマーク ---&quot;);

        var categories = data.Select(x =&gt; x.Category).Distinct()
            .Select(cat =&gt; new { Category = cat, Description = $&quot;Description for {cat}&quot; })
            .ToList();

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // LINQ Join
        stopwatch.Restart();
        var joined1 = data.Join(categories, 
            d =&gt; d.Category, 
            c =&gt; c.Category, 
            (d, c) =&gt; new { d.Id, d.Value, c.Description })
            .ToList();
        stopwatch.Stop();
        Console.WriteLine($&quot;LINQ Join: {stopwatch.ElapsedMilliseconds}ms ({joined1.Count} 件)&quot;);

        // Dictionary lookup (faster for large datasets)
        stopwatch.Restart();
        var categoryDict = categories.ToDictionary(c =&gt; c.Category, c =&gt; c.Description);
        var joined2 = data.Select(d =&gt; new 
        { 
            d.Id, 
            d.Value, 
            Description = categoryDict.TryGetValue(d.Category, out var desc) ? desc : &quot;Unknown&quot; 
        }).ToList();
        stopwatch.Stop();
        Console.WriteLine($&quot;Dictionary Lookup: {stopwatch.ElapsedMilliseconds}ms ({joined2.Count} 件)&quot;);
        Console.WriteLine();
    }

    private static void BenchmarkCustomExtensions(List&lt;TestEntity&gt; data)
    {
        Console.WriteLine(&quot;--- カスタム拡張メソッドベンチマーク ---&quot;);

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();

        // Batch processing
        stopwatch.Restart();
        var batches = data.Batch(1000).ToList();
        stopwatch.Stop();
        Console.WriteLine($&quot;Batch処理 (1000件ずつ): {stopwatch.ElapsedMilliseconds}ms ({batches.Count} バッチ)&quot;);

        // DistinctBy
        stopwatch.Restart();
        var distinct1 = data.DistinctBy(x =&gt; x.Category).ToList();
        stopwatch.Stop();
        Console.WriteLine($&quot;DistinctBy: {stopwatch.ElapsedMilliseconds}ms ({distinct1.Count} 件)&quot;);

        // Partition
        stopwatch.Restart();
        var (active, inactive) = data.PartitionBy(x =&gt; x.IsActive);
        var activeCount = active.Count();
        var inactiveCount = inactive.Count();
        stopwatch.Stop();
        Console.WriteLine($&quot;PartitionBy: {stopwatch.ElapsedMilliseconds}ms (Active: {activeCount}, Inactive: {inactiveCount})&quot;);
        Console.WriteLine();
    }
