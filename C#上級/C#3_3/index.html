    はじめに - デザインパターンとは

    デザインパターンは、ソフトウェア設計における「よくある問題に対する再利用可能な解決策」です。料理のレシピのように、特定の状況で使える「設計の型」を提供します。

    なぜデザインパターンが重要なのか

    • コードの再利用性: 似たような問題を効率的に解決
    • 保守性の向上: 標準的な構造で理解しやすい
    • 開発チームでの共通理解: パターン名で設計意図を伝達
    • 拡張性: 変更に強い柔軟な設計

    今回学習する5つのパターン： 1. Singleton - 唯一のインスタンスを保証 2. Factory - オブジェクト生成の複雑さを隠蔽 3. Observer - オブジェクト間の疎結合な通知 4. Strategy - アルゴリズムの動的な切り替え 5. Command - 操作のカプセル化

    1. Singletonパターン - 唯一のインスタンスを保証

    パターンの目的

    「システム全体で1つだけ存在すべきオブジェクト」を実現します。

    実際の使用例

    • データベース接続管理
    • ログ出力システム
    • アプリケーション設定
    • キャッシュシステム

    基本的な実装（スレッドセーフ版）

    using System;

// スレッドセーフなSingleton実装
public sealed class DatabaseConnection
{
    // Lazyを使用してスレッドセーフかつ遅延初期化を実現
    private static readonly Lazy&lt;DatabaseConnection&gt; _instance = 
        new Lazy&lt;DatabaseConnection&gt;(() =&gt; new DatabaseConnection());

    // privateコンストラクタで外部からのインスタンス化を防ぐ
    private DatabaseConnection()
    {
        ConnectionString = &quot;Server=localhost;Database=MyApp;&quot;;
        Console.WriteLine(&quot;データベース接続を初期化しました&quot;);
    }

    // 唯一のアクセスポイント
    public static DatabaseConnection Instance =&gt; _instance.Value;

    public string ConnectionString { get; private set; }

    public void ExecuteQuery(string query)
    {
        Console.WriteLine($&quot;クエリ実行: {query}&quot;);
    }
}

    実用的な例：アプリケーション設定管理

    public sealed class AppSettings
{
    private static readonly Lazy&lt;AppSettings&gt; _instance = 
        new Lazy&lt;AppSettings&gt;(() =&gt; new AppSettings());

    private readonly Dictionary&lt;string, string&gt; _settings;

    private AppSettings()
    {
        // 設定ファイルから読み込み（例）
        _settings = new Dictionary&lt;string, string&gt;
        {
            [&quot;ApiUrl&quot;] = &quot;https://api.example.com&quot;,
            [&quot;MaxRetries&quot;] = &quot;3&quot;,
            [&quot;TimeoutSeconds&quot;] = &quot;30&quot;,
            [&quot;LogLevel&quot;] = &quot;Info&quot;
        };
        Console.WriteLine(&quot;アプリケーション設定を読み込みました&quot;);
    }

    public static AppSettings Instance =&gt; _instance.Value;

    public string GetSetting(string key)
    {
        return _settings.TryGetValue(key, out var value) ? value : string.Empty;
    }

    public T GetSetting&lt;T&gt;(string key) where T : IConvertible
    {
        var value = GetSetting(key);
        if (string.IsNullOrEmpty(value))
            return default(T);

        try
        {
            return (T)Convert.ChangeType(value, typeof(T));
        }
        catch
        {
            return default(T);
        }
    }
}

    使用例とテスト

    class SingletonExample
{
    static void Main()
    {
        Console.WriteLine(&quot;=== Singletonパターンのテスト ===&quot;);

        // 複数回取得しても同じインスタンス
        var db1 = DatabaseConnection.Instance;
        var db2 = DatabaseConnection.Instance;

        Console.WriteLine($&quot;同じインスタンス: {ReferenceEquals(db1, db2)}&quot;);

        db1.ExecuteQuery(&quot;SELECT * FROM Users&quot;);
        db2.ExecuteQuery(&quot;SELECT * FROM Products&quot;);

        // 設定の取得テスト
        var settings = AppSettings.Instance;
        var apiUrl = settings.GetSetting(&quot;ApiUrl&quot;);
        var maxRetries = settings.GetSetting&lt;int&gt;(&quot;MaxRetries&quot;);

        Console.WriteLine($&quot;API URL: {apiUrl}&quot;);
        Console.WriteLine($&quot;最大リトライ回数: {maxRetries}&quot;);

        // 同じインスタンスかチェック
        var settings2 = AppSettings.Instance;
        Console.WriteLine($&quot;設定も同じインスタンス: {ReferenceEquals(settings, settings2)}&quot;);
    }
}

    2. Factoryパターン - オブジェクト生成の複雑さを隠蔽

    パターンの目的

    「どのクラスのインスタンスを作るか」をクライアントから隠し、オブジェクト生成を一元管理します。

    Factory Methodパターンの実装

    // 製品の抽象クラス
public abstract class Logger
{
    public abstract void Log(string message);
    public abstract void LogError(string message);
}

// 具象製品：ファイルログ
public class FileLogger : Logger
{
    private readonly string _filePath;

    public FileLogger(string filePath)
    {
        _filePath = filePath;
        Console.WriteLine($&quot;ファイルログ初期化: {_filePath}&quot;);
    }

    public override void Log(string message)
    {
        var logEntry = $&quot;[{DateTime.Now:yyyy-MM-dd HH:mm:ss}] INFO: {message}&quot;;
        File.AppendAllText(_filePath, logEntry + Environment.NewLine);
        Console.WriteLine($&quot;ファイルに記録: {message}&quot;);
    }

    public override void LogError(string message)
    {
        var logEntry = $&quot;[{DateTime.Now:yyyy-MM-dd HH:mm:ss}] ERROR: {message}&quot;;
        File.AppendAllText(_filePath, logEntry + Environment.NewLine);
        Console.WriteLine($&quot;エラーをファイルに記録: {message}&quot;);
    }
}

// 具象製品：コンソールログ
public class ConsoleLogger : Logger
{
    public ConsoleLogger()
    {
        Console.WriteLine(&quot;コンソールログ初期化&quot;);
    }

    public override void Log(string message)
    {
        Console.ForegroundColor = ConsoleColor.Green;
        Console.WriteLine($&quot;[{DateTime.Now:HH:mm:ss}] INFO: {message}&quot;);
        Console.ResetColor();
    }

    public override void LogError(string message)
    {
        Console.ForegroundColor = ConsoleColor.Red;
        Console.WriteLine($&quot;[{DateTime.Now:HH:mm:ss}] ERROR: {message}&quot;);
        Console.ResetColor();
    }
}

// 具象製品：データベースログ
public class DatabaseLogger : Logger
{
    private readonly string _connectionString;

    public DatabaseLogger(string connectionString)
    {
        _connectionString = connectionString;
        Console.WriteLine(&quot;データベースログ初期化&quot;);
    }

    public override void Log(string message)
    {
        // 実際のDBアクセスの代わりにシミュレート
        Console.WriteLine($&quot;DB保存: {message}&quot;);
    }

    public override void LogError(string message)
    {
        Console.WriteLine($&quot;DBにエラー保存: {message}&quot;);
    }
}

// Factory（生成の責任を持つ）
public static class LoggerFactory
{
    public static Logger CreateLogger(string type, string parameter = null)
    {
        return type.ToLower() switch
        {
            &quot;file&quot; =&gt; new FileLogger(parameter ?? &quot;app.log&quot;),
            &quot;console&quot; =&gt; new ConsoleLogger(),
            &quot;database&quot; =&gt; new DatabaseLogger(parameter ?? &quot;DefaultConnection&quot;),
            _ =&gt; throw new ArgumentException($&quot;不明なログタイプ: {type}&quot;)
        };
    }

    // より高度なFactory：環境に基づく生成
    public static Logger CreateLoggerForEnvironment(string environment)
    {
        return environment.ToLower() switch
        {
            &quot;development&quot; =&gt; new ConsoleLogger(),
            &quot;testing&quot; =&gt; new FileLogger(&quot;test.log&quot;),
            &quot;production&quot; =&gt; new DatabaseLogger(&quot;ProductionDB&quot;),
            _ =&gt; new ConsoleLogger() // デフォルト
        };
    }
}

    3. Observerパターン - オブジェクト間の疎結合な通知

    パターンの目的

    「あるオブジェクトの状態変化を、複数の関連オブジェクトに自動的に通知する」仕組みを提供します。

    C#のイベント機能を使った実装

    // Subject（監視される側）
public class TemperatureSensor
{
    private double _temperature;

    // C#のイベント機能を使用
    public event Action&lt;double&gt; TemperatureChanged;
    public event Action&lt;double&gt; HighTemperatureAlert;
    public event Action&lt;double&gt; LowTemperatureAlert;

    public double Temperature
    {
        get =&gt; _temperature;
        set
        {
            if (Math.Abs(_temperature - value) &gt; 0.1) // 0.1度以上の変化で通知
            {
                var oldTemperature = _temperature;
                _temperature = value;
                OnTemperatureChanged(value);
                CheckTemperatureAlerts(value);

                Console.WriteLine($&quot;温度変化: {oldTemperature:F1}℃ → {value:F1}℃&quot;);
            }
        }
    }

    protected virtual void OnTemperatureChanged(double newTemperature)
    {
        TemperatureChanged?.Invoke(newTemperature);
    }

    private void CheckTemperatureAlerts(double temperature)
    {
        if (temperature &gt; 30)
        {
            HighTemperatureAlert?.Invoke(temperature);
        }
        else if (temperature &lt; 5)
        {
            LowTemperatureAlert?.Invoke(temperature);
        }
    }
}

// Observer1：温度表示
public class TemperatureDisplay
{
    private readonly string _name;

    public TemperatureDisplay(string name)
    {
        _name = name;
    }

    public void Subscribe(TemperatureSensor sensor)
    {
        sensor.TemperatureChanged += OnTemperatureChanged;
        sensor.HighTemperatureAlert += OnHighTemperature;
        sensor.LowTemperatureAlert += OnLowTemperature;
        Console.WriteLine($&quot;{_name}が温度センサーを監視開始&quot;);
    }

    public void Unsubscribe(TemperatureSensor sensor)
    {
        sensor.TemperatureChanged -= OnTemperatureChanged;
        sensor.HighTemperatureAlert -= OnHighTemperature;
        sensor.LowTemperatureAlert -= OnLowTemperature;
        Console.WriteLine($&quot;{_name}が温度センサーの監視を終了&quot;);
    }

    private void OnTemperatureChanged(double temperature)
    {
        Console.WriteLine($&quot;  📺 {_name}: 現在温度 {temperature:F1}℃&quot;);
    }

    private void OnHighTemperature(double temperature)
    {
        Console.WriteLine($&quot;  🔥 {_name}: 高温警告！ {temperature:F1}℃&quot;);
    }

    private void OnLowTemperature(double temperature)
    {
        Console.WriteLine($&quot;  🧊 {_name}: 低温警告！ {temperature:F1}℃&quot;);
    }
}

    4. Strategyパターン - アルゴリズムの動的な切り替え

    パターンの目的

    「実行時にアルゴリズムを選択・変更できる」柔軟な設計を提供します。

    支払い処理の戦略パターン

    // 戦略インターフェース
public interface IPaymentStrategy
{
    PaymentResult ProcessPayment(decimal amount);
    string GetPaymentMethod();
    bool IsAvailable(decimal amount);
}

// 支払い結果
public class PaymentResult
{
    public bool IsSuccess { get; set; }
    public string Message { get; set; }
    public string TransactionId { get; set; }
    public DateTime ProcessedAt { get; set; }
}

// 具象戦略1：クレジットカード
public class CreditCardPayment : IPaymentStrategy
{
    private readonly string _cardNumber;
    private readonly string _cvv;
    private readonly decimal _limit;

    public CreditCardPayment(string cardNumber, string cvv, decimal limit = 100000)
    {
        _cardNumber = cardNumber;
        _cvv = cvv;
        _limit = limit;
    }

    public PaymentResult ProcessPayment(decimal amount)
    {
        Console.WriteLine($&quot;クレジットカード決済処理中...&quot;);
        Thread.Sleep(1500); // 処理時間をシミュレート

        if (amount &gt; _limit)
        {
            return new PaymentResult
            {
                IsSuccess = false,
                Message = $&quot;利用限度額（{_limit:C}）を超過しています&quot;,
                ProcessedAt = DateTime.Now
            };
        }

        Console.WriteLine($&quot;カード番号: {_cardNumber.Substring(0, 4)}****&quot;);
        return new PaymentResult
        {
            IsSuccess = true,
            Message = $&quot;{amount:C}の決済が完了しました&quot;,
            TransactionId = $&quot;CC{DateTime.Now:yyyyMMddHHmmss}&quot;,
            ProcessedAt = DateTime.Now
        };
    }

    public string GetPaymentMethod() =&gt; &quot;クレジットカード&quot;;

    public bool IsAvailable(decimal amount) =&gt; amount &lt;= _limit;
}

// コンテキスト：ショッピングカート
public class ShoppingCart
{
    private readonly List&lt;(string Item, decimal Price)&gt; _items = new();
    private IPaymentStrategy _paymentStrategy;

    public void AddItem(string item, decimal price)
    {
        _items.Add((item, price));
        Console.WriteLine($&quot;商品追加: {item} ({price:C})&quot;);
    }

    public void SetPaymentStrategy(IPaymentStrategy strategy)
    {
        _paymentStrategy = strategy;
        Console.WriteLine($&quot;支払い方法を設定: {strategy.GetPaymentMethod()}&quot;);
    }

    public decimal GetTotal()
    {
        return _items.Sum(item =&gt; item.Price);
    }

    public PaymentResult Checkout()
    {
        var total = GetTotal();

        if (_items.Count == 0)
        {
            Console.WriteLine(&quot;カートが空です&quot;);
            return new PaymentResult { IsSuccess = false, Message = &quot;商品がありません&quot; };
        }

        if (_paymentStrategy == null)
        {
            Console.WriteLine(&quot;支払い方法が設定されていません&quot;);
            return new PaymentResult { IsSuccess = false, Message = &quot;支払い方法未設定&quot; };
        }

        var result = _paymentStrategy.ProcessPayment(total);

        if (result.IsSuccess)
        {
            Console.WriteLine($&quot;✅ {result.Message}&quot;);
            Console.WriteLine($&quot;取引ID: {result.TransactionId}&quot;);
            _items.Clear(); // カートをクリア
        }
        else
        {
            Console.WriteLine($&quot;❌ 決済失敗: {result.Message}&quot;);
        }

        return result;
    }
}

    5. Commandパターン - 操作のカプセル化

    パターンの目的

    「リクエストをオブジェクトとしてカプセル化」し、アンドゥ/リドゥ、キューイング、ログ記録を可能にします。

    テキストエディタの実装

    // Commandインターフェース
public interface ICommand
{
    void Execute();
    void Undo();
    string GetDescription();
}

// Receiver：実際の処理を行うクラス
public class TextEditor
{
    private StringBuilder _content = new StringBuilder();

    public string Content =&gt; _content.ToString();

    public void InsertText(int position, string text)
    {
        if (position &lt; 0 || position &gt; _content.Length)
            throw new ArgumentOutOfRangeException(nameof(position));

        _content.Insert(position, text);
    }

    public void DeleteText(int position, int length)
    {
        if (position &lt; 0 || position + length &gt; _content.Length)
            throw new ArgumentOutOfRangeException();

        _content.Remove(position, length);
    }

    public void ShowContent()
    {
        Console.WriteLine($&quot;エディタ内容: \&quot;{Content}\&quot;&quot;);
    }
}

// 具象Command：テキスト挿入
public class InsertTextCommand : ICommand
{
    private readonly TextEditor _editor;
    private readonly int _position;
    private readonly string _text;

    public InsertTextCommand(TextEditor editor, int position, string text)
    {
        _editor = editor;
        _position = position;
        _text = text;
    }

    public void Execute()
    {
        _editor.InsertText(_position, _text);
    }

    public void Undo()
    {
        _editor.DeleteText(_position, _text.Length);
    }

    public string GetDescription() =&gt; $&quot;挿入: \&quot;{_text}\&quot; at {_position}&quot;;
}

// Invoker：コマンドを管理・実行するクラス
public class EditorInvoker
{
    private readonly Stack&lt;ICommand&gt; _undoStack = new();
    private readonly Stack&lt;ICommand&gt; _redoStack = new();

    public void ExecuteCommand(ICommand command)
    {
        try
        {
            command.Execute();
            _undoStack.Push(command);
            _redoStack.Clear(); // 新しいコマンド実行時はRedoスタックをクリア

            Console.WriteLine($&quot;実行: {command.GetDescription()}&quot;);
        }
        catch (Exception ex)
        {
            Console.WriteLine($&quot;コマンド実行失敗: {ex.Message}&quot;);
        }
    }

    public void Undo()
    {
        if (_undoStack.Count &gt; 0)
        {
            var command = _undoStack.Pop();
            try
            {
                command.Undo();
                _redoStack.Push(command);
                Console.WriteLine($&quot;アンドゥ: {command.GetDescription()}&quot;);
            }
            catch (Exception ex)
            {
                Console.WriteLine($&quot;アンドゥ失敗: {ex.Message}&quot;);
                _undoStack.Push(command);
            }
        }
        else
        {
            Console.WriteLine(&quot;アンドゥできる操作がありません&quot;);
        }
    }

    public void Redo()
    {
        if (_redoStack.Count &gt; 0)
        {
            var command = _redoStack.Pop();
            try
            {
                command.Execute();
                _undoStack.Push(command);
                Console.WriteLine($&quot;リドゥ: {command.GetDescription()}&quot;);
            }
            catch (Exception ex)
            {
                Console.WriteLine($&quot;リドゥ失敗: {ex.Message}&quot;);
                _redoStack.Push(command);
            }
        }
        else
        {
            Console.WriteLine(&quot;リドゥできる操作がありません&quot;);
        }
    }
}

    演習1: Singletonパターンの実装（基礎）

    問題: ゲームの設定を管理するGameSettingsクラスをSingletonパターンで実装してください。

    要件: - プレイヤー名、難易度、音量を管理 - スレッドセーフな実装 - 設定値の変更と取得メソッド

    // 解答例のヒント
public sealed class GameSettings

    パターン別特徴比較表

    パターン 主な目的 適用場面 メリット デメリット 実装の複雑さ
    Singleton 唯一インスタンス保証 設定管理、DB接続 リソース管理、グローバルアクセス テストが困難、隠れた依存関係 ⭐⭐
    Factory オブジェクト生成の抽象化 環境別生成、型の切り替え 生成ロジックの集約、拡張性 複雑性の増加 ⭐⭐⭐
    Observer 状態変化の通知 イベント処理、MVC 疎結合、動的な関係 デバッグの困難さ、循環参照 ⭐⭐⭐
    Strategy アルゴリズムの切り替え 支払い処理、ソート 実行時切り替え、テスト容易 クラス数の増加 ⭐⭐
    Command 操作のカプセル化 アンドゥ、マクロ、ログ 操作の記録・再実行 メモリ使用量の増加 ⭐⭐⭐⭐

    使い分けの判断フローチャート

    問題の性質を分析
├─ システム全体で1つのインスタンスが必要
│  └─→ Singleton パターン
├─ オブジェクトの生成が複雑・条件分岐が多い
│  └─→ Factory パターン
├─ 状態変化を複数のオブジェクトに通知が必要
│  └─→ Observer パターン
├─ 実行時にアルゴリズムを切り替えたい
│  └─→ Strategy パターン
└─ 操作の記録・取り消し・再実行が必要
   └─→ Command パターン

    パフォーマンスへの影響と最適化

    メモリ使用量の考慮

    // Commandパターンでのメモリ最適化
public class CommandHistory
{
    private readonly Queue&lt;ICommand&gt; _commands = new();
    private const int MaxHistorySize = 100;

    public void AddCommand(ICommand command)
    {
        _commands.Enqueue(command);

        // 古いコマンドを削除してメモリを節約
        while (_commands.Count &gt; MaxHistorySize)
        {
            _commands.Dequeue();
        }
    }
}

    Observerパターンでのメモリリーク防止

    // WeakReferenceを使用したObserver
public class WeakObserver&lt;T&gt;
{
    private readonly WeakReference&lt;T&gt; _target;
    private readonly Action&lt;object&gt; _callback;

    public WeakObserver(T target, Action&lt;object&gt; callback)
    {
        _target = new WeakReference&lt;T&gt;(target);
        _callback = callback;
    }

    public bool TryNotify(object data)
    {
        if (_target.TryGetTarget(out var target))
        {
            _callback(data);
            return true;
        }
        return false; // ターゲットがGCされた
    }
}

    よくある実装の間違いと対策

    1. Singletonの間違った実装

    // ❌ 悪い例：スレッドセーフでない
public class BadSingleton
{
    private static BadSingleton _instance;

    public static BadSingleton Instance
    {
        get
        {
            if (_instance == null) // 複数スレッドで問題
                _instance = new BadSingleton();
            return _instance;
        }
    }
}

// ✅ 良い例：Lazy&lt;T&gt;使用
public sealed class GoodSingleton
{
    private static readonly Lazy&lt;GoodSingleton&gt; _instance = 
        new Lazy&lt;GoodSingleton&gt;(() =&gt; new GoodSingleton());

    public static GoodSingleton Instance =&gt; _instance.Value;
}

    2. Observerパターンでの循環参照

    // ❌ 悪い例：循環参照が発生
public class Subject
{
    private readonly List&lt;IObserver&gt; _observers = new();

    public void AddObserver(IObserver observer)
    {
        _observers.Add(observer); // 強参照でメモリリーク
    }
}

// ✅ 良い例：適切な購読解除
public class Subject : IDisposable
{
    public event Action&lt;Data&gt; DataChanged;

    public void Dispose()
    {
        DataChanged = null; // イベントハンドラーをクリア
    }
}

    3. Strategyパターンでの戦略選択ロジック

    // ❌ 悪い例：if-else の連続
public IStrategy SelectStrategy(string type)
{
    if (type == &quot;A&quot;) return new StrategyA();
    if (type == &quot;B&quot;) return new StrategyB();
    if (type == &quot;C&quot;) return new StrategyC();
    // 新しい戦略追加のたびに修正が必要
}

// ✅ 良い例：ファクトリーとの組み合わせ
public class StrategyFactory
{
    private readonly Dictionary&lt;string, Func&lt;IStrategy&gt;&gt; _strategies;

    public StrategyFactory()
    {
        _strategies = new Dictionary&lt;string, Func&lt;IStrategy&gt;&gt;
        {
            [&quot;A&quot;] = () =&gt; new StrategyA(),
            [&quot;B&quot;] = () =&gt; new StrategyB(),
            [&quot;C&quot;] = () =&gt; new StrategyC()
        };
    }

    public IStrategy CreateStrategy(string type)
    {
        return _strategies.TryGetValue(type, out var factory) 
            ? factory() 
            : throw new ArgumentException($&quot;Unknown strategy: {type}&quot;);
    }
}

    実践での活用指針

    1. チーム開発での導入

    • 段階的導入: まず1つのパターンから始める
    • コードレビュー: パターンの適用が適切かチェック
    • ドキュメント化: パターンの使用理由を明記
    • 研修: チーム全体でパターンの理解を共有

    2. レガシーコードでの活用

    • リファクタリング: 段階的にパターンを適用
    • ラッパー: 既存コードをパターンでラップ
    • テスト追加: パターン適用前後でテストを追加

    3. パフォーマンス要件との両立

    • プロファイリング: パターン適用によるオーバーヘッドを測定
    • 最適化: 必要に応じてパターンを簡略化
    • トレードオフ: 保守性とパフォーマンスのバランス

    まとめ

    デザインパターンは強力なツールですが、適切な場面で適切に使用することが重要です。

    成功の鍵

    1. 問題の本質理解: パターンありきではなく、問題から始める
    2. 段階的適用: 小さく始めて徐々に拡張
    3. 継続的改善: 使用経験を積んでパターンを洗練
    4. チーム共有: パターンの知識をチーム全体で共有

    避けるべき罠

    • 過度な設計: 必要以上に複雑なパターンの適用
    • パターンの乱用: すべてをパターンで解決しようとする
    • 文脈無視: 問題の特性を考慮しないパターン選択

    適切に使用されたデザインパターンは、コードの品質向上、開発効率の改善、そしてチーム開発の円滑化に大きく貢献します。継続的な学習と実践を通じて、デザインパターンを効果的に活用できるエンジニアを目指しましょう。
