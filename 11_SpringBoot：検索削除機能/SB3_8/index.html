    🌐 Ajax技術の基礎理解

    Ajax とは何か

    Ajax（Asynchronous JavaScript and XML） は、ページを再読み込みすることなくサーバーと通信する技術です。

    📘 Ajax基礎理解

    従来の同期通信 vs Ajax（非同期通信）

    従来の方式（同期通信）： 

    &lt;!-- フォーム送信時 --&gt;
&lt;form action=&quot;/students/search&quot; method=&quot;GET&quot;&gt;
    &lt;input type=&quot;text&quot; name=&quot;name&quot; /&gt;
    &lt;button type=&quot;submit&quot;&gt;検索&lt;/button&gt;
&lt;/form&gt;
    ↓ ボタンクリック時 
    1. ページ全体が再読み込み
2. サーバーから新しいHTMLが返される
3. 画面が一瞬白くなる（UX低下）

    Ajaxの方式（非同期通信）： 

    // ボタンクリック時
$(&apos;#searchBtn&apos;).click(function() {
    $.ajax({
        url: &apos;/api/students/search&apos;,
        method: &apos;GET&apos;,
        data: { name: $(&apos;#searchName&apos;).val() },
        success: function(response) {
            // 結果のみを更新（ページ再読み込みなし）
            $(&apos;#searchResults&apos;).html(response);
        }
    });
});
    ↓ ボタンクリック時 
    1. ページはそのまま
2. 背景でサーバーと通信
3. 結果部分のみを更新（UX向上）

    Section A: 基本的なAjax検索実装

    src/main/resources/static/js/ajax-search.js を作成：

    /**
 * Ajax検索機能の実装
 * Week 10.1-10.4の検索機能をAjax化
 */
class AjaxSearchManager {

    constructor() {
        this.searchForm = $(&apos;#searchForm&apos;);
        this.searchButton = $(&apos;#searchButton&apos;);
        this.searchResults = $(&apos;#searchResults&apos;);
        this.loadingIndicator = $(&apos;#loadingIndicator&apos;);
    }

    /**
     * Ajax検索の初期化
     */
    initializeAjaxSearch() {
        // フォーム送信のデフォルト動作を無効化
        this.searchForm.on(&apos;submit&apos;, (e) =&gt; {
            e.preventDefault(); // ページ遷移を防ぐ
            this.performAjaxSearch();
        });

        // 検索ボタンクリック時の処理
        this.searchButton.on(&apos;click&apos;, (e) =&gt; {
            e.preventDefault();
            this.performAjaxSearch();
        });
    }

    /**
     * Ajax検索の実行
     */
    performAjaxSearch() {
        // 検索条件を取得
        const searchData = {
            name: $(&apos;#searchName&apos;).val().trim(),
            email: $(&apos;#searchEmail&apos;).val().trim(),
            courseId: $(&apos;#searchCourseId&apos;).val(),
            page: 0, // 最初のページから開始
            size: 10 // 1ページあたりの件数
        };

        // ローディング表示
        this.showLoading();

        // jQuery Ajax実行
        $.ajax({
            url: &apos;/api/students/search&apos;,
            method: &apos;GET&apos;,
            data: searchData,
            dataType: &apos;json&apos;,
            timeout: 10000, // 10秒でタイムアウト
            success: (response) =&gt; {
                this.handleSearchSuccess(response);
            },
            error: (xhr, status, error) =&gt; {
                this.handleSearchError(xhr, status, error);
            },
            complete: () =&gt; {
                this.hideLoading();
            }
        });
    }
}

    Section B: 成功・エラー処理の実装

        /**
     * 検索成功時の処理
     */
    handleSearchSuccess(response) {
        if (response.students &amp;&amp; response.students.length &gt; 0) {
            // 検索結果の表示
            this.displaySearchResults(response);

            // ページネーション情報の更新
            this.updatePaginationInfo(response.page);

            // 成功メッセージ
            this.showMessage(&apos;検索が完了しました&apos;, &apos;success&apos;);

        } else {
            // 結果が0件の場合
            this.displayNoResults();
            this.showMessage(&apos;該当する学生が見つかりませんでした&apos;, &apos;info&apos;);
        }
    }

    /**
     * 検索エラー時の処理
     */
    handleSearchError(xhr, status, error) {
        console.error(&apos;Ajax search error:&apos;, {
            status: xhr.status,
            statusText: xhr.statusText,
            responseText: xhr.responseText,
            error: error
        });

        let errorMessage = &apos;検索中にエラーが発生しました&apos;;

        // HTTPステータスコードに応じたメッセージ
        switch(xhr.status) {
            case 400:
                errorMessage = &apos;検索条件に不正な値が含まれています&apos;;
                break;
            case 500:
                errorMessage = &apos;サーバーエラーが発生しました。しばらく待ってから再度お試しください&apos;;
                break;
            case 0:
                if (status === &apos;timeout&apos;) {
                    errorMessage = &apos;検索がタイムアウトしました。条件を絞って再度お試しください&apos;;
                } else {
                    errorMessage = &apos;ネットワークエラーが発生しました。接続を確認してください&apos;;
                }
                break;
        }

        this.showMessage(errorMessage, &apos;error&apos;);
        this.displayErrorResults();
    }

    /**
     * ローディング表示
     */
    showLoading() {
        this.loadingIndicator.show();
        this.searchButton.prop(&apos;disabled&apos;, true).text(&apos;検索中...&apos;);
    }

    /**
     * ローディング非表示
     */
    hideLoading() {
        this.loadingIndicator.hide();
        this.searchButton.prop(&apos;disabled&apos;, false).text(&apos;検索&apos;);
    }
}

        // 基本チェック

            this.clearValidationState(&apos;email&apos;);

        // メール形式チェック

        this.showValidationSuccess(&apos;email&apos;);

    /**
     * メールアドレスの重複チェック
     */
    async validateEmailUniqueness(email, excludeStudentId = null) {
        // キャッシュチェック
        const cacheKey = `email_${email}_${excludeStudentId || &apos;new&apos;}`;
        if (this.validationCache.has(cacheKey)) {
            const cachedResult = this.validationCache.get(cacheKey);
            if (cachedResult.isDuplicate) {
                this.showValidationError(&apos;email&apos;, &apos;このメールアドレスは既に使用されています&apos;);
                return false;
            } else {
                this.showValidationSuccess(&apos;email&apos;);
                return true;
            }
        }

        try {
            this.showValidationLoading(&apos;email&apos;);

            const response = await fetch(&apos;/api/students/validate-email&apos;, {
                method: &apos;POST&apos;,
                headers: {
                    &apos;Content-Type&apos;: &apos;application/json&apos;,
                    &apos;X-Requested-With&apos;: &apos;XMLHttpRequest&apos;
                },
                body: JSON.stringify({
                    email: email,
                    excludeId: excludeStudentId
                })
            });

            if (!response.ok) {
                throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            }

            const result = await response.json();

            // 結果をキャッシュ
            this.validationCache.set(cacheKey, result);

            if (result.isDuplicate) {
                this.showValidationError(&apos;email&apos;, &apos;このメールアドレスは既に使用されています&apos;);
                return false;
            } else {
                this.showValidationSuccess(&apos;email&apos;);
                return true;

        } catch (error) {
            console.error(&apos;Email validation error:&apos;, error);
            this.showValidationError(&apos;email&apos;, &apos;バリデーションでエラーが発生しました&apos;);
            return false;
        }
    }
