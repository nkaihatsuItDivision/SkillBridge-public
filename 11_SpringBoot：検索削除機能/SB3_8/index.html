    🌐 Ajax技術の基礎理解

    Ajax とは何か

    Ajax（Asynchronous JavaScript and XML） は、ページを再読み込みすることなくサーバーと通信する技術です。

    📘 Ajax基礎理解

    従来の同期通信 vs Ajax（非同期通信）

    従来の方式（同期通信）： 

    <!-- フォーム送信時 -->
<form action="/students/search" method="GET">
    <input type="text" name="name" />
    <button type="submit">検索</button>
</form>
    ↓ ボタンクリック時 
    1. ページ全体が再読み込み
2. サーバーから新しいHTMLが返される
3. 画面が一瞬白くなる（UX低下）

    constructor() {

        this.lastSearchTerm = ''; // 前回の検索語
    }

    /**
     * リアルタイム検索の初期化
     */
    initializeRealtimeSearch() {
        // 検索フィールドでの入力イベント
        $('.realtime-search').on('input', (e) => {
            this.handleSearchInput(e.target);
        });

        // フォーカス時の処理
        $('.realtime-search').on('focus', (e) => {
            this.handleSearchFocus(e.target);
        });

    /**
     * 検索入力の処理（デバウンス処理付き）
     */
    handleSearchInput(input) {
        const searchValue = $(input).val().trim();

        // 前回と同じ検索語の場合はスキップ
        if (searchValue === this.lastSearchTerm) {
            return;
        }

        // 既存のタイマーをクリア
        clearTimeout(this.searchTimeout);

        // 最小文字数チェック
        if (searchValue.length < this.minSearchLength) {
            this.clearRealtimeResults();
            return;
        }

        // デバウンス処理：連続入力の場合は一定時間後に実行
        this.searchTimeout = setTimeout(() => {
            this.performRealtimeSearch(searchValue);
            this.lastSearchTerm = searchValue;
        }, this.searchDelay);
    }

    /**
     * 実際のリアルタイム検索実行
     */
    async performRealtimeSearch(searchValue) {
        try {
            // リアルタイム検索表示
            this.showRealtimeLoading();

            // Fetch APIを使用した非同期検索
            const response = await fetch('/api/students/realtime-search', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-Requested-With': 'XMLHttpRequest'
                },
                body: JSON.stringify({
                    searchTerm: searchValue,
                    maxResults: 10 // リアルタイム検索は結果を制限
                })
            });

            if (!response.ok) {
                throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            }

            const data = await response.json();
            this.displayRealtimeResults(data);

        } catch (error) {
            console.error('Realtime search error:', error);
            this.handleRealtimeSearchError(error);
        } finally {
            this.hideRealtimeLoading();
        }
    }
}

     * メールアドレスフィールドのバリデーション

        const studentId = $('#studentId').val(); // 編集時の除外ID

        // 基本チェック
        if (!email) {
            this.clearValidationState('email');
            return;
        }

        // メール形式チェック
        if (!this.isValidEmailFormat(email)) {
            this.showValidationError('email', 'メールアドレスの形式が正しくありません');
            return;
        }

        // 重複チェック
        const isUnique = await this.validateEmailUniqueness(email, studentId);
        if (!isUnique) {
            return; // エラーは validateEmailUniqueness 内で表示済み
        }

        this.showValidationSuccess('email');
    }

    /**
     * メールアドレスの重複チェック
     */
    async validateEmailUniqueness(email, excludeStudentId = null) {
        // キャッシュチェック
        const cacheKey = `email_${email}_${excludeStudentId || 'new'}`;
        if (this.validationCache.has(cacheKey)) {
            const cachedResult = this.validationCache.get(cacheKey);
            if (cachedResult.isDuplicate) {
                this.showValidationError('email', 'このメールアドレスは既に使用されています');
                return false;
            } else {
                this.showValidationSuccess('email');
                return true;
            }
        }

        try {
            this.showValidationLoading('email');

            const response = await fetch('/api/students/validate-email', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-Requested-With': 'XMLHttpRequest'
                },
                body: JSON.stringify({
                    email: email,
                    excludeId: excludeStudentId
                })
            });

            if (!response.ok) {
                throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            }

            const result = await response.json();

            // 結果をキャッシュ
            this.validationCache.set(cacheKey, result);

            if (result.isDuplicate) {
                this.showValidationError('email', 'このメールアドレスは既に使用されています');
                return false;
            } else {
                this.showValidationSuccess('email');
                return true;
            }

        } catch (error) {
            console.error('Email validation error:', error);
            this.showValidationError('email', 'バリデーションでエラーが発生しました');
            return false;
        }
    }
}

                headers: {
                    'Content-Type': 'application/json',
                    'X-Requested-With': 'XMLHttpRequest',
                    'X-CSRF-TOKEN': $('meta[name="_csrf"]').attr('content')
                }
            });

            if (!response.ok) {
                throw new Error(`HTTP ${response.status}: ${response.statusText}`);

            if (result.success) {

        }

        const modal = this.createBatchConfirmationModal(

            `選択された ${this.selectedStudents.size} 名の学生を削除してもよろしいですか？
             <small class="text-muted">※論理削除のため、後から復元可能です</small>`,
            'warning',
            () => this.performBatchDelete()
        );

        modal.show();
    }

    /**
     * 一括削除の実行
     */
    async performBatchDelete() {
        const studentIds = Array.from(this.selectedStudents);

        try {
            this.uxManager.showLoading('一括削除処理中...', true);

            let processed = 0;
            const total = studentIds.length;

            for (const studentId of studentIds) {
                const response = await fetch(`/api/students/${studentId}/delete`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-Requested-With': 'XMLHttpRequest',
                        'X-CSRF-TOKEN': $('meta[name="_csrf"]').attr('content')
                    }
                });

                processed++;
                const percentage = Math.round((processed / total) * 100);
                this.uxManager.updateProgress(percentage);

                if (!response.ok) {
                    console.error(`Failed to delete student ${studentId}`);
                }
            }

            // 成功処理
            this.uxManager.showSuccess(`${processed} 名の学生を削除しました`);
            this.refreshStudentTable();
            this.clearSelection();

        } catch (error) {
            console.error('Batch delete error:', error);
            this.uxManager.showError(`一括削除でエラーが発生しました: ${error.message}`);
        } finally {
            this.uxManager.hideLoading();
        }
    }

    /**
     * 一括エクスポート（Week 10.7統合）
     */
    async performBatchExport() {
        if (this.selectedStudents.size === 0) {
            this.uxManager.showWarning('エクスポートする学生を選択してください');
            return;
        }

        const studentIds = Array.from(this.selectedStudents);

        try {
            this.uxManager.showLoading('エクスポート準備中...');

            const response = await fetch('/api/students/export/selected', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-Requested-With': 'XMLHttpRequest'
                },
                body: JSON.stringify({
                    studentIds: studentIds,
                    format: 'excel' // Week 10.7のExcel出力を活用
                })
            });

            if (!response.ok) {
                throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            }

            // ファイルダウンロード処理（Week 10.7のロジックを活用）
            const blob = await response.blob();
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `selected_students_${new Date().toISOString().split('T')[0]}.xlsx`;
            document.body.appendChild(a);
            a.click();
            window.URL.revokeObjectURL(url);
            document.body.removeChild(a);

            this.uxManager.showSuccess(`${studentIds.length} 名の学生データをエクスポートしました`);

        } catch (error) {
            console.error('Batch export error:', error);
            this.uxManager.showError(`エクスポートでエラーが発生しました: ${error.message}`);
        } finally {
            this.uxManager.hideLoading();
        }
    }
