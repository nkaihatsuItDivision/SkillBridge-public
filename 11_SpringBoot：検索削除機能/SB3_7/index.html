    📄 CSV エクスポートサービスの実装

    🔧 ステップ1: CSV生成サービスの作成

    学習方法: 以下のコードをsection毎に分けて理解してから実装してください： - Section A: CSVヘッダー生成機能 - Section B: データ行生成機能
    - Section C: CSV文字列エスケープ処理 - 重要: 1つのメソッドを理解してから次に進んでください。

    service パッケージに CsvExportService クラスを作成します：

    package com.studentmanagement.service;

import com.studentmanagement.entity.Student;
import org.springframework.stereotype.Service;
import java.util.List;
import java.time.format.DateTimeFormatter;

@Service
public class CsvExportService {

    private static final DateTimeFormatter DATE_FORMATTER = 
        DateTimeFormatter.ofPattern(&quot;yyyy-MM-dd HH:mm:ss&quot;);

    /**
     * 学生リストをCSV形式の文字列に変換
     */
    public String generateStudentListCsv(List&lt;Student&gt; students) {
        StringBuilder csv = new StringBuilder();

        // Section A: ヘッダー行の生成
        csv.append(&quot;ID,名前,メールアドレス,専攻,登録日,削除フラグ\n&quot;);

        // Section B: データ行の生成
        for (Student student : students) {
            csv.append(student.getId()).append(&quot;,&quot;)
               .append(escapeCSV(student.getName())).append(&quot;,&quot;)
               .append(escapeCSV(student.getEmail())).append(&quot;,&quot;)
               .append(escapeCSV(student.getMajor())).append(&quot;,&quot;)
               .append(student.getCreatedAt().format(DATE_FORMATTER)).append(&quot;,&quot;)
               .append(student.isDeleted() ? &quot;削除済み&quot; : &quot;有効&quot;).append(&quot;\n&quot;);
        }

        return csv.toString();
    }

    /**
     * Section C: CSV文字列のエスケープ処理
     */
    private String escapeCSV(String value) {
        if (value == null) {
            return &quot;&quot;;
        }

        // カンマ、ダブルクォート、改行が含まれる場合はエスケープが必要
        if (value.contains(&quot;,&quot;) || value.contains(&quot;\&quot;&quot;) || value.contains(&quot;\n&quot;)) {
            // ダブルクォートを２重にしてエスケープ
            return &quot;\&quot;&quot; + value.replace(&quot;\&quot;&quot;, &quot;\&quot;\&quot;&quot;) + &quot;\&quot;&quot;;
        }

        return value;
    }

    /**
     * 検索結果をCSVエクスポート（Week 10.1-10.4連携）
     */
    public String generateSearchResultCsv(List&lt;Student&gt; students, String searchCondition) {
        StringBuilder csv = new StringBuilder();

        // 検索条件をコメントとして追加
        csv.append(&quot;# 検索条件: &quot;).append(searchCondition).append(&quot;\n&quot;);
        csv.append(&quot;# エクスポート日時: &quot;)
           .append(java.time.LocalDateTime.now().format(DATE_FORMATTER))
           .append(&quot;\n&quot;);
        csv.append(&quot;\n&quot;);

        // 通常のCSV生成
        csv.append(generateStudentListCsv(students));

        return csv.toString();
    }

    /**
     * 選択データをCSVエクスポート（Week 10.6連携）
     */
    public String generateSelectedStudentsCsv(List&lt;Student&gt; students) {
        StringBuilder csv = new StringBuilder();

        // 選択データであることを明記
        csv.append(&quot;# 選択されたデータのエクスポート\n&quot;);
        csv.append(&quot;# 選択件数: &quot;).append(students.size()).append(&quot;件\n&quot;);
        csv.append(&quot;# エクスポート日時: &quot;)
           .append(java.time.LocalDateTime.now().format(DATE_FORMATTER))
           .append(&quot;\n&quot;);
        csv.append(&quot;\n&quot;);

        // 通常のCSV生成
        csv.append(generateStudentListCsv(students));

        return csv.toString();
    }
}

    コードの理解ポイント： - StringBuilder: 大量の文字列結合を効率的に処理 - escapeCSV(): CSV形式で問題となる文字（カンマ、ダブルクォート、改行）の適切な処理 - DateTimeFormatter: 日時の統一フォーマット - Week 10.1-10.6連携: 既存機能との統合を意識した設計

    📊 Excel エクスポートサービスの実装

    🔧 ステップ2: Excel生成サービスの作成

    学習方法: 以下のコードをsection毎に分けて理解してから実装してください： - Section A: Workbook・Sheetの基本操作 - Section B: ヘッダー行とスタイル設定
    - Section C: データ行の生成処理 - Section D: バイト配列変換処理

    service パッケージに ExcelExportService クラスを作成します：

    package com.studentmanagement.service;

import com.studentmanagement.entity.Student;
import org.apache.poi.ss.usermodel.*;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;
import org.springframework.stereotype.Service;

import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.util.List;
import java.time.format.DateTimeFormatter;

@Service
public class ExcelExportService {

    private static final DateTimeFormatter DATE_FORMATTER = 
        DateTimeFormatter.ofPattern(&quot;yyyy-MM-dd HH:mm:ss&quot;);

    /**
     * 学生リストをExcel形式のバイト配列に変換
     */
    public byte[] generateStudentListExcel(List&lt;Student&gt; students) {
        try (Workbook workbook = new XSSFWorkbook()) {

            // Section A: シートの作成
            Sheet sheet = workbook.createSheet(&quot;学生一覧&quot;);

            // Section B: ヘッダー行の作成とスタイル設定
            createHeaderRow(sheet, workbook);

            // Section C: データ行の作成
            createDataRows(sheet, students);

            // 列幅の自動調整
            autoSizeColumns(sheet);

            // Section D: バイト配列として返す
            return convertToByteArray(workbook);

        } catch (IOException e) {
            throw new RuntimeException(&quot;Excel生成に失敗しました&quot;, e);
        }
    }

    /**
     * Section B: ヘッダー行の作成
     */
    private void createHeaderRow(Sheet sheet, Workbook workbook) {
        Row headerRow = sheet.createRow(0);

        // ヘッダースタイルの作成
        CellStyle headerStyle = workbook.createCellStyle();
        Font headerFont = workbook.createFont();
        headerFont.setBold(true);
        headerFont.setColor(IndexedColors.WHITE.getIndex());
        headerStyle.setFont(headerFont);
        headerStyle.setFillForegroundColor(IndexedColors.BLUE.getIndex());
        headerStyle.setFillPattern(FillPatternType.SOLID_FOREGROUND);
        headerStyle.setBorderBottom(BorderStyle.THIN);
        headerStyle.setBorderTop(BorderStyle.THIN);
        headerStyle.setBorderRight(BorderStyle.THIN);
        headerStyle.setBorderLeft(BorderStyle.THIN);

        // ヘッダー列の定義
        String[] headers = {&quot;ID&quot;, &quot;名前&quot;, &quot;メールアドレス&quot;, &quot;専攻&quot;, &quot;登録日&quot;, &quot;状態&quot;};

        for (int i = 0; i &lt; headers.length; i++) {
            Cell cell = headerRow.createCell(i);
            cell.setCellValue(headers[i]);
            cell.setCellStyle(headerStyle);
        }
    }

    /**
     * Section C: データ行の作成
     */
    private void createDataRows(Sheet sheet, List&lt;Student&gt; students) {
        int rowNum = 1; // ヘッダー行の次から開始

        for (Student student : students) {
            Row row = sheet.createRow(rowNum++);

            // データの設定
            row.createCell(0).setCellValue(student.getId());
            row.createCell(1).setCellValue(student.getName());
            row.createCell(2).setCellValue(student.getEmail());
            row.createCell(3).setCellValue(student.getMajor());
            row.createCell(4).setCellValue(
                student.getCreatedAt().format(DATE_FORMATTER));
            row.createCell(5).setCellValue(
                student.isDeleted() ? &quot;削除済み&quot; : &quot;有効&quot;);
        }
    }

    /**
     * 列幅の自動調整
     */
    private void autoSizeColumns(Sheet sheet) {
        for (int i = 0; i &lt; 6; i++) {
            sheet.autoSizeColumn(i);
        }
    }

    /**
     * Section D: Workbookをバイト配列に変換
     */
    private byte[] convertToByteArray(Workbook workbook) throws IOException {
        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
        workbook.write(outputStream);
        return outputStream.toByteArray();
    }

    /**
     * 検索結果をExcelエクスポート（Week 10.1-10.4連携）
     */
    public byte[] generateSearchResultExcel(List&lt;Student&gt; students, String searchCondition) {
        try (Workbook workbook = new XSSFWorkbook()) {

            // 検索結果シートの作成
            Sheet sheet = workbook.createSheet(&quot;検索結果&quot;);

            // 検索条件を最初の行に記載
            Row conditionRow = sheet.createRow(0);
            conditionRow.createCell(0).setCellValue(&quot;検索条件: &quot; + searchCondition);

            Row dateRow = sheet.createRow(1);
            dateRow.createCell(0).setCellValue(&quot;エクスポート日時: &quot; + 
                java.time.LocalDateTime.now().format(DATE_FORMATTER));

            // 空行
            sheet.createRow(2);

            // ヘッダー行（3行目から開始）
            createHeaderRowAtIndex(sheet, workbook, 3);

            // データ行の作成（4行目から開始）
            createDataRowsFromIndex(sheet, students, 4);

            // 列幅の自動調整
            autoSizeColumns(sheet);

            return convertToByteArray(workbook);

        } catch (IOException e) {
            throw new RuntimeException(&quot;検索結果Excel生成に失敗しました&quot;, e);
        }
    }

    /**
     * 指定行からヘッダーを作成
     */
    private void createHeaderRowAtIndex(Sheet sheet, Workbook workbook, int rowIndex) {
        Row headerRow = sheet.createRow(rowIndex);

        // ヘッダースタイル（前述と同じ）
        CellStyle headerStyle = workbook.createCellStyle();
        Font headerFont = workbook.createFont();
        headerFont.setBold(true);
        headerFont.setColor(IndexedColors.WHITE.getIndex());
        headerStyle.setFont(headerFont);
        headerStyle.setFillForegroundColor(IndexedColors.BLUE.getIndex());
        headerStyle.setFillPattern(FillPatternType.SOLID_FOREGROUND);

        String[] headers = {&quot;ID&quot;, &quot;名前&quot;, &quot;メールアドレス&quot;, &quot;専攻&quot;, &quot;登録日&quot;, &quot;状態&quot;};

        for (int i = 0; i &lt; headers.length; i++) {
            Cell cell = headerRow.createCell(i);
            cell.setCellValue(headers[i]);
            cell.setCellStyle(headerStyle);
        }
    }

    /**
     * 指定行からデータ行を作成
     */
    private void createDataRowsFromIndex(Sheet sheet, List&lt;Student&gt; students, int startRowIndex) {
        int rowNum = startRowIndex;

        for (Student student : students) {
            Row row = sheet.createRow(rowNum++);

            row.createCell(0).setCellValue(student.getId());
            row.createCell(1).setCellValue(student.getName());
            row.createCell(2).setCellValue(student.getEmail());
            row.createCell(3).setCellValue(student.getMajor());
            row.createCell(4).setCellValue(
                student.getCreatedAt().format(DATE_FORMATTER));
            row.createCell(5).setCellValue(
                student.isDeleted() ? &quot;削除済み&quot; : &quot;有効&quot;);
        }
    }
}

    コードの理解ポイント： - XSSFWorkbook: .xlsx形式のExcelファイルを生成 - CellStyle: セルの見た目（背景色、フォント、枠線など）の設定 - autoSizeColumn(): 内容に応じた列幅の自動調整 - ByteArrayOutputStream: ファイルをメモリ上でバイト配列として生成

    package com.studentmanagement.controller;

    private ExcelExportService excelExportService;

    /**
     * Section A: CSV形式でのエクスポート
     */
    @GetMapping(&quot;/export/csv&quot;)
    public ResponseEntity&lt;String&gt; exportToCsv(
            @ModelAttribute StudentSearchCriteria criteria) {

        try {
            // Week 10.1-10.4の検索機能を活用
            List&lt;Student&gt; students = studentSearchService.searchStudents(criteria);

            // CSV生成
            String csvData = csvExportService.generateSearchResultCsv(
                students, criteria.toString());

            // レスポンスヘッダー設定
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.TEXT_PLAIN);
            headers.setContentDispositionFormData(&quot;attachment&quot;, 
                generateFileName(&quot;students&quot;, &quot;csv&quot;));

            return ResponseEntity.ok()
                    .headers(headers)
                    .body(csvData);

        } catch (Exception e) {
            return handleExportError(&quot;CSV&quot;, e);
        }
    }

    /**
     * Section B: Excel形式でのエクスポート
     */
    @GetMapping(&quot;/export/excel&quot;)
    public ResponseEntity&lt;byte[]&gt; exportToExcel(
            @ModelAttribute StudentSearchCriteria criteria) {

        try {
            // Week 10.1-10.4の検索機能を活用
            List&lt;Student&gt; students = studentSearchService.searchStudents(criteria);

            // Excel生成
            byte[] excelData = excelExportService.generateSearchResultExcel(
                students, criteria.toString());

            // レスポンスヘッダー設定
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_OCTET_STREAM);
            headers.setContentDispositionFormData(&quot;attachment&quot;, 
                generateFileName(&quot;students&quot;, &quot;xlsx&quot;));

            return ResponseEntity.ok()
                    .headers(headers)
                    .body(excelData);

        } catch (Exception e) {
            return ResponseEntity.internalServerError().build();
        }
    }

    /**
     * Section C: 選択されたデータのCSVエクスポート（Week 10.6連携）
     */
    @PostMapping(&quot;/export/selected/csv&quot;)
    public ResponseEntity&lt;String&gt; exportSelectedToCsv(
            @RequestParam(&quot;selectedIds&quot;) List&lt;Long&gt; selectedIds) {

        try {
            // 選択されたIDの学生を取得
            List&lt;Student&gt; selectedStudents = selectedIds.stream()
                .map(id -&gt; studentService.findById(id))
                .filter(student -&gt; student != null)
                .collect(Collectors.toList());

            if (selectedStudents.isEmpty()) {
                return ResponseEntity.badRequest()
                    .body(&quot;選択されたデータが見つかりません。&quot;);
            }

            // CSV生成
            String csvData = csvExportService.generateSelectedStudentsCsv(selectedStudents);

            // レスポンスヘッダー設定
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.TEXT_PLAIN);
            headers.setContentDispositionFormData(&quot;attachment&quot;, 
                generateFileName(&quot;selected_students&quot;, &quot;csv&quot;));

            return ResponseEntity.ok()
                    .headers(headers)
                    .body(csvData);

        } catch (Exception e) {
            return handleExportError(&quot;選択データCSV&quot;, e);
        }
    }

    /**
     * 選択されたデータのExcelエクスポート（Week 10.6連携）
     */
    @PostMapping(&quot;/export/selected/excel&quot;)
    public ResponseEntity&lt;byte[]&gt; exportSelectedToExcel(
            @RequestParam(&quot;selectedIds&quot;) List&lt;Long&gt; selectedIds) {

        try {
            // 選択されたIDの学生を取得
            List&lt;Student&gt; selectedStudents = selectedIds.stream()
                .map(id -&gt; studentService.findById(id))
                .filter(student -&gt; student != null)
                .collect(Collectors.toList());

            if (selectedStudents.isEmpty()) {
                return ResponseEntity.badRequest().build();
            }

            // Excel生成
            byte[] excelData = excelExportService.generateStudentListExcel(selectedStudents);

            // レスポンスヘッダー設定
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_OCTET_STREAM);
            headers.setContentDispositionFormData(&quot;attachment&quot;, 
                generateFileName(&quot;selected_students&quot;, &quot;xlsx&quot;));

            return ResponseEntity.ok()
                    .headers(headers)
                    .body(excelData);

        } catch (Exception e) {
            return ResponseEntity.internalServerError().build();
        }
    }

    /**
     * ファイル名生成（日時付き）
     */
    private String generateFileName(String prefix, String extension) {
        String timestamp = LocalDateTime.now()
            .format(DateTimeFormatter.ofPattern(&quot;yyyyMMdd_HHmmss&quot;));
        return prefix + &quot;_&quot; + timestamp + &quot;.&quot; + extension;
    }

    /**
     * Section D: エラーハンドリング
     */
    private ResponseEntity&lt;String&gt; handleExportError(String exportType, Exception e) {
        String errorMessage = exportType + &quot; エクスポートでエラーが発生しました: &quot; + e.getMessage();
        return ResponseEntity.internalServerError().body(errorMessage);
    }
}

// Week 10.6のselectionManagerと連携して選択数を更新
document.addEventListener(&apos;DOMContentLoaded&apos;, function() {

    // 選択状態変更時のコールバック
    function updateExportButtonStates() {
        const selectedIds = selectionManager.getSelectedIds();
        const selectedCount = selectedIds.length;

        // 選択数の表示更新
        const selectedCountDisplay = document.getElementById(&apos;selectedCountDisplay&apos;);
        if (selectedCountDisplay) {
            selectedCountDisplay.textContent = `選択件数: ${selectedCount}件`;
        }

        // 選択エクスポートボタンの有効/無効制御
        const selectedCsvBtn = document.getElementById(&apos;selectedCsvBtn&apos;);
        const selectedExcelBtn = document.getElementById(&apos;selectedExcelBtn&apos;);

        if (selectedCsvBtn &amp;&amp; selectedExcelBtn) {
            const isEnabled = selectedCount &gt; 0;
            selectedCsvBtn.disabled = !isEnabled;
            selectedExcelBtn.disabled = !isEnabled;

            // ボタンのスタイル更新
            if (isEnabled) {
                selectedCsvBtn.classList.remove(&apos;btn-outline-secondary&apos;);
                selectedCsvBtn.classList.add(&apos;btn-outline-success&apos;);
                selectedExcelBtn.classList.remove(&apos;btn-outline-secondary&apos;);
                selectedExcelBtn.classList.add(&apos;btn-outline-primary&apos;);
            } else {
                selectedCsvBtn.classList.remove(&apos;btn-outline-success&apos;);
                selectedCsvBtn.classList.add(&apos;btn-outline-secondary&apos;);
                selectedExcelBtn.classList.remove(&apos;btn-outline-primary&apos;);
                selectedExcelBtn.classList.add(&apos;btn-outline-secondary&apos;);
            }
        }
    }

    // Week 10.6のselectionManagerにコールバック登録
    if (typeof selectionManager !== &apos;undefined&apos;) {
        // チェックボックス変更時のイベント監視
        document.addEventListener(&apos;change&apos;, function(e) {
            if (e.target.classList.contains(&apos;student-checkbox&apos;) || e.target.id === &apos;selectAll&apos;) {
                updateExportButtonStates();
            }
        });

        // 初期状態の設定
        updateExportButtonStates();
    }
});
