                    criteriaBuilder.lower(root.get("major")),

                );
                predicates.add(majorPredicate);
            }

            // 学年検索条件
            if (criteria.getGrade() != null) {
                Predicate gradePredicate = criteriaBuilder.equal(
                    root.get("grade"), 
                    criteria.getGrade()
                );
                predicates.add(gradePredicate);
            }

            // 学年範囲検索条件
            if (criteria.getMinGrade() != null) {
                Predicate minGradePredicate = criteriaBuilder.greaterThanOrEqualTo(
                    root.get("grade"), 
                    criteria.getMinGrade()
                );
                predicates.add(minGradePredicate);
            }

            if (criteria.getMaxGrade() != null) {
                Predicate maxGradePredicate = criteriaBuilder.lessThanOrEqualTo(
                    root.get("grade"), 
                    criteria.getMaxGrade()
                );
                predicates.add(maxGradePredicate);
            }

            // メールアドレス検索条件
            if (StringUtils.hasText(criteria.getEmail())) {
                Predicate emailPredicate = criteriaBuilder.like(
                    criteriaBuilder.lower(root.get("email")), 
                    "%" + criteria.getEmail().toLowerCase() + "%"
                );
                predicates.add(emailPredicate);
            }

            // Week 10.4の核心：AND/OR演算子による条件結合
            if (predicates.isEmpty()) {
                return criteriaBuilder.conjunction(); // 条件なしの場合は全件取得
            }

            if ("OR".equals(criteria.getSearchOperator())) {
                // OR条件：いずれかの条件を満たす
                return criteriaBuilder.or(predicates.toArray(new Predicate[0]));
            } else {
                // AND条件：全ての条件を満たす（デフォルト）
                return criteriaBuilder.and(predicates.toArray(new Predicate[0]));
            }
        };
    }

        List<Integer> gradeOptions = Arrays.asList(1, 2, 3, 4);
        List<String> statusOptions = studentSearchService.getAvailableStatuses();

        model.addAttribute("gradeOptions", gradeOptions);
        model.addAttribute("statusOptions", statusOptions);

    // Week 10.4で追加：高度検索専用バリデーション
    private void validateAdvancedSearchCriteria(StudentSearchForm searchForm, BindingResult bindingResult) {
        // 日付範囲の妥当性チェック
        if (searchForm.getStartDate() != null && searchForm.getEndDate() != null) {
            if (searchForm.getStartDate().isAfter(searchForm.getEndDate())) {
                bindingResult.rejectValue("endDate", "invalid.dateRange", 
                    "終了日は開始日以降で入力してください。");
            }
        }

        // 検索条件が1つも指定されていない場合のチェック
        if (!searchForm.hasAnySearchCriteria()) {
            bindingResult.reject("empty.searchCriteria", 
                "検索条件を1つ以上指定してください。");
        }
    }

    // Week 10.4で追加：検索条件概要作成
    private String buildAdvancedSearchSummary(StudentSearchCriteria criteria) {
        StringBuilder summary = new StringBuilder();
        summary.append("【高度検索結果】 ");
        summary.append(criteria.getSearchOperator()).append("条件 - ");

        List<String> conditions = new ArrayList<>();
        if (StringUtils.hasText(criteria.getName())) {
            conditions.add("名前: " + criteria.getName());
        }
        if (StringUtils.hasText(criteria.getMajor())) {
            conditions.add("専攻: " + criteria.getMajor());
        }
        if (criteria.getGrade() != null) {
            conditions.add("学年: " + criteria.getGrade() + "年");
        }
        if (criteria.getStartDate() != null || criteria.getEndDate() != null) {
            conditions.add("期間指定: あり");
        }
        if (criteria.hasMultipleSelection()) {
            conditions.add("複数選択: あり");
        }

        return summary.append(String.join(", ", conditions)).toString();
    }

/**

function saveSearchCriteria(searchName) {
    const form = document.querySelector('#advancedSearchForm');
    const formData = new FormData(form);

    // FormDataをオブジェクトに変換
    const criteria = {};
    for (const [key, value] of formData.entries()) {
        if (criteria[key]) {
            // 複数の値がある場合は配列に
            if (Array.isArray(criteria[key])) {
                criteria[key].push(value);
            } else {
                criteria[key] = [criteria[key], value];
            }
        } else {
            criteria[key] = value;
        }
    }

    fetch('/students/api/search-criteria/save', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            searchName: searchName,
            criteria: criteria
        })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            showAlert('success', '検索条件を保存しました: ' + searchName);
        } else {
            showAlert('error', '保存に失敗しました: ' + data.message);
        }
    })
    .catch(error => {
        console.error('検索条件保存エラー:', error);
        showAlert('error', '保存中にエラーが発生しました');
    });
}
