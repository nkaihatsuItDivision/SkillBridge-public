    🔍 1. 検索フォーム設計の基礎

    検索フォームの要件分析

    📘​ 検索フォーム設計

    企業システムの検索画面では、以下の要件を満たす必要があります：

    機能要件

    • 複数条件検索: 名前・学年・専攻・メールアドレス等での絞り込み
    • 検索タイプ切り替え: 部分一致・完全一致の選択
    • 範囲検索: 学年範囲・日付範囲等の指定
    • 条件リセット: 検索条件の一括クリア
    • 条件保存: よく使う検索条件の保存・呼び出し

    非機能要件

    • 使いやすさ: 直感的な操作・分かりやすいレイアウト
    • レスポンシブ: モバイル・タブレット対応
    • バリデーション: リアルタイムエラーチェック・適切なメッセージ表示
    • パフォーマンス: 高速な画面表示・スムーズな操作感

    検索フォーム用DTOの設計

    Week 10.1のStudentSearchCriteriaを画面用に拡張したStudentSearchFormを作成します：

    package com.example.form;  

import javax.validation.constraints.*; 
import org.hibernate.validator.constraints.Email;  

public class StudentSearchForm {  

    // 名前検索 
    @Size(max = 100, message = "学生名は100文字以内で入力してください") 
    private String name;  

    // 学年検索 
    @Min(value = 1, message = "学年は1以上で入力してください") 
    @Max(value = 6, message = "学年は6以下で入力してください") 
    private Integer grade;  

    // 専攻検索 
    @Pattern(regexp = "^[a-zA-Z\\u3040-\\u309F\\u30A0-\\u30FF\\u4E00-\\u9FAF\\s]*$",  
             message = "専攻は日本語・英語のみ入力可能です") 
    private String major;  

    // メールアドレス検索 
    @Email(message = "正しいメールアドレス形式で入力してください") 
    private String email;  

    // 学年範囲検索 
    private Integer minGrade; 
    private Integer maxGrade;  

    // 検索タイプ設定 
    private String nameSearchType = "partial"; // デフォルト：部分一致 
    private String majorSearchType = "partial";  

    // カスタムバリデーション 
    @AssertTrue(message = "最小学年は最大学年以下で設定してください") 
    public boolean isValidGradeRange() { 
        if (minGrade != null && maxGrade != null) { 
            return minGrade <= maxGrade; 
        } 
        return true; 
    }  

    // Week 10.1のStudentSearchCriteriaへの変換 
    public StudentSearchCriteria toCriteria() { 
        StudentSearchCriteria criteria = new StudentSearchCriteria(); 
        criteria.setName(this.name); 
        criteria.setGrade(this.grade); 
        criteria.setMajor(this.major); 
        criteria.setEmail(this.email); 
        criteria.setMinGrade(this.minGrade); 
        criteria.setMaxGrade(this.maxGrade); 
        return criteria; 
    }  

    // getter/setter省略 
}

    package com.example.controller;  

import javax.validation.Valid; 
import org.springframework.beans.factory.annotation.Autowired; 
import org.springframework.stereotype.Controller; 
import org.springframework.ui.Model; 
import org.springframework.validation.BindingResult; 
import org.springframework.web.bind.annotation.*; 
import org.springframework.web.servlet.mvc.support.RedirectAttributes; 
import javax.servlet.http.HttpServletRequest; 
import java.util.List;  

@Controller 
@RequestMapping("/students") 
public class StudentSearchController {  

    @Autowired 
    private StudentSearchService studentSearchService;  

    /** 
     * 検索画面表示 
     */ 
    @GetMapping("/search") 
    public String searchForm(Model model) { 
        // 空の検索フォームオブジェクトを作成 
        model.addAttribute("searchCriteria", new StudentSearchForm());  

        // 保存済み検索条件を追加（後で実装） 
        model.addAttribute("savedConditions", getSavedSearchConditions());  

        return "student/search"; 
    }  

    /** 
     * 検索実行 
     */ 
    @PostMapping("/search") 
    public String executeSearch(@Valid @ModelAttribute("searchCriteria") StudentSearchForm searchForm, 
                                 BindingResult bindingResult, 
                                 RedirectAttributes redirectAttributes, 
                                 Model model) {  

        // バリデーションエラーチェック 
        if (bindingResult.hasErrors()) { 
            model.addAttribute("savedConditions", getSavedSearchConditions()); 
            return "student/search";  // エラー時は検索画面に戻る 
        }  

        try { 
            // Week 10.1の検索サービスを活用 
            StudentSearchCriteria criteria = searchForm.toCriteria(); 
            List<StudentDto> searchResults = studentSearchService.searchStudents(criteria);  

            // 検索結果をセッションに保存（Week 10.3で活用） 
            redirectAttributes.addFlashAttribute("searchResults", searchResults); 
            redirectAttributes.addFlashAttribute("searchCriteria", criteria); 
            redirectAttributes.addFlashAttribute("successMessage",  
                searchResults.size() + "件の学生が見つかりました");  

            // PRGパターン：検索結果画面にリダイレクト 
            return "redirect:/students/search/results";  

        } catch (ServiceException e) { 
            // エラーハンドリング 
            bindingResult.reject("search.error", "検索でエラーが発生しました: " + e.getMessage()); 
            model.addAttribute("savedConditions", getSavedSearchConditions()); 
            return "student/search"; 
        } 
    }  

    /** 
     * 検索結果画面表示（Week 10.3で詳細実装） 
     */ 
    @GetMapping("/search/results") 
    public String searchResults() { 
        return "student/search-results"; 
    }  

    /** 
     * 条件リセット 
     */ 
    @PostMapping("/search/reset") 
    public String resetSearchConditions(RedirectAttributes redirectAttributes) { 
        redirectAttributes.addFlashAttribute("infoMessage", "検索条件をリセットしました"); 
        return "redirect:/students/search"; 
    }  

    /** 
     * 検索条件保存（Ajax対応） 
     */ 
    @PostMapping("/search/save") 
    @ResponseBody 
    public ResponseEntity<String> saveSearchCondition(@RequestBody StudentSearchForm searchForm, 
                                                          HttpServletRequest request) { 
        try { 
            // セッションまたはデータベースに保存 
            saveSearchConditionToSession(searchForm, request.getSession()); 
            return ResponseEntity.ok("検索条件を保存しました"); 
        } catch (Exception e) { 
            return ResponseEntity.badRequest().body("保存に失敗しました"); 
        } 
    }  

    /** 
     * 保存済み検索条件取得 
     */ 
    private List<StudentSearchForm> getSavedSearchConditions() { 
        // セッションまたはデータベースから取得（実装簡略化） 
        return new ArrayList<>(); 
    }  

    /** 
     * 検索条件セッション保存 
     */ 
    private void saveSearchConditionToSession(StudentSearchForm searchForm, HttpSession session) { 
        // セッションに保存（実装簡略化） 
        List<StudentSearchForm> savedConditions =  
            (List<StudentSearchForm>) session.getAttribute("savedSearchConditions"); 
        if (savedConditions == null) { 
            savedConditions = new ArrayList<>(); 
        } 
        savedConditions.add(searchForm); 
        session.setAttribute("savedSearchConditions", savedConditions); 
    } 
}

function initializeGradeRangeValidation() { 
    const minGradeSelect = document.getElementById('minGrade'); 
    const maxGradeSelect = document.getElementById('maxGrade');  

    function validateGradeRange() { 
        const minGrade = parseInt(minGradeSelect.value); 
        const maxGrade = parseInt(maxGradeSelect.value);  

        if (minGrade && maxGrade && minGrade > maxGrade) {

        } else { 
            hideValidationWarning(); 
            return true; 
        } 
    }

}

    const nameInput = document.getElementById('name'); 
    const emailInput = document.getElementById('email');

        } else { 
            hideFieldError(this); 
        } 
    });

    // 条件保存機能（既に実装済み）

            body: JSON.stringify(searchData) 
        }) 
        .then(response => response.text()) 
        .then(message => { 
            showSuccess(message); 
        }) 
        .catch(error => { 
            showError('保存に失敗しました'); 
        }); 
    };

    };

function showFieldError(field, message) { 
    field.classList.add('is-invalid'); 
    let errorDiv = field.nextElementSibling; 
    if (!errorDiv || !errorDiv.classList.contains('invalid-feedback')) { 
        errorDiv = document.createElement('div'); 
        errorDiv.className = 'invalid-feedback'; 
        field.parentNode.insertBefore(errorDiv, field.nextSibling); 
    } 
    errorDiv.textContent = message; 
}

function showMessage(message, type) { 
    const alertDiv = document.createElement('div'); 
    alertDiv.className = `alert alert-${type === 'success' ? 'success' : 'danger'} alert-dismissible fade show`; 
    alertDiv.innerHTML = `

    setTimeout(() => { 
        if (alertDiv.parentNode) { 
            alertDiv.parentNode.removeChild(alertDiv); 
        } 
    }, 5000); 
}
