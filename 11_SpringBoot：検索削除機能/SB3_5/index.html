public class Student {

    public void logicalDelete(String deletedBy, String deleteReason) {

    💡 論理削除と物理削除の使い分け

    📘 論理削除と物理削除

    論理削除（推奨）： - データを実際には削除せず、削除フラグで管理 - データの復元が可能 - 監査ログとして削除履歴を保持 - 関連データとの整合性を保てる

    物理削除： - データを完全に削除 - 復元不可能 - ストレージ容量の節約 - 機密情報の完全削除が必要な場合

    @Service

public class StudentDeleteServiceImpl implements StudentDeleteService {

    @Autowired
    private StudentRepository studentRepository;

    /**
     * 論理削除を実行する
     */
    @Override
    @Transactional
    public void logicalDeleteStudent(Long studentId, String deleteReason, String deletedBy) 
            throws ServiceException {

        logger.info("論理削除処理開始：学生ID={}, 削除者={}", studentId, deletedBy);

        try {
            // 1. 学生の存在確認
            Student student = studentRepository.findById(studentId)
                .orElseThrow(() -> new ServiceException("学生が見つかりません: " + studentId));

            // 2. 既に削除済みかチェック
            if (student.isDeleted()) {
                throw new ServiceException("既に削除済みの学生です: " + studentId);
            }

            // 3. 削除可能性チェック
            if (!canDeleteStudent(studentId)) {
                throw new ServiceException("この学生は削除できません: " + studentId);
            }

            // 4. 論理削除の実行
            student.logicalDelete(deletedBy, deleteReason);
            studentRepository.save(student);

            // 5. 削除ログの記録
            logDeleteOperation(studentId, "LOGICAL_DELETE", deletedBy, deleteReason);

            logger.info("論理削除完了：学生ID={}, 削除者={}", studentId, deletedBy);

        } catch (ServiceException e) {
            logger.error("論理削除でServiceExceptionが発生：学生ID=" + studentId, e);
            throw e;
        } catch (Exception e) {
            logger.error("論理削除で予期しないエラーが発生：学生ID=" + studentId, e);
            throw new ServiceException("論理削除の処理中にエラーが発生しました", e);
        }
    }

    /**
     * 物理削除を実行する
     */
    @Override
    @Transactional
    public void physicalDeleteStudent(Long studentId, String deleteReason, String deletedBy) 
            throws ServiceException {

        logger.info("物理削除処理開始：学生ID={}, 削除者={}", studentId, deletedBy);

        try {
            // 1. 学生の存在確認
            Student student = studentRepository.findById(studentId)
                .orElseThrow(() -> new ServiceException("学生が見つかりません: " + studentId));

            // 2. 削除可能性チェック
            if (!canDeleteStudent(studentId)) {
                throw new ServiceException("この学生は削除できません: " + studentId);
            }

            // 3. 削除ログの記録（削除前に実行）
            logDeleteOperation(studentId, "PHYSICAL_DELETE", deletedBy, deleteReason);

            // 4. 物理削除の実行
            studentRepository.deleteById(studentId);

            logger.info("物理削除完了：学生ID={}, 削除者={}", studentId, deletedBy);

        } catch (ServiceException e) {
            logger.error("物理削除でServiceExceptionが発生：学生ID=" + studentId, e);
            throw e;
        } catch (Exception e) {
            logger.error("物理削除で予期しないエラーが発生：学生ID=" + studentId, e);
            throw new ServiceException("物理削除の処理中にエラーが発生しました", e);
        }
    }

    /**
     * 論理削除を取り消す（復元）

    @Override
    @Transactional

        logger.info("復元処理開始：学生ID={}, 復元者={}", studentId, restoredBy);

        try {
            // 1. 学生の存在確認
            Student student = studentRepository.findById(studentId)
                .orElseThrow(() -> new ServiceException("学生が見つかりません: " + studentId));

            // 2. 削除済みかチェック
            if (!student.isDeleted()) {
                throw new ServiceException("この学生は削除されていません: " + studentId);
            }

            // 3. 復元の実行
            student.restore();
            studentRepository.save(student);

            // 4. 復元ログの記録
            logDeleteOperation(studentId, "RESTORE", restoredBy, "データ復元");

            logger.info("復元完了：学生ID={}, 復元者={}", studentId, restoredBy);

        } catch (ServiceException e) {
            logger.error("復元でServiceExceptionが発生：学生ID=" + studentId, e);
            throw e;
        } catch (Exception e) {
            logger.error("復元で予期しないエラーが発生：学生ID=" + studentId, e);

    }

    @Controller
@RequestMapping("/students")
public class StudentDeleteController {

    private static final Logger logger = LoggerFactory.getLogger(StudentDeleteController.class);

    @Autowired
    private StudentService studentService;

    @Autowired
    private StudentDeleteService studentDeleteService;

    /**
     * 削除確認画面を表示
     */
    @GetMapping("/{id}/delete/confirm")
    public String showDeleteConfirm(@PathVariable Long id, Model model) {
        try {
            StudentDto student = studentService.findById(id);

            // 削除フォームの初期化
            DeleteForm deleteForm = new DeleteForm();
            deleteForm.setStudentId(id);
            deleteForm.setDeleteType("LOGICAL"); // デフォルトは論理削除

            model.addAttribute("student", student);
            model.addAttribute("deleteForm", deleteForm);

            return "students/delete-confirm";

        } catch (ServiceException e) {
            logger.error("削除確認画面表示エラー: 学生ID=" + id, e);
            model.addAttribute("errorMessage", "学生情報の取得に失敗しました");
            return "redirect:/students/search/results";
        }
    }

    /**
     * 削除を実行
     */
    @PostMapping("/delete/execute")
    public String executeDelete(@Valid @ModelAttribute DeleteForm deleteForm,
                               BindingResult bindingResult,
                               RedirectAttributes redirectAttributes) {

        if (bindingResult.hasErrors()) {
            redirectAttributes.addFlashAttribute("errorMessage", "入力内容に不備があります");
            return "redirect:/students/" + deleteForm.getStudentId() + "/delete/confirm";
        }

        try {
            if ("LOGICAL".equals(deleteForm.getDeleteType())) {
                // 論理削除の実行
                studentDeleteService.logicalDeleteStudent(
                    deleteForm.getStudentId(),
                    deleteForm.getDeleteReason(),
                    deleteForm.getDeletedBy()
                );
                redirectAttributes.addFlashAttribute("successMessage", "学生の論理削除が完了しました");

            } else if ("PHYSICAL".equals(deleteForm.getDeleteType())) {
                // 物理削除の実行
                studentDeleteService.physicalDeleteStudent(
                    deleteForm.getStudentId(),
                    deleteForm.getDeleteReason(),
                    deleteForm.getDeletedBy()
                );
                redirectAttributes.addFlashAttribute("successMessage", "学生の物理削除が完了しました");

            } else {
                redirectAttributes.addFlashAttribute("errorMessage", "無効な削除タイプです");
                return "redirect:/students/" + deleteForm.getStudentId() + "/delete/confirm";
            }

            return "redirect:/students/search/results";

        } catch (ServiceException e) {
            logger.error("削除実行エラー: 学生ID=" + deleteForm.getStudentId(), e);
            redirectAttributes.addFlashAttribute("errorMessage", "削除処理中にエラーが発生しました: " + e.getMessage());
            return "redirect:/students/" + deleteForm.getStudentId() + "/delete/confirm";
        }
    }

    /**
     * 復元確認画面を表示
     */
    @GetMapping("/{id}/restore/confirm")
    public String showRestoreConfirm(@PathVariable Long id, Model model) {
        try {
            StudentDto student = studentService.findById(id);

            if (!student.isDeleted()) {
                model.addAttribute("errorMessage", "この学生は削除されていません");
                return "redirect:/students/search/results";
            }

            model.addAttribute("student", student);
            return "students/restore-confirm";

        } catch (ServiceException e) {
            logger.error("復元確認画面表示エラー: 学生ID=" + id, e);
            model.addAttribute("errorMessage", "学生情報の取得に失敗しました");
            return "redirect:/students/search/results";
        }
    }

    /**
     * 復元を実行
     */
    @PostMapping("/{id}/restore/execute")
    public String executeRestore(@PathVariable Long id,
                                @RequestParam String restoredBy,
                                RedirectAttributes redirectAttributes) {

        if (restoredBy == null || restoredBy.trim().isEmpty()) {
            redirectAttributes.addFlashAttribute("errorMessage", "復元実行者名を入力してください");
            return "redirect:/students/" + id + "/restore/confirm";
        }

        try {
            studentDeleteService.restoreStudent(id, restoredBy.trim());
            redirectAttributes.addFlashAttribute("successMessage", "学生の復元が完了しました");
            return "redirect:/students/search/results";

        } catch (ServiceException e) {
            logger.error("復元実行エラー: 学生ID=" + id, e);
            redirectAttributes.addFlashAttribute("errorMessage", "復元処理中にエラーが発生しました: " + e.getMessage());
            return "redirect:/students/" + id + "/restore/confirm";
        }
    }

    /**
     * 削除履歴を取得（Ajax用）
     */
    @GetMapping("/{id}/delete-history")
    @ResponseBody
    public String getDeleteHistory(@PathVariable Long id) {
        try {
            return studentDeleteService.getDeleteHistory(id);
        } catch (ServiceException e) {
            logger.error("削除履歴取得エラー: 学生ID=" + id, e);
            return "削除履歴の取得に失敗しました";
        }
    }
}
