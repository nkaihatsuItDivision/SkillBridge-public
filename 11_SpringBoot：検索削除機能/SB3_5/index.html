    🏗️ ステップ1: Entityクラスの削除対応拡張

    src/main/java/com/example/demo/entity/Student.javaを以下のように拡張してください：

    @Entity
@Table(name = &quot;students&quot;)
public class Student {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false, length = 100)
    private String name;

    @Column(nullable = false)
    private Integer grade;

    @Column(nullable = false, length = 100)
    private String major;

    @Column(nullable = false, length = 200)
    private String email;

    @Column(name = &quot;created_date&quot;)
    private LocalDate createdDate;

    // ========== Week 10.5新規追加：削除機能対応フィールド ==========

    /**
     * 論理削除フラグ
     * false: アクティブ（削除されていない）
     * true: 削除済み
     */
    @Column(name = &quot;deleted_flg&quot;, nullable = false)
    private Boolean deletedFlg = false;

    /**
     * 削除日時
     */
    @Column(name = &quot;deleted_at&quot;)
    private LocalDateTime deletedAt;

    /**
     * 削除実行者
     */
    @Column(name = &quot;deleted_by&quot;, length = 100)
    private String deletedBy;

    /**
     * 削除理由
     */
    @Column(name = &quot;delete_reason&quot;, length = 500)
    private String deleteReason;

    // ========== コンストラクタ ==========

    public Student() {
        this.createdDate = LocalDate.now();
        this.deletedFlg = false; // デフォルトはアクティブ状態
    }

    public Student(String name, Integer grade, String major, String email) {
        this();
        this.name = name;
        this.grade = grade;
        this.major = major;
        this.email = email;

    /**
     * 論理削除を実行する
     * @param deletedBy 削除実行者
     * @param deleteReason 削除理由
     */
    public void logicalDelete(String deletedBy, String deleteReason) {
        this.deletedFlg = true;
        this.deletedAt = LocalDateTime.now();

        this.deleteReason = deleteReason;
    }

    public void restore() {

    public boolean isActive() {

    💡 論理削除と物理削除の使い分け

    ​ 論理削除と物理削除

    論理削除（推奨）： - データを実際には削除せず、削除フラグで管理 - データの復元が可能 - 監査ログとして削除履歴を保持 - 関連データとの整合性を保てる

    物理削除： - データを完全に削除 - 復元不可能 - ストレージ容量の節約 - 機密情報の完全削除が必要な場合

    3. @Transactionalによる削除サービス実装

    ⚙️ ステップ3: 削除機能専用サービス作成

    安全な削除処理を管理する専用サービスクラスを作成しましょう。

    src/main/java/com/example/demo/service/StudentDeleteService.java（インターフェース）：

    public interface StudentDeleteService {

    /**
     * 論理削除を実行する
     * @param studentId 削除対象の学生ID
     * @param deleteReason 削除理由
     * @param deletedBy 削除実行者
     * @throws ServiceException 削除処理でエラーが発生した場合
     */
    void logicalDeleteStudent(Long studentId, String deleteReason, String deletedBy) 
        throws ServiceException;

    /**
     * 物理削除を実行する
     * @param studentId 削除対象の学生ID
     * @param deleteReason 削除理由
     * @param deletedBy 削除実行者
     * @throws ServiceException 削除処理でエラーが発生した場合
     */
    void physicalDeleteStudent(Long studentId, String deleteReason, String deletedBy) 
        throws ServiceException;

    /**
     * 論理削除を取り消す（復元）
     * @param studentId 復元対象の学生ID
     * @param restoredBy 復元実行者
     * @throws ServiceException 復元処理でエラーが発生した場合
     */
    void restoreStudent(Long studentId, String restoredBy) throws ServiceException;

    /**
     * 削除可能性をチェックする
     * @param studentId チェック対象の学生ID
     * @return 削除可能な場合true
     * @throws ServiceException チェック処理でエラーが発生した場合
     */
    boolean canDeleteStudent(Long studentId) throws ServiceException;

    /**
     * 削除履歴を取得する
     * @param studentId 学生ID
     * @return 削除履歴情報
     * @throws ServiceException 取得処理でエラーが発生した場合
     */
    String getDeleteHistory(Long studentId) throws ServiceException;
}

    src/main/java/com/example/demo/service/impl/StudentDeleteServiceImpl.java（実装クラス）：

    @Service
@Transactional
public class StudentDeleteServiceImpl implements StudentDeleteService {

    private static final Logger logger = LoggerFactory.getLogger(StudentDeleteServiceImpl.class);

    @Autowired
    private StudentRepository studentRepository;

    /**
     * 論理削除を実行する
     */
    @Override
    @Transactional
    public void logicalDeleteStudent(Long studentId, String deleteReason, String deletedBy) 
            throws ServiceException {

        logger.info(&quot;論理削除処理開始：学生ID={}, 削除者={}&quot;, studentId, deletedBy);

        try {
            // 1. 学生の存在確認
            Student student = studentRepository.findById(studentId)
                .orElseThrow(() -&gt; new ServiceException(&quot;学生が見つかりません: &quot; + studentId));

            // 2. 既に削除済みかチェック
            if (student.isDeleted()) {
                throw new ServiceException(&quot;既に削除済みの学生です: &quot; + studentId);
            }

            // 3. 削除可能性チェック
            if (!canDeleteStudent(studentId)) {
                throw new ServiceException(&quot;この学生は削除できません: &quot; + studentId);
            }

            // 4. 論理削除の実行
            student.logicalDelete(deletedBy, deleteReason);
            studentRepository.save(student);

            // 5. 削除ログの記録
            logDeleteOperation(studentId, &quot;LOGICAL_DELETE&quot;, deletedBy, deleteReason);

            logger.info(&quot;論理削除完了：学生ID={}, 削除者={}&quot;, studentId, deletedBy);

        } catch (ServiceException e) {
            logger.error(&quot;論理削除でServiceExceptionが発生：学生ID=&quot; + studentId, e);
            throw e;
        } catch (Exception e) {
            logger.error(&quot;論理削除で予期しないエラーが発生：学生ID=&quot; + studentId, e);
            throw new ServiceException(&quot;論理削除の処理中にエラーが発生しました&quot;, e);
        }
    }

    /**
     * 物理削除を実行する
     */
    @Override
    @Transactional
    public void physicalDeleteStudent(Long studentId, String deleteReason, String deletedBy) 
            throws ServiceException {

        logger.info(&quot;物理削除処理開始：学生ID={}, 削除者={}&quot;, studentId, deletedBy);

        try {
            // 1. 学生の存在確認
            Student student = studentRepository.findById(studentId)
                .orElseThrow(() -&gt; new ServiceException(&quot;学生が見つかりません: &quot; + studentId));

            // 2. 削除可能性チェック
            if (!canDeleteStudent(studentId)) {
                throw new ServiceException(&quot;この学生は削除できません: &quot; + studentId);
            }

            // 3. 削除ログの記録（削除前に実行）
            logDeleteOperation(studentId, &quot;PHYSICAL_DELETE&quot;, deletedBy, deleteReason);

            // 4. 物理削除の実行
            studentRepository.deleteById(studentId);

            logger.info(&quot;物理削除完了：学生ID={}, 削除者={}&quot;, studentId, deletedBy);

        } catch (ServiceException e) {
            logger.error(&quot;物理削除でServiceExceptionが発生：学生ID=&quot; + studentId, e);
            throw e;
        } catch (Exception e) {
            logger.error(&quot;物理削除で予期しないエラーが発生：学生ID=&quot; + studentId, e);
            throw new ServiceException(&quot;物理削除の処理中にエラーが発生しました&quot;, e);
        }
    }

    /**
     * 論理削除を取り消す（復元）
     */
    @Override
    @Transactional
    public void restoreStudent(Long studentId, String restoredBy) throws ServiceException {

        logger.info(&quot;復元処理開始：学生ID={}, 復元者={}&quot;, studentId, restoredBy);

        try {
            // 1. 学生の存在確認
            Student student = studentRepository.findById(studentId)
                .orElseThrow(() -&gt; new ServiceException(&quot;学生が見つかりません: &quot; + studentId));

            // 2. 削除済みかチェック
            if (!student.isDeleted()) {
                throw new ServiceException(&quot;この学生は削除されていません: &quot; + studentId);
            }

            // 3. 復元の実行
            student.restore();
            studentRepository.save(student);

            // 4. 復元ログの記録
            logDeleteOperation(studentId, &quot;RESTORE&quot;, restoredBy, &quot;データ復元&quot;);

            logger.info(&quot;復元完了：学生ID={}, 復元者={}&quot;, studentId, restoredBy);

        } catch (ServiceException e) {
            logger.error(&quot;復元でServiceExceptionが発生：学生ID=&quot; + studentId, e);
            throw e;
        } catch (Exception e) {
            logger.error(&quot;復元で予期しないエラーが発生：学生ID=&quot; + studentId, e);
            throw new ServiceException(&quot;復元の処理中にエラーが発生しました&quot;, e);
        }
    }

    @Controller
@RequestMapping(&quot;/students&quot;)
public class StudentDeleteController {

    private static final Logger logger = LoggerFactory.getLogger(StudentDeleteController.class);

    @Autowired
    private StudentService studentService;

    @Autowired
    private StudentDeleteService studentDeleteService;

    /**
     * 削除確認画面を表示
     */
    @GetMapping(&quot;/{id}/delete/confirm&quot;)
    public String showDeleteConfirm(@PathVariable Long id, Model model) {
        try {
            StudentDto student = studentService.findById(id);

            // 削除フォームの初期化
            DeleteForm deleteForm = new DeleteForm();
            deleteForm.setStudentId(id);
            deleteForm.setDeleteType(&quot;LOGICAL&quot;); // デフォルトは論理削除

            model.addAttribute(&quot;student&quot;, student);
            model.addAttribute(&quot;deleteForm&quot;, deleteForm);

            return &quot;students/delete-confirm&quot;;

        } catch (ServiceException e) {
            logger.error(&quot;削除確認画面表示エラー: 学生ID=&quot; + id, e);
            model.addAttribute(&quot;errorMessage&quot;, &quot;学生情報の取得に失敗しました&quot;);
            return &quot;redirect:/students/search/results&quot;;
        }
    }

    /**
     * 削除を実行
     */
    @PostMapping(&quot;/delete/execute&quot;)
    public String executeDelete(@Valid @ModelAttribute DeleteForm deleteForm,
                               BindingResult bindingResult,
                               RedirectAttributes redirectAttributes) {

        if (bindingResult.hasErrors()) {
            redirectAttributes.addFlashAttribute(&quot;errorMessage&quot;, &quot;入力内容に不備があります&quot;);
            return &quot;redirect:/students/&quot; + deleteForm.getStudentId() + &quot;/delete/confirm&quot;;
        }

        try {
            if (&quot;LOGICAL&quot;.equals(deleteForm.getDeleteType())) {
                // 論理削除の実行
                studentDeleteService.logicalDeleteStudent(
                    deleteForm.getStudentId(),
                    deleteForm.getDeleteReason(),
                    deleteForm.getDeletedBy()
                );
                redirectAttributes.addFlashAttribute(&quot;successMessage&quot;, &quot;学生の論理削除が完了しました&quot;);

            } else if (&quot;PHYSICAL&quot;.equals(deleteForm.getDeleteType())) {
                // 物理削除の実行
                studentDeleteService.physicalDeleteStudent(
                    deleteForm.getStudentId(),
                    deleteForm.getDeleteReason(),
                    deleteForm.getDeletedBy()
                );
                redirectAttributes.addFlashAttribute(&quot;successMessage&quot;, &quot;学生の物理削除が完了しました&quot;);

            } else {
                redirectAttributes.addFlashAttribute(&quot;errorMessage&quot;, &quot;無効な削除タイプです&quot;);
                return &quot;redirect:/students/&quot; + deleteForm.getStudentId() + &quot;/delete/confirm&quot;;
            }

            return &quot;redirect:/students/search/results&quot;;

        } catch (ServiceException e) {
            logger.error(&quot;削除実行エラー: 学生ID=&quot; + deleteForm.getStudentId(), e);
            redirectAttributes.addFlashAttribute(&quot;errorMessage&quot;, &quot;削除処理中にエラーが発生しました: &quot; + e.getMessage());
            return &quot;redirect:/students/&quot; + deleteForm.getStudentId() + &quot;/delete/confirm&quot;;
        }
    }

    /**
     * 復元確認画面を表示
     */
    @GetMapping(&quot;/{id}/restore/confirm&quot;)
    public String showRestoreConfirm(@PathVariable Long id, Model model) {
        try {
            StudentDto student = studentService.findById(id);

            if (!student.isDeleted()) {
                model.addAttribute(&quot;errorMessage&quot;, &quot;この学生は削除されていません&quot;);
                return &quot;redirect:/students/search/results&quot;;
            }

            model.addAttribute(&quot;student&quot;, student);
            return &quot;students/restore-confirm&quot;;

        } catch (ServiceException e) {
            logger.error(&quot;復元確認画面表示エラー: 学生ID=&quot; + id, e);
            model.addAttribute(&quot;errorMessage&quot;, &quot;学生情報の取得に失敗しました&quot;);
            return &quot;redirect:/students/search/results&quot;;
        }
    }

    /**
     * 復元を実行
     */
    @PostMapping(&quot;/{id}/restore/execute&quot;)
    public String executeRestore(@PathVariable Long id,
                                @RequestParam String restoredBy,
                                RedirectAttributes redirectAttributes) {

        if (restoredBy == null || restoredBy.trim().isEmpty()) {
            redirectAttributes.addFlashAttribute(&quot;errorMessage&quot;, &quot;復元実行者名を入力してください&quot;);
            return &quot;redirect:/students/&quot; + id + &quot;/restore/confirm&quot;;
        }

        try {
            studentDeleteService.restoreStudent(id, restoredBy.trim());
            redirectAttributes.addFlashAttribute(&quot;successMessage&quot;, &quot;学生の復元が完了しました&quot;);
            return &quot;redirect:/students/search/results&quot;;

        } catch (ServiceException e) {
            logger.error(&quot;復元実行エラー: 学生ID=&quot; + id, e);
            redirectAttributes.addFlashAttribute(&quot;errorMessage&quot;, &quot;復元処理中にエラーが発生しました: &quot; + e.getMessage());
            return &quot;redirect:/students/&quot; + id + &quot;/restore/confirm&quot;;
        }
    }

    /**
     * 削除履歴を取得（Ajax用）

    @GetMapping(&quot;/{id}/delete-history&quot;)
    @ResponseBody
    public String getDeleteHistory(@PathVariable Long id) {

            return &quot;削除履歴の取得に失敗しました&quot;;
        }
    }
