}

public class Student {

    学習方法: 以下のコードをsection毎に分けて理解してから実装してください： - Section A: N+1問題の発生パターン確認 - Section B: @EntityGraphによる解決 - 重要: N+1問題の仕組みを理解してから解決策を実装してください。

    Section A: N+1問題の発生例

    まず、問題のあるコードを確認しましょう：

    @Service
@Transactional(readOnly = true)
public class ProblematicStudentService {

    @Autowired
    private StudentRepository studentRepository;

    /**
     * ❌ N+1問題が発生するメソッド
     */
    public List<StudentDTO> findAllStudentsWithCourse() {
        // 1. 学生一覧を取得（1回のクエリ）
        List<Student> students = studentRepository.findByDeletedAtIsNull();

        List<StudentDTO> result = new ArrayList<>();

        // 2. 各学生のコース情報を取得（N回のクエリ）
        for (Student student : students) {
            StudentDTO dto = new StudentDTO();
            dto.setId(student.getId());
            dto.setName(student.getName());
            dto.setEmail(student.getEmail());

            // ❌ ここでN+1問題発生！
            Course course = student.getCourse(); // 遅延読み込みでクエリ発行
            if (course != null) {
                dto.setCourseName(course.getName());
            }

            result.add(dto);
        }

        return result;
    }
}

    このコードは以下のような問題があります：

    学生数が100人の場合：
1回目: SELECT * FROM students WHERE deleted_at IS NULL  (学生一覧取得)
2回目: SELECT * FROM courses WHERE id = ?               (学生1のコース)
3回目: SELECT * FROM courses WHERE id = ?               (学生2のコース)
...
101回目: SELECT * FROM courses WHERE id = ?             (学生100のコース)

合計: 101回のクエリ実行！

    Section B: @EntityGraphによる解決

    N+1問題を解決する最も効果的な方法：

    @Repository
public interface OptimizedStudentRepository extends JpaRepository<Student, Long> {

    /**
     * ✅ @EntityGraphでN+1問題を解決
     */
    @EntityGraph(attributePaths = {"course"})
    @Query("SELECT s FROM Student s WHERE s.deletedAt IS NULL")
    List<Student> findAllWithCourse();

    /**
     * ✅ ページネーション対応の効率的検索
     */
    @EntityGraph(attributePaths = {"course"})
    Page<Student> findByNameContainingAndDeletedAtIsNull(
        String name, Pageable pageable);

    /**
     * ✅ 複数の関連エンティティを同時取得
     */
    @EntityGraph(attributePaths = {"course", "grades"})
    @Query("SELECT s FROM Student s WHERE s.id = ?1")
    Optional<Student> findByIdWithCourseAndGrades(Long id);

    /**
     * ✅ JOIN FETCHによる明示的な結合
     */
    @Query("SELECT s FROM Student s " +
           "LEFT JOIN FETCH s.course " +
           "WHERE s.deletedAt IS NULL " +
           "ORDER BY s.createdAt DESC")
    List<Student> findAllWithJoinFetch();
}

    最適化されたサービス実装

    @Service
@Transactional(readOnly = true)
public class OptimizedStudentService {

    @Autowired
    private OptimizedStudentRepository studentRepository;

    /**
     * ✅ N+1問題を解決した効率的な学生取得
     */
    public List<StudentDTO> findStudentsWithDetails() {
        // 1回のクエリで学生とコース情報を同時取得
        List<Student> students = studentRepository.findAllWithCourse();

        return students.stream()
            .map(this::convertToDTO)
            .collect(Collectors.toList());
    }

    /**
     * ✅ ページネーション対応の効率的検索
     */
    public Page<StudentDTO> searchStudentsOptimized(
            String name, String email, Pageable pageable) {

        // EntityGraphでコースを一括取得
        Page<Student> studentPage;

        if (StringUtils.hasText(name)) {
            studentPage = studentRepository
                .findByNameContainingAndDeletedAtIsNull(name, pageable);
        } else {
            studentPage = studentRepository.findAll(pageable);
        }

        return studentPage.map(this::convertToDTO);
    }

    private StudentDTO convertToDTO(Student student) {
        StudentDTO dto = new StudentDTO();
        dto.setId(student.getId());
        dto.setName(student.getName());
        dto.setEmail(student.getEmail());

        // コース情報は既に取得済みなので追加クエリなし
        Course course = student.getCourse();
        if (course != null) {
            dto.setCourseName(course.getName());
        }

        return dto;
    }
}

    @Configuration
@EnableCaching
public class CacheConfig {

    @Bean
    public CacheManager cacheManager(RedisConnectionFactory connectionFactory) {
        RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
            // キャッシュの有効期間：10分
            .entryTtl(Duration.ofMinutes(10))
            // キーのシリアライザー
            .serializeKeysWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new StringRedisSerializer()))
            // 値のシリアライザー  
            .serializeValuesWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new GenericJackson2JsonRedisSerializer()))
            // nullの場合はキャッシュしない
            .disableCachingNullValues();

        return RedisCacheManager.builder(connectionFactory)
            .cacheDefaults(config)
            // 特定キャッシュの個別設定
            .withCacheConfiguration("students", 
                config.entryTtl(Duration.ofMinutes(30)))
            .withCacheConfiguration("courses", 
                config.entryTtl(Duration.ofHours(1)))
            .build();
    }

    @Bean
    public RedisTemplate<String, Object> redisTemplate(
            RedisConnectionFactory connectionFactory) {
        RedisTemplate<String, Object> template = new RedisTemplate<>();
        template.setConnectionFactory(connectionFactory);

        // キーのシリアライザー設定
        template.setKeySerializer(new StringRedisSerializer());
        template.setHashKeySerializer(new StringRedisSerializer());

        // 値のシリアライザー設定
        template.setValueSerializer(new GenericJackson2JsonRedisSerializer());
        template.setHashValueSerializer(new GenericJackson2JsonRedisSerializer());

        template.afterPropertiesSet();
        return template;

     * ✅ データベースクエリのパフォーマンス監視
     */
    @Around("execution(* com.example.studentmanagement.repository.*.*(..))")
    public Object monitorRepositoryPerformance(ProceedingJoinPoint joinPoint) 
            throws Throwable {

        String methodName = joinPoint.getSignature().getName();
        String className = joinPoint.getTarget().getClass().getSimpleName();

        Timer.Sample sample = Timer.start(meterRegistry);

        try {
            Object result = joinPoint.proceed();

            // 成功メトリクス
            sample.stop(Timer.builder("repository.query.duration")
                .tag("class", className)
                .tag("method", methodName)
                .tag("status", "success")
                .register(meterRegistry));

            log.debug("Repository method executed: {}.{}", className, methodName);
            return result;

        } catch (Exception e) {
            // エラーメトリクス
            sample.stop(Timer.builder("repository.query.duration")
                .tag("class", className)
                .tag("method", methodName)
                .tag("status", "error")
                .register(meterRegistry));

            meterRegistry.counter("repository.query.errors",
                "class", className,
                "method", methodName,
                "exception", e.getClass().getSimpleName())
                .increment();

            log.error("Repository method error: {}.{}", className, methodName, e);
            throw e;
        }
    }

    /**
     * ✅ サービス層のパフォーマンス監視
     */
    @Around("execution(* com.example.studentmanagement.service.*.*(..))")

            throws Throwable {

        String methodName = joinPoint.getSignature().getName();
        String className = joinPoint.getTarget().getClass().getSimpleName();

        Timer.Sample sample = Timer.start(meterRegistry);

                .tag("class", className)
                .tag("method", methodName)
                .register(meterRegistry));

        } catch (Exception e) {
            sample.stop(Timer.builder("service.method.duration")
                .tag("class", className)
                .tag("method", methodName)
                .tag("status", "error")

        }
    }

    @Service
@Component
public class CustomMetricsService {

    private final MeterRegistry meterRegistry;
    private final StudentRepository studentRepository;

    public CustomMetricsService(MeterRegistry meterRegistry, 
                               StudentRepository studentRepository) {
        this.meterRegistry = meterRegistry;
        this.studentRepository = studentRepository;

        // 定期的なメトリクス更新
        Gauge.builder("students.total.count")
            .description("Total number of students")
            .register(meterRegistry, this, CustomMetricsService::getTotalStudents);

        Gauge.builder("students.active.count")
            .description("Number of active students")
            .register(meterRegistry, this, CustomMetricsService::getActiveStudents);
    }

    public double getTotalStudents() {
        return studentRepository.count();
    }

    public double getActiveStudents() {
        return studentRepository.countByDeletedAtIsNull();
    }

    /**
     * ✅ 検索パフォーマンスの記録
     */
    public void recordSearchPerformance(String searchType, long duration, int resultCount) {
        Timer.builder("search.performance")
            .tag("type", searchType)
            .register(meterRegistry)
            .record(duration, TimeUnit.MILLISECONDS);

        meterRegistry.counter("search.results",
            "type", searchType,
            "count_range", getCountRange(resultCount))
            .increment();
    }

    private String getCountRange(int count) {
        if (count == 0) return "zero";
        if (count <= 10) return "small";
        if (count <= 100) return "medium";
        return "large";
    }
}

    @RestController
@RequestMapping("/api/performance")
public class PerformanceTestController {

    @Autowired
    private CachedStudentService studentService;

    @Autowired
    private CustomMetricsService metricsService;

    /**
     * ✅ 負荷テスト用：基本検索
     */
    @GetMapping("/students/search")
    public ResponseEntity<PerformanceTestResult> searchStudentsForLoadTest(
            @RequestParam(required = false) String name,
            @RequestParam(required = false) String email,
            @RequestParam(defaultValue = "0") int page,
            @RequestParam(defaultValue = "20") int size) {

        long startTime = System.currentTimeMillis();

        try {
            Page<StudentDTO> result = studentService.searchStudentsWithCache(
                name, email, page, size);

            long duration = System.currentTimeMillis() - startTime;

            // メトリクス記録
            metricsService.recordSearchPerformance("basic_search", duration, 
                                                 (int) result.getTotalElements());

            PerformanceTestResult testResult = new PerformanceTestResult();
            testResult.setSuccess(true);
            testResult.setDuration(duration);
            testResult.setResultCount(result.getTotalElements());
            testResult.setData(result.getContent());

            return ResponseEntity.ok(testResult);

        } catch (Exception e) {
            long duration = System.currentTimeMillis() - startTime;

            PerformanceTestResult testResult = new PerformanceTestResult();
            testResult.setSuccess(false);
            testResult.setDuration(duration);
            testResult.setError(e.getMessage());

            return ResponseEntity.status(500).body(testResult);
        }
    }
