    🔍 ステップ1: Pageableとは何か

    Pageableは、Spring Data JPAが提供するページネーション機能の中核です。大量のデータを効率的に分割表示するための仕組みです。

    なぜページネーションが重要なのか

    // 悪い例：全データを一度に取得（パフォーマンス問題）
List&lt;Student&gt; allStudents = studentRepository.findAll(); // 10,000件のデータ
// → メモリ大量消費、画面表示の遅延、ユーザビリティ悪化

// 良い例：ページネーション活用（企業システムレベル）
Pageable pageable = PageRequest.of(0, 20); // 1ページ目、20件ずつ
Page&lt;Student&gt; studentPage = studentRepository.findAll(pageable);
// → 効率的なメモリ使用、高速表示、優れたユーザビリティ

    public Page&lt;StudentDto&gt; searchStudentsWithPaging(StudentSearchCriteria criteria, &lt;br&gt;
                                                     Pageable pageable) &lt;br&gt;
            throws ServiceException {
        try {
            log.info(&quot;ページネーション検索開始: criteria={}, page={}, size={}&quot;, &lt;br&gt;
                    criteria, pageable.getPageNumber(), pageable.getPageSize());

            // Week 10.1のSpecification活用

            // ページネーション検索実行

            // Entity → DTO変換（Page対応）
            Page&lt;StudentDto&gt; result = studentPage.map(this::convertToDto);

            log.info(&quot;ページネーション検索完了: 総件数={}, 総ページ数={}&quot;, &lt;br&gt;
                    result.getTotalElements(), result.getTotalPages());

            return result;

        } catch (Exception e) {
            log.error(&quot;ページネーション検索でエラーが発生しました&quot;, e);
            throw new ServiceException(&quot;検索処理中にエラーが発生しました&quot;, e);
        }
    }

    // 検索結果統計情報取得
    public SearchResultSummary getSearchSummary(StudentSearchCriteria criteria) &lt;br&gt;
            throws ServiceException {
        try {
            // 基本統計
            long totalCount = studentRepository.count(&lt;br&gt;
                StudentSpecifications.searchByCriteria(criteria));

            // 専攻別統計
            Map&lt;String, Long&gt; majorStats = studentRepository.countByMajorGrouping(criteria);

            // 学年別統計
            Map&lt;Integer, Long&gt; gradeStats = studentRepository.countByGradeGrouping(criteria);

            return new SearchResultSummary(totalCount, majorStats, gradeStats);

        } catch (Exception e) {
            throw new ServiceException(&quot;統計情報取得でエラーが発生しました&quot;, e);
        }
    }
}

    @Controller
@RequestMapping(&quot;/students&quot;)
public class StudentSearchController {

    @Autowired
    private StudentSearchService studentSearchService;

    // Week 10.2で実装済み：検索フォーム表示
    @GetMapping(&quot;/search&quot;)
    public String searchForm(Model model) {
        // 既存実装（変更なし）
    }

    // Week 10.2で実装済み：検索実行（PRGパターン）
    @PostMapping(&quot;/search&quot;)
    public String executeSearch(@Valid @ModelAttribute(&quot;searchCriteria&quot;) StudentSearchForm searchForm,&lt;br&gt;
                               BindingResult bindingResult,&lt;br&gt;
                               RedirectAttributes redirectAttributes,&lt;br&gt;
                               Model model) {
        // 既存実装に検索結果画面リダイレクト追加
        if (bindingResult.hasErrors()) {
            return &quot;student/search-form&quot;;
        }

        try {
            StudentSearchCriteria criteria = searchForm.toCriteria();

            // 検索条件をFlashAttributeに保存（PRGパターン）
            redirectAttributes.addFlashAttribute(&quot;searchCriteria&quot;, criteria);
            redirectAttributes.addFlashAttribute(&quot;searchForm&quot;, searchForm);

            return &quot;redirect:/students/search/results&quot;;

        } catch (ServiceException e) {
            bindingResult.reject(&quot;search.error&quot;, e.getMessage());
            return &quot;student/search-form&quot;;
        }
    }

    // Week 10.3新規実装：検索結果表示（ページネーション対応）
    @GetMapping(&quot;/search/results&quot;)
    public String searchResults(@RequestParam(defaultValue = &quot;0&quot;) int page,&lt;br&gt;
                               @RequestParam(defaultValue = &quot;10&quot;) int size,&lt;br&gt;
                               @RequestParam(defaultValue = &quot;id&quot;) String sortBy,&lt;br&gt;
                               @RequestParam(defaultValue = &quot;asc&quot;) String sortDirection,&lt;br&gt;
                               @ModelAttribute(&quot;searchCriteria&quot;) StudentSearchCriteria criteria,&lt;br&gt;
                               Model model) {

        try {
            // パラメータバリデーション
            if (page &lt; 0) page = 0;
            if (size &lt; 1 || size &gt; 100) size = 10;

            // Pageableオブジェクト作成
            Sort.Direction direction = Sort.Direction.fromString(sortDirection);
            Sort sort = Sort.by(direction, sortBy);
            Pageable pageable = PageRequest.of(page, size, sort);

            // Week 10.1-10.2基盤活用：ページネーション検索実行
            Page&lt;StudentDto&gt; searchResults = studentSearchService.searchStudentsWithPaging(criteria, pageable);

            // 検索結果統計情報取得
            SearchResultSummary summary = studentSearchService.getSearchSummary(criteria);

            // 検索結果・ページ情報をモデルに設定
            model.addAttribute(&quot;searchResults&quot;, searchResults);
            model.addAttribute(&quot;searchSummary&quot;, summary);
            model.addAttribute(&quot;currentPage&quot;, page);
            model.addAttribute(&quot;totalPages&quot;, searchResults.getTotalPages());
            model.addAttribute(&quot;totalElements&quot;, searchResults.getTotalElements());
            model.addAttribute(&quot;sortBy&quot;, sortBy);
            model.addAttribute(&quot;sortDirection&quot;, sortDirection);
            model.addAttribute(&quot;pageSize&quot;, size);

            // ページネーション用のページ番号リスト生成
            List&lt;Integer&gt; pageNumbers = generatePageNumbers(page, searchResults.getTotalPages());
            model.addAttribute(&quot;pageNumbers&quot;, pageNumbers);

            return &quot;student/search-results&quot;;

        } catch (ServiceException e) {
            model.addAttribute(&quot;errorMessage&quot;, &quot;検索結果の表示中にエラーが発生しました: &quot; + e.getMessage());
            return &quot;student/search-form&quot;;
        }
    }

    // ページネーション用のページ番号リスト生成
    private List&lt;Integer&gt; generatePageNumbers(int currentPage, int totalPages) {
        List&lt;Integer&gt; pageNumbers = new ArrayList&lt;&gt;();
        int startPage = Math.max(0, currentPage - 2);
        int endPage = Math.min(totalPages - 1, currentPage + 2);

        for (int i = startPage; i &lt;= endPage; i++) {
            pageNumbers.add(i);
        }

        return pageNumbers;
    }
}

public enum SortOption {
    NAME_ASC(&quot;name&quot;, &quot;asc&quot;, &quot;学生名（昇順）&quot;),
    NAME_DESC(&quot;name&quot;, &quot;desc&quot;, &quot;学生名（降順）&quot;),

    MAJOR_ASC(&quot;major&quot;, &quot;asc&quot;, &quot;専攻（昇順）&quot;),
    MAJOR_DESC(&quot;major&quot;, &quot;desc&quot;, &quot;専攻（降順）&quot;),
    EMAIL_ASC(&quot;email&quot;, &quot;asc&quot;, &quot;メール（昇順）&quot;),
    EMAIL_DESC(&quot;email&quot;, &quot;desc&quot;, &quot;メール（降順）&quot;),
    CREATED_DATE_ASC(&quot;createdDate&quot;, &quot;asc&quot;, &quot;登録日（昇順）&quot;),
    CREATED_DATE_DESC(&quot;createdDate&quot;, &quot;desc&quot;, &quot;登録日（降順）&quot;);

    private final String sortBy;
    private final String direction;
    private final String displayName;

    SortOption(String sortBy, String direction, String displayName) {
        this.sortBy = sortBy;
        this.direction = direction;
        this.displayName = displayName;
    }

    // getter メソッド
    public String getSortBy() { return sortBy; }
    public String getDirection() { return direction; }
    public String getDisplayName() { return displayName; }

    // Sort オブジェクト生成
    public Sort createSort() {
        Sort.Direction dir = Sort.Direction.fromString(direction);
        return Sort.by(dir, sortBy);
    }

    // 表示用の文字列生成
    public static SortOption fromParameters(String sortBy, String direction) {
        for (SortOption option : values()) {
            if (option.sortBy.equals(sortBy) &amp;&amp; option.direction.equals(direction)) {
                return option;
            }
        }
        return NAME_ASC; // デフォルト
    }
}

    // 検索履歴管理
class SearchHistory {
    constructor() {
        this.maxHistory = 5;
        this.storageKey = &apos;studentSearchHistory&apos;;
    }

    // 検索履歴に追加
    addSearch(criteria) {
        const history = this.getHistory();
        const searchItem = {
            criteria: criteria,
            timestamp: new Date().toISOString(),
            displayText: this.generateDisplayText(criteria)
        };

        // 重複チェック
        const existingIndex = history.findIndex(item =&gt; 
            JSON.stringify(item.criteria) === JSON.stringify(criteria)
        );

        if (existingIndex &gt;= 0) {
            history.splice(existingIndex, 1);
        }

        history.unshift(searchItem);

        // 最大件数制限
        if (history.length &gt; this.maxHistory) {
            history.splice(this.maxHistory);
        }

        localStorage.setItem(this.storageKey, JSON.stringify(history));
    }

    // 検索履歴取得
    getHistory() {
        const stored = localStorage.getItem(this.storageKey);
        return stored ? JSON.parse(stored) : [];
    }

    // 表示テキスト生成
    generateDisplayText(criteria) {
        const parts = [];
        if (criteria.name) parts.push(`名前:${criteria.name}`);
        if (criteria.grade) parts.push(`学年:${criteria.grade}年`);
        if (criteria.major) parts.push(`専攻:${criteria.major}`);
        if (criteria.email) parts.push(`メール:${criteria.email}`);

        return parts.length &gt; 0 ? parts.join(&apos;, &apos;) : &apos;全件検索&apos;;
    }

    // 履歴表示HTML生成
    generateHistoryHTML() {
        const history = this.getHistory();
        if (history.length === 0) {
            return &apos;&lt;p class=&quot;text-muted&quot;&gt;検索履歴はありません&lt;/p&gt;&apos;;
        }

        return history.map((item, index) =&gt; `
            &lt;div class=&quot;search-history-item&quot;&gt;
                &lt;a href=&quot;#&quot; onclick=&quot;applySearchHistory(${index})&quot; class=&quot;text-decoration-none&quot;&gt;
                    &lt;i class=&quot;fas fa-history&quot;&gt;&lt;/i&gt; ${item.displayText}
                &lt;/a&gt;
                &lt;small class=&quot;text-muted d-block&quot;&gt;
                    ${new Date(item.timestamp).toLocaleString()}
                &lt;/small&gt;
            &lt;/div&gt;
        `).join(&apos;&apos;);
    }
}
