    @Test
@DisplayName("学生検索から削除までの統合フロー")
void testCompleteStudentManagementFlow() {
    // 1. 学生作成テスト（基本CRUD）
    Student testStudent = new Student();
    testStudent.setName("統合テスト太郎");
    testStudent.setEmail("integration@test.com");
    testStudent.setAge(22);
    testStudent.setMajor("コンピュータサイエンス");

    Student savedStudent = studentRepository.save(testStudent);
    assertThat(savedStudent.getId()).isNotNull();
    assertThat(savedStudent.getName()).isEqualTo("統合テスト太郎");

    // 2. 検索機能テスト（Week 10.1-10.4の機能）
    StudentSearchForm searchForm = new StudentSearchForm();
    searchForm.setName("統合テスト");
    searchForm.setNameMatchType("partial");

    Pageable pageable = PageRequest.of(0, 10);
    Page<Student> searchResults = studentService.searchStudents(searchForm, pageable);

    assertThat(searchResults.getTotalElements()).isGreaterThan(0);
    assertThat(searchResults.getContent().get(0).getName()).contains("統合テスト");

    // 3. Ajax検索機能テスト（Week 10.8の機能）
    // JavaScriptなしで Ajax レスポンスをテスト
    ResponseEntity<String> ajaxResponse = restTemplate
        .getForEntity("/api/students/search?name=統合テスト&ajax=true", String.class);

    assertThat(ajaxResponse.getStatusCode()).isEqualTo(HttpStatus.OK);
    assertThat(ajaxResponse.getBody()).contains("統合テスト太郎");

    // 4. パフォーマンス最適化確認（Week 10.9の機能）
    long startTime = System.currentTimeMillis();
    studentService.searchStudents(searchForm, pageable);
    long endTime = System.currentTimeMillis();

    // キャッシュ効果で2回目は高速化されることを確認
    long cachedStartTime = System.currentTimeMillis();
    studentService.searchStudents(searchForm, pageable);

import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.web.csrf.CookieCsrfTokenRepository;

@Configuration
@EnableWebSecurity
@EnableMethodSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .authorizeHttpRequests(authz -> authz
                // 静的リソースとログインページは許可
                .requestMatchers("/login", "/css/**", "/js/**", "/images/**").permitAll()
                // 管理者機能は ADMIN ロール必須
                .requestMatchers("/api/admin/**", "/admin/**").hasRole("ADMIN")
                // 学生管理機能は USER または ADMIN ロール
                .requestMatchers("/api/students/**", "/students/**").hasAnyRole("USER", "ADMIN")
                // その他は認証必要
                .anyRequest().authenticated()
            )
            .formLogin(form -> form
                .loginPage("/login")
                .defaultSuccessUrl("/students", true)
                .failureUrl("/login?error=true")
                .permitAll()
            )
            .logout(logout -> logout
                .logoutUrl("/logout")
                .logoutSuccessUrl("/login?logout=true")
                .invalidateHttpSession(true)
                .deleteCookies("JSESSIONID")
                .permitAll()
            )
            .csrf(csrf -> csrf
                .csrfTokenRepository(CookieCsrfTokenRepository.withHttpOnlyFalse())
            )
            .sessionManagement(session -> session
                .maximumSessions(1)
                .maxSessionsPreventsLogin(false)
            );

        return http.build();
    }

    @Bean
    public PasswordEncoder passwordEncoder() {
        return new BCryptPasswordEncoder();
    }
}

    private StudentRepository studentRepository;

    public Health health() {
        Health.Builder builder = Health.up();

            // データベース接続確認
            long studentCount = studentRepository.count();
            builder.withDetail("database", Map.of(
                "status", "UP",
                "studentCount", studentCount,
                "connection", "OK"
            ));

            redisTemplate.opsForValue().set("health_check", "ok");
            String redisCheck = (String) redisTemplate.opsForValue().get("health_check");
            builder.withDetail("redis", Map.of(
                "status", "UP",
                "connection", "ok".equals(redisCheck) ? "OK" : "ERROR"
            ));

            // パフォーマンスメトリクス
            Runtime runtime = Runtime.getRuntime();
            long totalMemory = runtime.totalMemory();
            long freeMemory = runtime.freeMemory();
            long usedMemory = totalMemory - freeMemory;

                "memory_total", totalMemory / (1024 * 1024) + " MB",
                "memory_used", usedMemory / (1024 * 1024) + " MB",
                "memory_free", freeMemory / (1024 * 1024) + " MB",

            ));

                .withDetail("exception", e.getClass().getSimpleName())
                .build();

            "timestamp", LocalDateTime.now(),
            "path", request.getRequestURI(),

        Map<String, Object> errorResponse = Map.of(
            "error", "Validation Error",
            "message", ex.getMessage(),
            "timestamp", LocalDateTime.now(),
            "path", request.getRequestURI(),
            "status", HttpStatus.BAD_REQUEST.value()
        );

        return ResponseEntity.status(HttpStatus.BAD_REQUEST).body(errorResponse);
    }

    @ExceptionHandler(Exception.class)
    public ModelAndView handleGeneralException(Exception ex, HttpServletRequest request) {

    public Page<StudentDTO> searchStudents(

            "deletedCount", deletedCount,

    }
