    /**
 * 一括操作用JavaScript
 * Week 10.6: チェックボックス制御と選択状態管理
 */

class SelectionManager {
    constructor() {
        this.selectedIds = new Set();
        this.currentPageIds = new Set();
        this.studentData = new Map(); // 学生データのキャッシュ
        this.initialize();
    }

    /**
     * 初期化処理
     */
    initialize() {
        // 現在ページの学生IDを収集
        this.collectCurrentPageIds();

        // 既存の選択状態を復元
        this.restoreSelectionState();

        // イベントリスナーを設定
        this.setupEventListeners();

        // UI状態を更新
        this.updateUI();
    }

    /**
     * 現在ページの学生IDを収集
     */
    collectCurrentPageIds() {
        this.currentPageIds.clear();
        this.studentData.clear();

        const checkboxes = document.querySelectorAll('.student-checkbox');
        checkboxes.forEach(checkbox => {
            const studentId = parseInt(checkbox.value);
            const row = checkbox.closest('tr');

            // 学生情報を収集
            const name = row.querySelector('td:nth-child(3)').textContent.trim();
            const email = row.querySelector('td:nth-child(4)').textContent.trim();
            const isDeleted = row.classList.contains('deleted-row');

            this.currentPageIds.add(studentId);
            this.studentData.set(studentId, {
                id: studentId,
                name: name,
                email: email,
                isDeleted: isDeleted
            });
        });
    }

    /**
     * 選択状態を復元（ページネーション対応）
     */
    restoreSelectionState() {
        this.currentPageIds.forEach(id => {
            const checkbox = document.querySelector(`input[value="${id}"]`);
            if (checkbox && this.selectedIds.has(id)) {
                checkbox.checked = true;
            }
        });
    }

    /**
     * イベントリスナーの設定
     */
    setupEventListeners() {
        // 全選択チェックボックス
        const selectAllCheckbox = document.getElementById('selectAll');
        if (selectAllCheckbox) {
            selectAllCheckbox.addEventListener('change', (e) => {
                this.toggleAllSelection(e.target.checked);
            });
        }

        // 個別チェックボックス
        document.querySelectorAll('.student-checkbox').forEach(checkbox => {
            checkbox.addEventListener('change', (e) => {
                this.toggleStudentSelection(parseInt(e.target.value), e.target.checked);
            });
        });
    }

    /**
     * 全選択/全解除の切り替え
     */
    toggleAllSelection(selected) {
        const enabledCheckboxes = document.querySelectorAll('.student-checkbox:not(:disabled)');

        enabledCheckboxes.forEach(checkbox => {
            const studentId = parseInt(checkbox.value);
            checkbox.checked = selected;

            if (selected) {
                this.selectedIds.add(studentId);
            } else {
                this.selectedIds.delete(studentId);
            }
        });

        this.updateUI();
    }

    /**
     * 個別学生の選択状態切り替え
     */
    toggleStudentSelection(studentId, selected) {
        if (selected) {
            this.selectedIds.add(studentId);
        } else {
            this.selectedIds.delete(studentId);
        }

        this.updateSelectAllState();
        this.updateBulkActionButtons();
    }

    /**
     * 全選択チェックボックスの状態を更新
     */
    updateSelectAllState() {
        const selectAllCheckbox = document.getElementById('selectAll');
        if (!selectAllCheckbox) return;

        const enabledCheckboxes = document.querySelectorAll('.student-checkbox:not(:disabled)');
        const enabledIds = Array.from(enabledCheckboxes).map(cb => parseInt(cb.value));
        const selectedEnabledIds = enabledIds.filter(id => this.selectedIds.has(id));

        if (selectedEnabledIds.length === 0) {
            selectAllCheckbox.checked = false;
            selectAllCheckbox.indeterminate = false;
        } else if (selectedEnabledIds.length === enabledIds.length) {
            selectAllCheckbox.checked = true;
            selectAllCheckbox.indeterminate = false;
        } else {
            selectAllCheckbox.checked = false;
            selectAllCheckbox.indeterminate = true;
        }
    }

    /**
     * 一括操作ボタンの有効/無効を更新
     */
    updateBulkActionButtons() {
        const hasSelection = this.selectedIds.size > 0;

        // 選択された学生の状態を確認
        const selectedStudents = this.getSelectedStudents();
        const hasActiveStudents = selectedStudents.some(student => !student.isDeleted);
        const hasDeletedStudents = selectedStudents.some(student => student.isDeleted);

        // ボタンの有効/無効を設定
        const bulkLogicalDeleteBtn = document.getElementById('bulkLogicalDeleteBtn');
        const bulkPhysicalDeleteBtn = document.getElementById('bulkPhysicalDeleteBtn');
        const bulkRestoreBtn = document.getElementById('bulkRestoreBtn');

        if (bulkLogicalDeleteBtn) {
            bulkLogicalDeleteBtn.disabled = !hasActiveStudents;
        }

        if (bulkPhysicalDeleteBtn) {
            bulkPhysicalDeleteBtn.disabled = !hasSelection;
        }

        if (bulkRestoreBtn) {
            bulkRestoreBtn.disabled = !hasDeletedStudents;
        }
    }

    /**
     * UI全体の更新
     */
    updateUI() {
        this.updateSelectAllState();
        this.updateBulkActionButtons();
    }

    /**
     * 選択された学生データを取得
     */
    getSelectedStudents() {
        return Array.from(this.selectedIds)
            .map(id => this.studentData.get(id))
            .filter(student => student !== undefined);
    }

    /**
     * 選択された学生IDを取得
     */
    getSelectedIds() {
        return Array.from(this.selectedIds);
    }

    /**
     * 選択をクリア
     */
    clearSelection() {
        this.selectedIds.clear();
        document.querySelectorAll('.student-checkbox').forEach(checkbox => {
            checkbox.checked = false;
        });
        this.updateUI();
    }
}

/**
 * 一括削除処理の結果を管理するクラス
 * Week 10.6: 一括削除機能

public class BulkDeleteResult {

    private List<Long> successIds = new ArrayList<>();
    private Map<Long, String> errorMap = new HashMap<>();

    /**
     * 成功した学生IDを追加
     */

     */
    public boolean hasErrors() {
        return !errorMap.isEmpty();
    }

     * 成功件数を取得
     */

                .map(Long::parseLong)
                .collect(Collectors.toList());

            response.put("success", false);
            response.put("message", "削除対象の学生が選択されていません");

            response.put("success", false);
            response.put("message", "不正な削除タイプです");
            return ResponseEntity.badRequest().body(response);
        }

        // 結果を返す
        response.put("success", true);
        response.put("successCount", result.getSuccessCount());
        response.put("errorCount", result.getErrorCount());

        if (result.hasErrors()) {
            response.put("errors", result.getErrorMap());
        }

        // ログ出力
        logger.info("一括{}削除完了: 成功={}, エラー={}, 実行者={}", 
                   deleteType, result.getSuccessCount(), result.getErrorCount(), deletedBy);

        return ResponseEntity.ok(response);

        response.put("success", false);

        return ResponseEntity.badRequest().body(response);
