    📖 1. @Queryアノテーションによる柔軟な検索

    1.1 JPQL（Java Persistence Query Language）の基本

    JPQLは、エンティティクラスを対象としたオブジェクト指向クエリ言語です：

    📘 @Query・JPQL実装パターン図

    @Repository
public interface StudentRepository extends JpaRepository<Student, Long> {

    // 1. 基本的なJPQLクエリ
    @Query("SELECT s FROM Student s WHERE s.name = :name")
    List<Student> findByNameExact(@Param("name") String name);

    // 2. 部分一致検索
    @Query("SELECT s FROM Student s WHERE s.name LIKE %:name%")
    List<Student> findByNameContaining(@Param("name") String name);

    // 3. 複数条件での検索
    @Query("SELECT s FROM Student s WHERE s.grade = :grade AND s.major = :major")
    List<Student> findByGradeAndMajor(@Param("grade") Integer grade, @Param("major") String major);

    // 4. 範囲検索
    @Query("SELECT s FROM Student s WHERE s.grade BETWEEN :minGrade AND :maxGrade")
    List<Student> findByGradeRange(@Param("minGrade") Integer minGrade, @Param("maxGrade") Integer maxGrade);

    // 5. 並び替え付き検索
    @Query("SELECT s FROM Student s WHERE s.major = :major ORDER BY s.name ASC")
    List<Student> findByMajorOrderByName(@Param("major") String major);
}

    1.2 ネイティブクエリの活用

    データベース固有の機能を使用したい場合は、ネイティブクエリを使用します：

    @Repository
public interface StudentRepository extends JpaRepository<Student, Long> {

    @Query(value = "SELECT * FROM students s WHERE DATE(s.created_date) = CURDATE()",

    List<Object[]> findMajorsWithMinStudentCount(@Param("minCount") Integer minCount);
}

    🔧 2. 動的検索条件の実装

    2.1 検索条件DTOの設計

    まず、検索条件を表現するDTO（Data Transfer Object）を作成します：

    📘 動的検索アーキテクチャ図

    public class StudentSearchCriteria {
    private String name;           // 名前（部分一致）
    private Integer grade;         // 学年（完全一致）
    private String major;          // 専攻（完全一致）
    private String email;          // メールアドレス（部分一致）
    private Integer minGrade;      // 学年範囲（最小値）
    private Integer maxGrade;      // 学年範囲（最大値）

    // デフォルトコンストラクタ
    public StudentSearchCriteria() {}

    // 全パラメータコンストラクタ
    public StudentSearchCriteria(String name, Integer grade, String major, 
                                String email, Integer minGrade, Integer maxGrade) {
        this.name = name;
        this.grade = grade;
        this.major = major;
        this.email = email;
        this.minGrade = minGrade;
        this.maxGrade = maxGrade;
    }

    // 検索条件が空かどうかをチェック
    public boolean isEmpty() {
        return (name == null || name.trim().isEmpty()) &&
               grade == null &&
               (major == null || major.trim().isEmpty()) &&
               (email == null || email.trim().isEmpty()) &&
               minGrade == null &&
               maxGrade == null;
    }

    // getter/setter
    public String getName() { return name; }
    public void setName(String name) { this.name = name; }

    public Integer getGrade() { return grade; }
    public void setGrade(Integer grade) { this.grade = grade; }

    public String getMajor() { return major; }
    public void setMajor(String major) { this.major = major; }

    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }

    public Integer getMinGrade() { return minGrade; }
    public void setMinGrade(Integer minGrade) { this.minGrade = minGrade; }

    public Integer getMaxGrade() { return maxGrade; }
    public void setMaxGrade(Integer maxGrade) { this.maxGrade = maxGrade; }
}

    2.2 Criteria APIによる動的クエリ実装

    Criteria APIを使用して、条件に応じて動的にクエリを構築します：

    @Repository
public interface StudentRepository extends JpaRepository<Student, Long>, JpaSpecificationExecutor<Student> {

    // Specification<Student>を受け取るメソッドが自動的に利用可能になる
    // List<Student> findAll(Specification<Student> spec);
    // Page<Student> findAll(Specification<Student> spec, Pageable pageable);
}

    Specificationクラスを作成して動的検索を実装：

    import org.springframework.data.jpa.domain.Specification;
import javax.persistence.criteria.*;
import java.util.ArrayList;
import java.util.List;

public class StudentSpecifications {

    public static Specification<Student> searchByCriteria(StudentSearchCriteria criteria) {
        return (root, query, criteriaBuilder) -> {
            List<Predicate> predicates = new ArrayList<>();

            // 名前での検索（部分一致）
            if (criteria.getName() != null && !criteria.getName().trim().isEmpty()) {
                predicates.add(criteriaBuilder.like(
                    criteriaBuilder.lower(root.get("name")), 
                    "%" + criteria.getName().toLowerCase() + "%"
                ));
            }

            // 学年での検索（完全一致）
            if (criteria.getGrade() != null) {
                predicates.add(criteriaBuilder.equal(root.get("grade"), criteria.getGrade()));
            }

            // 専攻での検索（完全一致）
            if (criteria.getMajor() != null && !criteria.getMajor().trim().isEmpty()) {
                predicates.add(criteriaBuilder.equal(root.get("major"), criteria.getMajor()));
            }

            // メールアドレスでの検索（部分一致）
            if (criteria.getEmail() != null && !criteria.getEmail().trim().isEmpty()) {
                predicates.add(criteriaBuilder.like(
                    criteriaBuilder.lower(root.get("email")), 
                    "%" + criteria.getEmail().toLowerCase() + "%"
                ));
            }

            // 学年範囲での検索
            if (criteria.getMinGrade() != null) {
                predicates.add(criteriaBuilder.greaterThanOrEqualTo(root.get("grade"), criteria.getMinGrade()));
            }
            if (criteria.getMaxGrade() != null) {
                predicates.add(criteriaBuilder.lessThanOrEqualTo(root.get("grade"), criteria.getMaxGrade()));
            }

            // すべての条件をANDで結合
            return criteriaBuilder.and(predicates.toArray(new Predicate[0]));
        };
    }

    // 専攻別の学生数を取得
    public static Specification<Student> byMajorWithMinStudents(String major, long minCount) {
        return (root, query, criteriaBuilder) -> {
            query.groupBy(root.get("major"));
            query.having(criteriaBuilder.greaterThan(criteriaBuilder.count(root), minCount));
            return criteriaBuilder.equal(root.get("major"), major);
        };
    }
}

    🏗️ 3. Repository拡張パターン

    3.1 カスタムリポジトリインターフェースの作成

    より高度な検索機能のために、カスタムリポジトリを作成します：

    📘 Repository拡張パターン図

    // カスタムリポジトリインターフェース
public interface StudentRepositoryCustom {

    // 複雑な検索クエリ
    List<Student> findStudentsWithComplexCriteria(StudentSearchCriteria criteria);

    // 統計情報の取得
    StudentStatistics getStudentStatistics();

    // 学年別学生数の取得
    Map<Integer, Long> getStudentCountByGrade();

    // 上位成績者の取得
    List<Student> findTopPerformers(int limit);
}

    3.2 カスタムリポジトリの実装

    @Repository
public class StudentRepositoryCustomImpl implements StudentRepositoryCustom {

    @PersistenceContext
    private EntityManager entityManager;

    @Override
    public List<Student> findStudentsWithComplexCriteria(StudentSearchCriteria criteria) {
        CriteriaBuilder cb = entityManager.getCriteriaBuilder();
        CriteriaQuery<Student> query = cb.createQuery(Student.class);
        Root<Student> root = query.from(Student.class);

        List<Predicate> predicates = new ArrayList<>();

        // 動的条件の構築
        if (criteria.getName() != null && !criteria.getName().trim().isEmpty()) {
            predicates.add(cb.like(cb.lower(root.get("name")), 
                                 "%" + criteria.getName().toLowerCase() + "%"));
        }

        if (criteria.getGrade() != null) {
            predicates.add(cb.equal(root.get("grade"), criteria.getGrade()));
        }

        // 複雑な条件の例：名前に特定の文字が含まれ、かつ学年が範囲内
        if (criteria.getMinGrade() != null && criteria.getMaxGrade() != null) {
            predicates.add(cb.between(root.get("grade"), 
                                    criteria.getMinGrade(), criteria.getMaxGrade()));
        }

        query.where(predicates.toArray(new Predicate[0]));
        query.orderBy(cb.asc(root.get("name")));

        return entityManager.createQuery(query).getResultList();
    }

    @Override
    public StudentStatistics getStudentStatistics() {
        String jpql = "SELECT new com.example.dto.StudentStatistics(" +
                     "COUNT(s), AVG(s.grade), MAX(s.grade), MIN(s.grade)) " +
                     "FROM Student s";

        return entityManager.createQuery(jpql, StudentStatistics.class)
                          .getSingleResult();
    }

    @Override
    public Map<Integer, Long> getStudentCountByGrade() {
        String jpql = "SELECT s.grade, COUNT(s) FROM Student s GROUP BY s.grade";

        List<Object[]> results = entityManager.createQuery(jpql).getResultList();
        Map<Integer, Long> gradeCountMap = new HashMap<>();

        for (Object[] result : results) {
            Integer grade = (Integer) result[0];
            Long count = (Long) result[1];
            gradeCountMap.put(grade, count);
        }

        return gradeCountMap;
    }

    @Override
    public List<Student> findTopPerformers(int limit) {
        String jpql = "SELECT s FROM Student s ORDER BY s.gpa DESC";

        return entityManager.createQuery(jpql, Student.class)
                          .setMaxResults(limit)
                          .getResultList();
    }
}

            }
        }

            throw new IllegalArgumentException("学年は1-6の範囲で指定してください");
        }
    }

    5.3 効率的なクエリ設計

    public class StudentSpecifications {

    /**
     * パフォーマンスを考慮した検索条件構築
     */
    public static Specification<Student> searchOptimized(StudentSearchCriteria criteria) {
        return (root, query, criteriaBuilder) -> {
            List<Predicate> predicates = new ArrayList<>();

            // インデックスが効く条件から優先的に適用
            // 1. 最も選択性の高い条件から
            if (criteria.getGrade() != null) {
                predicates.add(criteriaBuilder.equal(root.get("grade"), criteria.getGrade()));
            }

            if (criteria.getMajor() != null && !criteria.getMajor().trim().isEmpty()) {
                predicates.add(criteriaBuilder.equal(root.get("major"), criteria.getMajor()));
            }

            // 2. LIKE検索は後回し（コストが高いため）
            if (criteria.getName() != null && !criteria.getName().trim().isEmpty()) {
                predicates.add(criteriaBuilder.like(
                    criteriaBuilder.lower(root.get("name")), 
                    "%" + criteria.getName().toLowerCase() + "%"
                ));
            }

            return criteriaBuilder.and(predicates.toArray(new Predicate[0]));
        };
    }
}
