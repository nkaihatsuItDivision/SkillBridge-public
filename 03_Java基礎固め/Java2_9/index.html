    レッスン3.8で作成したStringUtilsクラスを確認し、今回のツール作成で活用する準備をしましょう。

    /**
 * Week 3で作成した文字列処理ユーティリティクラス
 * レッスン3.8で12個のオーバーロードメソッドを実装済み
 */
public class StringUtils {

    // === 検索機能群（4つのオーバーロード） ===

    /**
     * 基本検索：文字列内でキーワードを検索
     */
    public static boolean search(String text, String keyword) {
        if (text == null || keyword == null) return false;
        return text.contains(keyword);
    }

    /**
     * 大文字小文字を無視した検索
     */
    public static boolean search(String text, String keyword, boolean ignoreCase) {
        if (text == null || keyword == null) return false;
        if (ignoreCase) {
            return text.toLowerCase().contains(keyword.toLowerCase());
        }
        return text.contains(keyword);
    }

    /**
     * 位置指定検索：指定位置以降でキーワードを検索
     */
    public static int search(String text, String keyword, int startIndex) {
        if (text == null || keyword == null || startIndex &lt; 0) return -1;
        return text.indexOf(keyword, startIndex);
    }

    /**
     * 複数キーワード検索：いずれかのキーワードが見つかればtrue
     */
    public static boolean search(String text, String[] keywords) {
        if (text == null || keywords == null) return false;
        for (String keyword : keywords) {
            if (keyword != null &amp;&amp; text.contains(keyword)) {
                return true;
            }
        }
        return false;
    }

    // === 整形機能群（4つのオーバーロード） ===

    /**
     * 基本整形：前後の空白を削除
     */
    public static String format(String text) {
        return (text == null) ? &quot;&quot; : text.trim();
    }

    /**
     * 長さ制限付き整形
     */
    public static String format(String text, int maxLength) {
        if (text == null) return &quot;&quot;;
        String trimmed = text.trim();
        if (trimmed.length() &lt;= maxLength) return trimmed;
        return trimmed.substring(0, maxLength) + &quot;...&quot;;
    }

    /**
     * パディング付き整形：指定文字数になるまで空白で埋める
     */
    public static String format(String text, int width, char padChar) {
        if (text == null) text = &quot;&quot;;
        String trimmed = text.trim();
        if (trimmed.length() &gt;= width) return trimmed;

        StringBuilder sb = new StringBuilder(trimmed);
        while (sb.length() &lt; width) {
            sb.append(padChar);
        }
        return sb.toString();
    }

    /**
     * 複合オプション整形：長さ制限＋パディング＋大文字化
     */
    public static String format(String text, int maxLength, int width, boolean toUpper) {
        if (text == null) text = &quot;&quot;;
        String result = text.trim();

        // 大文字化
        if (toUpper) result = result.toUpperCase();

        // 長さ制限
        if (result.length() &gt; maxLength) {
            result = result.substring(0, maxLength) + &quot;...&quot;;
        }

        // パディング
        StringBuilder sb = new StringBuilder(result);
        while (sb.length() &lt; width) {
            sb.append(&apos; &apos;);
        }

        return sb.toString();
    }

    // === CSV処理機能群（4つのオーバーロード） ===

    /**
     * 基本CSV分割：カンマで分割
     */
    public static String[] parseCSV(String csvLine) {
        if (csvLine == null) return new String[0];
        return csvLine.split(&quot;,&quot;);
    }

    /**
     * 区切り文字指定CSV分割
     */
    public static String[] parseCSV(String csvLine, String delimiter) {
        if (csvLine == null || delimiter == null) return new String[0];
        return csvLine.split(delimiter);
    }

    /**
     * 引用符対応CSV分割：ダブルクォートで囲まれた値を適切に処理
     */
    public static String[] parseCSV(String csvLine, String delimiter, boolean handleQuotes) {
        if (csvLine == null || delimiter == null) return new String[0];

        if (!handleQuotes) {
            return csvLine.split(delimiter);
        }

        // 引用符を考慮した分割処理
        ArrayList&lt;String&gt; result = new ArrayList&lt;&gt;();
        boolean inQuotes = false;
        StringBuilder current = new StringBuilder();

        for (int i = 0; i &lt; csvLine.length(); i++) {
            char c = csvLine.charAt(i);

            if (c == &apos;&quot;&apos;) {
                inQuotes = !inQuotes;
            } else if (!inQuotes &amp;&amp; csvLine.substring(i).startsWith(delimiter)) {
                result.add(current.toString().trim());
                current = new StringBuilder();
                i += delimiter.length() - 1; // Skip delimiter
            } else {
                current.append(c);
            }
        }
        result.add(current.toString().trim());

        return result.toArray(new String[0]);
    }

    /**
     * 空値処理付きCSV分割：空文字列をnullに変換
     */
    public static String[] parseCSV(String csvLine, String delimiter, boolean handleQuotes, boolean nullifyEmpty) {
        String[] result = parseCSV(csvLine, delimiter, handleQuotes);

        if (nullifyEmpty) {
            for (int i = 0; i &lt; result.length; i++) {
                if (result[i] != null &amp;&amp; result[i].trim().isEmpty()) {
                    result[i] = null;
                }
            }
        }

        return result;
    }
}

    📖 テキストファイル読み込みの基本

    /**
 * テキストファイルを読み込んで内容を取得する
 */
public static String readTextFile(String fileName) {
    StringBuilder content = new StringBuilder();

    try (BufferedReader reader = new BufferedReader(new FileReader(fileName))) {
        String line;
        while ((line = reader.readLine()) != null) {
            content.append(line).append(&quot;\n&quot;);
        }
    } catch (IOException e) {
        System.out.println(&quot;ファイル読み込みエラー: &quot; + e.getMessage());
        return null;
    }

    return content.toString();
}

    💾 テキストファイル書き出しの基本

    /**
 * テキスト内容をファイルに書き出す
 */
public static boolean writeTextFile(String fileName, String content) {
    try (BufferedWriter writer = new BufferedWriter(new FileWriter(fileName))) {
        writer.write(content);
        return true;
    } catch (IOException e) {
        System.out.println(&quot;ファイル書き出しエラー: &quot; + e.getMessage());
        return false;
    }
}

    🔍 重要な概念説明

    BufferedReader/BufferedWriter とは？ - 効率的なファイル読み書きのためのクラス - バッファリング: データをまとめて処理することで高速化 - try-with-resources: 自動的にファイルを閉じる安全な仕組み

    IOException とは？ - 入出力例外: ファイルが見つからない、権限がない等のエラー - try-catch文: エラーが発生しても適切に処理する仕組み

    今回は難しく考えすぎず、「ファイルを読み書きする便利な道具」として活用しましょう！

/**

/**
 * CSV統計情報生成の実行
 */
private static void executeCSVStatistics(String content) {
    System.out.println(&quot;=== CSV統計情報 ===&quot;);

    String[] lines = content.split(&quot;\n&quot;);
    ArrayList&lt;String[]&gt; allData = new ArrayList&lt;&gt;();
    int maxColumns = 0;
    int totalRows = 0;

    // データの解析
    for (String line : lines) {
        if (line.trim().isEmpty()) continue;

        String[] fields = StringUtils.parseCSV(line, &quot;,&quot;, true, true); // 全オプション有効
        allData.add(fields);
        maxColumns = Math.max(maxColumns, fields.length);
        totalRows++;
    }

    // 統計情報の表示
    System.out.println(&quot;データ行数: &quot; + totalRows);
    System.out.println(&quot;最大列数: &quot; + maxColumns);

    // 列ごとの統計
    for (int col = 0; col &lt; maxColumns; col++) {
        System.out.printf(&quot;\n--- 列%d の統計 ---\n&quot;, col + 1);

        int nonEmptyCount = 0;
        int numericCount = 0;
        double sum = 0;
        ArrayList&lt;String&gt; values = new ArrayList&lt;&gt;();

        for (String[] row : allData) {
            if (col &lt; row.length &amp;&amp; row[col] != null &amp;&amp; !row[col].trim().isEmpty()) {
                String value = row[col].trim();
                values.add(value);
                nonEmptyCount++;

                // 数値かどうかチェック
                try {
                    double num = Double.parseDouble(value);
                    numericCount++;
                    sum += num;
                } catch (NumberFormatException e) {
                    // 数値ではない
                }
            }
        }

        System.out.println(&quot;  非空白データ数: &quot; + nonEmptyCount);
        if (numericCount &gt; 0) {
            System.out.println(&quot;  数値データ数: &quot; + numericCount);
            System.out.printf(&quot;  数値の合計: %.2f\n&quot;, sum);
            System.out.printf(&quot;  数値の平均: %.2f\n&quot;, sum / numericCount);
        }

        // よく出現する値（上位3件）
        if (!values.isEmpty()) {
            Map&lt;String, Integer&gt; frequency = new HashMap&lt;&gt;();
            for (String value : values) {
                frequency.put(value, frequency.getOrDefault(value, 0) + 1);
            }

            System.out.println(&quot;  よく出現する値:&quot;);
            frequency.entrySet().stream()
                .sorted((e1, e2) -&gt; e2.getValue().compareTo(e1.getValue()))
                .limit(3)
                .forEach(entry -&gt; 
                    System.out.println(&quot;    「&quot; + entry.getKey() + &quot;」: &quot; + entry.getValue() + &quot;回&quot;));
        }
    }
}

// 前述のファイル入出力メソッドを追加
private static String readTextFile(String fileName) {
    StringBuilder content = new StringBuilder();

    try (BufferedReader reader = new BufferedReader(new FileReader(fileName))) {
        String line;
        while ((line = reader.readLine()) != null) {
            content.append(line).append(&quot;\n&quot;);
        }
    } catch (IOException e) {
        System.out.println(&quot;ファイル読み込みエラー: &quot; + e.getMessage());
        return null;
    }

    return content.toString();
}

private static boolean writeTextFile(String fileName, String content) {
    try (BufferedWriter writer = new BufferedWriter(new FileWriter(fileName))) {
        writer.write(content);
        return true;
    } catch (IOException e) {
        System.out.println(&quot;ファイル書き出しエラー: &quot; + e.getMessage());
        return false;
    }
}
