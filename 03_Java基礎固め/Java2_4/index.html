    ⚡ 性能特性の定量的比較

    まずは実際の数値で性能差を確認しましょう。

    性能測定プログラムの作成

    新しいJavaクラス PerformanceComparator を作成してください：

    import java.util.ArrayList;
import java.text.DecimalFormat;

/**
 * 配列とArrayListの性能を定量的に比較するプログラム
 * 実際の処理時間を測定して、使い分けの判断材料を提供
 */
public class PerformanceComparator {

    // テストに使用するデータサイズ
    private static final int DATA_SIZE = 1000000;  // 100万件

    public static void main(String[] args) {
        System.out.println(&quot;=== 配列 vs ArrayList 性能比較テスト ===\n&quot;);

        // 1. データアクセス速度の比較
        compareAccessSpeed();

        // 2. データ追加処理の比較
        compareAddOperation();

        // 3. データ検索速度の比較  
        compareSearchSpeed();

        System.out.println(&quot;\n=== 比較テスト完了 ===&quot;);
    }

    /**
     * データアクセス速度の比較
     * 配列とArrayListで同じデータにアクセスする速度を測定
     */
    private static void compareAccessSpeed() {
        System.out.println(&quot;【1. データアクセス速度比較】&quot;);

        // テストデータの準備
        int[] array = new int[DATA_SIZE];
        ArrayList&lt;Integer&gt; list = new ArrayList&lt;&gt;();

        // データの初期化
        for (int i = 0; i &lt; DATA_SIZE; i++) {
            array[i] = i;
            list.add(i);
        }

        // 配列のアクセス速度測定
        long startTime = System.nanoTime();
        long sum1 = 0;
        for (int i = 0; i &lt; DATA_SIZE; i++) {
            sum1 += array[i];  // 配列からデータ取得
        }
        long arrayTime = System.nanoTime() - startTime;

        // ArrayListのアクセス速度測定
        startTime = System.nanoTime();
        long sum2 = 0;
        for (int i = 0; i &lt; DATA_SIZE; i++) {
            sum2 += list.get(i);  // ArrayListからデータ取得
        }
        long listTime = System.nanoTime() - startTime;

        // 結果の表示
        DecimalFormat df = new DecimalFormat(&quot;#,###&quot;);
        System.out.println(&quot;配列のアクセス時間: &quot; + df.format(arrayTime / 1000000) + &quot; ms&quot;);
        System.out.println(&quot;ArrayListのアクセス時間: &quot; + df.format(listTime / 1000000) + &quot; ms&quot;);
        System.out.println(&quot;速度差: &quot; + String.format(&quot;%.1f&quot;, (double)listTime / arrayTime) + &quot;倍\n&quot;);
    }

    /**
     * データ追加処理の比較
     * 配列とArrayListでデータ追加処理の違いを確認
     */
    private static void compareAddOperation() {
        System.out.println(&quot;【2. データ追加処理比較】&quot;);

        // ArrayListの追加処理測定
        long startTime = System.nanoTime();
        ArrayList&lt;Integer&gt; list = new ArrayList&lt;&gt;();
        for (int i = 0; i &lt; DATA_SIZE; i++) {
            list.add(i);  // 動的にデータ追加
        }
        long listAddTime = System.nanoTime() - startTime;

        DecimalFormat df = new DecimalFormat(&quot;#,###&quot;);
        System.out.println(&quot;ArrayList追加時間: &quot; + df.format(listAddTime / 1000000) + &quot; ms&quot;);
        System.out.println(&quot;配列の場合: 事前にサイズが分かっていれば高速、不明なら実装困難&quot;);
        System.out.println(&quot;→ データサイズが不明な場合はArrayListが圧倒的に有利\n&quot;);
    }

    /**
     * データ検索速度の比較
     * 特定の値を探す処理での性能差を測定
     */
    private static void compareSearchSpeed() {
        System.out.println(&quot;【3. データ検索速度比較】&quot;);

        // テストデータの準備
        int[] array = new int[DATA_SIZE];
        ArrayList&lt;Integer&gt; list = new ArrayList&lt;&gt;();

        for (int i = 0; i &lt; DATA_SIZE; i++) {
            array[i] = i;
            list.add(i);
        }

        int searchValue = DATA_SIZE - 1;  // 最後の要素を検索（最悪ケース）

        // 配列での線形検索
        long startTime = System.nanoTime();
        int arrayResult = -1;
        for (int i = 0; i &lt; array.length; i++) {
            if (array[i] == searchValue) {
                arrayResult = i;
                break;
            }
        }
        long arraySearchTime = System.nanoTime() - startTime;

        // ArrayListでの検索（indexOf使用）
        startTime = System.nanoTime();
        int listResult = list.indexOf(searchValue);
        long listSearchTime = System.nanoTime() - startTime;

        // 結果表示
        DecimalFormat df = new DecimalFormat(&quot;#,###&quot;);
        System.out.println(&quot;配列での検索時間: &quot; + df.format(arraySearchTime / 1000) + &quot; μs&quot;);
        System.out.println(&quot;ArrayListでの検索時間: &quot; + df.format(listSearchTime / 1000) + &quot; μs&quot;);
        System.out.println(&quot;ArrayList検索は内部最適化済みで高速\n&quot;);
    }
}

    💾 メモリ効率の比較

    次に、メモリ使用量の違いを確認しましょう。

    メモリ使用量測定プログラム

    新しいクラス MemoryComparator を作成：

    import java.util.ArrayList;

/**
 * 配列とArrayListのメモリ使用量を比較
 * 実際のメモリ効率を数値で確認
 */
public class MemoryComparator {

    private static final int TEST_SIZE = 100000;  // 10万件

    public static void main(String[] args) {
        System.out.println(&quot;=== メモリ使用量比較テスト ===\n&quot;);

        // メモリ使用量の測定
        compareMemoryUsage();

        // メモリ効率の分析
        analyzeMemoryEfficiency();
    }

    /**
     * 実際のメモリ使用量を測定
     */
    private static void compareMemoryUsage() {
        Runtime runtime = Runtime.getRuntime();

        // ガベージコレクション実行
        System.gc();
        long baseMemory = runtime.totalMemory() - runtime.freeMemory();

        // 配列でのメモリ使用量
        System.gc();
        long beforeArray = runtime.totalMemory() - runtime.freeMemory();
        int[] array = new int[TEST_SIZE];
        for (int i = 0; i &lt; TEST_SIZE; i++) {
            array[i] = i;
        }
        long afterArray = runtime.totalMemory() - runtime.freeMemory();
        long arrayMemory = afterArray - beforeArray;

        // ArrayListでのメモリ使用量  
        System.gc();
        long beforeList = runtime.totalMemory() - runtime.freeMemory();
        ArrayList&lt;Integer&gt; list = new ArrayList&lt;&gt;();
        for (int i = 0; i &lt; TEST_SIZE; i++) {
            list.add(i);
        }
        long afterList = runtime.totalMemory() - runtime.freeMemory();
        long listMemory = afterList - beforeList;

        // 結果表示
        System.out.println(&quot;【メモリ使用量比較】&quot;);
        System.out.println(&quot;配列のメモリ使用: &quot; + formatBytes(arrayMemory));
        System.out.println(&quot;ArrayListのメモリ使用: &quot; + formatBytes(listMemory));
        System.out.println(&quot;メモリ差: &quot; + String.format(&quot;%.1f&quot;, (double)listMemory / arrayMemory) + &quot;倍\n&quot;);
    }

    /**
     * メモリ効率の詳細分析
     */
    private static void analyzeMemoryEfficiency() {
        System.out.println(&quot;【メモリ効率分析】&quot;);

        // 配列の理論値
        int arrayBytes = TEST_SIZE * 4;  // int = 4バイト
        System.out.println(&quot;配列の理論メモリ: &quot; + formatBytes(arrayBytes));

        // ArrayListの追加オーバーヘッド
        System.out.println(&quot;ArrayListの追加要素:&quot;);
        System.out.println(&quot;- オブジェクトラッパー (Integer)&quot;);
        System.out.println(&quot;- 内部配列の余剰領域（容量の約25%）&quot;);
        System.out.println(&quot;- クラス構造のオーバーヘッド&quot;);

        System.out.println(&quot;\n【結論】&quot;);
        System.out.println(&quot;✓ 配列: メモリ効率が最高、大量データに最適&quot;);
        System.out.println(&quot;✓ ArrayList: 利便性重視、メモリより開発効率優先&quot;);
    }

    /**
     * バイト数を読みやすい形式にフォーマット
     */
    private static String formatBytes(long bytes) {
        if (bytes &lt; 1024) return bytes + &quot; B&quot;;
        if (bytes &lt; 1024 * 1024) return String.format(&quot;%.1f KB&quot;, bytes / 1024.0);
        return String.format(&quot;%.1f MB&quot;, bytes / 1024.0 / 1024.0);
    }
}

    • ArrayList: 約2-3倍のメモリ使用（利便性のコスト）
    • トレードオフ: メモリ効率 vs 開発効率

    import java.util.ArrayList;
import java.util.Scanner;

/**

public class DataStructureSelector {

    public static void main(String[] args) {

        System.out.println(&quot;- 開発効率: 重視（短期開発）&quot;);
        System.out.println(&quot;- ユーザー体験: リアルタイム更新&quot;);

        System.out.println(&quot;\n推奨: ArrayList&quot;);
        System.out.println(&quot;理由:&quot;);
        System.out.println(&quot;✓ データ数未知 → 動的サイズが必須&quot;);
        System.out.println(&quot;✓ 追加・削除頻繁 → ArrayListが簡単&quot;);
        System.out.println(&quot;✓ 開発効率優先 → 豊富なメソッドが有効&quot;);
        System.out.println(&quot;✓ 検索機能 → indexOf, containsが便利\n&quot;);
    }

    /**
     * シナリオ3: 大量データの分析処理
     */
    private static void bigDataScenario() {
        System.out.println(&quot;【シナリオ3: センサーデータの統計処理】&quot;);
        System.out.println(&quot;条件:&quot;);
        System.out.println(&quot;- データ数: 100万件（固定）&quot;);
        System.out.println(&quot;- 処理: 数値計算・統計処理が中心&quot;);
        System.out.println(&quot;- 性能要件: 高速処理が必須&quot;);
        System.out.println(&quot;- メモリ: 大量データでメモリ効率重要&quot;);

        System.out.println(&quot;\n推奨: 配列&quot;);
        System.out.println(&quot;理由:&quot;);
        System.out.println(&quot;✓ データ数確定 → 固定サイズで十分&quot;);
        System.out.println(&quot;✓ 数値計算中心 → 配列のアクセス速度活用&quot;);
        System.out.println(&quot;✓ 高速処理必須 → オーバーヘッド最小化&quot;);
        System.out.println(&quot;✓ 大量データ → メモリ効率が重要\n&quot;);
    }

    /**
     * 判断力チェック
     */
    private static void checkYourJudgment(Scanner scanner) {
        System.out.println(&quot;以下の状況では、どちらを選びますか？&quot;);

    実用判断のポイント

    各シナリオから学べる判断基準：

    1. データサイズ: 固定なら配列、動的ならArrayList
    2. 処理の種類: アクセス重視なら配列、操作重視ならArrayList
    3. 性能要件: 高速必須なら配列、開発効率ならArrayList
    4. メモリ制限: 厳しいなら配列、余裕ありならArrayList

    📋 実践的ガイドライン

    実際の開発で使える具体的な判断フローを構築しましょう。

    判断フローチャート

    データ構造選択フロー:

1. データサイズは事前に確定している？
   Yes → 2-A へ
   No  → ArrayList推奨（動的サイズ必須）

2-A. 処理の中心は何？
   - アクセス/計算中心 → 3-A へ
   - 追加/削除/検索中心 → ArrayList推奨

3-A. 性能要件は？
   - 高速処理必須 → 配列推奨
   - 開発効率優先 → ArrayList推奨

3-B. メモリ制限は？
   - 厳しい → 配列推奨
   - 問題なし → ArrayList推奨

    具体的な判断基準表

    項目 配列が有利 ArrayListが有利
    データサイズ 事前に確定 動的に変化
    主な処理 アクセス・計算 追加・削除・検索
    性能要件 高速処理必須 開発効率重視
    メモリ制限 厳しい 余裕あり
    開発期間 十分な時間 短期開発

    🔄 移行戦略：配列↔ArrayList変換

    実務では選択を変更する必要が出てきます。効率的な移行方法を学びましょう。

    移行戦略プログラム

    import java.util.ArrayList;
import java.util.Arrays;

/**
 * 配列とArrayListの相互変換戦略
 * 実務で必要な移行テクニック
 */
public class MigrationStrategy {

    public static void main(String[] args) {
        System.out.println(&quot;=== データ構造移行戦略 ===\n&quot;);

        // 配列からArrayListへの移行
        arrayToArrayListMigration();

        // ArrayListから配列への移行
        arrayListToArrayMigration();

        // 性能を考慮した移行判断
        migrationDecision();
    }

    /**
     * 配列からArrayListへの移行パターン
     */
    private static void arrayToArrayListMigration() {
        System.out.println(&quot;【配列 → ArrayList移行】&quot;);

        // 元の配列データ
        int[] originalArray = {10, 20, 30, 40, 50};
        System.out.println(&quot;元のデータ: &quot; + Arrays.toString(originalArray));

        // 移行方法1: Arrays.asListを使用（推奨）
        ArrayList&lt;Integer&gt; list1 = new ArrayList&lt;&gt;(Arrays.asList(
            Arrays.stream(originalArray).boxed().toArray(Integer[]::new)
        ));

        // 移行方法2: ループで移行（理解しやすい）
        ArrayList&lt;Integer&gt; list2 = new ArrayList&lt;&gt;();
        for (int value : originalArray) {
            list2.add(value);
        }

        System.out.println(&quot;移行後ArrayList: &quot; + list1);
        System.out.println(&quot;移行理由例: データサイズが動的に変化する要件追加\n&quot;);
    }

    /**
     * ArrayListから配列への移行パターン
     */
    private static void arrayListToArrayMigration() {
        System.out.println(&quot;【ArrayList → 配列移行】&quot;);

        // 元のArrayListデータ
        ArrayList&lt;Integer&gt; originalList = new ArrayList&lt;&gt;();
        originalList.add(100);
        originalList.add(200);
        originalList.add(300);
        System.out.println(&quot;元のデータ: &quot; + originalList);

        // 移行方法1: toArrayを使用（推奨）
        Integer[] array1 = originalList.toArray(new Integer[0]);

        // 移行方法2: 基本型配列が必要な場合
        int[] array2 = originalList.stream().mapToInt(Integer::intValue).toArray();

        System.out.println(&quot;移行後配列: &quot; + Arrays.toString(array2));
        System.out.println(&quot;移行理由例: 大量データ処理で性能問題が発生\n&quot;);
    }

    /**
     * 移行の判断基準
     */
    private static void migrationDecision() {
        System.out.println(&quot;【移行判断基準】&quot;);
        System.out.println(&quot;配列 → ArrayList移行のタイミング:&quot;);
        System.out.println(&quot;✓ データサイズが動的になった&quot;);
        System.out.println(&quot;✓ 頻繁な追加・削除が必要になった&quot;);
        System.out.println(&quot;✓ 検索・フィルタ機能が必要になった&quot;);
        System.out.println(&quot;✓ 開発効率を優先したい&quot;);

        System.out.println(&quot;\nArrayList → 配列移行のタイミング:&quot;);
        System.out.println(&quot;✓ 性能問題が発生した&quot;);
        System.out.println(&quot;✓ メモリ使用量を削減したい&quot;);
        System.out.println(&quot;✓ データサイズが確定した&quot;);
        System.out.println(&quot;✓ 単純なアクセス処理が中心になった&quot;);
    }
}

    性能特性の理解 ✅

    • アクセス速度: 配列が2-3倍高速
    • メモリ効率: 配列が2-3倍効率的
    • 開発効率: ArrayListが圧倒的に高い
    • 機能の豊富さ: ArrayListが多機能

    実践的判断フロー ✅

    🤔 開発場面での判断手順:

1️⃣ データサイズは事前に確定？
   → No なら ArrayList

2️⃣ 処理の中心は？
   → アクセス・計算なら配列
   → 追加・削除・検索ならArrayList

3️⃣ 性能とメモリどちらが重要？
   → 性能・メモリ重視なら配列
   → 開発効率重視ならArrayList

4️⃣ 将来の機能拡張は？
   → 拡張予定ありならArrayList
   → 機能固定なら配列
