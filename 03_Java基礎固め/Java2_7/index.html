    1. 高度な検索テクニック

    🔍 ステップ1: 複数条件での検索

    レッスン3.6では単純な検索を学びました。今度は、複数の条件を組み合わせた検索を実装してみましょう。

    新しいクラス AdvancedStringSearch を作成してください：

    import java.util.ArrayList;

public class AdvancedStringSearch {
    public static void main(String[] args) {
        // サンプルデータ：顧客情報
        String[] customers = {
            &quot;田中太郎,tanaka@email.com,東京都,VIP&quot;,
            &quot;佐藤花子,sato@email.com,大阪府,一般&quot;,
            &quot;高橋次郎,takahashi@email.com,東京都,一般&quot;, 
            &quot;田中美香,tanaka.mika@email.com,神奈川県,VIP&quot;,
            &quot;山田太郎,yamada@email.com,愛知県,VIP&quot;
        };

        System.out.println(&quot;=== 高度な検索テクニック ===\n&quot;);

        // 1. 単純検索（復習）
        System.out.println(&quot;1. 単純検索：「田中」を含む顧客&quot;);
        simpleSearch(customers, &quot;田中&quot;);

        // 2. 複数条件検索（AND条件）
        System.out.println(&quot;\n2. 複数条件検索：「東京都」かつ「VIP」の顧客&quot;);
        multipleConditionSearch(customers, &quot;東京都&quot;, &quot;VIP&quot;);

        // 3. 大文字小文字を無視した検索
        System.out.println(&quot;\n3. 大文字小文字無視：「EMAIL」を含む顧客&quot;);
        caseInsensitiveSearch(customers, &quot;EMAIL&quot;);

        // 4. 位置ベースの検索
        System.out.println(&quot;\n4. 位置ベース検索：メールアドレスが「.com」で終わる顧客&quot;);
        positionBasedSearch(customers);
    }

    // 単純検索（レッスン3.6の復習）
    public static void simpleSearch(String[] data, String keyword) {
        for (String item : data) {
            if (item.contains(keyword)) {
                System.out.println(&quot;  &quot; + item);
            }
        }
    }

    // 複数条件検索（AND条件）
    public static void multipleConditionSearch(String[] data, String condition1, String condition2) {
        for (String item : data) {
            // 両方の条件を満たす場合のみ表示
            if (item.contains(condition1) &amp;&amp; item.contains(condition2)) {
                System.out.println(&quot;  &quot; + item);
            }
        }
    }

    // 大文字小文字を無視した検索
    public static void caseInsensitiveSearch(String[] data, String keyword) {
        String lowerKeyword = keyword.toLowerCase();  // 検索キーワードを小文字に

        for (String item : data) {
            // データも小文字に変換して比較
            if (item.toLowerCase().contains(lowerKeyword)) {
                System.out.println(&quot;  &quot; + item);
            }
        }
    }

    // 位置ベースの検索
    public static void positionBasedSearch(String[] data) {
        for (String item : data) {
            // メールアドレス部分を抽出（2番目のカンマ区切りフィールド）
            String[] parts = item.split(&quot;,&quot;);
            if (parts.length &gt;= 2) {
                String email = parts[1];
                if (email.endsWith(&quot;.com&quot;)) {
                    System.out.println(&quot;  &quot; + item);
                }
            }
        }
    }
}

    🔍 ステップ2: OR条件とカスタム検索

    続いて、OR条件やより複雑な検索パターンを実装してみましょう：

    // AdvancedStringSearchクラスにメソッドを追加

// OR条件検索
public static void orConditionSearch(String[] data, String condition1, String condition2) {
    System.out.println(&quot;OR条件検索：「&quot; + condition1 + &quot;」または「&quot; + condition2 + &quot;」を含む顧客&quot;);

    for (String item : data) {
        // どちらかの条件を満たす場合に表示
        if (item.contains(condition1) || item.contains(condition2)) {
            System.out.println(&quot;  &quot; + item);
        }
    }
}

// 除外検索（NOT条件）
public static void excludeSearch(String[] data, String includeKeyword, String excludeKeyword) {
    System.out.println(&quot;除外検索：「&quot; + includeKeyword + &quot;」を含むが「&quot; + excludeKeyword + &quot;」は含まない顧客&quot;);

    for (String item : data) {
        // 含む条件は満たすが、除外条件は満たさない場合に表示
        if (item.contains(includeKeyword) &amp;&amp; !item.contains(excludeKeyword)) {
            System.out.println(&quot;  &quot; + item);
        }
    }
}

// 範囲指定検索（indexOf活用）
public static void rangeSearch(String[] data) {
    System.out.println(&quot;範囲指定検索：メールアドレスに「@」と「.」の両方を含む顧客&quot;);

    for (String item : data) {
        int atIndex = item.indexOf(&quot;@&quot;);
        int dotIndex = item.indexOf(&quot;.&quot;);

        // @マークが.マークより前にある場合（正しいメール形式）
        if (atIndex != -1 &amp;&amp; dotIndex != -1 &amp;&amp; atIndex &lt; dotIndex) {
            System.out.println(&quot;  &quot; + item);
        }
    }
}

    2. 効率的な置換テクニック

    🔧 ステップ3: 複数パターンの置換とStringBuilder

    レッスン3.6では基本的なreplace()メソッドを学びました。今度は、効率的で高度な置換技術を習得しましょう。

    新しいクラス AdvancedStringReplace を作成してください：

    public class AdvancedStringReplace {
    public static void main(String[] args) {
        System.out.println(&quot;=== 効率的な置換テクニック ===\n&quot;);

        // サンプルデータ：ユーザー入力（よくある問題文字を含む）
        String userInput = &quot;こんにちは！！！　今日は良い天気ですね～～？？？　　よろしく！！&quot;;
        System.out.println(&quot;元の文字列：\&quot;&quot; + userInput + &quot;\&quot;&quot;);

        // 1. 基本的な置換（復習）
        System.out.println(&quot;\n1. 基本的な置換&quot;);
        basicReplace(userInput);

        // 2. 複数パターンの一括置換
        System.out.println(&quot;\n2. 複数パターン一括置換&quot;);
        multipleReplace(userInput);

        // 3. StringBuilderを使った効率的な置換
        System.out.println(&quot;\n3. StringBuilder効率的置換&quot;);
        efficientReplace(userInput);

        // 4. 条件付き置換
        System.out.println(&quot;\n4. 条件付き置換&quot;);
        conditionalReplace(userInput);
    }

    // 基本的な置換（レッスン3.6の復習）
    public static void basicReplace(String text) {
        String result = text.replace(&quot;！！！&quot;, &quot;！&quot;);  // 連続感嘆符を単一に
        System.out.println(&quot;  結果：\&quot;&quot; + result + &quot;\&quot;&quot;);
    }

    // 複数パターンの一括置換
    public static void multipleReplace(String text) {
        String result = text;

        // 複数の置換を順次実行
        result = result.replace(&quot;！！！&quot;, &quot;！&quot;);      // 連続感嘆符
        result = result.replace(&quot;？？？&quot;, &quot;？&quot;);      // 連続疑問符
        result = result.replace(&quot;～～&quot;, &quot;～&quot;);        // 連続波線
        result = result.replace(&quot;　　&quot;, &quot; &quot;);         // 全角スペース連続を半角1つに
        result = result.replace(&quot;　&quot;, &quot; &quot;);          // 残りの全角スペースを半角に

        System.out.println(&quot;  結果：\&quot;&quot; + result + &quot;\&quot;&quot;);
    }

    // StringBuilderを使った効率的な置換
    public static void efficientReplace(String text) {
        StringBuilder sb = new StringBuilder();

        for (int i = 0; i &lt; text.length(); i++) {
            char current = text.charAt(i);

            // 連続する同じ記号をチェック
            if (current == &apos;！&apos; || current == &apos;？&apos; || current == &apos;～&apos;) {
                sb.append(current);  // 1つ目は追加

                // 連続する同じ文字をスキップ
                while (i + 1 &lt; text.length() &amp;&amp; text.charAt(i + 1) == current) {
                    i++;  // 連続する文字を飛ばす
                }
            } else if (current == &apos;　&apos;) {
                // 全角スペースを半角に
                sb.append(&apos; &apos;);

                // 連続する全角スペースをスキップ
                while (i + 1 &lt; text.length() &amp;&amp; text.charAt(i + 1) == &apos;　&apos;) {
                    i++;
                }
            } else {
                sb.append(current);  // 通常の文字はそのまま追加
            }
        }

        String result = sb.toString();
        System.out.println(&quot;  結果：\&quot;&quot; + result + &quot;\&quot;&quot;);
        System.out.println(&quot;  (StringBuilderを使用した効率的な処理)&quot;);
    }

    // 条件付き置換
    public static void conditionalReplace(String text) {
        StringBuilder sb = new StringBuilder();

        // 文字数をカウントしながら置換
        int consecutiveCount = 1;
        char previousChar = &apos;\0&apos;;

        for (int i = 0; i &lt; text.length(); i++) {
            char current = text.charAt(i);

            if (current == previousChar &amp;&amp; (current == &apos;！&apos; || current == &apos;？&apos; || current == &apos;～&apos;)) {
                consecutiveCount++;
                // 3つ以上連続している場合のみ置換（2つまでは許可）
                if (consecutiveCount &lt;= 2) {
                    sb.append(current);
                }
            } else {
                consecutiveCount = 1;
                if (current == &apos;　&apos;) {
                    sb.append(&apos; &apos;);  // 全角スペースは常に半角に
                } else {
                    sb.append(current);
                }
            }

            previousChar = current;
        }

        String result = sb.toString();
        System.out.println(&quot;  結果：\&quot;&quot; + result + &quot;\&quot;&quot;);
        System.out.println(&quot;  (3つ以上の連続記号のみ削除、2つまでは許可)&quot;);
    }
}

    🔧 ステップ4: 正規表現の基本活用

    正規表現は強力な文字列処理ツールです。Java初心者でも使える基本パターンを学びましょう：

    // AdvancedStringReplaceクラスにメソッドを追加

// 正規表現を使った置換（基本パターン）
public static void regexReplace(String text) {
    System.out.println(&quot;5. 正規表現を使った置換&quot;);

    String result = text;

    // [！]{2,} = 感嘆符が2つ以上連続している部分を1つに
    result = result.replaceAll(&quot;[！]{2,}&quot;, &quot;！&quot;);

    // [？]{2,} = 疑問符が2つ以上連続している部分を1つに
    result = result.replaceAll(&quot;[？]{2,}&quot;, &quot;？&quot;);

    // [～]{2,} = 波線が2つ以上連続している部分を1つに
    result = result.replaceAll(&quot;[～]{2,}&quot;, &quot;～&quot;);

    // [　\\s]+ = 全角スペースまたは半角スペースが1つ以上連続している部分を半角スペース1つに
    result = result.replaceAll(&quot;[　\\s]+&quot;, &quot; &quot;);

    System.out.println(&quot;  結果：\&quot;&quot; + result + &quot;\&quot;&quot;);
    System.out.println(&quot;  (正規表現でスマートに処理)&quot;);
}

    💡 重要なポイント

    なぜStringBuilderが効率的なのか？

    1. Stringは不変: 毎回新しいオブジェクトが作成される
    2. StringBuilderは可変: 同じオブジェクト内で変更可能
    3. メモリ効率: 大量の置換処理で威力を発揮
    4. 実行速度: 連続置換ではStringBuilderが圧倒的に高速

    📊 ステップ5: CSVデータの安全な分割

    実際の業務でよく扱うCSVデータを安全に処理する技術を学びましょう。

    新しいクラス CSVDataProcessor を作成してください：

    import java.util.ArrayList;

public class CSVDataProcessor {
    public static void main(String[] args) {
        System.out.println(&quot;=== CSVデータ処理の実践技術 ===\n&quot;);

        // サンプルCSVデータ（商品情報）

    // 基本的なCSV分割
    public static void basicCSVSplit(String[] data) {
        for (int i = 0; i &lt; data.length; i++) {
            if (i == 0) {
                System.out.println(&quot;  ヘッダー：&quot; + data[i]);
                continue;
            }

            String[] columns = data[i].split(&quot;,&quot;);
            System.out.println(&quot;  行&quot; + i + &quot;：&quot; + java.util.Arrays.toString(columns));
        }
    }

    // 安全なCSV分割（エラーハンドリング付き）
    public static void safeCSVSplit(String[] data) {
        for (int i = 0; i &lt; data.length; i++) {
            // 空行をスキップ
            if (data[i] == null || data[i].trim().isEmpty()) {
                System.out.println(&quot;  行&quot; + i + &quot;：空行をスキップ&quot;);
                continue;
            }

            // ヘッダー行の処理
            if (i == 0) {
                System.out.println(&quot;  ヘッダー：&quot; + data[i]);
                continue;
            }

            try {
                String[] columns = parseSafeCSV(data[i]);
                System.out.printf(&quot;  行%d：ID=%s, 商品名=%s, 価格=%s, 在庫=%s, カテゴリ=%s%n&quot;, 
                    i, 
                    getColumnSafe(columns, 0),
                    getColumnSafe(columns, 1), 
                    getColumnSafe(columns, 2),
                    getColumnSafe(columns, 3),
                    getColumnSafe(columns, 4)
                );
            } catch (Exception e) {
                System.out.println(&quot;  行&quot; + i + &quot;：エラー - &quot; + e.getMessage());
            }
        }
    }

    // 安全なCSV解析（引用符考慮）
    public static String[] parseSafeCSV(String line) {
        ArrayList&lt;String&gt; columns = new ArrayList&lt;&gt;();
        StringBuilder currentColumn = new StringBuilder();
        boolean insideQuotes = false;

        for (int i = 0; i &lt; line.length(); i++) {
            char c = line.charAt(i);

            if (c == &apos;&quot;&apos;) {
                insideQuotes = !insideQuotes;  // 引用符の内外を切り替え
            } else if (c == &apos;,&apos; &amp;&amp; !insideQuotes) {
                // 引用符外のカンマで分割
                columns.add(currentColumn.toString().trim());
                currentColumn = new StringBuilder();
            } else {
                currentColumn.append(c);
            }
        }

        // 最後の列を追加
        columns.add(currentColumn.toString().trim());

        return columns.toArray(new String[0]);
    }

    // 安全な列取得
    public static String getColumnSafe(String[] columns, int index) {
        if (columns == null || index &gt;= columns.length) {
            return &quot;(未設定)&quot;;
        }

        String value = columns[index];
        return (value == null || value.isEmpty()) ? &quot;(空)&quot; : value;
    }

    // データ変換と集計
    public static void dataAnalysis(String[] data) {
        int totalProducts = 0;
        int totalStock = 0;
        long totalValue = 0;  // 在庫価値合計

        for (int i = 1; i &lt; data.length; i++) {  // ヘッダーをスキップ
            if (data[i] == null || data[i].trim().isEmpty()) continue;

            try {
                String[] columns = parseSafeCSV(data[i]);

                // 価格と在庫数を数値に変換
                String priceStr = getColumnSafe(columns, 2);
                String stockStr = getColumnSafe(columns, 3);

                if (!priceStr.equals(&quot;(空)&quot;) &amp;&amp; !priceStr.equals(&quot;(未設定)&quot;) &amp;&amp;
                    !stockStr.equals(&quot;(空)&quot;) &amp;&amp; !stockStr.equals(&quot;(未設定)&quot;)) {

                    int price = Integer.parseInt(priceStr);
                    int stock = Integer.parseInt(stockStr);

                    totalProducts++;
                    totalStock += stock;
                    totalValue += (long)price * stock;
                }
            } catch (NumberFormatException e) {
                // 数値変換エラーは無視して次へ
                continue;
            }
        }

        System.out.println(&quot;  商品数：&quot; + totalProducts + &quot;件&quot;);
        System.out.println(&quot;  総在庫数：&quot; + totalStock + &quot;個&quot;);
        System.out.println(&quot;  在庫価値合計：&quot; + String.format(&quot;%,d&quot;, totalValue) + &quot;円&quot;);
    }

    // 検索とフィルタリング
    public static void searchAndFilter(String[] data) {
        System.out.println(&quot;  電子機器カテゴリの商品：&quot;);

        for (int i = 1; i &lt; data.length; i++) {  // ヘッダーをスキップ
            if (data[i] == null || data[i].trim().isEmpty()) continue;

            String[] columns = parseSafeCSV(data[i]);
            String category = getColumnSafe(columns, 4);

            if (category.contains(&quot;電子機器&quot;)) {
                String productName = getColumnSafe(columns, 1);
                String price = getColumnSafe(columns, 2);
                System.out.println(&quot;    &quot; + productName + &quot; - &quot; + price + &quot;円&quot;);
            }
        }
    }

    public static String cleanText(String text) {
        StringBuilder sb = new StringBuilder();
        char previousChar = &apos;\0&apos;;

        for (int i = 0; i &lt; text.length(); i++) {
            char current = text.charAt(i);

            // 連続する記号の処理
            if (current == &apos;！&apos; || current == &apos;？&apos; || current == &apos;～&apos;) {
                if (current != previousChar) {
                    sb.append(current);
                }
                // 連続する同じ記号は無視
            }
            // 全角スペースを半角に
            else if (current == &apos;　&apos;) {
                if (previousChar != &apos; &apos; &amp;&amp; previousChar != &apos;　&apos;) {
                    sb.append(&apos; &apos;);
                }
                // 連続するスペースは無視
            }
            // 通常の文字
            else {
                sb.append(current);
            }

            previousChar = current;
        }

        // 前後の空白を除去
        return sb.toString().trim();
    }

    public static void performAdvancedSearch(String[] texts, String keyword) {
        int matchCount = 0;

        System.out.println(&quot;検索キーワード: \&quot;&quot; + keyword + &quot;\&quot;&quot;);
        System.out.println();

        for (int i = 0; i &lt; texts.length; i++) {
            String text = texts[i];

            // 大文字小文字を無視して検索
            if (text.toLowerCase().contains(keyword.toLowerCase())) {
                matchCount++;

                // マッチした部分をハイライト表示（簡易版）
                String highlightedText = highlightMatch(text, keyword);
                System.out.println(&quot;行&quot; + (i + 1) + &quot;: &quot; + highlightedText);
            }
        }

        System.out.println(&quot;\n検索結果: &quot; + matchCount + &quot;件のマッチが見つかりました。&quot;);
    }

    // マッチ部分のハイライト（簡易版）
    public static String highlightMatch(String text, String keyword) {
        // 大文字小文字を無視して置換
        StringBuilder result = new StringBuilder();
        String lowerText = text.toLowerCase();
        String lowerKeyword = keyword.toLowerCase();

        int lastIndex = 0;
        int index = lowerText.indexOf(lowerKeyword);

        while (index != -1) {
            // マッチ前の部分を追加
            result.append(text.substring(lastIndex, index));

            // マッチ部分を大括弧で囲んで追加
            result.append(&quot;[&quot;).append(text.substring(index, index + keyword.length())).append(&quot;]&quot;);

            lastIndex = index + keyword.length();
            index = lowerText.indexOf(lowerKeyword, lastIndex);
        }

        // 残りの部分を追加
        result.append(text.substring(lastIndex));

        return result.toString();
    }

    💡 よくある質問

    Q1: 正規表現は覚える必要がありますか？

    A: 基本パターンだけで十分です。今回学んだ以下があれば実用的： - [文字] : 指定文字のマッチ - {回数} : 繰り返し回数指定 - + : 1回以上の繰り返し - 覚えられない部分は都度調べて使用すればOK

    Q2: StringBuilderはいつ使うべきですか？

    A: 以下の場合にStringBuilderを使用： - 大量の文字列結合（100回以上の目安） - ループ内での文字列構築 - 動的に文字列を構築する場合 - 少量の処理ならString.replace()で十分

    Q3: CSVデータの引用符処理が複雑です

    A: 引用符処理は確かに複雑です： - 基本的には今回のパターンで対応可能 - より複雑な場合は専用ライブラリ（OpenCSV等）を使用 - 自作する場合は段階的にテストすることが重要

    Q4: エラーハンドリングはどこまで必要ですか？

    A: 実用性を考慮して段階的に： 1. 最低限: null チェック、空文字チェック 2. 基本: NumberFormatException等の主要例外 3. 実用: 想定外データのスキップ処理 4. プロ: 詳細なログ出力とリカバリ処理
