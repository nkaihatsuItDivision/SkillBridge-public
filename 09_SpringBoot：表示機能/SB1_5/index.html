    2. @Table - テーブル名の明示的指定

    @Entity
@Table(name = "students")
public class Student {
    // クラス名とテーブル名が異なる場合に使用
}

    @Tableの効果: - テーブル名指定: クラス名 Student をテーブル名 students にマッピング - 命名規則対応: Javaのキャメルケースとデータベースのスネークケースを橋渡し - 複数環境対応: 開発・本番環境でのテーブル名違いに対応

    3. @Id - 主キーの指定

    @Id
@GeneratedValue(strategy = GenerationType.IDENTITY)
@Column(name = "student_id")
private Long studentId;

    @Idの効果: - 主キー指定: このフィールドがテーブルの主キーであることを宣言 - 一意性保証: データベースレベルでの一意性制約 - 検索最適化: 主キーによる高速検索の実現

@Column(name = "email", nullable = false, length = 100, unique = true)

    import javax.validation.ConstraintViolation;
import javax.validation.Validator;
import javax.validation.ValidatorFactory;
import javax.validation.Validation;
import java.util.Set;

public class ValidationTest {
    public static void main(String[] args) {
        // Validatorの取得
        ValidatorFactory factory = Validation.buildDefaultValidatorFactory();
        Validator validator = factory.getValidator();

        // 不正なデータでStudentオブジェクト作成
        Student invalidStudent = Student.builder()
            .studentName("") // 空文字（@NotBlankでエラー）
            .email("invalid-email") // 不正な形式（@Emailでエラー）
            .birthDate(LocalDate.of(2030, 1, 1)) // 未来日付（@Pastでエラー）
            .departmentId(-1) // 負の数（@Positiveでエラー）
            .enrollmentYear(1800) // 範囲外（@Minでエラー）
            .build();

        // バリデーション実行
        Set<ConstraintViolation<Student>> violations = validator.validate(invalidStudent);

        // エラーメッセージ表示
        if (!violations.isEmpty()) {
            System.out.println("=== バリデーションエラー ===");
            for (ConstraintViolation<Student> violation : violations) {
                System.out.println("フィールド: " + violation.getPropertyPath());
                System.out.println("エラーメッセージ: " + violation.getMessage());
                System.out.println("入力値: " + violation.getInvalidValue());
                System.out.println("---");
            }
        } else {
            System.out.println("バリデーション成功！");
        }

        // 正しいデータでのテスト
        Student validStudent = Student.builder()
            .studentName("田中太郎")
            .email("tanaka@example.com")
            .birthDate(LocalDate.of(2000, 4, 15))
            .departmentId(1)
            .enrollmentYear(2020)
            .build();

        Set<ConstraintViolation<Student>> validViolations = validator.validate(validStudent);
        if (validViolations.isEmpty()) {
            System.out.println("正しいデータ - バリデーション成功！");
        }
    }
}

    package com.example.entity;

import lombok.*;
import javax.persistence.*;
import javax.validation.constraints.*;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.math.BigDecimal;

@Entity
@Table(name = "students")
@Data
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Student {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    @Column(name = "student_id")
    private Long studentId;

    @NotBlank(message = "学生名は必須です")
    @Size(min = 1, max = 100, message = "学生名は1文字以上100文字以内で入力してください")
    @Pattern(regexp = "^[\\p{IsHan}\\p{IsHiragana}\\p{IsKatakana}\\p{IsAlphabetic}\\s]+$", 
             message = "学生名は日本語、英語、スペースのみ使用可能です")
    @Column(name = "student_name", nullable = false, length = 100)
    private String studentName;

    @NotBlank(message = "メールアドレスは必須です")
    @Email(message = "正しいメールアドレス形式で入力してください")
    @Size(max = 100, message = "メールアドレスは100文字以内で入力してください")
    @Column(name = "email", nullable = false, length = 100, unique = true)
    private String email;

    @Past(message = "生年月日は過去の日付を入力してください")
    @Column(name = "birth_date")
    private LocalDate birthDate;

    @Positive(message = "学科IDは正の数値を入力してください")
    @Column(name = "department_id")
    private Integer departmentId;

    @NotNull(message = "入学年度は必須です")
    @Min(value = 1900, message = "入学年度は1900年以降を入力してください")
    @Max(value = 2100, message = "入学年度は2100年以前を入力してください")
    @Column(name = "enrollment_year", nullable = false)
    private Integer enrollmentYear;

    @DecimalMin(value = "0.0", message = "GPAは0.0以上である必要があります")
    @DecimalMax(value = "4.0", message = "GPAは4.0以下である必要があります")
    @Digits(integer = 1, fraction = 2, message = "GPAは小数点以下2桁までです")
    @Column(name = "gpa", precision = 3, scale = 2)
    private BigDecimal gpa;

    @Pattern(regexp = "^(ACTIVE|INACTIVE|GRADUATED|SUSPENDED)$", 
             message = "学生ステータスはACTIVE、INACTIVE、GRADUATED、SUSPENDEDのいずれかです")
    @Builder.Default
    @Column(name = "status", length = 20)
    private String status = "ACTIVE";

    @Size(max = 500, message = "備考は500文字以内で入力してください")
    @Column(name = "remarks", length = 500)
    private String remarks;

    @Column(name = "created_at")
    private LocalDateTime createdAt;

    @Column(name = "updated_at")
    private LocalDateTime updatedAt;

    // ライフサイクルコールバック
    @PrePersist
    protected void onCreate() {
        createdAt = LocalDateTime.now();
        updatedAt = LocalDateTime.now();
        if (status == null) {
            status = "ACTIVE";
        }
    }

    @PreUpdate
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }

    // ビジネスロジック メソッド
    public int getAge() {
        if (birthDate == null) {
            return 0;
        }
        return LocalDate.now().getYear() - birthDate.getYear();
    }

    public int getStudyYears() {
        return LocalDate.now().getYear() - enrollmentYear + 1;
    }

    public boolean isActive() {
        return "ACTIVE".equals(status);
    }

    public String getGradeLevel() {
        if (gpa == null) return "未評価";
        double gpaValue = gpa.doubleValue();
        if (gpaValue >= 3.5) return "優秀";
        if (gpaValue >= 3.0) return "良好";
        if (gpaValue >= 2.5) return "普通";
        return "要改善";
    }
}
