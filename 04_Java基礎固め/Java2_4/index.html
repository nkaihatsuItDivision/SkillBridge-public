    ⚡ 性能特性の定量的比較

    まずは実際の数値で性能差を確認しましょう。

    性能測定プログラムの作成

    新しいJavaクラス PerformanceComparator を作成してください：

    import java.util.ArrayList;
import java.text.DecimalFormat;

/**
 * 配列とArrayListの性能を定量的に比較するプログラム
 * 実際の処理時間を測定して、使い分けの判断材料を提供
 */
public class PerformanceComparator {

    // テストに使用するデータサイズ
    private static final int DATA_SIZE = 1000000;  // 100万件

    public static void main(String[] args) {
        System.out.println("=== 配列 vs ArrayList 性能比較テスト ===\n");

        // 1. データアクセス速度の比較
        compareAccessSpeed();

        // 2. データ追加処理の比較
        compareAddOperation();

        // 3. データ検索速度の比較  
        compareSearchSpeed();

        System.out.println("\n=== 比較テスト完了 ===");
    }

    /**
     * データアクセス速度の比較
     * 配列とArrayListで同じデータにアクセスする速度を測定
     */
    private static void compareAccessSpeed() {
        System.out.println("【1. データアクセス速度比較】");

        // テストデータの準備
        int[] array = new int[DATA_SIZE];
        ArrayList<Integer> list = new ArrayList<>();

        // データの初期化
        for (int i = 0; i < DATA_SIZE; i++) {
            array[i] = i;
            list.add(i);
        }

        // 配列のアクセス速度測定
        long startTime = System.nanoTime();
        long sum1 = 0;
        for (int i = 0; i < DATA_SIZE; i++) {
            sum1 += array[i];  // 配列からデータ取得
        }
        long arrayTime = System.nanoTime() - startTime;

        // ArrayListのアクセス速度測定
        startTime = System.nanoTime();
        long sum2 = 0;
        for (int i = 0; i < DATA_SIZE; i++) {
            sum2 += list.get(i);  // ArrayListからデータ取得
        }
        long listTime = System.nanoTime() - startTime;

        // 結果の表示
        DecimalFormat df = new DecimalFormat("#,###");
        System.out.println("配列のアクセス時間: " + df.format(arrayTime / 1000000) + " ms");
        System.out.println("ArrayListのアクセス時間: " + df.format(listTime / 1000000) + " ms");
        System.out.println("速度差: " + String.format("%.1f", (double)listTime / arrayTime) + "倍\n");
    }

    /**
     * データ追加処理の比較
     * 配列とArrayListでデータ追加処理の違いを確認
     */
    private static void compareAddOperation() {
        System.out.println("【2. データ追加処理比較】");

        // ArrayListの追加処理測定
        long startTime = System.nanoTime();
        ArrayList<Integer> list = new ArrayList<>();
        for (int i = 0; i < DATA_SIZE; i++) {
            list.add(i);  // 動的にデータ追加
        }
        long listAddTime = System.nanoTime() - startTime;

        DecimalFormat df = new DecimalFormat("#,###");
        System.out.println("ArrayList追加時間: " + df.format(listAddTime / 1000000) + " ms");
        System.out.println("配列の場合: 事前にサイズが分かっていれば高速、不明なら実装困難");
        System.out.println("→ データサイズが不明な場合はArrayListが圧倒的に有利\n");
    }

    /**
     * データ検索速度の比較
     * 特定の値を探す処理での性能差を測定
     */
    private static void compareSearchSpeed() {
        System.out.println("【3. データ検索速度比較】");

        // テストデータの準備
        int[] array = new int[DATA_SIZE];
        ArrayList<Integer> list = new ArrayList<>();

        for (int i = 0; i < DATA_SIZE; i++) {
            array[i] = i;
            list.add(i);
        }

        int searchValue = DATA_SIZE - 1;  // 最後の要素を検索（最悪ケース）

        // 配列での線形検索
        long startTime = System.nanoTime();
        int arrayResult = -1;
        for (int i = 0; i < array.length; i++) {
            if (array[i] == searchValue) {
                arrayResult = i;
                break;
            }
        }
        long arraySearchTime = System.nanoTime() - startTime;

        // ArrayListでの検索（indexOf使用）
        startTime = System.nanoTime();
        int listResult = list.indexOf(searchValue);
        long listSearchTime = System.nanoTime() - startTime;

        // 結果表示
        DecimalFormat df = new DecimalFormat("#,###");
        System.out.println("配列での検索時間: " + df.format(arraySearchTime / 1000) + " μs");
        System.out.println("ArrayListでの検索時間: " + df.format(listSearchTime / 1000) + " μs");
        System.out.println("ArrayList検索は内部最適化済みで高速\n");
    }
}

    プログラムを実行して、実際の性能差を確認してください。

    典型的な結果例： - アクセス速度: 配列が約2-3倍高速 - 追加処理: ArrayListが圧倒的に柔軟 - 検索処理: ArrayListの内部最適化により実用的

    💾 メモリ効率の比較

    次に、メモリ使用量の違いを確認しましょう。

    メモリ使用量測定プログラム

    新しいクラス MemoryComparator を作成：

    import java.util.ArrayList;

/**
 * 配列とArrayListのメモリ使用量を比較
 * 実際のメモリ効率を数値で確認
 */
public class MemoryComparator {

    private static final int TEST_SIZE = 100000;  // 10万件

    public static void main(String[] args) {
        System.out.println("=== メモリ使用量比較テスト ===\n");

        // メモリ使用量の測定
        compareMemoryUsage();

        // メモリ効率の分析
        analyzeMemoryEfficiency();
    }

    /**
     * 実際のメモリ使用量を測定
     */
    private static void compareMemoryUsage() {
        Runtime runtime = Runtime.getRuntime();

        // ガベージコレクション実行
        System.gc();
        long baseMemory = runtime.totalMemory() - runtime.freeMemory();

        // 配列でのメモリ使用量
        System.gc();
        long beforeArray = runtime.totalMemory() - runtime.freeMemory();
        int[] array = new int[TEST_SIZE];
        for (int i = 0; i < TEST_SIZE; i++) {
            array[i] = i;
        }
        long afterArray = runtime.totalMemory() - runtime.freeMemory();
        long arrayMemory = afterArray - beforeArray;

        // ArrayListでのメモリ使用量  
        System.gc();
        long beforeList = runtime.totalMemory() - runtime.freeMemory();
        ArrayList<Integer> list = new ArrayList<>();
        for (int i = 0; i < TEST_SIZE; i++) {
            list.add(i);
        }
        long afterList = runtime.totalMemory() - runtime.freeMemory();
        long listMemory = afterList - beforeList;

        // 結果表示
        System.out.println("【メモリ使用量比較】");
        System.out.println("配列のメモリ使用: " + formatBytes(arrayMemory));
        System.out.println("ArrayListのメモリ使用: " + formatBytes(listMemory));
        System.out.println("メモリ差: " + String.format("%.1f", (double)listMemory / arrayMemory) + "倍\n");
    }

    /**
     * メモリ効率の詳細分析
     */
    private static void analyzeMemoryEfficiency() {
        System.out.println("【メモリ効率分析】");

        // 配列の理論値
        int arrayBytes = TEST_SIZE * 4;  // int = 4バイト
        System.out.println("配列の理論メモリ: " + formatBytes(arrayBytes));

        // ArrayListの追加オーバーヘッド
        System.out.println("ArrayListの追加要素:");
        System.out.println("- オブジェクトラッパー (Integer)");
        System.out.println("- 内部配列の余剰領域（容量の約25%）");
        System.out.println("- クラス構造のオーバーヘッド");

        System.out.println("\n【結論】");
        System.out.println("✓ 配列: メモリ効率が最高、大量データに最適");
        System.out.println("✓ ArrayList: 利便性重視、メモリより開発効率優先");
    }

    /**
     * バイト数を読みやすい形式にフォーマット
     */
    private static String formatBytes(long bytes) {
        if (bytes < 1024) return bytes + " B";
        if (bytes < 1024 * 1024) return String.format("%.1f KB", bytes / 1024.0);
        return String.format("%.1f MB", bytes / 1024.0 / 1024.0);
    }
}

    import java.util.ArrayList;
import java.util.Arrays;

/**
 * 配列とArrayListの相互変換戦略
 * 実務で必要な移行テクニック
 */
public class MigrationStrategy {

    public static void main(String[] args) {
        System.out.println("=== データ構造移行戦略 ===\n");

        // 配列からArrayListへの移行
        arrayToArrayListMigration();

        // ArrayListから配列への移行
        arrayListToArrayMigration();

        // 性能を考慮した移行判断
        migrationDecision();
    }

    /**
     * 配列からArrayListへの移行パターン
     */
    private static void arrayToArrayListMigration() {
        System.out.println("【配列 → ArrayList移行】");

        // 元の配列データ
        int[] originalArray = {10, 20, 30, 40, 50};
        System.out.println("元のデータ: " + Arrays.toString(originalArray));

        // 移行方法1: Arrays.asListを使用（推奨）
        ArrayList<Integer> list1 = new ArrayList<>(Arrays.asList(
            Arrays.stream(originalArray).boxed().toArray(Integer[]::new)
        ));

        // 移行方法2: ループで移行（理解しやすい）
        ArrayList<Integer> list2 = new ArrayList<>();
        for (int value : originalArray) {
            list2.add(value);
        }

        System.out.println("移行後ArrayList: " + list1);
        System.out.println("移行理由例: データサイズが動的に変化する要件追加\n");
    }

    /**
     * ArrayListから配列への移行パターン
     */
    private static void arrayListToArrayMigration() {
        System.out.println("【ArrayList → 配列移行】");

        // 元のArrayListデータ
        ArrayList<Integer> originalList = new ArrayList<>();
        originalList.add(100);
        originalList.add(200);
        originalList.add(300);
        System.out.println("元のデータ: " + originalList);

        // 移行方法1: toArrayを使用（推奨）
        Integer[] array1 = originalList.toArray(new Integer[0]);

        // 移行方法2: 基本型配列が必要な場合
        int[] array2 = originalList.stream().mapToInt(Integer::intValue).toArray();

        System.out.println("移行後配列: " + Arrays.toString(array2));
        System.out.println("移行理由例: 大量データ処理で性能問題が発生\n");
    }

    /**
     * 移行の判断基準
     */
    private static void migrationDecision() {
        System.out.println("【移行判断基準】");
        System.out.println("配列 → ArrayList移行のタイミング:");
        System.out.println("✓ データサイズが動的になった");
