    1. 高度な検索テクニック

    🔍 ステップ1: 複数条件での検索

    レッスン3.6では単純な検索を学びました。今度は、複数の条件を組み合わせた検索を実装してみましょう。

    新しいクラス AdvancedStringSearch を作成してください：

    import java.util.ArrayList;

public class AdvancedStringSearch {
    public static void main(String[] args) {
        // サンプルデータ：顧客情報
        String[] customers = {
            "田中太郎,tanaka@email.com,東京都,VIP",
            "佐藤花子,sato@email.com,大阪府,一般",
            "高橋次郎,takahashi@email.com,東京都,一般", 
            "田中美香,tanaka.mika@email.com,神奈川県,VIP",
            "山田太郎,yamada@email.com,愛知県,VIP"
        };

        System.out.println("=== 高度な検索テクニック ===\n");

        // 1. 単純検索（復習）
        System.out.println("1. 単純検索：「田中」を含む顧客");
        simpleSearch(customers, "田中");

        // 2. 複数条件検索（AND条件）
        System.out.println("\n2. 複数条件検索：「東京都」かつ「VIP」の顧客");
        multipleConditionSearch(customers, "東京都", "VIP");

        // 3. 大文字小文字を無視した検索
        System.out.println("\n3. 大文字小文字無視：「EMAIL」を含む顧客");
        caseInsensitiveSearch(customers, "EMAIL");

        // 4. 位置ベースの検索
        System.out.println("\n4. 位置ベース検索：メールアドレスが「.com」で終わる顧客");
        positionBasedSearch(customers);
    }

    // 単純検索（レッスン3.6の復習）
    public static void simpleSearch(String[] data, String keyword) {
        for (String item : data) {
            if (item.contains(keyword)) {
                System.out.println("  " + item);
            }
        }
    }

    // 複数条件検索（AND条件）
    public static void multipleConditionSearch(String[] data, String condition1, String condition2) {
        for (String item : data) {
            // 両方の条件を満たす場合のみ表示
            if (item.contains(condition1) && item.contains(condition2)) {
                System.out.println("  " + item);
            }
        }
    }

    // 大文字小文字を無視した検索
    public static void caseInsensitiveSearch(String[] data, String keyword) {
        String lowerKeyword = keyword.toLowerCase();  // 検索キーワードを小文字に

        for (String item : data) {
            // データも小文字に変換して比較
            if (item.toLowerCase().contains(lowerKeyword)) {
                System.out.println("  " + item);
            }
        }
    }

    // 位置ベースの検索
    public static void positionBasedSearch(String[] data) {
        for (String item : data) {
            // メールアドレス部分を抽出（2番目のカンマ区切りフィールド）
            String[] parts = item.split(",");
            if (parts.length >= 2) {
                String email = parts[1];
                if (email.endsWith(".com")) {
                    System.out.println("  " + item);
                }
            }
        }
    }
}

    レッスン3.6では基本的なreplace()メソッドを学びました。今度は、効率的で高度な置換技術を習得しましょう。

    新しいクラス AdvancedStringReplace を作成してください：

    public class AdvancedStringReplace {
    public static void main(String[] args) {
        System.out.println("=== 効率的な置換テクニック ===\n");

        // サンプルデータ：ユーザー入力（よくある問題文字を含む）
        String userInput = "こんにちは！！！　今日は良い天気ですね～～？？？　　よろしく！！";
        System.out.println("元の文字列：\"" + userInput + "\"");

        // 1. 基本的な置換（復習）
        System.out.println("\n1. 基本的な置換");
        basicReplace(userInput);

        // 2. 複数パターンの一括置換
        System.out.println("\n2. 複数パターン一括置換");
        multipleReplace(userInput);

        // 3. StringBuilderを使った効率的な置換
        System.out.println("\n3. StringBuilder効率的置換");
        efficientReplace(userInput);

        // 4. 条件付き置換
        System.out.println("\n4. 条件付き置換");
        conditionalReplace(userInput);
    }

    // 基本的な置換（レッスン3.6の復習）
    public static void basicReplace(String text) {
        String result = text.replace("！！！", "！");  // 連続感嘆符を単一に
        System.out.println("  結果：\"" + result + "\"");
    }

    // 複数パターンの一括置換
    public static void multipleReplace(String text) {
        String result = text;

        // 複数の置換を順次実行
        result = result.replace("！！！", "！");      // 連続感嘆符
        result = result.replace("？？？", "？");      // 連続疑問符
        result = result.replace("～～", "～");        // 連続波線
        result = result.replace("　　", " ");         // 全角スペース連続を半角1つに
        result = result.replace("　", " ");          // 残りの全角スペースを半角に

        System.out.println("  結果：\"" + result + "\"");
    }

    // StringBuilderを使った効率的な置換
    public static void efficientReplace(String text) {
        StringBuilder sb = new StringBuilder();

        for (int i = 0; i < text.length(); i++) {
            char current = text.charAt(i);

            // 連続する同じ記号をチェック
            if (current == '！' || current == '？' || current == '～') {
                sb.append(current);  // 1つ目は追加

                // 連続する同じ文字をスキップ
                while (i + 1 < text.length() && text.charAt(i + 1) == current) {
                    i++;  // 連続する文字を飛ばす
                }
            } else if (current == '　') {
                // 全角スペースを半角に
                sb.append(' ');

                // 連続する全角スペースをスキップ
                while (i + 1 < text.length() && text.charAt(i + 1) == '　') {
                    i++;
                }
            } else {
                sb.append(current);  // 通常の文字はそのまま追加
            }
        }

        String result = sb.toString();
        System.out.println("  結果：\"" + result + "\"");
        System.out.println("  (StringBuilderを使用した効率的な処理)");
    }

    // 条件付き置換
    public static void conditionalReplace(String text) {
        StringBuilder sb = new StringBuilder();

        // 文字数をカウントしながら置換
        int consecutiveCount = 1;
        char previousChar = '\0';

        for (int i = 0; i < text.length(); i++) {
            char current = text.charAt(i);

            if (current == previousChar && (current == '！' || current == '？' || current == '～')) {
                consecutiveCount++;
                // 3つ以上連続している場合のみ置換（2つまでは許可）
                if (consecutiveCount <= 2) {
                    sb.append(current);
                }
            } else {
                consecutiveCount = 1;
                if (current == '　') {
                    sb.append(' ');  // 全角スペースは常に半角に
                } else {
                    sb.append(current);
                }
            }

            previousChar = current;
        }

        String result = sb.toString();
        System.out.println("  結果：\"" + result + "\"");
        System.out.println("  (3つ以上の連続記号のみ削除、2つまでは許可)");
    }
}

    📊 ステップ5: CSVデータの安全な分割

    実際の業務でよく扱うCSVデータを安全に処理する技術を学びましょう。

    新しいクラス CSVDataProcessor を作成してください：

    import java.util.ArrayList;

public class CSVDataProcessor {
    public static void main(String[] args) {
        System.out.println("=== CSVデータ処理の実践技術 ===\n");

        // サンプルCSVデータ（商品情報）
        String[] csvData = {
            "商品ID,商品名,価格,在庫数,カテゴリ",
            "P001,ノートパソコン,98000,5,電子機器",

        basicCSVSplit(csvData);

        // 2. 安全なCSV分割（エラーハンドリング付き）
        System.out.println("\n2. 安全なCSV分割");
        safeCSVSplit(csvData);

        // 3. データ変換と集計
        System.out.println("\n3. データ変換と集計");
        dataAnalysis(csvData);

        // 4. 検索とフィルタリング
        System.out.println("\n4. 検索とフィルタリング");

    }

    // 基本的なCSV分割
    public static void basicCSVSplit(String[] data) {
        for (int i = 0; i < data.length; i++) {
            if (i == 0) {
                System.out.println("  ヘッダー：" + data[i]);
                continue;
            }

            String[] columns = data[i].split(",");
            System.out.println("  行" + i + "：" + java.util.Arrays.toString(columns));
        }
    }

    // 安全なCSV分割（エラーハンドリング付き）
    public static void safeCSVSplit(String[] data) {
        for (int i = 0; i < data.length; i++) {
            // 空行をスキップ
            if (data[i] == null || data[i].trim().isEmpty()) {
                System.out.println("  行" + i + "：空行をスキップ");
                continue;
            }

            // ヘッダー行の処理
            if (i == 0) {
                System.out.println("  ヘッダー：" + data[i]);
                continue;
            }

            try {
                String[] columns = parseSafeCSV(data[i]);
                System.out.printf("  行%d：ID=%s, 商品名=%s, 価格=%s, 在庫=%s, カテゴリ=%s%n", 
                    i, 
                    getColumnSafe(columns, 0),
                    getColumnSafe(columns, 1), 
                    getColumnSafe(columns, 2),
                    getColumnSafe(columns, 3),
                    getColumnSafe(columns, 4)
                );
            } catch (Exception e) {
                System.out.println("  行" + i + "：エラー - " + e.getMessage());
            }
        }
    }

    // 安全なCSV解析（引用符考慮）
    public static String[] parseSafeCSV(String line) {
        ArrayList<String> columns = new ArrayList<>();
        StringBuilder currentColumn = new StringBuilder();
        boolean insideQuotes = false;

        for (int i = 0; i < line.length(); i++) {
            char c = line.charAt(i);

            if (c == '"') {
                insideQuotes = !insideQuotes;  // 引用符の内外を切り替え
            } else if (c == ',' && !insideQuotes) {
                // 引用符外のカンマで分割
                columns.add(currentColumn.toString().trim());
                currentColumn = new StringBuilder();
            } else {
                currentColumn.append(c);
            }
        }

        // 最後の列を追加
        columns.add(currentColumn.toString().trim());

        return columns.toArray(new String[0]);
    }

    // 安全な列取得
    public static String getColumnSafe(String[] columns, int index) {
        if (columns == null || index >= columns.length) {
            return "(未設定)";
        }

        String value = columns[index];
        return (value == null || value.isEmpty()) ? "(空)" : value;
    }

    // データ変換と集計
    public static void dataAnalysis(String[] data) {
        int totalProducts = 0;
        int totalStock = 0;
        long totalValue = 0;  // 在庫価値合計

        for (int i = 1; i < data.length; i++) {  // ヘッダーをスキップ
            if (data[i] == null || data[i].trim().isEmpty()) continue;

            try {
                String[] columns = parseSafeCSV(data[i]);

                // 価格と在庫数を数値に変換
                String priceStr = getColumnSafe(columns, 2);
                String stockStr = getColumnSafe(columns, 3);

                if (!priceStr.equals("(空)") && !priceStr.equals("(未設定)") &&
                    !stockStr.equals("(空)") && !stockStr.equals("(未設定)")) {

                    int price = Integer.parseInt(priceStr);
                    int stock = Integer.parseInt(stockStr);

                    totalProducts++;
                    totalStock += stock;
                    totalValue += (long)price * stock;
                }
            } catch (NumberFormatException e) {
                // 数値変換エラーは無視して次へ
                continue;
            }
        }

        System.out.println("  商品数：" + totalProducts + "件");
        System.out.println("  総在庫数：" + totalStock + "個");
        System.out.println("  在庫価値合計：" + String.format("%,d", totalValue) + "円");
    }

    // 検索とフィルタリング
    public static void searchAndFilter(String[] data) {
        System.out.println("  電子機器カテゴリの商品：");

        for (int i = 1; i < data.length; i++) {  // ヘッダーをスキップ
            if (data[i] == null || data[i].trim().isEmpty()) continue;

            String[] columns = parseSafeCSV(data[i]);
            String category = getColumnSafe(columns, 4);

            if (category.contains("電子機器")) {
                String productName = getColumnSafe(columns, 1);
                String price = getColumnSafe(columns, 2);
                System.out.println("    " + productName + " - " + price + "円");
            }
        }
    }
}

        for (String log : logs) {
            if (log.contains("[ERROR]")) {
                // ログを構造化して解析
                String[] parts = log.split(" ", 4);  // 最大4分割

                if (parts.length >= 4) {
                    String date = parts[0];
                    String time = parts[1];
                    String level = parts[2];
                    String message = parts[3];

                    System.out.println("  時刻: " + date + " " + time);

                // 時刻とアクションを抽出

                String action = extractAction(log);

                System.out.println("  " + time + " - " + action);

    // 時刻抽出
    public static String extractTime(String log) {
        int spaceIndex = log.indexOf(" ");
        if (spaceIndex != -1) {
            int secondSpaceIndex = log.indexOf(" ", spaceIndex + 1);
            if (secondSpaceIndex != -1) {
                return log.substring(spaceIndex + 1, secondSpaceIndex);
            }
        }
        return "(時刻不明)";
    }

    // アクション抽出
    public static String extractAction(String log) {
        int bracketStart = log.indexOf("]");
        if (bracketStart != -1) {
            String remaining = log.substring(bracketStart + 1).trim();
            int userIdIndex = remaining.indexOf("user_id=");
            if (userIdIndex != -1) {
                return remaining.substring(0, userIdIndex).trim();
            }
            return remaining;
        }
        return "(アクション不明)";
    }

        System.out.println("元のテキスト:");

        String cleaned = cleanText(input);

        System.out.println("\n整形後のテキスト:");
        System.out.println("\"" + cleaned + "\"");

        // 変更内容の説明
        if (!input.equals(cleaned)) {
            System.out.println("\n適用された処理:");

        } else {
            System.out.println("\n変更は必要ありませんでした。");
        }
    }

    // テキストクリーニング実行
    public static String cleanText(String text) {
        StringBuilder sb = new StringBuilder();
        char previousChar = '\0';

        for (int i = 0; i < text.length(); i++) {
            char current = text.charAt(i);

            // 連続する記号の処理
            if (current == '！' || current == '？' || current == '～') {
                if (current != previousChar) {
                    sb.append(current);
                }
                // 連続する同じ記号は無視
            }
            // 全角スペースを半角に
            else if (current == '　') {
                if (previousChar != ' ' && previousChar != '　') {
                    sb.append(' ');
                }
                // 連続するスペースは無視
            }
            // 通常の文字
            else {
                sb.append(current);
            }

            previousChar = current;
        }

        // 前後の空白を除去
        return sb.toString().trim();
    }

    // クリーニング詳細の表示
    public static void showCleaningDetails(String original, String cleaned) {
        if (original.contains("！！") || original.contains("？？") || original.contains("～～")) {
            System.out.println("- 連続する記号を単一に変換");
        }
        if (original.contains("　")) {
            System.out.println("- 全角スペースを半角スペースに変換");
        }
        if (!original.trim().equals(original)) {
            System.out.println("- 前後の不要な空白を除去");
        }
    }

    // ログ解析実行
    public static void analyzeLogData(String[] logs) {
        int infoCount = 0, warnCount = 0, errorCount = 0, otherCount = 0;

        for (String log : logs) {
            if (log.contains("[INFO]") || log.contains("INFO")) {
                infoCount++;
            } else if (log.contains("[WARN]") || log.contains("WARN")) {
                warnCount++;
            } else if (log.contains("[ERROR]") || log.contains("ERROR")) {
                errorCount++;
            } else {
                otherCount++;
            }
        }

        System.out.println("総ログ行数: " + logs.length + "行");
        System.out.println("INFOレベル: " + infoCount + "行");
        System.out.println("WARNレベル: " + warnCount + "行");
        System.out.println("ERRORレベル: " + errorCount + "行");
        System.out.println("その他: " + otherCount + "行");

        // エラーログがある場合は詳細表示
        if (errorCount > 0) {
            System.out.println("\nエラーログの詳細:");
            for (String log : logs) {
                if (log.contains("[ERROR]") || log.contains("ERROR")) {
                    System.out.println("  " + log);
                }
            }
        }
    }

    // 高度検索実行
    public static void performAdvancedSearch(String[] texts, String keyword) {
        int matchCount = 0;

        System.out.println("検索キーワード: \"" + keyword + "\"");
        System.out.println();

        for (int i = 0; i < texts.length; i++) {
            String text = texts[i];

            // 大文字小文字を無視して検索
            if (text.toLowerCase().contains(keyword.toLowerCase())) {
                matchCount++;

                // マッチした部分をハイライト表示（簡易版）
                String highlightedText = highlightMatch(text, keyword);
                System.out.println("行" + (i + 1) + ": " + highlightedText);
            }
        }

        System.out.println("\n検索結果: " + matchCount + "件のマッチが見つかりました。");
    }

    // マッチ部分のハイライト（簡易版）
    public static String highlightMatch(String text, String keyword) {
        // 大文字小文字を無視して置換
        StringBuilder result = new StringBuilder();
        String lowerText = text.toLowerCase();
        String lowerKeyword = keyword.toLowerCase();

        int lastIndex = 0;
        int index = lowerText.indexOf(lowerKeyword);

        while (index != -1) {
            // マッチ前の部分を追加
            result.append(text.substring(lastIndex, index));

            // マッチ部分を大括弧で囲んで追加
            result.append("[").append(text.substring(index, index + keyword.length())).append("]");

            lastIndex = index + keyword.length();
            index = lowerText.indexOf(lowerKeyword, lastIndex);
        }

        // 残りの部分を追加
        result.append(text.substring(lastIndex));

        return result.toString();
    }

    💡 よくある質問

    Q1: 正規表現は覚える必要がありますか？

    A: 基本パターンだけで十分です。今回学んだ以下があれば実用的： - [文字] : 指定文字のマッチ - {回数} : 繰り返し回数指定 - + : 1回以上の繰り返し - 覚えられない部分は都度調べて使用すればOK

    Q2: StringBuilderはいつ使うべきですか？

    A: 以下の場合にStringBuilderを使用： - 大量の文字列結合（100回以上の目安） - ループ内での文字列構築 - 動的に文字列を構築する場合 - 少量の処理ならString.replace()で十分

    Q3: CSVデータの引用符処理が複雑です

    A: 引用符処理は確かに複雑です： - 基本的には今回のパターンで対応可能 - より複雑な場合は専用ライブラリ（OpenCSV等）を使用 - 自作する場合は段階的にテストすることが重要

    Q4: エラーハンドリングはどこまで必要ですか？

    A: 実用性を考慮して段階的に： 1. 最低限: null チェック、空文字チェック 2. 基本: NumberFormatException等の主要例外 3. 実用: 想定外データのスキップ処理 4. プロ: 詳細なログ出力とリカバリ処理
