    📘 ToDoシステム設計の概要

    2. データモデルとクラス設計

    🛠️ ステップ3: Todoクラスの設計

    実際のシステム開発では、データを表現するクラスを最初に設計します。

    新しいクラスを作成

    Eclipse で Todo.java クラスを作成してください：

    import java.time.LocalDate;
import java.time.format.DateTimeFormatter;

/**
 * ToDoアイテムを表現するクラス
 * 1つのToDoに必要な全ての情報を管理
 */
public class Todo {
    // ToDo項目の属性
    private int id;                    // 一意識別子
    private String title;              // タイトル
    private String description;        // 詳細説明
    private boolean completed;         // 完了状態
    private Priority priority;         // 優先度
    private LocalDate dueDate;        // 期限
    private LocalDate createdDate;    // 作成日

    // 優先度の列挙型（enum）
    public enum Priority {
        HIGH("高"), MEDIUM("中"), LOW("低");

        private final String displayName;

        Priority(String displayName) {
            this.displayName = displayName;
        }

        public String getDisplayName() {
            return displayName;
        }
    }

    // コンストラクタ
    public Todo(int id, String title, String description, Priority priority, LocalDate dueDate) {
        this.id = id;
        this.title = title;
        this.description = description;
        this.completed = false;  // 初期状態は未完了
        this.priority = priority;
        this.dueDate = dueDate;
        this.createdDate = LocalDate.now();
    }

    // ゲッター・セッターメソッド
    public int getId() { return id; }
    public String getTitle() { return title; }
    public void setTitle(String title) { this.title = title; }

    public String getDescription() { return description; }
    public void setDescription(String description) { this.description = description; }

    public boolean isCompleted() { return completed; }
    public void setCompleted(boolean completed) { this.completed = completed; }

    public Priority getPriority() { return priority; }
    public void setPriority(Priority priority) { this.priority = priority; }

    public LocalDate getDueDate() { return dueDate; }
    public void setDueDate(LocalDate dueDate) { this.dueDate = dueDate; }

    public LocalDate getCreatedDate() { return createdDate; }

    // 期限切れかどうかを判定
    public boolean isOverdue() {
        if (dueDate == null || completed) {
            return false;
        }
        return LocalDate.now().isAfter(dueDate);
    }

    // 表示用の文字列生成
    @Override
    public String toString() {
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy/MM/dd");
        String status = completed ? "✓" : " ";
        String dueDateStr = (dueDate != null) ? dueDate.format(formatter) : "未設定";
        String overdueWarning = isOverdue() ? " [期限切れ]" : "";

        return String.format("[%s] %s (%s) - 期限: %s%s", 
                           status, title, priority.getDisplayName(), dueDateStr, overdueWarning);
    }
}

    ✅ 実行してテスト

    一度保存して、コンパイルエラーがないか確認してください。

    🎯 ステップ4: システム設計のポイント解説

    なぜこの設計にしたのか？

    1. enum（列挙型）の採用 

    public enum Priority {
    HIGH("高"), MEDIUM("中"), LOW("低");
}
    - 優先度の値を制限（HIGH/MEDIUM/LOW のみ） - タイプセーフ（コンパイル時にエラー検出） - 表示名との分離（内部値と表示値を別管理）

    2. LocalDateの採用 

    private LocalDate dueDate;        // 期限
private LocalDate createdDate;    // 作成日
    - Java 8以降の現代的な日付API - null安全性の考慮 - 日付計算の簡単化

    3. メソッド設計の工夫 

    public boolean isOverdue() {
    // 期限切れ判定のビジネスロジック
}
    - 判定ロジックをメソッドに集約 - 再利用性の向上 - テストの容易化

    この設計により、保守性と拡張性の高いシステムが実現できます！

    3. メインシステムの実装

    🛠️ ステップ5: ToDoManagerクラスの実装

    システムの中核となる管理クラスを実装します。

    新しいクラスを作成

    Eclipse で TodoManager.java クラスを作成してください：

    import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.Scanner;

/**
 * ToDoリスト管理システムのメインクラス
 * レッスン1-4で学んだ技術を統合的に活用
 */
public class TodoManager {

    /**
     * 有効な選択肢の入力を取得
     */
    private static int getValidChoice(int min, int max) {
        while (true) {
            try {
                int choice = Integer.parseInt(scanner.nextLine());
                if (choice >= min && choice <= max) {
                    return choice;
                }
                System.out.printf("エラー: %d〜%d の範囲で入力してください: ", min, max);
            } catch (NumberFormatException e) {
                System.out.print("エラー: 数値を入力してください: ");
            }
        }
    }

    /**
     * 新しいToDoの追加
     * レッスン2-3の技術を活用
     */
    private static void addTodo() {
        System.out.println("\n=== 📝 新しいToDo追加 ===");

        // タイトル入力
        System.out.print("タイトル: ");
        String title = scanner.nextLine().trim();
        if (title.isEmpty()) {
            System.out.println("エラー: タイトルは必須です。");
            return;
        }

        // 説明入力
        System.out.print("説明 (省略可): ");
        String description = scanner.nextLine().trim();

        // 優先度選択
        Todo.Priority priority = selectPriority();

        // 期限設定
        LocalDate dueDate = inputDueDate();

        // ToDoオブジェクト作成とリストに追加
        Todo newTodo = new Todo(nextId++, title, description, priority, dueDate);
        todoList.add(newTodo);  // ArrayListの動的サイズ活用

        System.out.println("✅ ToDoを追加しました: " + newTodo.getTitle());
        System.out.printf("📊 現在のToDo数: %d件\n", todoList.size());
    }

    /**
     * 優先度の選択
     */
    private static Todo.Priority selectPriority() {
        System.out.println("優先度を選択してください:");
        System.out.println("1. 高");
        System.out.println("2. 中");
        System.out.println("3. 低");
        System.out.print("選択 (1-3): ");

        int choice = getValidChoice(1, 3);
        switch (choice) {
            case 1: return Todo.Priority.HIGH;
            case 2: return Todo.Priority.MEDIUM;
            case 3: return Todo.Priority.LOW;
            default: return Todo.Priority.MEDIUM;
        }
    }

    /**
     * 期限日の入力
     */
    private static LocalDate inputDueDate() {
        System.out.print("期限 (yyyy-mm-dd形式、省略可): ");
        String dueDateStr = scanner.nextLine().trim();

        if (dueDateStr.isEmpty()) {
            return null;
        }

        try {
            DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd");
            return LocalDate.parse(dueDateStr, formatter);
        } catch (DateTimeParseException e) {
            System.out.println("警告: 日付形式が正しくありません。期限を設定しませんでした。");
            return null;
        }
    }

    private static void displayAllTodos() {

    /**
     * ToDo完了状態の切り替え

    private static void toggleTodoStatus() {

            System.out.println("📭 ToDoリストは空です。");

        System.out.println("\n=== ✅ ToDo完了状態の切り替え ===");

     * ToDoの削除

        displayAllTodos();

    /**
     * ToDoの検索機能
     * レッスン3の検索技術を活用
     */
    private static void searchTodos() {
        if (todoList.isEmpty()) {
            System.out.println("📭 ToDoリストは空です。");
            return;
        }

        System.out.println("\n=== 🔍 ToDo検索 ===");
        System.out.println("1. タイトルで検索");
        System.out.println("2. 完了状態で検索");
        System.out.println("3. 優先度で検索");
        System.out.println("4. 期限切れで検索");
        System.out.print("検索方法を選択 (1-4): ");

        int searchType = getValidChoice(1, 4);
        ArrayList<Todo> results = new ArrayList<>();

        switch (searchType) {
            case 1: // タイトル検索
                System.out.print("検索するタイトル（部分一致）: ");
                String keyword = scanner.nextLine().trim().toLowerCase();
                for (Todo todo : todoList) {
                    if (todo.getTitle().toLowerCase().contains(keyword)) {
                        results.add(todo);
                    }
                }
                break;

            case 2: // 完了状態検索
                System.out.println("1. 未完了のみ");
                System.out.println("2. 完了済みのみ");
                System.out.print("選択 (1-2): ");
                boolean searchCompleted = getValidChoice(1, 2) == 2;
                for (Todo todo : todoList) {
                    if (todo.isCompleted() == searchCompleted) {
                        results.add(todo);
                    }
                }
                break;

            case 3: // 優先度検索
                Todo.Priority targetPriority = selectPriority();
                for (Todo todo : todoList) {
                    if (todo.getPriority() == targetPriority) {
                        results.add(todo);
                    }
                }
                break;

            case 4: // 期限切れ検索
                for (Todo todo : todoList) {
                    if (todo.isOverdue()) {
                        results.add(todo);
                    }
                }
                break;
        }

        // 検索結果の表示
        displaySearchResults(results);
    }

    /**
     * ToDoの並び替え機能
     * レッスン3のソート技術を活用
     */
    private static void sortTodos() {
        if (todoList.isEmpty()) {
            System.out.println("📭 ToDoリストは空です。");
            return;
        }

        System.out.println("\n=== 📊 ToDo並び替え ===");
        System.out.println("1. 優先度順（高→低）");
        System.out.println("2. 期限順（近い→遠い）");
        System.out.println("3. 作成日順（新しい→古い）");
        System.out.println("4. 完了状態順（未完了→完了済み）");
        System.out.print("並び替え方法を選択 (1-4): ");

        int sortType = getValidChoice(1, 4);

        switch (sortType) {
            case 1: // 優先度順
                Collections.sort(todoList, new Comparator<Todo>() {
                    @Override
                    public int compare(Todo t1, Todo t2) {
                        return t1.getPriority().ordinal() - t2.getPriority().ordinal();
                    }
                });
                System.out.println("✅ 優先度順に並び替えました。");
                break;

            case 2: // 期限順
                Collections.sort(todoList, new Comparator<Todo>() {
                    @Override
                    public int compare(Todo t1, Todo t2) {
                        if (t1.getDueDate() == null && t2.getDueDate() == null) return 0;
                        if (t1.getDueDate() == null) return 1;
                        if (t2.getDueDate() == null) return -1;
                        return t1.getDueDate().compareTo(t2.getDueDate());
                    }
                });
                System.out.println("✅ 期限順に並び替えました。");
                break;

            case 3: // 作成日順
                Collections.sort(todoList, new Comparator<Todo>() {
                    @Override
                    public int compare(Todo t1, Todo t2) {
                        return t2.getCreatedDate().compareTo(t1.getCreatedDate());
                    }
                });
                System.out.println("✅ 作成日順に並び替えました。");
                break;

            case 4: // 完了状態順
                Collections.sort(todoList, new Comparator<Todo>() {
                    @Override
                    public int compare(Todo t1, Todo t2) {
                        return Boolean.compare(t1.isCompleted(), t2.isCompleted());
                    }
                });
                System.out.println("✅ 完了状態順に並び替えました。");
                break;
        }

        displayAllTodos();
    }

    /**
     * 統計情報の表示
     */
    private static void displayStatistics() {
        System.out.println("\n=== 📈 統計情報 ===");

        if (todoList.isEmpty()) {
            System.out.println("📭 ToDoリストは空です。");
            return;
        }

        int total = todoList.size();
        int completed = 0;
        int overdue = 0;
        int high = 0, medium = 0, low = 0;

        for (Todo todo : todoList) {
            if (todo.isCompleted()) completed++;
            if (todo.isOverdue()) overdue++;

            switch (todo.getPriority()) {
                case HIGH: high++; break;
                case MEDIUM: medium++; break;
                case LOW: low++; break;
            }
        }

        int pending = total - completed;
        double completionRate = total > 0 ? (double) completed / total * 100 : 0;

        System.out.println("📊 全体統計");
        System.out.printf("   総ToDo数: %d件\n", total);
        System.out.printf("   完了済み: %d件\n", completed);
        System.out.printf("   未完了  : %d件\n", pending);
        System.out.printf("   完了率  : %.1f%%\n", completionRate);
        System.out.printf("   期限切れ: %d件\n", overdue);

        System.out.println("\n📊 優先度別統計");
        System.out.printf("   高優先度: %d件\n", high);
        System.out.printf("   中優先度: %d件\n", medium);
        System.out.printf("   低優先度: %d件\n", low);

        if (overdue > 0) {
            System.out.println("\n⚠️ 期限切れのToDoがあります！");
        }
    }

    // データの永続化
public static void saveTodoList(String filename) {
    try (PrintWriter writer = new PrintWriter(new FileWriter(filename))) {
        for (Todo todo : todoList) {
            writer.println(todo.getId() + "," + 
                          todo.getTitle() + "," + 
                          todo.getDescription() + "," + 
                          todo.isCompleted() + "," + 
                          todo.getPriority() + "," + 
                          todo.getDueDate());
        }
        System.out.println("✅ データを保存しました: " + filename);
    } catch (IOException e) {
        System.out.println("❌ 保存に失敗しました: " + e.getMessage());
    }
}
