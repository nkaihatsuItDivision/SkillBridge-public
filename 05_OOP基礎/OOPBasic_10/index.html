public class DataAnalyzer {

    public void performClassAnalysis(List&lt;Student&gt; students) {
        System.out.println(&quot;=== „ÇØ„É©„ÇπÂàÜÊûêÁµêÊûú ===&quot;);

        // Âü∫Êú¨Áµ±Ë®à
        double avgGPA = calculateAverageGPA(students);
        Student topStudent = findTopPerformer(students);
        Student bottomStudent = findBottomPerformer(students);

        System.out.println(&quot;Âπ≥ÂùáGPA: &quot; + String.format(&quot;%.2f&quot;, avgGPA));
        System.out.println(&quot;ÊúÄÈ´òGPA: &quot; + String.format(&quot;%.2f&quot;, topStudent.calculateGPA()));
        System.out.println(&quot;ÊúÄ‰ΩéGPA: &quot; + String.format(&quot;%.2f&quot;, bottomStudent.calculateGPA()));
        System.out.println(&quot;ÊúÄÂÑ™ÁßÄÂ≠¶Áîü: &quot; + topStudent.getName());

        // ÊàêÁ∏æÂàÜÂ∏É
        System.out.println(&quot;\n--- ÊàêÁ∏æÂàÜÂ∏É ---&quot;);
        Map&lt;String, Long&gt; levelDistribution = students.stream()
                                                     .collect(Collectors.groupingBy(Student::getPerformanceLevel,
                                                                                  Collectors.counting()));

        levelDistribution.forEach((level, count) -&gt; 
            System.out.println(level + &quot;: &quot; + count + &quot;Âêç&quot;));

        System.out.println(&quot;==================&quot;);
    }

    private double calculateAverageGPA(List&lt;Student&gt; students) {
        return students.stream()
                      .mapToDouble(Student::calculateGPA)
                      .average()
                      .orElse(0.0);
    }

    private Student findTopPerformer(List&lt;Student&gt; students) {
        return students.stream()
                      .max((s1, s2) -&gt; Double.compare(s1.calculateGPA(), s2.calculateGPA()))
                      .orElse(null);
    }

    private Student findBottomPerformer(List&lt;Student&gt; students) {
        return students.stream()
                      .min((s1, s2) -&gt; Double.compare(s1.calculateGPA(), s2.calculateGPA()))
                      .orElse(null);
    }
}
