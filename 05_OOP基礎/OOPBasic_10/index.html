    📋 開発する学生管理システムの概要

    今回開発するのは、実際の教育現場で使えるレベルの包括的な学生管理システムです。

public class DataAnalyzer {

    // クラス全体の分析
    public void performClassAnalysis(List<Student> students) {
        double avgGPA = calculateAverageGPA(students);
        Student topStudent = findTopPerformer(students);

        System.out.println("=== クラス分析結果 ===");
        System.out.println("平均GPA: " + String.format("%.2f", avgGPA));
        System.out.println("最優秀学生: " + topStudent.getName() + 
                         " (GPA: " + String.format("%.2f", topStudent.calculateGPA()) + ")");
    }

    private double calculateAverageGPA(List<Student> students) {
        return students.stream()
                      .mapToDouble(Student::calculateGPA)
                      .average()
                      .orElse(0.0);
    }

    private Student findTopPerformer(List<Student> students) {
        return students.stream()
                      .max((s1, s2) -> Double.compare(s1.calculateGPA(), s2.calculateGPA()))
                      .orElse(null);
    }
}

public class Student {
    // レッスン4.7-4.8：カプセル化の実践

    private String name;
    private int age;
    private String major;
    private List<Grade> grades;

    // レッスン4.5：コンストラクタ設計
    public Student(String studentId, String name, int age, String major) {
        this.studentId = studentId;
        this.name = name;
        this.age = age;
        this.major = major;
        this.grades = new ArrayList<>();
    }

    // レッスン4.6：インスタンスメソッド（メソッドチェーン対応）
    public Student addGrade(Grade grade) {
        this.grades.add(grade);
        return this;
    }

    // レッスン4.8：getter/setterの適切な実装
    public String getStudentId() { return studentId; }
    public String getName() { return name; }
    public int getAge() { return age; }
    public String getMajor() { return major; }
    public List<Grade> getGrades() { return new ArrayList<>(grades); } // 防御的コピー

    // 実用機能：GPA計算
    public double calculateGPA() {
        if (grades.isEmpty()) return 0.0;

        double totalPoints = grades.stream()
                                  .mapToDouble(Grade::getScore)
                                  .sum();
        return totalPoints / grades.size() / 20.0; // 100点満点を5.0満点に変換
    }

    // 成績レベル判定
    public String getPerformanceLevel() {
        double gpa = calculateGPA();
        if (gpa >= 4.0) return "優秀";
        else if (gpa >= 3.0) return "良好";
        else if (gpa >= 2.0) return "普通";
        else return "要努力";
    }
}

    public void generateStudentReport(String studentId) {
        Student student = findStudentById(studentId);
        if (student != null) {
            reportGenerator.generateIndividualReport(student);
        } else {
            System.out.println("❌ 学生ID " + studentId + " が見つかりません");
        }
    }

    public void generateAllReports() {
        System.out.println("=== 全学生レポート一括生成 ===\n");
        students.forEach(reportGenerator::generateIndividualReport);
        System.out.println();
        reportGenerator.generateClassReport(students);
    }

    // 分析機能
    public void analyzeClassPerformance() {
        dataAnalyzer.performClassAnalysis(students);
    }

    // 検索機能
    public List<Student> searchStudentsByMajor(String major) {
        return students.stream()
                      .filter(s -> s.getMajor().equals(major))
                      .collect(Collectors.toList());
    }

    public List<Student> findHighPerformers(double minGPA) {
        return students.stream()
                      .filter(s -> s.calculateGPA() >= minGPA)
                      .sorted((s1, s2) -> Double.compare(s2.calculateGPA(), s1.calculateGPA()))
                      .collect(Collectors.toList());
    }

    // システム統計
    public void showSystemSummary() {
        System.out.println("=== システム統計サマリー ===");
        System.out.println("総学生数: " + students.size() + "名");

        int totalGrades = students.stream()
                                 .mapToInt(s -> s.getGrades().size())
                                 .sum();
        System.out.println("総成績数: " + totalGrades + "件");

        Map<String, Long> majorCount = students.stream()
                                              .collect(Collectors.groupingBy(Student::getMajor, 
                                                                           Collectors.counting()));
        System.out.println("専攻数: " + majorCount.size() + "種類");

        System.out.println("\n--- 専攻別学生数 ---");
        majorCount.forEach((major, count) -> 
            System.out.println(major + ": " + count + "名"));
        System.out.println("==========================");
    }
}
