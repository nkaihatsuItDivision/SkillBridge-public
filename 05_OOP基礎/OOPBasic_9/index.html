    前回のレッスンでは、getter/setterによる安全なデータアクセス方法を習得しました：

    public class Student {
    private String name;
    private int age;
    private double gpa;

    // getter: 読み取り専用アクセス
    public String getName() {
        return name;
    }

    // setter: バリデーション付き書き込みアクセス
    public void setName(String name) {
        if (name != null && !name.trim().isEmpty()) {
            this.name = name;
        } else {
            throw new IllegalArgumentException("名前が無効です");
        }
    }

    // 読み取り専用プロパティ
    public boolean isAdult() {
        return age >= 18;
    }
}

    🏗️ 複数クラス協調設計とは何か？

    現実世界での協調関係

    単一クラス設計は「一人で作業する職人」のようなものでした。 複数クラス協調設計は「チームで連携する組織」のようなものです。

    現実世界の例： - 学生と成績：学生は複数の成績を持ち、成績は特定の学生に属する - 車とエンジン：車はエンジンを持ち、エンジンは車の一部として動作する - 図書館と本：図書館は多くの本を管理し、本は図書館に属する

    1. StudentクラスとGradeクラスの基本設計

    Gradeクラス（成績クラス）の作成

    まず、成績情報を管理するGradeクラスを作成しましょう：

    /**
 * 成績情報を管理するクラス
 * 科目名、点数、学期などの成績に関する責任を担う
 */
public class Grade {
    private String subject;    // 科目名
    private int score;         // 点数
    private String semester;   // 学期

    // コンストラクタ
    public Grade(String subject, int score, String semester) {
        setSubject(subject);
        setScore(score);
        setSemester(semester);
    }

    // getter/setter（レッスン4.8の技術を活用）
    public String getSubject() {
        return subject;
    }

    public void setSubject(String subject) {
        if (subject != null && !subject.trim().isEmpty()) {
            this.subject = subject;
        } else {
            throw new IllegalArgumentException("科目名が無効です");
        }
    }

    public int getScore() {
        return score;
    }

    public void setScore(int score) {
        if (score >= 0 && score <= 100) {
            this.score = score;
        } else {
            throw new IllegalArgumentException("点数は0-100の範囲で入力してください");
        }
    }

    public String getSemester() {
        return semester;
    }

    public void setSemester(String semester) {
        if (semester != null && !semester.trim().isEmpty()) {
            this.semester = semester;
        } else {
            throw new IllegalArgumentException("学期が無効です");
        }
    }

    // 成績に関する処理はGradeクラスの責任
    public String getLetterGrade() {
        if (score >= 90) return "A";
        else if (score >= 80) return "B";
        else if (score >= 70) return "C";
        else if (score >= 60) return "D";
        else return "F";
    }

    public boolean isPassing() {
        return score >= 60;
    }

    public double getGradePoint() {
        if (score >= 90) return 4.0;
        else if (score >= 80) return 3.0;
        else if (score >= 70) return 2.0;
        else if (score >= 60) return 1.0;
        else return 0.0;
    }

    // 情報表示
    public void showGradeInfo() {
        System.out.printf("%s: %d点 (%s) - %s%n", 
            subject, score, getLetterGrade(), semester);
    }

    @Override
    public String toString() {
        return String.format("%s:%d点(%s)", subject, score, getLetterGrade());
    }
}

import java.util.List;

/**
 * 学生情報を管理し、Gradeクラスと協調するクラス
 * 学生の基本情報と成績管理の責任を担う

public class Student {
    private String name;
    private int age;
    private String major;
    private List<Grade> grades; // Gradeクラスとの関連

    // コンストラクタ
    public Student(String name, int age, String major) {
        setName(name);
        setAge(age);
        setMajor(major);
        this.grades = new ArrayList<>();
    }

    // 基本的なgetter/setter
    public String getName() {
        return name;
    }

    public void setName(String name) {
        if (name != null && !name.trim().isEmpty()) {
            this.name = name;
        } else {
            throw new IllegalArgumentException("名前が無効です");
        }
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        if (age > 0 && age < 150) {
            this.age = age;
        } else {
            throw new IllegalArgumentException("年齢が無効です");
        }
    }

    public String getMajor() {
        return major;
    }

    public void setMajor(String major) {
        if (major != null && !major.trim().isEmpty()) {
            this.major = major;
        } else {
            throw new IllegalArgumentException("専攻が無効です");
        }
    }

    // ✅ クラス間協調メソッド：成績の追加
    public void addGrade(Grade grade) {
        if (grade != null) {
            grades.add(grade);
            System.out.println(name + "の成績を追加しました: " + grade.toString());
        } else {
            throw new IllegalArgumentException("成績が無効です");
        }
    }

    // ✅ クラス間協調メソッド：GPA計算
    public double calculateGPA() {
        if (grades.isEmpty()) {
            return 0.0;
        }

        double totalPoints = 0.0;
        for (Grade grade : grades) {
            totalPoints += grade.getGradePoint();
        }

        return Math.round(totalPoints / grades.size() * 100.0) / 100.0;
    }

    // ✅ クラス間協調メソッド：科目別成績の取得
    public List<Grade> getGradesBySubject(String subject) {
        List<Grade> result = new ArrayList<>();
        for (Grade grade : grades) {
            if (grade.getSubject().equals(subject)) {
                result.add(grade);
            }
        }
        return result;
    }

    // ✅ クラス間協調メソッド：合格科目数の取得
    public int getPassingGradeCount() {
        int count = 0;
        for (Grade grade : grades) {
            if (grade.isPassing()) {
                count++;
            }
        }
        return count;
    }

    // 防御的コピーによる安全な成績リスト取得
    public List<Grade> getGrades() {
        return new ArrayList<>(grades);
    }

    // 学生情報の表示
    public void showStudentInfo() {

        System.out.printf("名前: %s (%d歳)%n", name, age);

        System.out.printf("合格科目数: %d科目%n", getPassingGradeCount());

            System.out.println("\n=== 成績一覧 ===");
            for (Grade grade : grades) {
                System.out.print("  ");
                grade.showGradeInfo();
            }
        }
    }
}

    📘 責務・結合度・凝集度

    結合度の管理

    結合度（クラス間の依存関係の強さ）を適切に管理することが重要です：

    // ✅ 良い例：疎結合（Loose Coupling）
public class Student {
    private List<Grade> grades;

    // Gradeクラスのpublicメソッドのみを使用
    public double calculateGPA() {
        double totalPoints = 0.0;
        for (Grade grade : grades) {
            totalPoints += grade.getGradePoint(); // publicメソッド使用
        }
        return totalPoints / grades.size();
    }
}

// ❌ 悪い例：密結合（Tight Coupling）
public class Student {
    private List<Grade> grades;

    // Gradeクラスの内部実装に依存
    public double calculateGPA() {
        double totalPoints = 0.0;
        for (Grade grade : grades) {
            // Gradeクラスの内部フィールドに直接アクセス（危険）
            int score = grade.score; // privateフィールドへのアクセス
            // 複雑な計算ロジックをStudentクラスに記述（責任の混在）
            if (score >= 90) totalPoints += 4.0;
            else if (score >= 80) totalPoints += 3.0;
            // ...
        }
        return totalPoints / grades.size();
    }
}

    凝集度の向上

    凝集度（クラス内の要素がどれだけ関連しているか）を高めることで、理解しやすいクラスを作成できます：

    // ✅ 高凝集度：関連する処理がまとまっている
public class Grade {
    private String subject;
    private int score;
    private String semester;

    // すべてのメソッドが「成績」に関連
    public String getLetterGrade() { /* 成績評価 */ }
    public boolean isPassing() { /* 合格判定 */ }
    public double getGradePoint() { /* GPA値 */ }
    public void showGradeInfo() { /* 成績表示 */ }
}

// ❌ 低凝集度：関連しない処理が混在
public class Grade {
    private String subject;
    private int score;

    // 成績とは関係ない処理が混在（低凝集度）
    public void sendEmail() { /* メール送信？ */ }
    public void printReport() { /* レポート印刷？ */ }
    public void calculateTax() { /* 税金計算？ */ }
}

    💡 よくある質問

    Q1: どのクラスにどのメソッドを配置すべきかわかりません

    A: 以下の「責任の原則」で判断してください：

    // 「誰の責任？」で判断する
public class Student {
    // ✅ 学生に関する責任
    public void addGrade(Grade grade) { /* 学生が成績を追加 */ }
    public double calculateGPA() { /* 学生のGPAを計算 */ }
    public void showStudentInfo() { /* 学生情報を表示 */ }
}

public class Grade {
    // ✅ 成績に関する責任
    public String getLetterGrade() { /* 成績の評価を返す */ }
    public boolean isPassing() { /* 成績の合格判定 */ }
    public double getGradePoint() { /* 成績のGPA値を返す */ }
}

    判断基準： - そのデータを「誰が持っているか」 - その処理を「誰がするのが自然か」 - 将来の変更が「どこに集中するか」

    Q2: クラス間でデータを渡すときの注意点は？

    A: 制御されたアクセスを心がけてください：

    // ✅ 良い例：getter/setterを使用
public class Student {
    public void addGrade(Grade grade) {
        // Gradeクラスのpublicメソッドを使用
        if (grade.getScore() >= 0 && grade.getScore() <= 100) {
            grades.add(grade);
        }
    }
}

// ❌ 悪い例：直接フィールドアクセス
public class Student {
    public void addGrade(Grade grade) {
        // privateフィールドに直接アクセス（不可能）
        if (grade.score >= 0 && grade.score <= 100) { // コンパイルエラー
            grades.add(grade);
        }
    }
}

    Q3: どこまで機能を分割すべきですか？

    A: 単一責任の原則を基準にしてください：

    // ✅ 適切な分割
public class Grade {
    // 成績に関する単一の責任
    public String getLetterGrade() { return ...; }
    public boolean isPassing() { return ...; }
    public double getGradePoint() { return ...; }
}

// ❌ 過度な分割（かえって複雑）
public class LetterGradeCalculator { ... }
public class PassingJudgment { ... }
public class GradePointCalculator { ... }
// 小さすぎる責任で分割しすぎ

    Q4: 2つのクラスが互いを参照するのは問題ありますか？

    A: 双方向参照は避けることを推奨します：

    // ✅ 良い例：単方向の関係
public class Student {
    private List<Grade> grades; // StudentがGradeを知っている
}

public class Grade {
    // GradeはStudentを直接参照しない
}

// ❌ 避けるべき例：双方向の関係
public class Student {
    private List<Grade> grades;
}

public class Grade {
    private Student student; // 循環参照で複雑になりやすい
}

    import java.util.ArrayList;
import java.util.List;

public class ClassManager {
    private List<Student> students;
    private String className;
    private String semester;

    public ClassManager(String className, String semester) {
        this.className = className;
        this.semester = semester;
        this.students = new ArrayList<>();
    }

    // 学生管理メソッド
    public void addStudent(Student student) {
        if (student != null && !students.contains(student)) {
            students.add(student);
            System.out.println(className + "に" + student.getName() + "を追加しました");
        }
    }

    public Student findStudent(String name) {
        for (Student student : students) {
            if (student.getName().equals(name)) {
                return student;
            }
        }
        return null;
    }

    public List<Student> getStudents() {
        return new ArrayList<>(students);
    }

    // 統計メソッド
    public double calculateClassGPA() {
        if (students.isEmpty()) return 0.0;

        double totalGPA = 0.0;
        for (Student student : students) {
            totalGPA += student.calculateGPA();
        }
        return Math.round(totalGPA / students.size() * 100.0) / 100.0;
    }

    public Student getTopStudent() {
        if (students.isEmpty()) return null;

        Student topStudent = students.get(0);
        for (Student student : students) {
            if (student.calculateGPA() > topStudent.calculateGPA()) {
                topStudent = student;
            }
        }
        return topStudent;
    }

    public List<Student> getStudentsByGPARange(double minGPA, double maxGPA) {
        List<Student> result = new ArrayList<>();
        for (Student student : students) {
            double gpa = student.calculateGPA();
            if (gpa >= minGPA && gpa <= maxGPA) {
                result.add(student);
            }
        }
        return result;
    }

    // レポートメソッド
    public void showClassReport() {
        System.out.println("=== " + className + " (" + semester + ") クラスレポート ===");
        System.out.printf("総学生数: %d人%n", students.size());
        System.out.printf("クラス平均GPA: %.2f%n", calculateClassGPA());

        Student topStudent = getTopStudent();
        if (topStudent != null) {
            System.out.printf("最優秀学生: %s (GPA: %.2f)%n", 
                topStudent.getName(), topStudent.calculateGPA());
        }

        System.out.println("\n=== 学生一覧 ===");
        for (Student student : students) {
            System.out.printf("%s: GPA %.2f (%d科目履修)%n",
                student.getName(), 
                student.calculateGPA(),
                student.getGrades().size());
        }
    }

    public void showSubjectReport(String subject) {
        System.out.println("=== " + subject + " 科目別レポート ===");

        List<Grade> subjectGrades = new ArrayList<>();
        for (Student student : students) {
            List<Grade> grades = student.getGradesBySubject(subject);
            subjectGrades.addAll(grades);
        }

        if (subjectGrades.isEmpty()) {
            System.out.println("該当する成績がありません");
            return;
        }

        double totalScore = 0.0;
        for (Grade grade : subjectGrades) {
            totalScore += grade.getScore();
        }
        double averageScore = totalScore / subjectGrades.size();

        System.out.printf("履修者数: %d人%n", subjectGrades.size());
        System.out.printf("平均点: %.1f点%n", averageScore);

        System.out.println("\n=== 個別成績 ===");
        for (Grade grade : subjectGrades) {
