    public class Student {
    private String name;
    private int age;
    private double gpa;
    private List<String> subjects;
    private static int totalStudents = 0;

    // デフォルトコンストラクタ
    public Student() {
        this("未設定", 0, 0.0);
    }

    // 引数ありコンストラクタ
    public Student(String name, int age, double gpa) {
        this.name = name;
        this.age = age;
        this.gpa = gpa;
        this.subjects = new ArrayList<>();
        totalStudents++;
    }
}

    🔧 インスタンスメソッドとは何か？

    オブジェクトの「動作」を定義する技術

    インスタンスメソッドは、「特定のオブジェクト（インスタンス）が持つ動作」を定義したものです。

    現実世界の例： - 車のオブジェクト: start()（エンジン始動）、stop()（停止）、accelerate()（加速） - スマホのオブジェクト: makeCall()（電話をかける）、sendMessage()（メッセージ送信） - 学生のオブジェクト: study()（勉強する）、takeExam()（試験を受ける）

    📘 インスタンスメソッドの概念

    staticメソッドとの違い

    public class Student {
    private String name;
    private double gpa;

    // ✅ インスタンスメソッド：特定の学生の動作
    public void showInfo() {
        System.out.println("名前：" + this.name + ", GPA：" + this.gpa);
    }

    // ✅ staticメソッド：学生クラス全体に関わる動作
    public static void showStudentCount() {
        System.out.println("総学生数：" + totalStudents);
    }
}

    ステップ4: thisの高度な使い方

    thisキーワードをより深く理解し、実用的に活用しましょう：

    public class Student {
    private String name;
    private int age;
    private double gpa;
    private List<String> subjects;

    // コンストラクタでのthis活用（レッスン4.5の復習）
    public Student(String name, int age, double gpa) {
        this.name = name;    // this.name（フィールド） = name（引数）
        this.age = age;
        this.gpa = gpa;
        this.subjects = new ArrayList<>();
    }

    // 同名引数とthisの使い分け
    public void updateInfo(String name, int age) {
        // thisがないと引数同士の代入になってしまう
        this.name = name;  // フィールドに代入
        this.age = age;    // フィールドに代入

        System.out.println("情報を更新しました：" + this.name + "(" + this.age + "歳)");
    }

    // 他のオブジェクトとの比較でthisを活用
    public boolean isSameAge(Student other) {
        return this.age == other.age;
    }

    public boolean hasHigherGpaThan(Student other) {
        return this.gpa > other.gpa;
    }

    // thisを使った現在オブジェクトの返却（後述のメソッドチェーンで活用）
    public Student introduceYourself() {
        System.out.println("はじめまして、" + this.name + "です。" + 
                         this.age + "歳で、GPAは" + this.gpa + "です。");
        return this;  // 自分自身を返す
    }
}

    import java.util.*;

public class Student {
    private String name;
    private int age;
    private double gpa;
    private List<String> subjects;
    private String major;

    // コンストラクタ
    public Student(String name, int age, double gpa, String major) {
        this.name = name;
        this.age = age;
        this.gpa = gpa;
        this.major = major;
        this.subjects = new ArrayList<>();
    }

    // === 情報表示メソッド群 ===

    public void showBasicInfo() {
        System.out.println("=== 基本情報 ===");
        System.out.println("名前：" + this.name);
        System.out.println("年齢：" + this.age + "歳");
        System.out.println("GPA：" + this.gpa);
        System.out.println("専攻：" + this.major);
        System.out.println("成績レベル：" + getGradeLevel());
    }

    public void showDetailedInfo() {
        showBasicInfo();
        System.out.println("=== 履修情報 ===");
        System.out.println("履修科目数：" + this.subjects.size());
        System.out.println("学生タイプ：" + (isFullTimeStudent() ? "正規生" : "聴講生"));

        if (!this.subjects.isEmpty()) {
            System.out.println("履修科目：");
            for (int i = 0; i < this.subjects.size(); i++) {
                System.out.println("  " + (i + 1) + ". " + this.subjects.get(i));
            }
        }
    }

    public String getSummary() {
        return this.name + "（" + this.age + "歳・" + this.major + 
               "・GPA:" + this.gpa + "・" + getGradeLevel() + "）";
    }

    // === 状態変更メソッド群（メソッドチェーン対応） ===

    public Student updateGpa(double newGpa) {
        if (newGpa < 0.0 || newGpa > 4.0) {
            System.out.println("エラー：GPAは0.0-4.0の範囲で入力してください");
        } else {
            double oldGpa = this.gpa;
            this.gpa = newGpa;
            System.out.println(this.name + "のGPA更新：" + oldGpa + " → " + newGpa);
        }
        return this;
    }

    public Student addSubject(String subject) {
        if (subject == null || subject.trim().isEmpty()) {
            System.out.println("エラー：科目名を正しく入力してください");
        } else if (this.subjects.contains(subject)) {
            System.out.println(this.name + "は既に" + subject + "を履修済みです");
        } else {
            this.subjects.add(subject);
            System.out.println(this.name + "が" + subject + "を履修開始しました");
        }
        return this;
    }

    public Student removeSubject(String subject) {
        if (this.subjects.remove(subject)) {
            System.out.println(this.name + "が" + subject + "の履修を終了しました");
        } else {
            System.out.println(this.name + "は" + subject + "を履修していません");
        }
        return this;
    }

    public Student updateAge(int newAge) {
        if (newAge > 0 && newAge != this.age) {
            this.age = newAge;
            System.out.println(this.name + "の年齢を更新しました：" + newAge + "歳");
        }
        return this;
    }

    // === 判定・比較メソッド群 ===

    public String getGradeLevel() {
        if (this.gpa >= 3.7) return "最優秀";
        else if (this.gpa >= 3.3) return "優秀";
        else if (this.gpa >= 3.0) return "良好";
        else if (this.gpa >= 2.7) return "普通";
        else if (this.gpa >= 2.0) return "要注意";
        else return "要指導";
    }

    public boolean isFullTimeStudent() {
        return this.subjects.size() >= 4;
    }

    public boolean hasHigherGpaThan(Student other) {
        return this.gpa > other.gpa;
    }

    public boolean isSameAge(Student other) {
        return this.age == other.age;
    }

    // === ユーティリティメソッド ===

    public Student introduceSelf() {
        System.out.println("はじめまして、" + this.name + "です。");
        System.out.println(this.major + "を専攻している" + this.age + "歳で、");
        System.out.println("現在のGPAは" + this.gpa + "（" + getGradeLevel() + "レベル）です。");
        return this;
    }

    // テスト用メインメソッド
    public static void main(String[] args) {
        System.out.println("=== 多機能学生管理システム ===\n");

        // 学生オブジェクトの作成
        Student tanaka = new Student("田中太郎", 20, 3.2, "コンピュータ科学");
        Student sato = new Student("佐藤花子", 19, 3.8, "数学");
        Student yamada = new Student("山田次郎", 21, 2.9, "物理学");

        // メソッドチェーンによる連続処理
        System.out.println("=== 田中太郎の学習履歴 ===");
        tanaka.introduceSelf()
              .addSubject("プログラミング基礎")
              .addSubject("データ構造")
              .addSubject("アルゴリズム")
              .addSubject("データベース")
              .updateGpa(3.5);

        System.out.println("\n=== 佐藤花子の学習履歴 ===");
        sato.addSubject("微積分")
            .addSubject("線形代数")
            .addSubject("統計学")
            .updateAge(20)
            .showDetailedInfo();

        System.out.println("\n=== 学生間比較 ===");
        System.out.println("田中のGPAは佐藤より高い？ " + tanaka.hasHigherGpaThan(sato));
        System.out.println("田中と山田は同い年？ " + tanaka.isSameAge(yamada));
        System.out.println("佐藤と山田は同じ専攻？ " + sato.isSameMajor(yamada));

        System.out.println("\n=== 全学生の要約 ===");
        System.out.println("1. " + tanaka.getSummary());
        System.out.println("2. " + sato.getSummary());
        System.out.println("3. " + yamada.getSummary());

        System.out.println("\n=== システム完成 ===");
        System.out.println("インスタンスメソッドを活用した学生管理システムが完成しました！");
    }

    💡 よくある質問

    Q1: staticメソッドとインスタンスメソッドの使い分けがわかりません

    A: 以下の判断基準で区別してください：

    public class Student {
    private String name;
    private static int totalStudents = 0;

    // ✅ インスタンスメソッド：特定の学生に関する処理
    public void showInfo() {
        System.out.println("名前：" + this.name); // thisが必要
    }

    // ✅ staticメソッド：学生クラス全体に関する処理
    public static void showTotalCount() {
        System.out.println("総学生数：" + totalStudents); // thisは不要
    }
}

    判断ポイント： - thisキーワードが必要なら → インスタンスメソッド - クラス全体の処理なら → staticメソッド

    Q2: メソッドチェーンはいつ使うべきですか？

    A: 以下の場面で有効です：

    // ✅ 良い例：設定系の連続処理
student.setName("田中")
       .setAge(20)
       .setGpa(3.5)
       .showInfo();

// ❌ 避けるべき例：意味の異なる処理の混在
student.showInfo()     // 表示処理
       .addSubject("数学")  // 状態変更
       .calculateGpa();     // 計算処理（混乱を招く）

    適用基準： - 設定・更新系の処理が連続する場合 - 処理の流れが自然で読みやすい場合

    Q3: thisキーワードは常に必要ですか？

    A: 必須ではありませんが、推奨します：

    public class Student {
    private String name;

    // 引数とフィールドが同名の場合：thisが必須
    public void setName(String name) {
        this.name = name; // thisがないとエラー
    }

    // 引数とフィールドが異なる名前の場合：thisは省略可能だが推奨
    public void setName(String newName) {
        this.name = newName; // thisがあると分かりやすい
        name = newName;      // thisなしでも動作するが非推奨
    }
}

    Q4: メソッドの戻り値はいつvoidにして、いつ値を返すべきですか？

    A: 以下の基準で判断してください：

    // ✅ void：表示・出力系の処理
public void showInfo() {
    System.out.println(...);
}

// ✅ 値を返す：計算・判定・取得系の処理
public String getName() {
    return this.name;
}

public boolean isAdult() {
    return this.age >= 18;
}

// ✅ 自分自身を返す：メソッドチェーン対応
public Student setAge(int age) {
    this.age = age;
    return this;
}
